017package org.apache.logging.log4j.spring.boot;
018
019import java.util.concurrent.locks.Lock;
020import java.util.concurrent.locks.ReentrantLock;
021
022import org.apache.logging.log4j.LogManager;
023import org.apache.logging.log4j.internal.LogManagerStatus;
024import org.springframework.core.env.Environment;
025
026/**
027 * Provides access to the Spring Environment.
028 */
029public class SpringEnvironmentHolder {
030
031    private volatile Environment environment;
032    private Lock lock = new ReentrantLock();
033
034
035    protected Environment getEnvironment() {
036        if (environment == null && LogManagerStatus.isInitialized() && LogManager.getFactory() != null &&
037                LogManager.getFactory().hasContext(SpringEnvironmentHolder.class.getName(), null, false)) {
038            lock.lock();
039            try {
040                if (environment == null) {
041                    Object obj = LogManager.getContext(false).getObject(Log4j2CloudConfigLoggingSystem.ENVIRONMENT_KEY);
042                    environment = obj instanceof Environment ? (Environment) obj : null;
043                }
044            } finally {
045                lock.unlock();
046            }
047        }
048        return environment;
049    }
050}

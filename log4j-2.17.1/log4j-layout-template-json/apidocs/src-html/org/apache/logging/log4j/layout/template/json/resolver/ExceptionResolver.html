032/**
033 * Exception resolver.
034 * <p>
035 * Note that this resolver is toggled by {@link
036 * JsonTemplateLayout.Builder#setStackTraceEnabled(boolean) stackTraceEnabled}
037 * layout configuration, which is by default populated from <tt>log4j.layout.jsonTemplate.stackTraceEnabled</tt>
038 * system property.
039 *
040 * <h3>Configuration</h3>
041 *
042 * <pre>
043 * config              = field , [ stringified ] , [ stackTrace ]
044 * field               = "field" -> ( "className" | "message" | "stackTrace" )
045 *
046 * stackTrace          = "stackTrace" -> (
047 *                         [ stringified ]
048 *                       , [ elementTemplate ]
049 *                       )
050 *
051 * stringified         = "stringified" -> ( boolean | truncation )
052 * truncation          = "truncation" -> (
053 *                         [ suffix ]
054 *                       , [ pointMatcherStrings ]
055 *                       , [ pointMatcherRegexes ]
056 *                       )
057 * suffix              = "suffix" -> string
058 * pointMatcherStrings = "pointMatcherStrings" -> string[]
059 * pointMatcherRegexes = "pointMatcherRegexes" -> string[]
060 *
061 * elementTemplate     = "elementTemplate" -> object
062 * </pre>
063 *
064 * <tt>stringified</tt> is set to <tt>false</tt> by default.
065 * <tt>stringified</tt> at the root level is <b>deprecated</b> in favor of
066 * <tt>stackTrace.stringified</tt>, which has precedence if both are provided.
067 * <p>
068 * <tt>pointMatcherStrings</tt> and <tt>pointMatcherRegexes</tt> enable the
069 * truncation of stringified stack traces after the given matching point. If
070 * both parameters are provided, <tt>pointMatcherStrings</tt> will be checked
071 * first.
072 * <p>
073 * If a stringified stack trace truncation takes place, it will be indicated
074 * with <tt>suffix</tt>, which by default is set to the configured
075 * <tt>truncatedStringSuffix</tt> in the layout, unless explicitly provided.
076 * <p>
077 * <tt>elementTemplate</tt> is an object describing the template to be used
078 * while resolving the {@link StackTraceElement} array. If <tt>stringified</tt>
079 * is set to <tt>true</tt>, <tt>elementTemplate</tt> will be discarded. By
080 * default, <tt>elementTemplate</tt> is set to <tt>null</tt> and rather
081 * populated from the layout configuration. That is, the stack trace element
082 * template can also be provided using {@link JsonTemplateLayout.Builder#setStackTraceElementTemplate(String) stackTraceElementTemplate}
083 * and {@link JsonTemplateLayout.Builder#setStackTraceElementTemplateUri(String) setStackTraceElementTemplateUri}
084 * layout configuration parameters. The template to be employed is determined
085 * in the following order:
086 * <ol>
087 * <li><tt>elementTemplate</tt> provided in the resolver configuration
088 * <li><tt>stackTraceElementTemplate</tt> parameter from layout configuration
089 * (the default is populated from <tt>log4j.layout.jsonTemplate.stackTraceElementTemplate</tt>
090 * system property)
091 * <li><tt>stackTraceElementTemplateUri</tt> parameter from layout configuration
092 * (the default is populated from <tt>log4j.layout.jsonTemplate.stackTraceElementTemplateUri</tt>
093 * system property)
094 * </ol>
095 * <p>
096 * See {@link StackTraceElementResolver}
097 * for the list of available resolvers in a stack trace element template.
098 *
099 * <h3>Examples</h3>
100 *
101 * Resolve <tt>logEvent.getThrown().getClass().getCanonicalName()</tt>:
102 *
103 * <pre>
104 *  {
105 *   "$resolver": "exception",
106 *   "field": "className"
107 * }
108 * </pre>
109 *
110 * Resolve the stack trace into a list of <tt>StackTraceElement</tt> objects:
111 *
112 * <pre>
113 *  {
114 *   "$resolver": "exception",
115 *   "field": "stackTrace"
116 * }
117 * </pre>
118 *
119 * Resolve the stack trace into a string field:
120 *
121 * <pre>
122 *  {
123 *   "$resolver": "exception",
124 *   "field": "stackTrace",
125 *   "stackTrace": {
126 *     "stringified": true
127 *   }
128 * }
129 * </pre>
130 *
131 * Resolve the stack trace into a string field
132 * such that the content will be truncated after the given point matcher:
133 *
134 * <pre>
135 *  {
136 *   "$resolver": "exception",
137 *   "field": "stackTrace",
138 *   "stackTrace": {
139 *     "stringified": {
140 *       "truncation": {
141 *         "suffix": ">",
142 *         "pointMatcherStrings": ["at javax.servlet.http.HttpServlet.service"]
143 *       }
144 *     }
145 *   }
146 * }
147 * </pre>
148 *
149 * Resolve the stack trace into an object described by the provided stack trace
150 * element template:
151 *
152 * <pre>
153 *  {
154 *   "$resolver": "exception",
155 *   "field": "stackTrace",
156 *   "stackTrace": {
157 *     "elementTemplate": {
158 *       "class": {
159 *         "$resolver": "stackTraceElement",
160 *         "field": "className"
161 *       },
162 *       "method": {
163 *         "$resolver": "stackTraceElement",
164 *         "field": "methodName"
165 *       },
166 *       "file": {
167 *         "$resolver": "stackTraceElement",
168 *         "field": "fileName"
169 *       },
170 *       "line": {
171 *         "$resolver": "stackTraceElement",
172 *         "field": "lineNumber"
173 *       }
174 *     }
175 *   }
176 * }
177 * </pre>
178 *
179 * @see JsonTemplateLayout.Builder#getTruncatedStringSuffix()
180 * @see JsonTemplateLayoutDefaults#getTruncatedStringSuffix()
181 * @see JsonTemplateLayout.Builder#getStackTraceElementTemplate()
182 * @see JsonTemplateLayoutDefaults#getStackTraceElementTemplate()
183 * @see JsonTemplateLayout.Builder#getStackTraceElementTemplateUri()
184 * @see JsonTemplateLayoutDefaults#getStackTraceElementTemplateUri()
185 * @see ExceptionRootCauseResolver
186 * @see StackTraceElementResolver
187 */
188public class ExceptionResolver implements EventResolver {
189
190    private static final Logger LOGGER = StatusLogger.getLogger();
191
192    private static final EventResolver NULL_RESOLVER =
193            (ignored, jsonGenerator) -> jsonGenerator.writeNull();
194
195    private final boolean stackTraceEnabled;
196
197    private final EventResolver internalResolver;
198
199    ExceptionResolver(
200            final EventResolverContext context,
201            final TemplateResolverConfig config) {
202        this.stackTraceEnabled = context.isStackTraceEnabled();
203        this.internalResolver = createInternalResolver(context, config);
204    }
205
206    EventResolver createInternalResolver(
207            final EventResolverContext context,
208            final TemplateResolverConfig config) {
209        final String fieldName = config.getString("field");
210        if ("className".equals(fieldName)) {
211            return createClassNameResolver();
212        } else if ("message".equals(fieldName)) {
213            return createMessageResolver();
214        } else if ("stackTrace".equals(fieldName)) {
215            return createStackTraceResolver(context, config);
216        }
217        throw new IllegalArgumentException("unknown field: " + config);
218    }
219
220    private EventResolver createClassNameResolver() {
221        return (final LogEvent logEvent, final JsonWriter jsonWriter) -> {
222            final Throwable exception = extractThrowable(logEvent);
223            if (exception == null) {
224                jsonWriter.writeNull();
225            } else {
226                String exceptionClassName = exception.getClass().getCanonicalName();
227                jsonWriter.writeString(exceptionClassName);
228            }
229        };
230    }
231
232    private EventResolver createMessageResolver() {
233        return (final LogEvent logEvent, final JsonWriter jsonWriter) -> {
234            final Throwable exception = extractThrowable(logEvent);
235            if (exception == null) {
236                jsonWriter.writeNull();
237            } else {
238                String exceptionMessage = exception.getMessage();
239                jsonWriter.writeString(exceptionMessage);
240            }
241        };
242    }
243
244    private EventResolver createStackTraceResolver(
245            final EventResolverContext context,
246            final TemplateResolverConfig config) {
247        if (!context.isStackTraceEnabled()) {
248            return NULL_RESOLVER;
249        }
250        final boolean stringified = isStackTraceStringified(config);
251        return stringified
252                ? createStackTraceStringResolver(context, config)
253                : createStackTraceObjectResolver(context, config);
254    }
255
256    private static boolean isStackTraceStringified(
257            final TemplateResolverConfig config) {
258        final Boolean stringifiedOld = config.getBoolean("stringified");
259        if (stringifiedOld != null) {
260            LOGGER.warn(
261                    "\"stringified\" flag at the root level of an exception " +
262                            "[root cause] resolver is deprecated in favor of " +
263                            "\"stackTrace.stringified\"");
264        }
265        final Object stringifiedNew =
266                config.getObject(new String[]{"stackTrace", "stringified"});
267        if (stringifiedOld == null && stringifiedNew == null) {
268            return false;
269        } else if (stringifiedNew == null) {
270            return stringifiedOld;
271        } else {
272            return !(stringifiedNew instanceof Boolean) || (boolean) stringifiedNew;
273        }
274    }
275
276    private EventResolver createStackTraceStringResolver(
277            final EventResolverContext context,
278            final TemplateResolverConfig config) {
279
280        // Read the configuration.
281        final String truncationSuffix =
282                readTruncationSuffix(context, config);
283        final List<String> truncationPointMatcherStrings =
284                readTruncationPointMatcherStrings(config);
285        final List<String> truncationPointMatcherRegexes =
286                readTruncationPointMatcherRegexes(config);
287
288        // Create the resolver.
289        final StackTraceStringResolver resolver =
290                new StackTraceStringResolver(
291                        context,
292                        truncationSuffix,
293                        truncationPointMatcherStrings,
294                        truncationPointMatcherRegexes);
295
296        // Create the null-protected resolver.
297        return (final LogEvent logEvent, final JsonWriter jsonWriter) -> {
298            final Throwable exception = extractThrowable(logEvent);
299            if (exception == null) {
300                jsonWriter.writeNull();
301            } else {
302                resolver.resolve(exception, jsonWriter);
303            }
304        };
305
306    }
307
308    private static String readTruncationSuffix(
309            final EventResolverContext context,
310            final TemplateResolverConfig config) {
311        final String suffix = config.getString(
312                new String[]{"stackTrace", "stringified", "truncation", "suffix"});
313        return suffix != null
314                ? suffix
315                : context.getTruncatedStringSuffix();
316    }
317
318    private static List<String> readTruncationPointMatcherStrings(
319            final TemplateResolverConfig config) {
320        List<String> strings = config.getList(
321                new String[]{"stackTrace", "stringified", "truncation", "pointMatcherStrings"},
322                String.class);
323        if (strings == null) {
324            strings = Collections.emptyList();
325        }
326        return strings;
327    }
328
329    private static List<String> readTruncationPointMatcherRegexes(
330            final TemplateResolverConfig config) {
331
332        // Extract the regexes.
333        List<String> regexes = config.getList(
334                new String[]{"stackTrace", "stringified", "truncation", "pointMatcherRegexes"},
335                String.class);
336        if (regexes == null) {
337            regexes = Collections.emptyList();
338        }
339
340        // Check the regex syntax.
341        for (int i = 0; i < regexes.size(); i++) {
342            final String regex = regexes.get(i);
343            try {
344                Pattern.compile(regex);
345            } catch (final PatternSyntaxException error) {
346                final String message = String.format(
347                        "invalid truncation point matcher regex at index %d: %s",
348                        i, regex);
349                throw new IllegalArgumentException(message, error);
350            }
351        }
352
353        // Return the extracted regexes.
354        return regexes;
355
356    }
357
358    private static final Map<String, StackTraceElementResolverFactory> STACK_TRACE_ELEMENT_RESOLVER_FACTORY_BY_NAME;
359
360    static {
361        final StackTraceElementResolverFactory stackTraceElementResolverFactory =
362                StackTraceElementResolverFactory.getInstance();
363        STACK_TRACE_ELEMENT_RESOLVER_FACTORY_BY_NAME =
364                Collections.singletonMap(
365                        stackTraceElementResolverFactory.getName(),
366                        stackTraceElementResolverFactory);
367    }
368
369    private EventResolver createStackTraceObjectResolver(
370            final EventResolverContext context,
371            final TemplateResolverConfig config) {
372        final TemplateResolver<StackTraceElement> stackTraceElementResolver =
373                createStackTraceElementResolver(context, config);
374        final StackTraceObjectResolver stackTraceResolver =
375                new StackTraceObjectResolver(stackTraceElementResolver);
376        return (final LogEvent logEvent, final JsonWriter jsonWriter) -> {
377            final Throwable throwable = extractThrowable(logEvent);
378            if (throwable == null) {
379                jsonWriter.writeNull();
380            } else {
381                stackTraceResolver.resolve(throwable, jsonWriter);
382            }
383        };
384    }
385
386    private static TemplateResolver<StackTraceElement> createStackTraceElementResolver(
387            final EventResolverContext context,
388            final TemplateResolverConfig config) {
389        final StackTraceElementResolverStringSubstitutor substitutor =
390                new StackTraceElementResolverStringSubstitutor(
391                        context.getSubstitutor().getInternalSubstitutor());
392        final StackTraceElementResolverContext stackTraceElementResolverContext =
393                StackTraceElementResolverContext
394                        .newBuilder()
395                        .setResolverFactoryByName(STACK_TRACE_ELEMENT_RESOLVER_FACTORY_BY_NAME)
396                        .setSubstitutor(substitutor)
397                        .setJsonWriter(context.getJsonWriter())
398                        .build();
399        final String stackTraceElementTemplate =
400                findEffectiveStackTraceElementTemplate(context, config);
401        return TemplateResolvers.ofTemplate(
402                stackTraceElementResolverContext,
403                stackTraceElementTemplate);
404    }
405
406    private static String findEffectiveStackTraceElementTemplate(
407            final EventResolverContext context,
408            final TemplateResolverConfig config) {
409
410        // First, check the template configured in the resolver configuration.
411        final Object stackTraceElementTemplateObject =
412                config.getObject(new String[]{"stackTrace", "elementTemplate"});
413        if (stackTraceElementTemplateObject != null) {
414            final JsonWriter jsonWriter = context.getJsonWriter();
415            return jsonWriter.use(() -> jsonWriter.writeValue(stackTraceElementTemplateObject));
416        }
417
418        // Otherwise, use the template provided in the context.
419        return context.getStackTraceElementTemplate();
420
421    }
422
423    Throwable extractThrowable(final LogEvent logEvent) {
424        return logEvent.getThrown();
425    }
426
427    static String getName() {
428        return "exception";
429    }
430
431    @Override
432    public boolean isResolvable() {
433        return stackTraceEnabled;
434    }
435
436    @Override
437    public boolean isResolvable(final LogEvent logEvent) {
438        return stackTraceEnabled && logEvent.getThrown() != null;
439    }
440
441    @Override
442    public void resolve(
443            final LogEvent logEvent,
444            final JsonWriter jsonWriter) {
445        internalResolver.resolve(logEvent, jsonWriter);
446    }

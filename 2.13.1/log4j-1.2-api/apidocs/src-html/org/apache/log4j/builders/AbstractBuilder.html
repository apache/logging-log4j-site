017package org.apache.log4j.builders;
018
019import org.apache.log4j.bridge.FilterAdapter;
020import org.apache.log4j.bridge.FilterWrapper;
021import org.apache.log4j.helpers.OptionConverter;
022import org.apache.log4j.spi.Filter;
023import org.apache.logging.log4j.Level;
024import org.apache.logging.log4j.Logger;
025import org.apache.logging.log4j.core.filter.CompositeFilter;
026import org.apache.logging.log4j.core.filter.ThresholdFilter;
027import org.apache.logging.log4j.status.StatusLogger;
028
029import java.util.ArrayList;
030import java.util.List;
031import java.util.Properties;
032
033/**
034 * Base class for Log4j 1 component builders.
035 */
036public abstract class AbstractBuilder {
037
038    private static Logger LOGGER = StatusLogger.getLogger();
039    protected static final String FILE_PARAM = "File";
040    protected static final String APPEND_PARAM = "Append";
041    protected static final String BUFFERED_IO_PARAM = "BufferedIO";
042    protected static final String BUFFER_SIZE_PARAM = "BufferSize";
043    protected static final String MAX_SIZE_PARAM = "MaxFileSize";
044    protected static final String MAX_BACKUP_INDEX = "MaxBackupIndex";
045    protected static final String RELATIVE = "RELATIVE";
046
047    private final String prefix;
048    private final Properties props;
049
050    public AbstractBuilder() {
051        this.prefix = null;
052        this.props = new Properties();
053    }
054
055    public AbstractBuilder(String prefix, Properties props) {
056        this.prefix = prefix + ".";
057        this.props = props;
058    }
059
060    public String getProperty(String key) {
061        return props.getProperty(prefix + key);
062    }
063
064    public String getProperty(String key, String defaultValue) {
065        return props.getProperty(prefix + key, defaultValue);
066    }
067
068    public boolean getBooleanProperty(String key) {
069        return Boolean.parseBoolean(props.getProperty(prefix + key, Boolean.FALSE.toString()));
070    }
071
072    public int getIntegerProperty(String key, int defaultValue) {
073        String value = props.getProperty(key);
074        try {
075            if (value != null) {
076                return Integer.parseInt(value);
077            }
078        } catch (Exception ex) {
079            LOGGER.warn("Error converting value {} of {} to an integer: {}", value, key, ex.getMessage());
080        }
081        return defaultValue;
082    }
083
084    public Properties getProperties() {
085        return props;
086    }
087
088
089    protected org.apache.logging.log4j.core.Filter buildFilters(String level, Filter filter) {
090        if (level != null && filter != null) {
091            List<org.apache.logging.log4j.core.Filter> filterList = new ArrayList<>();
092            org.apache.logging.log4j.core.Filter thresholdFilter =
093                    ThresholdFilter.createFilter(OptionConverter.convertLevel(level, Level.TRACE),
094                            org.apache.logging.log4j.core.Filter.Result.NEUTRAL,
095                            org.apache.logging.log4j.core.Filter.Result.DENY);
096            filterList.add(thresholdFilter);
097            Filter f = filter;
098            while (f != null) {
099                if (filter instanceof FilterWrapper) {
100                    filterList.add(((FilterWrapper) f).getFilter());
101                } else {
102                    filterList.add(new FilterAdapter(f));
103                }
104                f = f.next;
105            }
106            return CompositeFilter.createFilters(filterList.toArray(new org.apache.logging.log4j.core.Filter[0]));
107        } else if (level != null) {
108            return ThresholdFilter.createFilter(OptionConverter.convertLevel(level, Level.TRACE),
109                    org.apache.logging.log4j.core.Filter.Result.NEUTRAL,
110                    org.apache.logging.log4j.core.Filter.Result.DENY);
111        } else if (filter != null) {
112            if (filter instanceof FilterWrapper) {
113                return ((FilterWrapper) filter).getFilter();
114            } else {
115                return new FilterAdapter(filter);
116            }
117        }
118        return null;
119    }
120}

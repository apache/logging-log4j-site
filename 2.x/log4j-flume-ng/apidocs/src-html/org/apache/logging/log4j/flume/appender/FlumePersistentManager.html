122        /**
123         * Returns a FlumeAvroManager.
124         * @param name The name of the manager.
125         * @param agents The agents to use.
126         * @param properties Properties to pass to the Manager.
127         * @param batchSize The number of events to include in a batch.
128         * @param retries The number of times to retry connecting before giving up.
129         * @param connectionTimeout The amount of time to wait to establish a connection.
130         * @param requestTimeout The amount of time to wait for a response to a request.
131         * @param delay Amount of time to delay before delivering a batch.
132         * @param dataDir The location of the Berkeley database.
133         * @return A FlumeAvroManager.
134         */
135        public static FlumePersistentManager getManager(final String name, final Agent[] agents,
136                                                        final Property[] properties, int batchSize, final int retries,
137                                                        final int connectionTimeout, final int requestTimeout,
138                                                        final int delay, final String dataDir) {
139            if (agents == null || agents.length == 0) {
140                throw new IllegalArgumentException("At least one agent is required");
141            }
142    
143            if (batchSize <= 0) {
144                batchSize = 1;
145            }
146            final String dataDirectory = Strings.isEmpty(dataDir) ? DEFAULT_DATA_DIR : dataDir;
147    
148            final StringBuilder sb = new StringBuilder("FlumePersistent[");
149            boolean first = true;
150            for (final Agent agent : agents) {
151                if (!first) {
152                    sb.append(",");
153                }
154                sb.append(agent.getHost()).append(":").append(agent.getPort());
155                first = false;
156            }
157            sb.append("]");
158            sb.append(" ").append(dataDirectory);
159            return getManager(sb.toString(), factory, new FactoryData(name, agents, batchSize, retries,
160                connectionTimeout, requestTimeout, delay, dataDir, properties));
161        }

163        @Override
164        public void send(final Event event)  {
165            if (worker.isShutdown()) {
166                throw new LoggingException("Unable to record event");
167            }
168    
169            final Map<String, String> headers = event.getHeaders();
170            final byte[] keyData = headers.get(FlumeEvent.GUID).getBytes(UTF8);
171            try {
172                final ByteArrayOutputStream baos = new ByteArrayOutputStream();
173                final DataOutputStream daos = new DataOutputStream(baos);
174                daos.writeInt(event.getBody().length);
175                daos.write(event.getBody(), 0, event.getBody().length);
176                daos.writeInt(event.getHeaders().size());
177                for (final Map.Entry<String, String> entry : headers.entrySet()) {
178                    daos.writeUTF(entry.getKey());
179                    daos.writeUTF(entry.getValue());
180                }
181                byte[] eventData = baos.toByteArray();
182                if (secretKey != null) {
183                    final Cipher cipher = Cipher.getInstance("AES");
184                    cipher.init(Cipher.ENCRYPT_MODE, secretKey);
185                    eventData = cipher.doFinal(eventData);
186                }
187                final Future<Integer> future = threadPool.submit(new BDBWriter(keyData, eventData, environment, database,
188                    gate));
189                boolean interrupted = false;
190                int count = 0;
191                do {
192                    try {
193                        future.get();
194                    } catch (final InterruptedException ie) {
195                        interrupted = true;
196                        ++count;
197                    }
198                } while (interrupted && count <= 1);
199    
200            } catch (final Exception ex) {
201                throw new LoggingException("Exception occurred writing log event", ex);
202            }
203        }
204    
205        @Override
206        protected void releaseSub() {
207            LOGGER.debug("Shutting down FlumePersistentManager");
208            worker.shutdown();
209            threadPool.shutdown();
210            try {
211                threadPool.awaitTermination(SHUTDOWN_WAIT, TimeUnit.SECONDS);
212            } catch (final InterruptedException ie) {
213                LOGGER.warn("PersistentManager Thread pool failed to shut down");
214            }
215            try {
216                worker.join();
217            } catch (final InterruptedException ex) {
218                LOGGER.debug("Interrupted while waiting for worker to complete");
219            }
220            try {
221                LOGGER.debug("FlumePersistenceManager dataset status: {}", database.getStats(new StatsConfig()));
222                database.close();
223            } catch (final Exception ex) {
224                LOGGER.warn("Failed to close database", ex);
225            }
226            try {
227                environment.cleanLog();
228                environment.close();
229            } catch (final Exception ex) {
230                LOGGER.warn("Failed to close environment", ex);
231            }
232            super.releaseSub();
233        }

235        private void doSend(final SimpleEvent event) {
236            LOGGER.debug("Sending event to Flume");
237            super.send(event);
238        }

240        /**
241         * Thread for writing to Berkeley DB to avoid having interrupts close the database.
242         */
243        private static class BDBWriter implements Callable<Integer> {
244            private final byte[] eventData;
245            private final byte[] keyData;
246            private final Environment environment;
247            private final Database database;
248            private final Gate gate;
249    
250            public BDBWriter(final byte[] keyData, final byte[] eventData, final Environment environment,
251                             final Database database, final Gate gate) {
252                this.keyData = keyData;
253                this.eventData = eventData;
254                this.environment = environment;
255                this.database = database;
256                this.gate = gate;
257            }
258    
259            @Override
260            public  Integer call() throws Exception {
261                final DatabaseEntry key = new DatabaseEntry(keyData);
262                final DatabaseEntry data = new DatabaseEntry(eventData);
263                final Transaction txn = environment.beginTransaction(null, null);
264                try {
265                    database.put(txn, key, data);
266                    txn.commit();
267                    gate.open();
268                } catch (final Exception ex) {
269                    if (txn != null) {
270                        txn.abort();
271                    }
272                    throw ex;
273                }
274                return eventData.length;
275            }
276        }

314        /**
315         * Avro Manager Factory.
316         */
317        private static class BDBManagerFactory implements ManagerFactory<FlumePersistentManager, FactoryData> {
318    
319            /**
320             * Create the FlumeKratiManager.
321             * @param name The name of the entity to manage.
322             * @param data The data required to create the entity.
323             * @return The FlumeKratiManager.
324             */
325            @Override
326            public FlumePersistentManager createManager(final String name, final FactoryData data) {
327                SecretKey secretKey = null;
328    
329                Database database;
330                Environment environment;
331    
332                final Map<String, String> properties = new HashMap<String, String>();
333                if (data.properties != null) {
334                    for (final Property property : data.properties) {
335                        properties.put(property.getName(), property.getValue());
336                    }
337                }
338    
339                try {
340    
341                    final File dir = new File(data.dataDir);
342                    FileUtils.mkdir(dir, true);
343                    final EnvironmentConfig dbEnvConfig = new EnvironmentConfig();
344                    dbEnvConfig.setTransactional(true);
345                    dbEnvConfig.setAllowCreate(true);
346                    dbEnvConfig.setLockTimeout(5, TimeUnit.SECONDS);
347                    environment = new Environment(dir, dbEnvConfig);
348                    final DatabaseConfig dbConfig = new DatabaseConfig();
349                    dbConfig.setTransactional(true);
350                    dbConfig.setAllowCreate(true);
351                    database = environment.openDatabase(null, name, dbConfig);
352                } catch (final Exception ex) {
353                    LOGGER.error("Could not create FlumePersistentManager", ex);
354                    return null;
355                }
356    
357                try {
358                    String key = null;
359                    for (final Map.Entry<String, String> entry : properties.entrySet()) {
360                        if (entry.getKey().equalsIgnoreCase(KEY_PROVIDER)) {
361                            key = entry.getValue();
362                            break;
363                        }
364                    }
365                    if (key != null) {
366                        final PluginManager manager = new PluginManager("KeyProvider", SecretKeyProvider.class);
367                        manager.collectPlugins();
368                        final Map<String, PluginType> plugins = manager.getPlugins();
369                        if (plugins != null) {
370                            boolean found = false;
371                            for (final Map.Entry<String, PluginType> entry : plugins.entrySet()) {
372                                if (entry.getKey().equalsIgnoreCase(key)) {
373                                    found = true;
374                                    final Class cl = entry.getValue().getPluginClass();
375                                    try {
376                                        final SecretKeyProvider provider = (SecretKeyProvider) cl.newInstance();
377                                        secretKey = provider.getSecretKey();
378                                        LOGGER.debug("Persisting events using SecretKeyProvider {}", cl.getName());
379                                    } catch (final Exception ex) {
380                                        LOGGER.error("Unable to create SecretKeyProvider {}, encryption will be disabled",
381                                            cl.getName());
382                                    }
383                                    break;
384                                }
385                            }
386                            if (!found) {
387                                LOGGER.error("Unable to locate SecretKey provider {}, encryption will be disabled", key);
388                            }
389                        } else {
390                            LOGGER.error("Unable to locate SecretKey provider {}, encryption will be disabled", key);
391                        }
392                    }
393                } catch (final Exception ex) {
394                    LOGGER.warn("Error setting up encryption - encryption will be disabled", ex);
395                }
396                return new FlumePersistentManager(name, data.name, data.agents, data.batchSize, data.retries,
397                    data.connectionTimeout, data.requestTimeout, data.delay, database, environment, secretKey);
398            }
399        }

401        /**
402         * Thread that sends data to Flume and pulls it from Berkeley DB.
403         */
404        private static class WriterThread extends Thread  {
405            private volatile boolean shutdown = false;
406            private final Database database;
407            private final Environment environment;
408            private final FlumePersistentManager manager;
409            private final Gate gate;
410            private final SecretKey secretKey;
411            private final int batchSize;
412    
413            public WriterThread(final Database database, final Environment environment,
414                                final FlumePersistentManager manager, final Gate gate, final int batchsize, final SecretKey secretKey) {
415                this.database = database;
416                this.environment = environment;
417                this.manager = manager;
418                this.gate = gate;
419                this.batchSize = batchsize;
420                this.secretKey = secretKey;
421                this.setDaemon(true);
422            }
423    
424            public void shutdown() {
425                LOGGER.debug("Writer thread shutting down");
426                this.shutdown = true;
427                gate.open();
428            }
429    
430            public boolean isShutdown() {
431                return shutdown;
432            }
433    
434            @Override
435            public void run() {
436                LOGGER.trace("WriterThread started - batch size = " + batchSize + ", delay = " + manager.delay);
437                long nextBatch = System.currentTimeMillis() + manager.delay;
438                while (!shutdown) {
439                    long now = System.currentTimeMillis();
440                    if (database.count() >= batchSize || (database.count() > 0 && nextBatch < now)) {
441                        nextBatch = now + manager.delay;
442                        try {
443                            boolean errors = false;
444                            DatabaseEntry key = new DatabaseEntry();
445                            final DatabaseEntry data = new DatabaseEntry();
446    
447                            gate.close();
448                            OperationStatus status;
449                            if (batchSize > 1) {
450                                Cursor cursor = database.openCursor(null, CursorConfig.DEFAULT);
451                                try {
452                                    status = cursor.getFirst(key, data, null);
453    
454                                    final BatchEvent batch = new BatchEvent();
455                                    for (int i = 0; status == OperationStatus.SUCCESS && i < batchSize; ++i) {
456                                        final SimpleEvent event = createEvent(data);
457                                        if (event != null) {
458                                            batch.addEvent(event);
459                                        }
460                                        status = cursor.getNext(key, data, null);
461                                    }
462                                    try {
463                                        manager.send(batch);
464                                    } catch (final Exception ioe) {
465                                        LOGGER.error("Error sending events", ioe);
466                                        errors = true;
467                                    }
468                                    if (!errors) {
469                                        cursor.close();
470                                        cursor = null;
471                                        final Transaction txn = environment.beginTransaction(null, null);
472                                        try {
473                                            for (final Event event : batch.getEvents()) {
474                                                try {
475                                                    final Map<String, String> headers = event.getHeaders();
476                                                    key = new DatabaseEntry(headers.get(FlumeEvent.GUID).getBytes(UTF8));
477                                                    database.delete(txn, key);
478                                                } catch (final Exception ex) {
479                                                    LOGGER.error("Error deleting key from database", ex);
480                                                }
481                                            }
482                                            txn.commit();
483                                        } catch (final Exception ex) {
484                                            LOGGER.error("Unable to commit transaction", ex);
485                                            if (txn != null) {
486                                                txn.abort();
487                                            }
488                                        }
489                                    }
490                                } catch (final Exception ex) {
491                                    LOGGER.error("Error reading database", ex);
492                                    shutdown = true;
493                                    break;
494                                } finally {
495                                    if (cursor != null) {
496                                        cursor.close();
497                                    }
498                                }
499                            } else {
500                                Transaction txn = environment.beginTransaction(null, null);
501                                Cursor cursor = database.openCursor(txn, null);
502                                try {
503                                    status = cursor.getFirst(key, data, LockMode.RMW);
504                                    while (status == OperationStatus.SUCCESS) {
505                                        final SimpleEvent event = createEvent(data);
506                                        if (event != null) {
507                                            try {
508                                                manager.doSend(event);
509                                            } catch (final Exception ioe) {
510                                                errors = true;
511                                                LOGGER.error("Error sending event", ioe);
512                                                break;
513                                            }
514                                            try {
515                                                cursor.delete();
516                                            } catch (final Exception ex) {
517                                                LOGGER.error("Unable to delete event", ex);
518                                            }
519                                        }
520                                        status = cursor.getNext(key, data, LockMode.RMW);
521                                    }
522                                    if (cursor != null) {
523                                        cursor.close();
524                                        cursor = null;
525                                    }
526                                    txn.commit();
527                                    txn = null;
528                                } catch (final Exception ex) {
529                                    LOGGER.error("Error reading or writing to database", ex);
530                                    shutdown = true;
531                                    break;
532                                } finally {
533                                    if (cursor != null) {
534                                        cursor.close();
535                                    }
536                                    if (txn != null) {
537                                        txn.abort();
538                                    }
539                                }
540                            }
541                            if (errors) {
542                                Thread.sleep(manager.delay);
543                                continue;
544                            }
545                        } catch (final Exception ex) {
546                            LOGGER.warn("WriterThread encountered an exception. Continuing.", ex);
547                        }
548                    } else {
549                        while (!shutdown && (database.count() == 0 || database.count() < batchSize && nextBatch > now)) {
550                            try {
551                                final long interval = nextBatch - now;
552                                gate.waitForOpen(interval, TimeUnit.MILLISECONDS);
553                            } catch (final InterruptedException ie) {
554                                LOGGER.warn("WriterThread interrupted, continuing");
555                            } catch (final Exception ex) {
556                                LOGGER.error("WriterThread encountered an exception waiting for work", ex);
557                                break;
558                            }
559                            now = System.currentTimeMillis();
560                            if (database.count() == 0) {
561                                nextBatch = now + manager.delay;
562                            }
563                        }
564                        LOGGER.debug("WriterThread ready to work");
565                    }
566                }
567                LOGGER.trace("WriterThread exiting");
568            }

570            private SimpleEvent createEvent(final DatabaseEntry data) {
571                final SimpleEvent event = new SimpleEvent();
572                try {
573                    byte[] eventData = data.getData();
574                    if (secretKey != null) {
575                        final Cipher cipher = Cipher.getInstance("AES");
576                        cipher.init(Cipher.DECRYPT_MODE, secretKey);
577                        eventData = cipher.doFinal(eventData);
578                    }
579                    final ByteArrayInputStream bais = new ByteArrayInputStream(eventData);
580                    final DataInputStream dais = new DataInputStream(bais);
581                    int length = dais.readInt();
582                    final byte[] bytes = new byte[length];
583                    dais.read(bytes, 0, length);
584                    event.setBody(bytes);
585                    length = dais.readInt();
586                    final Map<String, String> map = new HashMap<String, String>(length);
587                    for (int i = 0; i < length; ++i) {
588                        final String headerKey = dais.readUTF();
589                        final String value = dais.readUTF();
590                        map.put(headerKey, value);
591                    }
592                    event.setHeaders(map);
593                    return event;
594                } catch (final Exception ex) {
595                    LOGGER.error("Error retrieving event", ex);
596                    return null;
597                }
598            }

602        /**
603         * Factory that creates Daemon threads that can be properly shut down.
604         */
605        private static class DaemonThreadFactory implements ThreadFactory {
606            private static final AtomicInteger POOL_NUMBER = new AtomicInteger(1);
607            private final ThreadGroup group;
608            private final AtomicInteger threadNumber = new AtomicInteger(1);
609            private final String namePrefix;
610    
611            public DaemonThreadFactory() {
612                final SecurityManager securityManager = System.getSecurityManager();
613                group = (securityManager != null) ? securityManager.getThreadGroup() :
614                    Thread.currentThread().getThreadGroup();
615                namePrefix = "DaemonPool-" + POOL_NUMBER.getAndIncrement() + "-thread-";
616            }
617    
618            @Override
619                    public Thread newThread(final Runnable r) {
620                final Thread thread = new Thread(group, r, namePrefix + threadNumber.getAndIncrement(), 0);
621                thread.setDaemon(true);
622                if (thread.getPriority() != Thread.NORM_PRIORITY) {
623                    thread.setPriority(Thread.NORM_PRIORITY);
624                }
625                return thread;
626            }
627        }

629        private static class Gate {
630            private static class Synchronizer extends AbstractQueuedSynchronizer {
631                boolean isSignalled() { return getState() != 0; }
632    
633                protected int tryAcquireShared(int ignore) {
634                    return isSignalled()? 1 : -1;
635                }
636    
637                protected boolean tryReleaseShared(int state) {
638                    setState(state);
639                    return true;
640                }
641            }
642    
643            private final Synchronizer sync = new Synchronizer();
644            public boolean isSignalled() { return sync.isSignalled(); }
645            public void open()         { sync.releaseShared(1); }
646            public void close()        { sync.releaseShared(0); }
647            public void waitForOpen(long timeout, TimeUnit timeUnit) throws InterruptedException {
648                sync.tryAcquireSharedNanos(1, timeUnit.toNanos(timeout));
649            }
650        }

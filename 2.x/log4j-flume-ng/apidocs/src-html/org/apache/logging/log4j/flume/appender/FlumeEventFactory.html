017    package org.apache.logging.log4j.flume.appender;
018    
019    import org.apache.logging.log4j.core.LogEvent;
020    
021    /**
022     * Factory to create Flume events.
023     */
024    public interface FlumeEventFactory {
025        /**
026         * Create a Flume event.
027         * @param event The Log4j LogEvent.
028         * @param includes A comma separated list of MDC elements to include.
029         * @param excludes A comma separated list of MDC elements to exclude.
030         * @param required A comma separated list of MDC elements that are required.
031         * @param mdcPrefix The value to prefix to MDC keys.
032         * @param eventPrefix The value to prefix to event keys.
033         * @param compress If true the event body should be compressed.
034         * @return A FlumeEvent.
035         */
036        FlumeEvent createEvent(LogEvent event, String includes, String excludes, String required,
037                          String mdcPrefix, String eventPrefix, boolean compress);
038    }

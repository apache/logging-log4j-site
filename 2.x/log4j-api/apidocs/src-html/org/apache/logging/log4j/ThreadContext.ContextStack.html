020    import java.io.Serializable;
021    import java.util.ArrayList;

028    /**
029     * The ThreadContext allows applications to store information either in a Map
030     * <p/>
031     * <p><b><em>The MDC is managed on a per thread basis</em></b>. A
032     * child thread automatically inherits a <em>copy</em> of the mapped
033     * diagnostic context of its parent.
034     */
035    public final class ThreadContext {
036    
037        /**
038         * Empty, immutable Map.
039         */
040        public static final Map<String, String> EMPTY_MAP = new ImmutableMap();
041    
042        /**
043         * Empty, immutable ContextStack.
044         */
045        public static final ContextStack EMPTY_STACK = new ImmutableStack();
046    
047        private static final String DISABLE_MAP = "disableThreadContextMap";
048    
049        private static final String DISABLE_STACK = "disableThreadContextStack";
050    
051        private static final String DISABLE_ALL = "disableThreadContext";
052    
053        private static boolean all = Boolean.getBoolean(DISABLE_ALL);
054    
055        private static boolean useMap = !(Boolean.getBoolean(DISABLE_MAP) || all);
056    
057        private static boolean useStack = !(Boolean.getBoolean(DISABLE_STACK) || all);
058    
059        private static ThreadLocal<Map<String, String>> localMap =
060            new InheritableThreadLocal<Map<String, String>>() {
061                @Override
062                protected Map<String, String> childValue(Map<String, String> parentValue) {
063                    return parentValue == null || !useMap ? null : new HashMap<String, String>(parentValue);
064                }
065            };
066    
067        private static ThreadLocal<ContextStack> localStack = new ThreadLocal<ContextStack>();
068    
069        private ThreadContext() {
070    
071        }
072    
073        /**
074         * Put a context value (the <code>o</code> parameter) as identified
075         * with the <code>key</code> parameter into the current thread's
076         * context map.
077         * <p/>
078         * <p>If the current thread does not have a context map it is
079         * created as a side effect.
080         * @param key The key name.
081         * @param value The key value.
082         */
083        public static void put(String key, String value) {
084            if (!useMap) {
085                return;
086            }
087            Map<String, String> map = localMap.get();
088            if (map == null) {
089                map = new HashMap<String, String>();
090                localMap.set(map);
091            }
092            map.put(key, value);
093        }
094    
095        /**
096         * Get the context identified by the <code>key</code> parameter.
097         * <p/>
098         * <p>This method has no side effects.
099         * @param key The key to locate.
100         * @return The value associated with the key or null.
101         */
102        public static String get(String key) {
103            Map<String, String> map = localMap.get();
104            return map == null ? null : map.get(key);
105        }
106    
107        /**
108         * Remove the the context identified by the <code>key</code>
109         * parameter.
110         * @param key The key to remove.
111         */
112        public static void remove(String key) {
113            Map<String, String> map = localMap.get();
114            if (map != null) {
115                map.remove(key);
116            }
117        }
118    
119        /**
120         * Clear the context.
121         */
122        public static void clear() {
123            localMap.remove();
124        }
125    
126        /**
127         * Determine if the key is in the context.
128         * @param key The key to locate.
129         * @return True if the key is in the context, false otherwise.
130         */
131        public static boolean containsKey(String key) {
132            Map<String, String> map = localMap.get();
133            return map == null ? false : map.containsKey(key);
134        }
135    
136        /**
137         * Get a copy of current thread's context Map.
138         * @return a copy of the context.
139         */
140        public static Map<String, String> getContext() {
141            Map<String, String> map = localMap.get();
142            return map == null ? new HashMap<String, String>() : new HashMap<String, String>(map);
143        }
144    
145        /**
146         * Get an immutable copy of the current thread's context Map.
147         * @return An immutable copy of the ThreadContext Map.
148         */
149        public static Map<String, String> getImmutableContext() {
150            Map<String, String> map = localMap.get();
151            return map == null ? new ImmutableMap() : new ImmutableMap(map);
152        }
153    
154        /**
155         * Return true if the Map is empty.
156         * @return true if the Map is empty, false otherwise.
157         */
158        public static boolean isEmpty() {
159            Map<String, String> map = localMap.get();
160            return map == null || map.size() == 0;
161        }
162    
163        /**
164         * Clear the stack for this thread.
165         */
166        public static void clearStack() {
167            localStack.remove();
168        }
169    
170        /**
171         * Return a copy of this thread's stack.
172         * @return A copy of this thread's stack.
173         */
174        public static ContextStack cloneStack() {
175            ContextStack stack = localStack.get();
176            return stack == null ? new ThreadContextStack() : new ThreadContextStack(stack.asList());
177        }
178    
179        /**
180         * Get an immutable copy of this current thread's context stack.
181         * @return an immutable copy of the ThreadContext stack.
182         */
183        public static ContextStack getImmutableStack() {
184            ContextStack stack = localStack.get();
185            return stack == null ? EMPTY_STACK : new ImmutableStack(stack.asList());
186        }
187    
188        /**
189         * Set this thread's stack.
190         * @param stack The stack to use.
191         */
192        public static void setStack(Collection<String> stack) {
193            if (stack.size() == 0 || !useStack) {
194                return;
195            }
196            localStack.set(new ThreadContextStack(stack));
197        }
198    
199        /**
200         * Get the current nesting depth of this thread's stack.
201         * @return the number of items in the stack.
202         *
203         * @see #trim
204         */
205        public static int getDepth() {
206            ContextStack stack = localStack.get();
207            return stack == null ? 0 : stack.getDepth();
208        }
209    
210        /**
211         * Returns the value of the last item placed on the stack.
212         * <p/>
213         * <p>The returned value is the value that was pushed last. If no
214         * context is available, then the empty string "" is returned.
215         *
216         * @return String The innermost diagnostic context.
217         */
218        public static String pop() {
219            ContextStack s = localStack.get();
220            if (s == null || s.getDepth() == 0) {
221                return "";
222            }
223            return s.pop();
224        }
225    
226        /**
227         * Looks at the last diagnostic context at the top of this NDC
228         * without removing it.
229         * <p/>
230         * <p>The returned value is the value that was pushed last. If no
231         * context is available, then the empty string "" is returned.
232         *
233         * @return String The innermost diagnostic context.
234         */
235        public static String peek() {
236            ContextStack s = localStack.get();
237            if (s == null || s.getDepth() == 0) {
238                return "";
239            }
240            return s.peek();
241        }
242    
243        /**
244         * Push new diagnostic context information for the current thread.
245         * <p/>
246         * <p>The contents of the <code>message</code> parameter is
247         * determined solely by the client.
248         *
249         * @param message The new diagnostic context information.
250         */
251        public static void push(String message) {
252            if (!useStack) {
253                return;
254            }
255            ContextStack stack = localStack.get();
256            if (stack == null) {
257                stack = new ThreadContextStack();
258                localStack.set(stack);
259            }
260            stack.push(message);
261        }
262    
263        /**
264         * Remove the diagnostic context for this thread.
265         * <p/>
266         * <p>Each thread that created a diagnostic context by calling
267         * {@link #push} should call this method before exiting. Otherwise,
268         * the memory used by the <b>thread</b> cannot be reclaimed by the
269         * VM.
270         * <p/>
271         * <p>As this is such an important problem in heavy duty systems and
272         * because it is difficult to always guarantee that the remove
273         * method is called before exiting a thread, this method has been
274         * augmented to lazily remove references to dead threads. In
275         * practice, this means that you can be a little sloppy and
276         * occasionally forget to call {@link #remove} before exiting a
277         * thread. However, you must call <code>remove</code> sometime. If
278         * you never call it, then your application is sure to run out of
279         * memory.
280         */
281        public static void removeStack() {
282            localStack.remove();
283        }
284    
285        /**
286         * Trims elements from this diagnostic context. If the current
287         * depth is smaller or equal to <code>maxDepth</code>, then no
288         * action is taken. If the current depth is larger than newDepth
289         * then all elements at maxDepth or higher are discarded.
290         * <p/>
291         * <p>This method is a convenient alternative to multiple {@link
292         * #pop} calls. Moreover, it is often the case that at the end of
293         * complex call sequences, the depth of the ThreadContext is
294         * unpredictable. The <code>trim</code> method circumvents
295         * this problem.
296         * <p/>
297         * <p>For example, the combination
298         * <pre>
299         * void foo() {
300         * &nbsp;  int depth = ThreadContext.getDepth();
301         * <p/>
302         * &nbsp;  ... complex sequence of calls
303         * <p/>
304         * &nbsp;  ThreadContext.trim(depth);
305         * }
306         * </pre>
307         * <p/>
308         * ensures that between the entry and exit of foo the depth of the
309         * diagnostic stack is conserved.
310         *
311         * @see #getDepth
312         * @param depth The number of elements to keep.
313         */
314        public static void trim(int depth) {
315            ContextStack stack = localStack.get();
316            if (stack != null) {
317                stack.trim(depth);
318            }
319        }
320    
321        /**
322         * The ThreadContext Stack interface.
323         */
324        public interface ContextStack extends Serializable {
325    
326            /**
327             * Clears all elements from the stack.
328             */
329            void clear();
330    
331            /**
332             * Return the element at the top of the stack.
333             * @return The element at the top of the stack.
334             * @throws java.util.NoSuchElementException if the stack is empty.
335             */
336            String pop();
337    
338            /**
339             * Return the element at the top of the stack without removing it or null if the stack is empty.
340             * @return the element at the top of the stack or null if the stack is empty.
341             */
342            String peek();
343    
344            /**
345             * Add an element to the stack.
346             * @param message The element to add.
347             */
348            void push(String message);
349    
350            /**
351             * Return the number of elements in the stack.
352             * @return the number of elements in the stack.
353             */
354            int getDepth();
355    
356            /**
357             * Returns all the elements in the stack in a List.
358             * @return all the elements in the stack in a List.
359             */
360            List<String> asList();
361    
362            /**
363             * Trims elements from the end of the stack.
364             * @param depth The maximum number of items in the stack to keep.
365             */
366            void trim(int depth);
367    
368            /**
369             * Returns a copy of the ContextStack.
370             * @return a copy of the ContextStack.
371             */
372            ContextStack copy();
373        }
374    
375        /**
376         * The ContextStack implementation.
377         */
378        private static class ThreadContextStack extends ArrayList<String> implements ContextStack {
379    
380            private static final long serialVersionUID = 5050501L;
381    
382            public ThreadContextStack() {
383                super();
384            }
385    
386            public ThreadContextStack(Collection<String> collection) {
387                super(collection);
388            }
389    
390            public String pop() {
391                int index = size() - 1;
392                if (index >= 0) {
393                    String result = get(index);
394                    remove(index);
395                    return result;
396                }
397                throw new NoSuchElementException("The ThreadContext stack is empty");
398            }
399    
400            public String peek() {
401                int index = size() - 1;
402                if (index >= 0) {
403                    return get(index);
404                }
405                return null;
406            }
407    
408            public void push(String message) {
409                add(message);
410            }
411    
412            public int getDepth() {
413                return size();
414            }
415    
416            public List<String> asList() {
417                return this;
418            }
419    
420            public void trim(int depth) {
421                if (depth < 0) {
422                    throw new IllegalArgumentException("Maximum stack depth cannot be negative");
423                }
424                while (size() > depth) {
425                    remove(size() - 1);
426                }
427    
428            }
429    
430            public ContextStack copy() {
431                return new ThreadContextStack(this);
432            }
433        }

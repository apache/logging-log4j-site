017    package org.apache.logging.log4j.spi;
018    
019    import org.apache.logging.log4j.Logger;
020    
021    /**
022     * Anchor point for logging implementations.
023     */
024    public interface LoggerContext {
025    
026        /**
027         * Returns a Logger.
028         * @param name The name of the Logger to return.
029         * @return The logger with the specified name.
030         */
031        Logger getLogger(String name);
032    
033        /**
034         * Detects if a Logger with the specified name exists.
035         * @param name The Logger name to search for.
036         * @return true if the Logger exists, false otherwise.
037         */
038        boolean hasLogger(String name);
039    
040        /**
041         * An anchor for some other context, such as a ServletContext.
042         * @return The external context.
043         */
044        Object getExternalContext();
045    }

017package org.apache.logging.log4j.util;

019import org.apache.logging.log4j.Logger;
020import org.apache.logging.log4j.status.StatusLogger;

022/**
023 * Utility for preventing primitive parameter values from being auto-boxed. Auto-boxing creates temporary objects
024 * which contribute to pressure on the garbage collector. With this utility users can convert primitive values directly
025 * into text without allocating temporary objects.
026 * <p>
027 * Example usage:
028 * </p><pre>
029 * import static org.apache.logging.log4j.util.Unbox.box;
030 * ...
031 * long longValue = 123456L;
032 * double doubleValue = 3.14;
033 * // prevent primitive values from being auto-boxed
034 * logger.debug("Long value={}, double value={}", box(longValue), box(doubleValue));
035 * </pre>
036 * <p>
037 * This class manages a small thread-local ring buffer of StringBuilders.
038 * Each time one of the {@code box()} methods is called, the next slot in the ring buffer is used, until the ring
039 * buffer is full and the first slot is reused. By default the Unbox ring buffer has 32 slots, so user code can
040 * have up to 32 boxed primitives in a single logger call.
041 * </p>
042 * <p>
043 * If more slots are required, set system property {@code log4j.unbox.ringbuffer.size} to the desired ring buffer size.
044 * Note that the specified number will be rounded up to the nearest power of 2.
045 * </p>
046 */
047@PerformanceSensitive("allocation")
048public class Unbox {
049    private static final Logger LOGGER = StatusLogger.getLogger();
050    private static final int BITS_PER_INT = 32;
051    private static final int RINGBUFFER_MIN_SIZE = 32;
052    private static final int RINGBUFFER_SIZE = calculateRingBufferSize("log4j.unbox.ringbuffer.size");
053    private static final int MASK = RINGBUFFER_SIZE - 1;
054
055    /**
056     * State implementation that only puts JDK classes in ThreadLocals, so this is safe to be used from
057     * web applications. Web application containers have thread pools that may hold on to ThreadLocal objects
058     * after the application was stopped. This may prevent the classes of the application from being unloaded,
059     * causing memory leaks.
060     * <p>
061     * Such memory leaks will not occur if only JDK classes are stored in ThreadLocals.
062     * </p>
063     */
064    private static class WebSafeState {
065        private final ThreadLocal<StringBuilder[]> ringBuffer = new ThreadLocal<>();
066        private final ThreadLocal<int[]> current = new ThreadLocal<>();
067
068        public StringBuilder getStringBuilder() {
069            StringBuilder[] array = ringBuffer.get();
070            if (array == null) {
071                array = new StringBuilder[RINGBUFFER_SIZE];
072                for (int i = 0; i < array.length; i++) {
073                    array[i] = new StringBuilder(21);
074                }
075                ringBuffer.set(array);
076                current.set(new int[1]);
077            }
078            final int[] index = current.get();
079            final StringBuilder result = array[MASK & index[0]++];
080            result.setLength(0);
081            return result;
082        }
083
084        public boolean isBoxedPrimitive(final StringBuilder text) {
085            final StringBuilder[] array = ringBuffer.get();
086            if (array == null) {
087                return false;
088            }
089            for (int i = 0; i < array.length; i++) {
090                if (text == array[i]) {
091                    return true;
092                }
093            }
094            return false;
095        }
096    }
097
098    private static class State {
099        private final StringBuilder[] ringBuffer = new StringBuilder[RINGBUFFER_SIZE];
100        private int current;
101        State() {
102            for (int i = 0; i < ringBuffer.length; i++) {
103                ringBuffer[i] = new StringBuilder(21);
104            }
105        }
106
107        public StringBuilder getStringBuilder() {
108            final StringBuilder result = ringBuffer[MASK & current++];
109            result.setLength(0);
110            return result;
111        }
112
113        public boolean isBoxedPrimitive(final StringBuilder text) {
114            for (int i = 0; i < ringBuffer.length; i++) {
115                if (text == ringBuffer[i]) {
116                    return true;
117                }
118            }
119            return false;
120        }
121    }
122    private static ThreadLocal<State> threadLocalState = new ThreadLocal<>();
123    private static WebSafeState webSafeState = new WebSafeState();
124
125    private Unbox() {
126        // this is a utility
127    }
128
129    private static int calculateRingBufferSize(final String propertyName) {
130        final String userPreferredRBSize = PropertiesUtil.getProperties().getStringProperty(propertyName,
131                String.valueOf(RINGBUFFER_MIN_SIZE));
132        try {
133            int size = Integer.parseInt(userPreferredRBSize);
134            if (size < RINGBUFFER_MIN_SIZE) {
135                size = RINGBUFFER_MIN_SIZE;
136                LOGGER.warn("Invalid {} {}, using minimum size {}.", propertyName, userPreferredRBSize,
137                        RINGBUFFER_MIN_SIZE);
138            }
139            return ceilingNextPowerOfTwo(size);
140        } catch (final Exception ex) {
141            LOGGER.warn("Invalid {} {}, using default size {}.", propertyName, userPreferredRBSize,
142                    RINGBUFFER_MIN_SIZE);
143            return RINGBUFFER_MIN_SIZE;
144        }
145    }
146
147    /**
148     * Calculate the next power of 2, greater than or equal to x.
149     * <p>
150     * From Hacker's Delight, Chapter 3, Harry S. Warren Jr.
151     *
152     * @param x Value to round up
153     * @return The next power of 2 from x inclusive
154     */
155    private static int ceilingNextPowerOfTwo(final int x) {
156        return 1 << (BITS_PER_INT - Integer.numberOfLeadingZeros(x - 1));
157    }

159    /**
160     * Returns a {@code StringBuilder} containing the text representation of the specified primitive value.
161     * This method will not allocate temporary objects.
162     *
163     * @param value the value whose text representation to return
164     * @return a {@code StringBuilder} containing the text representation of the specified primitive value
165     */
166    @PerformanceSensitive("allocation")
167    public static StringBuilder box(final float value) {
168        return getSB().append(value);
169    }

255    private static State getState() {
256        State state = threadLocalState.get();
257        if (state == null) {
258            state = new State();
259            threadLocalState.set(state);
260        }
261        return state;
262    }
263
264    private static StringBuilder getSB() {
265        return Constants.ENABLE_THREADLOCALS ? getState().getStringBuilder() : webSafeState.getStringBuilder();
266    }

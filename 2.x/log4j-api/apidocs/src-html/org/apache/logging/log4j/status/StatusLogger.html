017    package org.apache.logging.log4j.status;
018    
019    import org.apache.logging.log4j.spi.AbstractLogger;
020    import org.apache.logging.log4j.Level;
021    import org.apache.logging.log4j.Logger;
022    import org.apache.logging.log4j.Marker;
023    import org.apache.logging.log4j.message.Message;
024    
025    import java.util.ArrayList;
026    import java.util.Iterator;
027    import java.util.List;
028    import java.util.Queue;
029    import java.util.concurrent.ConcurrentLinkedQueue;
030    import java.util.concurrent.CopyOnWriteArrayList;
031    import java.util.concurrent.locks.ReentrantLock;
032    import java.util.concurrent.locks.ReentrantReadWriteLock;
033    
034    /**
035     * Mechanism to record events that occur in the logging system.
036     */
037    public final class StatusLogger extends AbstractLogger {
038    
039        /**
040         * System property that can be configured with the number of entries in the queue. Once the limit
041         * is reached older entries will be removed as new entries are added.
042         */
043        public static final String MAX_STATUS_ENTRIES = "log4j2.status.entries";
044    
045        private static final String NOT_AVAIL = "?";
046    
047        private static final int MAX_ENTRIES = Integer.getInteger(MAX_STATUS_ENTRIES, 200);
048    
049        // private static final String FQCN = AbstractLogger.class.getName();
050    
051        private static StatusLogger statusLogger = new StatusLogger();
052    
053        private Logger logger = null;
054    
055        private CopyOnWriteArrayList<StatusListener> listeners = new CopyOnWriteArrayList<StatusListener>();
056        private ReentrantReadWriteLock listenersLock = new ReentrantReadWriteLock();
057    
058        private Queue<StatusData> messages = new BoundedQueue<StatusData>(MAX_ENTRIES);
059        private ReentrantLock msgLock = new ReentrantLock();
060    
061        private StatusLogger() {
062        }
063    
064        /**
065         * Retrieve the StatusLogger.
066         * @return The StatusLogger.
067         */
068        public static StatusLogger getLogger() {
069            return statusLogger;
070        }
071    
072        /**
073         * Register a new listener.
074         * @param listener The StatusListener to register.
075         */
076        public void registerListener(StatusListener listener) {
077            listenersLock.writeLock().lock();
078            try {
079                listeners.add(listener);
080            } finally {
081                listenersLock.writeLock().unlock();
082            }
083        }
084    
085        /**
086         * Remove a StatusListener.
087         * @param listener The StatusListener to remove.
088         */
089        public void removeListener(StatusListener listener) {
090            listenersLock.writeLock().lock();
091            try {
092                listeners.remove(listener);
093            } finally {
094                listenersLock.writeLock().unlock();
095            }
096        }
097    
098        /**
099         * Returns a thread safe Iterator for the StatusListener.
100         * @return An Iterator for the list of StatusListeners.
101         */
102        public Iterator<StatusListener> getListeners() {
103            return listeners.iterator();
104        }
105    
106        /**
107         * Clears the list of status events and listeners.
108         */
109        public void reset() {
110            listeners.clear();
111            clear();
112        }
113    
114        /**
115         * Returns a List of all events as StatusData objects.
116         * @return The list of StatusData objects.
117         */
118        public List<StatusData> getStatusData() {
119            msgLock.lock();
120            try {
121                return new ArrayList<StatusData>(messages);
122            } finally {
123                msgLock.unlock();
124            }
125        }
126    
127        /**
128         * Clears the list of status events.
129         */
130        public void clear() {
131            msgLock.lock();
132            try {
133                messages.clear();
134            } finally {
135                msgLock.unlock();
136            }
137        }
138    
139    
140        /**
141         * Add an event.
142         * @param marker The Marker
143         * @param fqcn   The fully qualified class name of the <b>caller</b>
144         * @param level  The logging level
145         * @param msg    The message associated with the event.
146         * @param t      A Throwable or null.
147         */
148        @Override
149        public void log(Marker marker, String fqcn, Level level, Message msg, Throwable t) {
150            StackTraceElement element = null;
151            if (fqcn != null) {
152                element = getStackTraceElement(fqcn, Thread.currentThread().getStackTrace());
153            }
154            StatusData data = new StatusData(element, level, msg, t);
155            msgLock.lock();
156            try {
157                messages.add(data);
158            } finally {
159                msgLock.unlock();
160            }
161            for (StatusListener listener : listeners) {
162                listener.log(data);
163            }
164        }
165    
166        private StackTraceElement getStackTraceElement(String fqcn, StackTraceElement[] stackTrace) {
167            if (fqcn == null) {
168                return null;
169            }
170            boolean next = false;
171            for (StackTraceElement element : stackTrace) {
172                if (next) {
173                    return element;
174                }
175                String className = element.getClassName();
176                if (fqcn.equals(className)) {
177                    next = true;
178                } else if (NOT_AVAIL.equals(className)) {
179                    break;
180                }
181            }
182            return null;
183        }
184    
185        @Override
186        protected boolean isEnabled(Level level, Marker marker, String data) {
187            return isEnabled(level, marker);
188        }
189    
190        @Override
191        protected boolean isEnabled(Level level, Marker marker, String data, Throwable t) {
192            return isEnabled(level, marker);
193        }
194    
195        @Override
196        protected boolean isEnabled(Level level, Marker marker, String data, Object p1) {
197            return isEnabled(level, marker);
198        }
199    
200        @Override
201        protected boolean isEnabled(Level level, Marker marker, String data, Object p1, Object p2) {
202            return isEnabled(level, marker);
203        }
204    
205        @Override
206        protected boolean isEnabled(Level level, Marker marker, String data, Object p1, Object p2, Object p3) {
207            return isEnabled(level, marker);
208        }
209    
210        @Override
211        protected boolean isEnabled(Level level, Marker marker, String data, Object p1, Object p2, Object p3,
212                                    Object... params) {
213            return isEnabled(level, marker);
214        }
215    
216        @Override
217        protected boolean isEnabled(Level level, Marker marker, Object data, Throwable t) {
218            return isEnabled(level, marker);
219        }
220    
221        @Override
222        protected boolean isEnabled(Level level, Marker marker, Message data, Throwable t) {
223            return isEnabled(level, marker);
224        }
225    
226        protected boolean isEnabled(Level level, Marker marker) {
227            if (logger == null) {
228                return true;
229            }
230            switch (level) {
231                case FATAL:
232                    return logger.isFatalEnabled(marker);
233                case TRACE:
234                    return logger.isTraceEnabled(marker);
235                case DEBUG:
236                    return logger.isDebugEnabled(marker);
237                case INFO:
238                    return logger.isInfoEnabled(marker);
239                case WARN:
240                    return logger.isWarnEnabled(marker);
241                case ERROR:
242                    return logger.isErrorEnabled(marker);
243            }
244            return false;
245        }
246    
247        /**
248         * Queue for status events.
249         * @param <E> Object type to be stored in the queue.
250         */
251        private class BoundedQueue<E> extends ConcurrentLinkedQueue<E> {
252    
253            private final int size;
254    
255            public BoundedQueue(int size) {
256                this.size = size;
257            }
258    
259            public boolean add(E object) {
260                while (messages.size() > size) {
261                    messages.poll();
262                }
263                return super.add(object);
264            }
265        }
266    }

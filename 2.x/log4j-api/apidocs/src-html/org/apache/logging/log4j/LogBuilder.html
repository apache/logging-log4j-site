017package org.apache.logging.log4j;
018
019import org.apache.logging.log4j.message.Message;
020import org.apache.logging.log4j.util.Supplier;
021
022
023/**
024 * Interface for constructing log events before logging them. Instances of LogBuilders should only be created
025 * by calling one of the Logger methods that return a LogBuilder.
026 */
027public interface LogBuilder {
028
029    public static final LogBuilder NOOP = new LogBuilder() {};
030
031    default LogBuilder withMarker(Marker marker) {
032        return this;
033    }
034
035    default LogBuilder withThrowable(Throwable throwable) {
036        return this;
037    }
038
039    default LogBuilder withLocation() {
040        return this;
041    }
042
043    default LogBuilder withLocation(StackTraceElement location) {
044        return this;
045    }
046
047    default void log(CharSequence message) {
048    }
049
050    default void log(String message) {
051    }
052
053    /**
054     * Logs a message with parameters.
055     *
056     * @param message the message to log; the format depends on the message factory.
057     * @param params parameters to the message.
058     *
059     * @see org.apache.logging.log4j.util.Unbox
060     */
061    default void log(String message, Object... params) {
062    }
063
064    default void log(String message, Supplier<?>... params) {
065    }
066
067    default void log(Message message) {
068    }
069
070    default void log(Supplier<Message> messageSupplier) {
071    }
072
073    default void log(Object message) {
074
075    }
076
077    /**
078     * Logs a message with parameters.
079     *
080     * @param message the message to log; the format depends on the message factory.
081     * @param p0 parameter to the message.
082     *
083     * @see org.apache.logging.log4j.util.Unbox
084     */
085    default void log(String message, Object p0) {
086    }
087
088    /**
089     * Logs a message with parameters.
090     *
091     * @param message the message to log; the format depends on the message factory.
092     * @param p0 parameter to the message.
093     * @param p1 parameter to the message.
094     *
095     * @see org.apache.logging.log4j.util.Unbox
096     */
097    default void log(String message, Object p0, Object p1) {
098    }
099
100    /**
101     * Logs a message with parameters.
102     *
103     * @param message the message to log; the format depends on the message factory.
104     * @param p0 parameter to the message.
105     * @param p1 parameter to the message.
106     * @param p2 parameter to the message.
107     *
108     * @see org.apache.logging.log4j.util.Unbox
109     */
110    default void log(String message, Object p0, Object p1, Object p2) {
111    }
112
113    /**
114     * Logs a message with parameters.
115     *
116     * @param message the message to log; the format depends on the message factory.
117     * @param p0 parameter to the message.
118     * @param p1 parameter to the message.
119     * @param p2 parameter to the message.
120     * @param p3 parameter to the message.
121     *
122     * @see org.apache.logging.log4j.util.Unbox
123     */
124    default void log(String message, Object p0, Object p1, Object p2, Object p3) {
125    }
126
127    /**
128     * Logs a message with parameters.
129     *
130     * @param message the message to log; the format depends on the message factory.
131     * @param p0 parameter to the message.
132     * @param p1 parameter to the message.
133     * @param p2 parameter to the message.
134     * @param p3 parameter to the message.
135     * @param p4 parameter to the message.
136     *
137     * @see org.apache.logging.log4j.util.Unbox
138     */
139    default void log(String message, Object p0, Object p1, Object p2, Object p3, Object p4) {
140    }
141
142    /**
143     * Logs a message with parameters.
144     *
145     * @param message the message to log; the format depends on the message factory.
146     * @param p0 parameter to the message.
147     * @param p1 parameter to the message.
148     * @param p2 parameter to the message.
149     * @param p3 parameter to the message.
150     * @param p4 parameter to the message.
151     * @param p5 parameter to the message.
152     *
153     * @see org.apache.logging.log4j.util.Unbox
154     */
155    default void log(String message, Object p0, Object p1, Object p2, Object p3, Object p4, Object p5) {
156    }
157
158    /**
159     * Logs a message with parameters.
160     *
161     * @param message the message to log; the format depends on the message factory.
162     * @param p0 parameter to the message.
163     * @param p1 parameter to the message.
164     * @param p2 parameter to the message.
165     * @param p3 parameter to the message.
166     * @param p4 parameter to the message.
167     * @param p5 parameter to the message.
168     * @param p6 parameter to the message.
169     *
170     * @see org.apache.logging.log4j.util.Unbox
171     */
172    default void log(String message, Object p0, Object p1, Object p2, Object p3, Object p4, Object p5, Object p6) {
173    }
174
175    /**
176     * Logs a message with parameters.
177     *
178     * @param message the message to log; the format depends on the message factory.
179     * @param p0 parameter to the message.
180     * @param p1 parameter to the message.
181     * @param p2 parameter to the message.
182     * @param p3 parameter to the message.
183     * @param p4 parameter to the message.
184     * @param p5 parameter to the message.
185     * @param p6 parameter to the message.
186     * @param p7 parameter to the message.
187     *
188     * @see org.apache.logging.log4j.util.Unbox
189     */
190    default void log(String message, Object p0, Object p1, Object p2, Object p3, Object p4, Object p5, Object p6,
191            Object p7) {
192    }
193
194    /**
195     * Logs a message with parameters.
196     *
197     * @param message the message to log; the format depends on the message factory.
198     * @param p0 parameter to the message.
199     * @param p1 parameter to the message.
200     * @param p2 parameter to the message.
201     * @param p3 parameter to the message.
202     * @param p4 parameter to the message.
203     * @param p5 parameter to the message.
204     * @param p6 parameter to the message.
205     * @param p7 parameter to the message.
206     * @param p8 parameter to the message.
207     *
208     * @see org.apache.logging.log4j.util.Unbox
209     */
210    default void log(String message, Object p0, Object p1, Object p2, Object p3, Object p4, Object p5, Object p6,
211            Object p7, Object p8) {
212    }
213
214    /**
215     * Logs a message with parameters.
216     *
217     * @param message the message to log; the format depends on the message factory.
218     * @param p0 parameter to the message.
219     * @param p1 parameter to the message.
220     * @param p2 parameter to the message.
221     * @param p3 parameter to the message.
222     * @param p4 parameter to the message.
223     * @param p5 parameter to the message.
224     * @param p6 parameter to the message.
225     * @param p7 parameter to the message.
226     * @param p8 parameter to the message.
227     * @param p9 parameter to the message.
228     *
229     * @see org.apache.logging.log4j.util.Unbox
230     */
231    default void log(String message, Object p0, Object p1, Object p2, Object p3, Object p4, Object p5, Object p6,
232            Object p7, Object p8, Object p9) {
233    }
234}

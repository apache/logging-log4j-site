017package org.apache.logging.log4j.spi;
018
019import org.apache.logging.log4j.LogManager;
020import org.apache.logging.log4j.Logger;
021import org.apache.logging.log4j.status.StatusLogger;
022import org.apache.logging.log4j.util.Constants;
023import org.apache.logging.log4j.util.PropertiesUtil;
024import org.apache.logging.log4j.util.ProviderUtil;
025
026/**
027 * Creates the ThreadContextMap instance used by the ThreadContext.
028 *
029 * @see ThreadContextMap
030 * @see org.apache.logging.log4j.ThreadContext
031 * @since 2.7
032 */
033public final class ThreadContextMapFactory {
034    private static final Logger LOGGER = StatusLogger.getLogger();
035    private static final String THREAD_CONTEXT_KEY = "log4j2.threadContextMap";
036    private static final String GC_FREE_THREAD_CONTEXT_KEY = "log4j2.garbagefree.threadContextMap";
037
038    private ThreadContextMapFactory() {
039    }
040
041    public static ThreadContextMap createThreadContextMap() {
042        final PropertiesUtil managerProps = PropertiesUtil.getProperties();
043        final String threadContextMapName = managerProps.getStringProperty(THREAD_CONTEXT_KEY);
044        final ClassLoader cl = ProviderUtil.findClassLoader();
045        ThreadContextMap result = null;
046        if (threadContextMapName != null) {
047            try {
048                final Class<?> clazz = cl.loadClass(threadContextMapName);
049                if (ThreadContextMap.class.isAssignableFrom(clazz)) {
050                    result = (ThreadContextMap) clazz.newInstance();
051                }
052            } catch (final ClassNotFoundException cnfe) {
053                LOGGER.error("Unable to locate configured ThreadContextMap {}", threadContextMapName);
054            } catch (final Exception ex) {
055                LOGGER.error("Unable to create configured ThreadContextMap {}", threadContextMapName, ex);
056            }
057        }
058        if (result == null && ProviderUtil.hasProviders()) {
059            final String factoryClassName = LogManager.getFactory().getClass().getName();
060            for (final Provider provider : ProviderUtil.getProviders()) {
061                if (factoryClassName.equals(provider.getClassName())) {
062                    final Class<? extends ThreadContextMap> clazz = provider.loadThreadContextMap();
063                    if (clazz != null) {
064                        try {
065                            result = clazz.newInstance();
066                            break;
067                        } catch (final Exception e) {
068                            LOGGER.error("Unable to locate or load configured ThreadContextMap {}",
069                                    provider.getThreadContextMap(), e);
070                            result = createDefaultThreadContextMap();
071                        }
072                    }
073                }
074            }
075        }
076        if (result == null) {
077            result = createDefaultThreadContextMap();
078        }
079        return result;
080    }
081
082    private static ThreadContextMap createDefaultThreadContextMap() {
083        if (Constants.ENABLE_THREADLOCALS) {
084            if (PropertiesUtil.getProperties().getBooleanProperty(GC_FREE_THREAD_CONTEXT_KEY)) {
085                return new GarbageFreeSortedArrayThreadContextMap();
086            }
087            return new CopyOnWriteSortedArrayThreadContextMap();
088        }
089        return new DefaultThreadContextMap(true);
090    }
091}

017package org.apache.logging.log4j.internal;
018
019import java.io.Serializable;
020
021import org.apache.logging.log4j.Level;
022import org.apache.logging.log4j.LogBuilder;
023import org.apache.logging.log4j.Logger;
024import org.apache.logging.log4j.Marker;
025import org.apache.logging.log4j.message.Message;
026import org.apache.logging.log4j.message.SimpleMessage;
027import org.apache.logging.log4j.status.StatusLogger;
028import org.apache.logging.log4j.util.LambdaUtil;
029import org.apache.logging.log4j.util.StackLocatorUtil;
030import org.apache.logging.log4j.util.Supplier;
031
032
033/**
034 * Collects data for a log event and then logs it. This class should be considered private.
035 */
036public class DefaultLogBuilder implements LogBuilder {
037
038    private static Message EMPTY_MESSAGE = new SimpleMessage("");
039    private static final String FQCN = DefaultLogBuilder.class.getName();
040    private static final Logger LOGGER = StatusLogger.getLogger();
041
042    private final Logger logger;
043    private Level level;
044    private Marker marker;
045    private Throwable throwable;
046    private StackTraceElement location;
047    private volatile boolean inUse;
048    private long threadId;
049
050    public DefaultLogBuilder(Logger logger, Level level) {
051        this.logger = logger;
052        this.level = level;
053        this.threadId = Thread.currentThread().getId();
054        this.inUse = true;
055    }
056
057    public DefaultLogBuilder(Logger logger) {
058        this.logger = logger;
059        this.inUse = false;
060        this.threadId = Thread.currentThread().getId();
061    }
062
063    /**
064     * This method should be considered internal. It is used to reset the LogBuilder for a new log message.
065     * @param level The logging level for this event.
066     * @return This LogBuilder instance.
067     */
068    public LogBuilder reset(Level level) {
069        this.inUse = true;
070        this.level = level;
071        this.marker = null;
072        this.throwable = null;
073        this.location = null;
074        return this;
075    }
076
077    public LogBuilder withMarker(Marker marker) {
078        this.marker = marker;
079        return this;
080    }
081
082    public LogBuilder withThrowable(Throwable throwable) {
083        this.throwable = throwable;
084        return this;
085    }
086
087    public LogBuilder withLocation() {
088        location = StackLocatorUtil.getStackTraceElement(2);
089        return this;
090    }
091
092    public LogBuilder withLocation(StackTraceElement location) {
093        this.location = location;
094        return this;
095    }
096
097    public boolean isInUse() {
098        return inUse;
099    }
100
101    @Override
102    public void log(Message message) {
103        if (isValid()) {
104            logMessage(message);
105        }
106    }
107
108    @Override
109    public void log(CharSequence message) {
110        if (isValid()) {
111            logMessage(logger.getMessageFactory().newMessage(message));
112        }
113    }
114
115    @Override
116    public void log(String message) {
117        if (isValid()) {
118            logMessage(logger.getMessageFactory().newMessage(message));
119        }
120    }
121
122    @Override
123    public void log(String message, Object... params) {
124        if (isValid()) {
125            logMessage(logger.getMessageFactory().newMessage(message, params));
126        }
127    }
128
129    @Override
130    public void log(String message, Supplier<?>... params) {
131        if (isValid()) {
132            logMessage(logger.getMessageFactory().newMessage(message, LambdaUtil.getAll(params)));
133        }
134    }
135
136    @Override
137    public void log(Supplier<Message> messageSupplier) {
138        if (isValid()) {
139            logMessage(messageSupplier.get());
140        }
141    }
142
143    @Override
144    public void log(Object message) {
145        if (isValid()) {
146            logMessage(logger.getMessageFactory().newMessage(message));
147        }
148    }
149
150    @Override
151    public void log(String message, Object p0) {
152        if (isValid()) {
153            logMessage(logger.getMessageFactory().newMessage(message, p0));
154        }
155    }
156
157    @Override
158    public void log(String message, Object p0, Object p1) {
159        if (isValid()) {
160            logMessage(logger.getMessageFactory().newMessage(message, p0, p1));
161        }
162    }
163
164    @Override
165    public void log(String message, Object p0, Object p1, Object p2) {
166        if (isValid()) {
167            logMessage(logger.getMessageFactory().newMessage(message, p0, p1, p2));
168        }
169    }
170
171    @Override
172    public void log(String message, Object p0, Object p1, Object p2, Object p3) {
173        if (isValid()) {
174            logMessage(logger.getMessageFactory().newMessage(message, p0, p1, p2, p3));
175        }
176    }
177
178    @Override
179    public void log(String message, Object p0, Object p1, Object p2, Object p3, Object p4) {
180        if (isValid()) {
181            logMessage(logger.getMessageFactory().newMessage(message, p0, p1, p2, p3, p4));
182        }
183    }
184
185    @Override
186    public void log(String message, Object p0, Object p1, Object p2, Object p3, Object p4, Object p5) {
187        if (isValid()) {
188            logMessage(logger.getMessageFactory().newMessage(message, p0, p1, p2, p3, p4, p5));
189        }
190    }
191
192    @Override
193    public void log(String message, Object p0, Object p1, Object p2, Object p3, Object p4, Object p5, Object p6) {
194        if (isValid()) {
195            logMessage(logger.getMessageFactory().newMessage(message, p0, p1, p2, p3, p4, p5, p6));
196        }
197    }
198
199    @Override
200    public void log(String message, Object p0, Object p1, Object p2, Object p3, Object p4, Object p5, Object p6,
201            Object p7) {
202        if (isValid()) {
203            logMessage(logger.getMessageFactory().newMessage(message, p0, p1, p2, p3, p4, p5, p6, p7));
204        }
205    }
206
207    @Override
208    public void log(String message, Object p0, Object p1, Object p2, Object p3, Object p4, Object p5, Object p6,
209            Object p7, Object p8) {
210        if (isValid()) {
211            logMessage(logger.getMessageFactory().newMessage(message, p0, p1, p2, p3, p4, p5, p6, p7, p8));
212        }
213    }
214
215    @Override
216    public void log(String message, Object p0, Object p1, Object p2, Object p3, Object p4, Object p5, Object p6,
217            Object p7, Object p8, Object p9) {
218        if (isValid()) {
219            logMessage(logger.getMessageFactory().newMessage(message, p0, p1, p2, p3, p4, p5, p6, p7, p8, p9));
220        }
221    }
222
223    private void logMessage(Message message) {
224        try {
225            logger.logMessage(level, marker, FQCN, location, message, throwable);
226        } finally {
227            inUse = false;
228        }
229    }
230
231    private boolean isValid() {
232        if (!inUse) {
233            LOGGER.warn("Attempt to reuse LogBuilder was ignored. {}",
234                    StackLocatorUtil.getCallerClass(2));
235            return false ;
236        }
237        if (this.threadId != Thread.currentThread().getId()) {
238            LOGGER.warn("LogBuilder can only be used on the owning thread. {}",
239                    StackLocatorUtil.getCallerClass(2));
240            return false;
241        }
242        return true;
243    }
244}

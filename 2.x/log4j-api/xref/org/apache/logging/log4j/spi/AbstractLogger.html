111     /**
112      * Checks that the message factory a logger was created with is the same as the given messageFactory. If they are
113      * different log a warning to the {@linkplain StatusLogger}. A null MessageFactory translates to the default
114      * MessageFactory {@link #DEFAULT_MESSAGE_FACTORY_CLASS}.
115      *
116      * @param logger
117      *            The logger to check
118      * @param messageFactory
119      *            The message factory to check.
120      */
121     public static void checkMessageFactory(final Logger logger, final MessageFactory messageFactory) {
122         final String name = logger.getName();
123         final MessageFactory loggerMessageFactory = logger.getMessageFactory();
124         if (messageFactory != null && !loggerMessageFactory.equals(messageFactory)) {
125             StatusLogger
126                 .getLogger()
127                 .warn("The Logger {} was created with the message factory {} and is now requested with the " +
128                     "message factory {}, which may create log events with unexpected formatting.",
129                     name, loggerMessageFactory, messageFactory);
130         } else if (messageFactory == null
131             && !loggerMessageFactory.getClass().equals(DEFAULT_MESSAGE_FACTORY_CLASS)) {
132             StatusLogger
133                 .getLogger()
134                 .warn("The Logger {} was created with the message factory {} and is now requested with a null " +
135                     "message factory (defaults to {}), which may create log events with unexpected formatting.",
136                     name, loggerMessageFactory, DEFAULT_MESSAGE_FACTORY_CLASS.getName());
137         }
138     }

391     /**
392      * Logs entry to a method with location information.
393      *
394      * @param fqcn The fully qualified class name of the <b>caller</b>.
395      * @param params The parameters to the method.
396      */
397     protected void entry(final String fqcn, final Object... params) {
398         if (isEnabled(Level.TRACE, ENTRY_MARKER, (Object) null, null)) {
399             log(ENTRY_MARKER, fqcn, Level.TRACE, entryMsg(params.length, params), null);
400         }
401     }
402 
403     private Message entryMsg(final int count, final Object... params) {
404         if (count == 0) {
405             return messageFactory.newMessage("entry");
406         }
407         final StringBuilder sb = new StringBuilder("entry params(");
408         int i = 0;
409         for (final Object parm : params) {
410             if (parm != null) {
411                 sb.append(parm.toString());
412             } else {
413                 sb.append("null");
414             }
415             if (++i < params.length) {
416                 sb.append(", ");
417             }
418         }
419         sb.append(")");
420         return messageFactory.newMessage(sb.toString());
421     }

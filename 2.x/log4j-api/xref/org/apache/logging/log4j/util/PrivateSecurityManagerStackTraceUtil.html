17  package org.apache.logging.log4j.util;
18  
19  import java.util.ArrayDeque;
20  import java.util.Collections;
21  import java.util.Deque;
22  
23  /**
24   * Internal utility to share a fast implementation of {@code #getCurrentStackTrace()}
25   * with the java 9 implementation of {@link StackLocator}.
26   */
27  final class PrivateSecurityManagerStackTraceUtil {
28  
29      private static final PrivateSecurityManager SECURITY_MANAGER;
30  
31      static {
32          PrivateSecurityManager psm;
33          try {
34              final SecurityManager sm = System.getSecurityManager();
35              if (sm != null) {
36                  sm.checkPermission(new RuntimePermission("createSecurityManager"));
37              }
38              psm = new PrivateSecurityManager();
39          } catch (final SecurityException ignored) {
40              psm = null;
41          }
42  
43          SECURITY_MANAGER = psm;
44      }
45  
46      private PrivateSecurityManagerStackTraceUtil() {
47          // Utility Class
48      }
49  
50      static boolean isEnabled() {
51          return SECURITY_MANAGER != null;
52      }
53  
54      /**
55       * Returns the current execution stack as a Deque of classes.
56       * <p>
57       * The size of the Deque is the number of methods on the execution stack. The first element is the class that started
58       * execution on this thread, the next element is the class that was called next, and so on, until the last element: the
59       * method that called {@link SecurityManager#getClassContext()} to capture the stack.
60       * </p>
61       *
62       * @return the execution stack.
63       */
64      // benchmarks show that using the SecurityManager is much faster than looping through getCallerClass(int)
65      static Deque<Class<?>> getCurrentStackTrace() {
66          final Class<?>[] array = SECURITY_MANAGER.getClassContext();
67          final Deque<Class<?>> classes = new ArrayDeque<>(array.length);
68          Collections.addAll(classes, array);
69          return classes;
70      }
71  
72      private static final class PrivateSecurityManager extends SecurityManager {
73  
74          @Override
75          protected Class<?>[] getClassContext() {
76              return super.getClassContext();
77          }
78  
79      }
80  }

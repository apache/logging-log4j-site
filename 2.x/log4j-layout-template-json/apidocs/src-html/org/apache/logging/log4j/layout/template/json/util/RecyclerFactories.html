017package org.apache.logging.log4j.layout.template.json.util;
018
019import org.apache.logging.log4j.core.config.plugins.Plugin;
020import org.apache.logging.log4j.core.config.plugins.convert.TypeConverter;
021import org.apache.logging.log4j.core.config.plugins.convert.TypeConverters;
022import org.apache.logging.log4j.core.util.Constants;
023import org.apache.logging.log4j.util.LoaderUtil;
024import org.jctools.queues.MpmcArrayQueue;
025
026import java.lang.reflect.Constructor;
027import java.lang.reflect.Method;
028import java.util.Arrays;
029import java.util.LinkedHashSet;
030import java.util.Map;
031import java.util.Queue;
032import java.util.concurrent.ArrayBlockingQueue;
033import java.util.function.Supplier;
034
035public enum RecyclerFactories {;
036
037    private static final String JCTOOLS_QUEUE_CLASS_SUPPLIER_PATH =
038            "org.jctools.queues.MpmcArrayQueue.new";
039
040    private static final boolean JCTOOLS_QUEUE_CLASS_AVAILABLE =
041            isJctoolsQueueClassAvailable();
042
043    private static boolean isJctoolsQueueClassAvailable() {
044        try {
045            final String className = JCTOOLS_QUEUE_CLASS_SUPPLIER_PATH
046                    .replaceAll("\\.new$", "");
047            LoaderUtil.loadClass(className);
048            return true;
049        } catch (final ClassNotFoundException ignored) {
050            return false;
051        }
052    }
053
054    @Plugin(name = "RecyclerFactory", category = TypeConverters.CATEGORY)
055    public static final class RecyclerFactoryConverter implements TypeConverter<RecyclerFactory> {
056        @Override
057        public RecyclerFactory convert(final String recyclerFactorySpec) {
058            return ofSpec(recyclerFactorySpec);
059        }
060    }
061
062    public static RecyclerFactory ofSpec(final String recyclerFactorySpec) {
063
064        // Determine the default capacity.
065        int defaultCapacity = Math.max(
066                2 * Runtime.getRuntime().availableProcessors() + 1,
067                8);
068
069        // TLA-, MPMC-, or ABQ-based queueing factory -- if nothing is specified.
070        if (recyclerFactorySpec == null) {
071            if (Constants.ENABLE_THREADLOCALS) {
072                return ThreadLocalRecyclerFactory.getInstance();
073            } else {
074                final Supplier<Queue<Object>> queueSupplier =
075                        JCTOOLS_QUEUE_CLASS_AVAILABLE
076                                ? () -> new MpmcArrayQueue<>(defaultCapacity)
077                                : () -> new ArrayBlockingQueue<>(defaultCapacity);
078                return new QueueingRecyclerFactory(queueSupplier);
079            }
080        }
081
082        // Is a dummy factory requested?
083        else if (recyclerFactorySpec.equals("dummy")) {
084            return DummyRecyclerFactory.getInstance();
085        }
086
087        // Is a TLA factory requested?
088        else if (recyclerFactorySpec.equals("threadLocal")) {
089            return ThreadLocalRecyclerFactory.getInstance();
090        }
091
092        // Is a queueing factory requested?
093        else if (recyclerFactorySpec.startsWith("queue")) {
094            return readQueueingRecyclerFactory(recyclerFactorySpec, defaultCapacity);
095        }
096
097        // Bogus input, bail out.
098        else {
099            throw new IllegalArgumentException(
100                    "invalid recycler factory: " + recyclerFactorySpec);
101        }
102
103    }
104
105    private static RecyclerFactory readQueueingRecyclerFactory(
106            final String recyclerFactorySpec,
107            final int defaultCapacity) {
108
109        // Parse the spec.
110        final String queueFactorySpec = recyclerFactorySpec.substring(
111                "queue".length() +
112                        (recyclerFactorySpec.startsWith("queue:")
113                                ? 1
114                                : 0));
115        final Map<String, StringParameterParser.Value> parsedValues =
116                StringParameterParser.parse(
117                        queueFactorySpec,
118                        new LinkedHashSet<>(Arrays.asList("supplier", "capacity")));
119
120        // Read the supplier path.
121        final StringParameterParser.Value supplierValue = parsedValues.get("supplier");
122        final String supplierPath;
123        if (supplierValue == null || supplierValue instanceof StringParameterParser.NullValue) {
124            supplierPath = JCTOOLS_QUEUE_CLASS_AVAILABLE
125                    ? JCTOOLS_QUEUE_CLASS_SUPPLIER_PATH
126                    : "java.util.concurrent.ArrayBlockingQueue.new";
127        } else {
128            supplierPath = supplierValue.toString();
129        }
130
131        // Read the capacity.
132        final StringParameterParser.Value capacityValue = parsedValues.get("capacity");
133        final int capacity;
134        if (capacityValue == null || capacityValue instanceof StringParameterParser.NullValue) {
135            capacity = defaultCapacity;
136        } else {
137            try {
138                capacity = Integer.parseInt(capacityValue.toString());
139            } catch (final NumberFormatException error) {
140                throw new IllegalArgumentException(
141                        "failed reading capacity in queueing recycler " +
142                                "factory: " + queueFactorySpec, error);
143            }
144        }
145
146        // Execute the read spec.
147        return createRecyclerFactory(queueFactorySpec, supplierPath, capacity);
148
149    }
150
151    private static RecyclerFactory createRecyclerFactory(
152            final String queueFactorySpec,
153            final String supplierPath,
154            final int capacity) {
155        final int supplierPathSplitterIndex = supplierPath.lastIndexOf('.');
156        if (supplierPathSplitterIndex < 0) {
157            throw new IllegalArgumentException(
158                    "invalid supplier in queueing recycler factory: " +
159                            queueFactorySpec);
160        }
161        final String supplierClassName = supplierPath.substring(0, supplierPathSplitterIndex);
162        final String supplierMethodName = supplierPath.substring(supplierPathSplitterIndex + 1);
163        try {
164            final Class<?> supplierClass = LoaderUtil.loadClass(supplierClassName);
165            final Supplier<Queue<Object>> queueSupplier;
166            if ("new".equals(supplierMethodName)) {
167                final Constructor<?> supplierCtor =
168                        supplierClass.getDeclaredConstructor(int.class);
169                queueSupplier = () -> {
170                    try {
171                        @SuppressWarnings("unchecked")
172                        final Queue<Object> typedQueue =
173                                (Queue<Object>) supplierCtor.newInstance(capacity);
174                        return typedQueue;
175                    } catch (final Exception error) {
176                        throw new RuntimeException(
177                                "recycler queue construction failed for factory: " +
178                                        queueFactorySpec, error);
179                    }
180                };
181            } else {
182                final Method supplierMethod =
183                        supplierClass.getMethod(supplierMethodName, int.class);
184                queueSupplier = () -> {
185                    try {
186                        @SuppressWarnings("unchecked")
187                        final Queue<Object> typedQueue =
188                                (Queue<Object>) supplierMethod.invoke(null, capacity);
189                        return typedQueue;
190                    } catch (final Exception error) {
191                        throw new RuntimeException(
192                                "recycler queue construction failed for factory: " +
193                                        queueFactorySpec, error);
194                    }
195                };
196            }
197            return new QueueingRecyclerFactory(queueSupplier);
198        } catch (final Exception error) {
199            throw new RuntimeException(
200                    "failed executing queueing recycler factory: " +
201                            queueFactorySpec, error);
202        }
203    }

036    private static abstract class UnresolvableTemplateResolver
037            implements TemplateResolver<Object> {
038
039        @Override
040        public final boolean isResolvable() {
041            return false;
042        }
043
044        @Override
045        public final boolean isResolvable(Object value) {
046            return false;
047        }

051    private static final TemplateResolver<?> EMPTY_ARRAY_RESOLVER =
052            new UnresolvableTemplateResolver() {
053                @Override
054                public void resolve(final Object value, final JsonWriter jsonWriter) {
055                    jsonWriter.writeArrayStart();
056                    jsonWriter.writeArrayEnd();
057                }

062                @Override
063                public void resolve(final Object value, final JsonWriter jsonWriter) {
064                    jsonWriter.writeObjectStart();
065                    jsonWriter.writeObjectEnd();

069    private static final TemplateResolver<?> NULL_RESOLVER =
070            new UnresolvableTemplateResolver() {
071                @Override
072                public void resolve(final Object value, final JsonWriter jsonWriter) {

077    public static <V, C extends TemplateResolverContext<V, C>> TemplateResolver<V> ofTemplate(
078            final C context,
079            final String template) {
080
081        // Read the template.
082        Object node;
083        try {
084            node = JsonReader.read(template);
085        } catch (final Exception error) {
086            final String message = String.format("failed parsing template (template=%s)", template);
087            throw new RuntimeException(message, error);
088        }
089
090        if (context instanceof EventResolverContext) {
091
092            // Append the additional fields.
093            final EventResolverContext eventResolverContext = (EventResolverContext) context;
094            final EventTemplateAdditionalField[] additionalFields = eventResolverContext.getAdditionalFields();
095            appendAdditionalFields(node, additionalFields);
096
097            // Set the root object key, if given.
098            final String rootObjectKey = eventResolverContext.getEventTemplateRootObjectKey();
099            if (rootObjectKey != null) {
100                node = Collections.singletonMap(rootObjectKey, node);
101            }
102
103        }
104
105        // Resolve the template.
106        return ofObject(context, node);
107
108    }

110    private static void appendAdditionalFields(
111            final Object node,
112            EventTemplateAdditionalField[] additionalFields) {
113        if (additionalFields.length > 0) {
114
115            // Check that the root is an object node.
116            final Map<String, Object> objectNode;
117            try {
118                @SuppressWarnings("unchecked")
119                final Map<String, Object> map = (Map<String, Object>) node;
120                objectNode = map;
121            } catch (final ClassCastException error) {
122                final String message = String.format(
123                        "was expecting an object to merge additional fields: %s",
124                        node.getClass().getName());
125                throw new IllegalArgumentException(message);
126            }
127
128            // Merge additional fields.
129            for (final EventTemplateAdditionalField additionalField : additionalFields) {
130                final String additionalFieldKey = additionalField.getKey();
131                final Object additionalFieldValue;
132                switch (additionalField.getFormat()) {
133                    case STRING:
134                        additionalFieldValue = additionalField.getValue();
135                        break;
136                    case JSON:
137                        try {
138                            additionalFieldValue =  JsonReader.read(additionalField.getValue());
139                        } catch (final Exception error) {
140                            final String message = String.format(
141                                    "failed reading JSON provided by additional field: %s",
142                                    additionalFieldKey);
143                            throw new IllegalArgumentException(message, error);
144                        }
145                        break;
146                    default: {
147                        final String message = String.format(
148                                "unknown format %s for additional field: %s",
149                                additionalFieldKey, additionalField.getFormat());
150                        throw new IllegalArgumentException(message);
151                    }
152                }
153                objectNode.put(additionalFieldKey, additionalFieldValue);
154            }
155
156        }
157    }

159    private static <V, C extends TemplateResolverContext<V, C>> TemplateResolver<V> ofObject(
160            final C context,
161            final Object object) {
162        if (object == null) {
163            @SuppressWarnings("unchecked")
164            final TemplateResolver<V> nullResolver = (TemplateResolver<V>) NULL_RESOLVER;
165            return nullResolver;
166        } else if (object instanceof List) {
167            @SuppressWarnings("unchecked")
168            final List<Object> list = (List<Object>) object;
169            return ofList(context, list);
170        } else if (object instanceof Map) {
171            @SuppressWarnings("unchecked")
172            final Map<String, Object> map = (Map<String, Object>) object;
173            return ofMap(context, map);
174        } else if (object instanceof String) {
175            final String string = (String) object;
176            return ofString(context, string);
177        } else if (object instanceof Number) {
178            final Number number = (Number) object;
179            return ofNumber(number);
180        } else if (object instanceof Boolean) {
181            final boolean value = (boolean) object;
182            return ofBoolean(value);
183        } else {
184            final String message = String.format(
185                    "invalid JSON node type (class=%s)",
186                    object.getClass().getName());
187            throw new IllegalArgumentException(message);
188        }
189    }

191    private static <V, C extends TemplateResolverContext<V, C>> TemplateResolver<V> ofList(
192            final C context,
193            final List<Object> list) {
194
195        // Create resolver for each children.
196        final List<TemplateResolver<V>> itemResolvers = list
197                .stream()
198                .map(item -> {
199                    final TemplateResolver<V> itemResolver = ofObject(context, item);
200                    if (itemResolver.isFlattening()) {
201                        throw new IllegalArgumentException(
202                                "flattening resolvers are not allowed in lists");
203                    }
204                    return itemResolver;
205                })
206                .collect(Collectors.toList());
207
208        // Short-circuit if the array is empty.
209        if (itemResolvers.isEmpty()) {
210            @SuppressWarnings("unchecked")
211            final TemplateResolver<V> emptyArrayResolver =
212                    (TemplateResolver<V>) EMPTY_ARRAY_RESOLVER;
213            return emptyArrayResolver;
214        }
215
216        // Create a parent resolver collecting each child resolver execution.
217        return (final V value, final JsonWriter jsonWriter) -> {
218            jsonWriter.writeArrayStart();
219            for (int itemResolverIndex = 0;
220                 itemResolverIndex < itemResolvers.size();
221                 itemResolverIndex++) {
222                if (itemResolverIndex > 0) {
223                    jsonWriter.writeSeparator();
224                }
225                final TemplateResolver<V> itemResolver = itemResolvers.get(itemResolverIndex);
226                itemResolver.resolve(value, jsonWriter);
227            }
228            jsonWriter.writeArrayEnd();
229        };
230
231    }

233    private static <V, C extends TemplateResolverContext<V, C>> TemplateResolver<V> ofMap(
234            final C context,
235            final Map<String, Object> map) {
236
237        // Check if this is a resolver request.
238        if (map.containsKey(RESOLVER_FIELD_NAME)) {
239            return ofResolver(context, map);
240        }
241
242        // Create resolver for each object field.
243        final List<String> fieldNames = new ArrayList<>();
244        final List<TemplateResolver<V>> fieldResolvers = new ArrayList<>();
245        map.forEach((fieldName, fieldValue) -> {
246            final TemplateResolver<V> fieldResolver = ofObject(context, fieldValue);
247            final boolean resolvable = fieldResolver.isResolvable();
248            if (resolvable) {
249                fieldNames.add(fieldName);
250                fieldResolvers.add(fieldResolver);
251            }
252        });
253
254        // Short-circuit if the object is empty.
255        final int fieldCount = fieldNames.size();
256        if (fieldCount == 0) {
257            @SuppressWarnings("unchecked")
258            final TemplateResolver<V> emptyObjectResolver =
259                    (TemplateResolver<V>) EMPTY_OBJECT_RESOLVER;
260            return emptyObjectResolver;
261        }
262
263        // Prepare field names to avoid escape and truncation costs at runtime.
264        final List<String> fieldPrefixes = fieldNames
265                .stream()
266                .map(fieldName -> {
267                    try (JsonWriter jsonWriter = context.getJsonWriter()) {
268                        jsonWriter.writeString(fieldName);
269                        jsonWriter.getStringBuilder().append(':');
270                        return jsonWriter.getStringBuilder().toString();
271                    }
272                })
273                .collect(Collectors.toList());
274
275        return new TemplateResolver<V>() {
276
277            @Override
278            public boolean isResolvable() {
279                // We have already excluded unresolvable ones while collecting
280                // the resolvers. Hence it is safe to return true here.
281                return true;
282            }
283
284            /**
285             * The parent resolver checking if each child is resolvable given
286             * the passed {@code value}.
287             *
288             * This is an optimization to skip the rendering of a parent if all
289             * its children are not resolvable given the passed {@code value}.
290             */
291            @Override
292            public boolean isResolvable(final V value) {
293                for (int fieldIndex = 0; fieldIndex < fieldCount; fieldIndex++) {
294                    final TemplateResolver<V> fieldResolver = fieldResolvers.get(fieldIndex);
295                    final boolean resolvable = fieldResolver.isResolvable(value);
296                    if (resolvable) {
297                        return true;
298                    }
299                }
300                return false;
301            }
302
303            /**
304             * The parent resolver combining all child resolver executions.
305              */
306            @Override
307            public void resolve(final V value, final JsonWriter jsonWriter) {
308                final StringBuilder jsonWriterStringBuilder = jsonWriter.getStringBuilder();
309                jsonWriter.writeObjectStart();
310                for (int resolvedFieldCount = 0, fieldIndex = 0; fieldIndex < fieldCount; fieldIndex++) {
311                    final TemplateResolver<V> fieldResolver = fieldResolvers.get(fieldIndex);
312                    final boolean resolvable = fieldResolver.isResolvable(value);
313                    if (!resolvable) {
314                        continue;
315                    }
316                    final boolean succeedingEntry = resolvedFieldCount > 0;
317                    final boolean flattening = fieldResolver.isFlattening();
318                    if (flattening) {
319                        final int initLength = jsonWriterStringBuilder.length();
320                        fieldResolver.resolve(value, jsonWriter, succeedingEntry);
321                        final boolean resolved = jsonWriterStringBuilder.length() > initLength;
322                        if (resolved) {
323                            resolvedFieldCount++;
324                        }
325                    } else {
326                        if (succeedingEntry) {
327                            jsonWriter.writeSeparator();
328                        }
329                        final String fieldPrefix = fieldPrefixes.get(fieldIndex);
330                        jsonWriter.writeRawString(fieldPrefix);
331                        fieldResolver.resolve(value, jsonWriter, succeedingEntry);
332                        resolvedFieldCount++;
333                    }
334                }
335                jsonWriter.writeObjectEnd();
336            }
337
338        };
339
340    }

342    private static <V, C extends TemplateResolverContext<V, C>> TemplateResolver<V> ofResolver(
343            final C context,
344            final Map<String, Object> map) {
345
346        // Extract the resolver name.
347        final Object resolverNameObject = map.get(RESOLVER_FIELD_NAME);
348        if (!(resolverNameObject instanceof String)) {
349            throw new IllegalArgumentException(
350                    "invalid resolver name: " + resolverNameObject);
351        }
352        final String resolverName = (String) resolverNameObject;
353
354        // Retrieve the resolver.
355        final TemplateResolverFactory<V, C, ? extends TemplateResolver<V>> resolverFactory =
356                context.getResolverFactoryByName().get(resolverName);
357        if (resolverFactory == null) {
358            throw new IllegalArgumentException("unknown resolver: " + resolverName);
359        }
360        final TemplateResolverConfig resolverConfig = new TemplateResolverConfig(map);
361        return resolverFactory.create(context, resolverConfig);
362
363    }

365    private static <V, C extends TemplateResolverContext<V, C>> TemplateResolver<V> ofString(
366            final C context,
367            final String fieldValue) {
368
369        // Check if substitution needed at all. (Copied logic from
370        // AbstractJacksonLayout.valueNeedsLookup() method.)
371        final boolean substitutionNeeded = fieldValue.contains("${");
372        final JsonWriter contextJsonWriter = context.getJsonWriter();
373        if (substitutionNeeded) {
374
375            // Use Log4j substitutor with LogEvent.
376            if (EventResolverContext.class.isAssignableFrom(context.getContextClass())) {
377                return (final V value, final JsonWriter jsonWriter) -> {
378                    final LogEvent logEvent = (LogEvent) value;
379                    final String replacedText = context.getSubstitutor().replace(logEvent, fieldValue);
380                    jsonWriter.writeString(replacedText);
381                };
382            }
383
384            // Use standalone Log4j substitutor.
385            else {
386                final String replacedText = context.getSubstitutor().replace(null, fieldValue);
387                if (replacedText == null) {
388                    @SuppressWarnings("unchecked")
389                    final TemplateResolver<V> resolver =
390                            (TemplateResolver<V>) NULL_RESOLVER;
391                    return resolver;
392                } else {
393                    // Prepare the escaped replacement first.
394                    final String escapedReplacedText =
395                            contextJsonWriter.use(() ->
396                                    contextJsonWriter.writeString(replacedText));
397                    // Create a resolver dedicated to the escaped replacement.
398                    return (final V value, final JsonWriter jsonWriter) ->
399                            jsonWriter.writeRawString(escapedReplacedText);
400                }
401            }
402
403        }
404
405        // Write the field value as is.
406        else {
407            final String escapedFieldValue =
408                    contextJsonWriter.use(() ->
409                            contextJsonWriter.writeString(fieldValue));
410            return (final V value, final JsonWriter jsonWriter) ->
411                    jsonWriter.writeRawString(escapedFieldValue);
412        }
413
414    }

416    private static <V> TemplateResolver<V> ofNumber(final Number number) {
417        final String numberString = String.valueOf(number);
418        return (final V ignored, final JsonWriter jsonWriter) ->
419                jsonWriter.writeRawString(numberString);
420    }

422    private static <V> TemplateResolver<V> ofBoolean(final boolean value) {
423        return (final V ignored, final JsonWriter jsonWriter) ->
424                jsonWriter.writeBoolean(value);

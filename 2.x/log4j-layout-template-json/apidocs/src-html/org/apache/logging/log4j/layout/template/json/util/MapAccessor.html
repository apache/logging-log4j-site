017package org.apache.logging.log4j.layout.template.json.util;
018
019import java.util.Arrays;
020import java.util.Map;
021import java.util.Objects;
022
023public class MapAccessor {
024
025    private final Map<String, Object> map;
026
027    public MapAccessor(final Map<String, Object> map) {
028        this.map = Objects.requireNonNull(map, "map");
029    }
030
031    public String getString(final String key) {
032        final String[] path = {key};
033        return getObject(path, String.class);
034    }
035
036    public String getString(final String[] path) {
037        return getObject(path, String.class);
038    }
039
040    public boolean getBoolean(final String key, final boolean defaultValue) {
041        final String[] path = {key};
042        return getBoolean(path, defaultValue);
043    }
044
045    public boolean getBoolean(final String[] path, final boolean defaultValue) {
046        final Boolean value = getObject(path, Boolean.class);
047        return value == null ? defaultValue : value;
048    }
049
050    public Boolean getBoolean(final String key) {
051        final String[] path = {key};
052        return getObject(path, Boolean.class);
053    }
054
055    public Boolean getBoolean(final String[] path) {
056        return getObject(path, Boolean.class);
057    }
058
059    public Integer getInteger(final String key) {
060        final String[] path = {key};
061        return getInteger(path);
062    }
063
064    public Integer getInteger(final String[] path) {
065        return getObject(path, Integer.class);
066    }
067
068    public boolean exists(final String key) {
069        final String[] path = {key};
070        return exists(path);
071    }
072
073    public boolean exists(final String[] path) {
074        final Object value = getObject(path, Object.class);
075        return value != null;
076    }
077
078    public Object getObject(final String key) {
079        final String[] path = {key};
080        return getObject(path, Object.class);
081    }
082
083    public <T> T getObject(final String key, final Class<T> clazz) {
084        final String[] path = {key};
085        return getObject(path, clazz);
086    }
087
088    public Object getObject(final String[] path) {
089        return getObject(path, Object.class);
090    }
091
092    public <T> T getObject(final String[] path, final Class<T> clazz) {
093        Objects.requireNonNull(path, "path");
094        Objects.requireNonNull(clazz, "clazz");
095        if (path.length == 0) {
096            throw new IllegalArgumentException("empty path");
097        }
098        Object parent = map;
099        for (final String key : path) {
100            if (!(parent instanceof Map)) {
101                return null;
102            }
103            @SuppressWarnings("unchecked")
104            final Map<String, Object> parentMap = (Map<String, Object>) parent;
105            parent = parentMap.get(key);
106        }
107        if (parent != null && !clazz.isInstance(parent)) {
108            final String message = String.format(
109                    "was expecting %s at path %s: %s (of type %s)",
110                    clazz.getSimpleName(),
111                    Arrays.asList(path),
112                    parent,
113                    parent.getClass().getCanonicalName());
114            throw new IllegalArgumentException(message);
115        }
116        @SuppressWarnings("unchecked")
117        final T typedValue = (T) parent;
118        return typedValue;
119    }
120
121    @Override
122    public boolean equals(final Object instance) {
123        if (this == instance) return true;
124        if (instance == null || getClass() != instance.getClass()) return false;
125        final MapAccessor that = (MapAccessor) instance;
126        return map.equals(that.map);
127    }
128
129    @Override
130    public int hashCode() {
131        return Objects.hash(map);
132    }
133
134    @Override
135    public String toString() {
136        return map.toString();
137    }
138
139}

077                final JsonWriter jsonWriter,
078                final Encoder<StringBuilder> encoder) {
079            this.jsonWriter = jsonWriter;

090    private JsonTemplateLayout(final Builder builder) {
091        this.charset = builder.charset;
092        this.contentType = "application/json; charset=" + charset;
093        final String eventDelimiterSuffix = builder.isNullEventDelimiterEnabled() ? "\0" : "";
094        this.eventDelimiter = builder.eventDelimiter + eventDelimiterSuffix;
095        final Configuration configuration = builder.configuration;
096        final StrSubstitutor substitutor = configuration.getStrSubstitutor();
097        final JsonWriter jsonWriter = JsonWriter
098                .newBuilder()
099                .setMaxStringLength(builder.maxStringLength)
100                .setTruncatedStringSuffix(builder.truncatedStringSuffix)
101                .build();
102        final TemplateResolver<StackTraceElement> stackTraceElementObjectResolver =
103                builder.stackTraceEnabled
104                        ? createStackTraceElementResolver(builder, substitutor, jsonWriter)
105                        : null;
106        this.eventResolver = createEventResolver(
107                builder,
108                configuration,
109                substitutor,
110                charset,
111                jsonWriter,
112                stackTraceElementObjectResolver);
113        this.contextRecycler = createContextRecycler(builder, jsonWriter);
114    }
115
116    private static TemplateResolver<StackTraceElement> createStackTraceElementResolver(
117            final Builder builder,
118            final StrSubstitutor substitutor,
119            final JsonWriter jsonWriter) {
120        final StackTraceElementObjectResolverContext stackTraceElementObjectResolverContext =
121                StackTraceElementObjectResolverContext
122                        .newBuilder()
123                        .setSubstitutor(substitutor)
124                        .setJsonWriter(jsonWriter)
125                        .build();
126        final String stackTraceElementTemplate = readStackTraceElementTemplate(builder);
127        return TemplateResolvers.ofTemplate(stackTraceElementObjectResolverContext, stackTraceElementTemplate);
128    }
129
130    private TemplateResolver<LogEvent> createEventResolver(
131            final Builder builder,
132            final Configuration configuration,
133            final StrSubstitutor substitutor,
134            final Charset charset,
135            final JsonWriter jsonWriter,
136            final TemplateResolver<StackTraceElement> stackTraceElementObjectResolver) {
137        final String eventTemplate = readEventTemplate(builder);
138        final float maxByteCountPerChar = builder.charset.newEncoder().maxBytesPerChar();
139        final int maxStringByteCount =
140                Math.toIntExact(Math.round(Math.ceil(
141                        maxByteCountPerChar * builder.maxStringLength)));
142        final EventTemplateAdditionalField[] eventTemplateAdditionalFields =
143                builder.eventTemplateAdditionalFields != null
144                        ? builder.eventTemplateAdditionalFields
145                        : new EventTemplateAdditionalField[0];
146        final EventResolverContext resolverContext = EventResolverContext
147                .newBuilder()
148                .setConfiguration(configuration)
149                .setSubstitutor(substitutor)
150                .setCharset(charset)
151                .setJsonWriter(jsonWriter)
152                .setRecyclerFactory(builder.recyclerFactory)
153                .setMaxStringByteCount(maxStringByteCount)
154                .setTruncatedStringSuffix(builder.truncatedStringSuffix)
155                .setLocationInfoEnabled(builder.locationInfoEnabled)
156                .setStackTraceEnabled(builder.stackTraceEnabled)
157                .setStackTraceElementObjectResolver(stackTraceElementObjectResolver)
158                .setEventTemplateRootObjectKey(builder.eventTemplateRootObjectKey)
159                .setEventTemplateAdditionalFields(eventTemplateAdditionalFields)
160                .build();
161        return TemplateResolvers.ofTemplate(resolverContext, eventTemplate);
162    }
163
164    private static String readEventTemplate(final Builder builder) {
165        return readTemplate(
166                builder.eventTemplate,
167                builder.eventTemplateUri,
168                builder.charset);
169    }
170
171    private static String readStackTraceElementTemplate(final Builder builder) {
172        return readTemplate(
173                builder.stackTraceElementTemplate,
174                builder.stackTraceElementTemplateUri,
175                builder.charset);
176    }
177
178    private static String readTemplate(
179            final String template,
180            final String templateUri,
181            final Charset charset) {
182        return Strings.isBlank(template)
183                ? Uris.readUri(templateUri, charset)
184                : template;
185    }
186
187    private static Recycler<Context> createContextRecycler(
188            final Builder builder,
189            final JsonWriter jsonWriter) {
190        final Supplier<Context> supplier =
191                createContextSupplier(builder.charset, jsonWriter);
192        return builder
193                .recyclerFactory
194                .create(supplier, Context::close);
195    }
196
197    private static Supplier<Context> createContextSupplier(
198            final Charset charset,
199            final JsonWriter jsonWriter) {
200        return () -> {
201            final JsonWriter clonedJsonWriter = jsonWriter.clone();
202            final Encoder<StringBuilder> encoder =
203                    Constants.ENABLE_DIRECT_ENCODERS
204                            ? new LockingStringBuilderEncoder(charset)
205                            : null;
206            return new Context(clonedJsonWriter, encoder);
207        };
208    }

210    @Override
211    public byte[] toByteArray(final LogEvent event) {
212        final String eventJson = toSerializable(event);
213        return StringEncoder.toBytes(eventJson, charset);
214    }
215
216    @Override
217    public String toSerializable(final LogEvent event) {
218        final Context context = acquireContext();
219        final JsonWriter jsonWriter = context.jsonWriter;
220        final StringBuilder stringBuilder = jsonWriter.getStringBuilder();
221        try {
222            eventResolver.resolve(event, jsonWriter);
223            stringBuilder.append(eventDelimiter);
224            return stringBuilder.toString();
225        } finally {
226            contextRecycler.release(context);
227        }
228    }
229
230    @Override
231    public void encode(final LogEvent event, final ByteBufferDestination destination) {
232
233        // Acquire a context.
234        final Context context = acquireContext();
235        final JsonWriter jsonWriter = context.jsonWriter;
236        final StringBuilder stringBuilder = jsonWriter.getStringBuilder();
237        final Encoder<StringBuilder> encoder = context.encoder;
238
239        try {
240
241            // Render the JSON.
242            eventResolver.resolve(event, jsonWriter);
243            stringBuilder.append(eventDelimiter);
244
245            // Write to the destination.
246            if (encoder == null) {
247                final String eventJson = stringBuilder.toString();
248                final byte[] eventJsonBytes = StringEncoder.toBytes(eventJson, charset);
249                destination.writeBytes(eventJsonBytes, 0, eventJsonBytes.length);
250            } else {
251                encoder.encode(stringBuilder, destination);
252            }
253
254        }
255
256        // Release the context.
257        finally {
258            contextRecycler.release(context);
259        }
260
261    }
262
263    // Visible for tests.
264    Context acquireContext() {
265        return contextRecycler.acquire();
266    }

506        private void validate() {
507            Objects.requireNonNull(configuration, "config");
508            if (Strings.isBlank(eventTemplate) && Strings.isBlank(eventTemplateUri)) {
509                    throw new IllegalArgumentException(
510                            "both eventTemplate and eventTemplateUri are blank");
511            }
512            if (stackTraceEnabled &&
513                    Strings.isBlank(stackTraceElementTemplate)
514                    && Strings.isBlank(stackTraceElementTemplateUri)) {
515                throw new IllegalArgumentException(
516                        "both stackTraceElementTemplate and stackTraceElementTemplateUri are blank");
517            }
518            if (maxStringLength <= 0) {
519                throw new IllegalArgumentException(
520                        "was expecting a non-zero positive maxStringLength: " +
521                                maxStringLength);
522            }
523            Objects.requireNonNull(truncatedStringSuffix, "truncatedStringSuffix");
524            Objects.requireNonNull(recyclerFactory, "recyclerFactory");
525        }

529    @Plugin(name = "EventTemplateAdditionalField",
530            category = Node.CATEGORY,
531            printObject = true)
532    public static final class EventTemplateAdditionalField {
533
534        public enum Format { STRING, JSON }
535
536        private final String key;
537
538        private final String value;
539
540        private final Format format;
541
542        private EventTemplateAdditionalField(final Builder builder) {
543            this.key = builder.key;
544            this.value = builder.value;
545            this.format = builder.format;
546        }
547
548        public String getKey() {
549            return key;
550        }
551
552        public String getValue() {
553            return value;
554        }
555
556        public Format getFormat() {
557            return format;
558        }
559
560        @Override
561        public boolean equals(Object object) {
562            if (this == object) {
563                return true;
564            }
565            if (object == null || getClass() != object.getClass()) {
566                return false;
567            }
568            EventTemplateAdditionalField that = (EventTemplateAdditionalField) object;
569            return key.equals(that.key) &&
570                    value.equals(that.value) &&
571                    format == that.format;
572        }
573
574        @Override
575        public int hashCode() {
576            return Objects.hash(key, value, format);
577        }
578
579        @Override
580        public String toString() {
581            final String formattedValue = Format.STRING.equals(format)
582                    ? String.format("\"%s\"", value)
583                    : value;
584            return String.format("%s=%s", key, formattedValue);
585        }
586
587        @PluginBuilderFactory
588        public static EventTemplateAdditionalField.Builder newBuilder() {
589            return new EventTemplateAdditionalField.Builder();
590        }

592        public static class Builder
593                implements org.apache.logging.log4j.core.util.Builder<EventTemplateAdditionalField> {
594
595            @PluginBuilderAttribute
596            private String key;
597
598            @PluginBuilderAttribute
599            private String value;
600
601            @PluginBuilderAttribute
602            private Format format = Format.STRING;
603
604            public Builder setKey(final String key) {
605                this.key = key;
606                return this;
607            }
608
609            public Builder setValue(final String value) {
610                this.value = value;
611                return this;
612            }

615                this.format = format;
616                return this;

621                validate();

623            }

627                    throw new IllegalArgumentException("blank key");
628                }
629                if (Strings.isBlank(value)) {
630                    throw new IllegalArgumentException("blank value");
631                }
632                Objects.requireNonNull(format, "format");
633            }
634
635        }

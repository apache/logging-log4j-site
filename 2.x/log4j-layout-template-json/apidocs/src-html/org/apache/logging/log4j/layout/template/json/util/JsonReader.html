017package org.apache.logging.log4j.layout.template.json.util;
018
019import java.math.BigDecimal;
020import java.math.BigInteger;
021import java.text.CharacterIterator;
022import java.text.StringCharacterIterator;
023import java.util.LinkedHashMap;
024import java.util.LinkedList;
025import java.util.List;
026import java.util.Map;
027import java.util.Objects;
028
029/**
030 * A simple JSON parser mapping tokens to basic Java types.
031 * <p>
032 * The type mapping is as follows:
033 * <p>
034 * <ul>
035 * <li><tt>object</tt>s are mapped to {@link LinkedHashMap LinkedHashMap&lt;String,Object&gt;}
036 * <li><tt>array</tt>s are mapped to {@link LinkedList}
037 * <li><tt>string</tt>s are mapped to {@link String} with proper Unicode and
038 * escape character conversion
039 * <li><tt>true</tt>, <tt>false</tt>, and <tt>null</tt> are mapped to their Java
040 * counterparts
041 * <li>floating point <tt>number</tt>s are mapped to {@link BigDecimal}
042 * <li>integral <tt>number</tt>s are mapped to either primitive types
043 * (<tt>int</tt>, <tt>long</tt>) or {@link BigInteger}
044 * </ul>
045 * <p>
046 * This code is heavily influenced by the reader of
047 * <a href="https://github.com/bolerio/mjson/blob/e7a4da2daa6e17a63ec057948bc30818e8f44686/src/java/mjson/Json.java#L2684">mjson</a>.
048 */
049public final class JsonReader {
050
051    private enum Delimiter {
052
053        OBJECT_START("{"),
054
055        OBJECT_END("}"),
056
057        ARRAY_START("["),
058
059        ARRAY_END("]"),
060
061        COLON(":"),
062
063        COMMA(",");
064
065        private final String string;
066
067        Delimiter(final String string) {
068            this.string = string;
069        }
070
071        private static boolean exists(final Object token) {
072            for (Delimiter delimiter : values()) {
073                if (delimiter.string.equals(token)) {
074                    return true;
075                }
076            }
077            return false;
078        }
079
080    }
081
082    private CharacterIterator it;
083
084    private int readCharIndex = -1;
085
086    private char readChar;
087
088    private int readTokenStartIndex = -1;
089
090    private Object readToken;
091
092    private final StringBuilder buffer;
093
094    private JsonReader() {
095         this.buffer = new StringBuilder();
096    }
097
098    public static Object read(final String json) {
099        Objects.requireNonNull(json, "json");
100        final JsonReader reader = new JsonReader();
101        return reader.read(new StringCharacterIterator(json));
102    }
103
104    private Object read(final CharacterIterator ci) {
105        it = ci;
106        readCharIndex = 0;
107        readChar = it.first();
108        final Object token = readToken();
109        if (token instanceof Delimiter) {
110            final String message = String.format(
111                    "was not expecting %s at index %d",
112                    readToken, readTokenStartIndex);
113            throw new IllegalArgumentException(message);
114        }
115        skipWhiteSpace();
116        if (it.getIndex() != it.getEndIndex()) {
117            final String message = String.format(
118                    "was not expecting input at index %d: %c",
119                    readCharIndex, readChar);
120            throw new IllegalArgumentException(message);
121        }
122        return token;
123    }
124
125    private Object readToken() {
126        skipWhiteSpace();
127        readTokenStartIndex = readCharIndex;
128        final char prevChar = readChar;
129        readChar();
130        switch (prevChar) {
131
132            case '"':
133                readToken = readString();
134                break;
135
136            case '[':
137                readToken = readArray();
138                break;
139
140            case ']':
141                readToken = Delimiter.ARRAY_END;
142                break;
143
144            case ',':
145                readToken = Delimiter.COMMA;
146                break;
147
148            case '{':
149                readToken = readObject();
150                break;
151
152            case '}':
153                readToken = Delimiter.OBJECT_END;
154                break;
155
156            case ':':
157                readToken = Delimiter.COLON;
158                break;
159
160            case 't':
161                readToken = readTrue();
162                break;
163
164            case 'f':
165                readToken = readFalse();
166                break;
167
168            case 'n':
169                readToken = readNull();
170                break;
171
172            default:
173                unreadChar();
174                if (Character.isDigit(readChar) || readChar == '-') {
175                    readToken = readNumber();
176                } else {
177                    String message = String.format(
178                            "invalid character at index %d: %c",
179                            readCharIndex, readChar);
180                    throw new IllegalArgumentException(message);
181                }
182
183        }
184        return readToken;
185    }
186
187    private void skipWhiteSpace() {
188        do {
189            if (!Character.isWhitespace(readChar)) {
190                break;
191            }
192        } while (readChar() != CharacterIterator.DONE);
193    }
194
195    private char readChar() {
196        if (it.getIndex() == it.getEndIndex()) {
197            throw new IllegalArgumentException("premature end of input");
198        }
199        readChar = it.next();
200        readCharIndex = it.getIndex();
201        return readChar;
202    }
203
204    private void unreadChar() {
205        readChar = it.previous();
206        readCharIndex = it.getIndex();
207    }
208
209    private String readString() {
210        buffer.setLength(0);
211        while (readChar != '"') {
212            if (readChar == '\\') {
213                readChar();
214                if (readChar == 'u') {
215                    final char unicodeChar = readUnicodeChar();
216                    bufferChar(unicodeChar);
217                } else {
218                    switch (readChar) {
219                        case '"':
220                        case '\\':
221                            bufferReadChar();
222                            break;
223                        case 'b':
224                            bufferChar('\b');
225                            break;
226                        case 'f':
227                            bufferChar('\f');
228                            break;
229                        case 'n':
230                            bufferChar('\n');
231                            break;
232                        case 'r':
233                            bufferChar('\r');
234                            break;
235                        case 't':
236                            bufferChar('\t');
237                            break;
238                        default: {
239                            final String message = String.format(
240                                    "was expecting an escape character at index %d: %c",
241                                    readCharIndex, readChar);
242                            throw new IllegalArgumentException(message);
243                        }
244                    }
245                }
246            } else {
247                bufferReadChar();
248            }
249        }
250        readChar();
251        return buffer.toString();
252    }
253
254    private void bufferReadChar() {
255        bufferChar(readChar);
256    }
257
258    private void bufferChar(final char c) {
259        buffer.append(c);
260        readChar();
261    }
262
263    private char readUnicodeChar() {
264        int value = 0;
265        for (int i = 0; i < 4; i++) {
266            readChar();
267            if (readChar >= '0' && readChar <= '9') {
268                value = (value << 4) + readChar - '0';
269            } else if (readChar >= 'a' && readChar <= 'f') {
270                value = (value << 4) + (readChar - 'a') + 10;
271            } else if (readChar >= 'A' && readChar <= 'F') {
272                value = (value << 4) + (readChar - 'A') + 10;
273            } else {
274                final String message = String.format(
275                        "was expecting a unicode character at index %d: %c",
276                        readCharIndex, readChar);
277                throw new IllegalArgumentException(message);
278            }
279        }
280        return (char) value;
281    }
282
283    private Map<String, Object> readObject() {
284        final Map<String, Object> object = new LinkedHashMap<>();
285        String key = readObjectKey();
286        while (readToken != Delimiter.OBJECT_END) {
287            expectDelimiter(Delimiter.COLON, readToken());
288            if (readToken != Delimiter.OBJECT_END) {
289                Object value = readToken();
290                object.put(key, value);
291                if (readToken() == Delimiter.COMMA) {
292                    key = readObjectKey();
293                    if (key == null || Delimiter.exists(key)) {
294                        String message = String.format(
295                                "was expecting an object key at index %d: %s",
296                                readTokenStartIndex, readToken);
297                        throw new IllegalArgumentException(message);
298                    }
299                } else {
300                    expectDelimiter(Delimiter.OBJECT_END, readToken);
301                }
302            }
303        }
304        return object;
305    }
306
307    private List<Object> readArray() {
308        @SuppressWarnings("JdkObsolete")
309        final List<Object> array = new LinkedList<>();
310        readToken();
311        while (readToken != Delimiter.ARRAY_END) {
312            if (readToken instanceof Delimiter) {
313                final String message = String.format(
314                        "was expecting an array element at index %d: %s",
315                        readTokenStartIndex, readToken);
316                throw new IllegalArgumentException(message);
317            }
318            array.add(readToken);
319            if (readToken() == Delimiter.COMMA) {
320                if (readToken() == Delimiter.ARRAY_END) {
321                    final String message = String.format(
322                            "was expecting an array element at index %d: %s",
323                            readTokenStartIndex, readToken);
324                    throw new IllegalArgumentException(message);
325                }
326            } else {
327                expectDelimiter(Delimiter.ARRAY_END, readToken);
328            }
329        }
330        return array;
331    }
332
333    private String readObjectKey() {
334        readToken();
335        if (readToken == Delimiter.OBJECT_END) {
336            return null;
337        } else if (readToken instanceof String) {
338            return (String) readToken;
339        } else {
340            final String message = String.format(
341                    "was expecting an object key at index %d: %s",
342                    readTokenStartIndex, readToken);
343            throw new IllegalArgumentException(message);
344        }
345    }
346
347    private void expectDelimiter(
348            final Delimiter expectedDelimiter,
349            final Object actualToken) {
350        if (!expectedDelimiter.equals(actualToken)) {
351            String message = String.format(
352                    "was expecting %s at index %d: %s",
353                    expectedDelimiter, readTokenStartIndex, actualToken);
354            throw new IllegalArgumentException(message);
355        }
356    }
357
358    private boolean readTrue() {
359        if (readChar != 'r' || readChar() != 'u' || readChar() != 'e') {
360            String message = String.format(
361                    "was expecting keyword 'true' at index %d: %s",
362                    readCharIndex, readChar);
363            throw new IllegalArgumentException(message);
364        }
365        readChar();
366        return true;
367    }
368
369    private boolean readFalse() {
370        if (readChar != 'a' || readChar() != 'l' || readChar() != 's' || readChar() != 'e') {
371            String message = String.format(
372                    "was expecting keyword 'false' at index %d: %s",
373                    readCharIndex, readChar);
374            throw new IllegalArgumentException(message);
375        }
376        readChar();
377        return false;
378    }
379
380    private Object readNull() {
381        if (readChar != 'u' || readChar() != 'l' || readChar() != 'l') {
382            String message = String.format(
383                    "was expecting keyword 'null' at index %d: %s",
384                    readCharIndex, readChar);
385            throw new IllegalArgumentException(message);
386        }
387        readChar();
388        return null;
389    }
390
391    private Number readNumber() {
392
393        // Read sign.
394        buffer.setLength(0);
395        if (readChar == '-') {
396            bufferReadChar();
397        }
398
399        // Read fraction.
400        boolean floatingPoint = false;
401        bufferDigits();
402        if (readChar == '.') {
403            bufferReadChar();
404            bufferDigits();
405            floatingPoint = true;
406        }
407
408        // Read exponent.
409        if (readChar == 'e' || readChar == 'E') {
410            floatingPoint = true;
411            bufferReadChar();
412            if (readChar == '+' || readChar == '-') {
413                bufferReadChar();
414            }
415            bufferDigits();
416        }
417
418        // Convert the read number.
419        final String string = buffer.toString();
420        if (floatingPoint) {
421            return new BigDecimal(string);
422        } else {
423            final BigInteger bigInteger = new BigInteger(string);
424            try {
425                return bigInteger.intValueExact();
426            } catch (ArithmeticException ignoredIntOverflow) {
427                try {
428                    return bigInteger.longValueExact();
429                } catch (ArithmeticException ignoredLongOverflow) {
430                    return bigInteger;
431                }
432            }
433        }
434
435    }
436
437    private void bufferDigits() {
438        boolean found = false;
439        while (Character.isDigit(readChar)) {
440            found = true;
441            bufferReadChar();
442        }
443        if (!found) {
444            final String message = String.format(
445                    "was expecting a digit at index %d: %c",
446                    readCharIndex, readChar);
447            throw new IllegalArgumentException(message);
448        }
449    }

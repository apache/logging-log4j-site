56      public static String[] concatanateArrays(String[] l, String[] r) {
57          int len = l.length + r.length;
58          String[] a = new String[len];
59  
60          System.arraycopy(l, 0, a, 0, l.length);
61          System.arraycopy(r, 0, a, l.length, r.length);
62  
63          return a;
64      }

66      public static String convertSpecialChars(String s) {
67          char c;
68          int len = s.length();
69          StringBuilder sbuf = new StringBuilder(len);
70  
71          int i = 0;
72          while (i < len) {
73              c = s.charAt(i++);
74              if (c == '\\') {
75                  c = s.charAt(i++);
76                  for (CharMap entry : charMap) {
77                      if (entry.key == c) {
78                          c = entry.replacement;
79                      }
80                  }
81              }
82              sbuf.append(c);
83          }
84          return sbuf.toString();
85      }

88      /**
89       * Very similar to <code>System.getProperty</code> except
90       * that the {@link SecurityException} is hidden.
91       *
92       * @param key The key to search for.
93       * @param def The default value to return.
94       * @return the string value of the system property, or the default
95       * value if there is no property with that key.
96       * @since 1.1
97       */
98      public static String getSystemProperty(String key, String def) {
99          try {
100             return System.getProperty(key, def);
101         } catch (Throwable e) { // MS-Java throws com.ms.security.SecurityExceptionEx
102             LOGGER.debug("Was not allowed to read system property \"{}\".", key);
103             return def;
104         }
105     }

107     /**
108      * If <code>value</code> is "true", then <code>true</code> is
109      * returned. If <code>value</code> is "false", then
110      * <code>true</code> is returned. Otherwise, <code>default</code> is
111      * returned.
112      *
113      * <p>Case of value is unimportant.
114      */
115     public static boolean toBoolean(String value, boolean dEfault) {
116         if (value == null) {
117             return dEfault;
118         }
119         String trimmedVal = value.trim();
120         if ("true".equalsIgnoreCase(trimmedVal)) {
121             return true;
122         }
123         if ("false".equalsIgnoreCase(trimmedVal)) {
124             return false;
125         }
126         return dEfault;
127     }

129     /**
130      * Converts a standard or custom priority level to a Level
131      * object.  <p> If <code>value</code> is of form
132      * "level#classname", then the specified class' toLevel method
133      * is called to process the specified level string; if no '#'
134      * character is present, then the default {@link org.apache.log4j.Level}
135      * class is used to process the level value.
136      *
137      * <p>As a special case, if the <code>value</code> parameter is
138      * equal to the string "NULL", then the value <code>null</code> will
139      * be returned.
140      *
141      * <p> If any error occurs while converting the value to a level,
142      * the <code>defaultValue</code> parameter, which may be
143      * <code>null</code>, is returned.
144      *
145      * <p> Case of <code>value</code> is insignificant for the level level, but is
146      * significant for the class name part, if present.
147      *
148      * @since 1.1
149      */
150     public static Level toLevel(String value, Level defaultValue) {
151         if (value == null) {
152             return defaultValue;
153         }
154 
155         value = value.trim();
156 
157         int hashIndex = value.indexOf('#');
158         if (hashIndex == -1) {
159             if ("NULL".equalsIgnoreCase(value)) {
160                 return null;
161             } else {
162                 // no class name specified : use standard Level class
163                 return Level.toLevel(value, defaultValue);
164             }
165         }
166 
167         Level result = defaultValue;
168 
169         String clazz = value.substring(hashIndex + 1);
170         String levelName = value.substring(0, hashIndex);
171 
172         // This is degenerate case but you never know.
173         if ("NULL".equalsIgnoreCase(levelName)) {
174             return null;
175         }
176 
177         LOGGER.debug("toLevel" + ":class=[" + clazz + "]"
178                 + ":pri=[" + levelName + "]");
179 
180         try {
181             Class customLevel = LoaderUtil.loadClass(clazz);
182 
183             // get a ref to the specified class' static method
184             // toLevel(String, org.apache.log4j.Level)
185             Class[] paramTypes = new Class[]{String.class,
186                     org.apache.log4j.Level.class
187             };
188             java.lang.reflect.Method toLevelMethod =
189                     customLevel.getMethod("toLevel", paramTypes);
190 
191             // now call the toLevel method, passing level string + default
192             Object[] params = new Object[]{levelName, defaultValue};
193             Object o = toLevelMethod.invoke(null, params);
194 
195             result = (Level) o;
196         } catch (ClassNotFoundException e) {
197             LOGGER.warn("custom level class [" + clazz + "] not found.");
198         } catch (NoSuchMethodException e) {
199             LOGGER.warn("custom level class [" + clazz + "]"
200                     + " does not have a class function toLevel(String, Level)", e);
201         } catch (java.lang.reflect.InvocationTargetException e) {
202             if (e.getTargetException() instanceof InterruptedException
203                     || e.getTargetException() instanceof InterruptedIOException) {
204                 Thread.currentThread().interrupt();
205             }
206             LOGGER.warn("custom level class [" + clazz + "]"
207                     + " could not be instantiated", e);
208         } catch (ClassCastException e) {
209             LOGGER.warn("class [" + clazz
210                     + "] is not a subclass of org.apache.log4j.Level", e);
211         } catch (IllegalAccessException e) {
212             LOGGER.warn("class [" + clazz +
213                     "] cannot be instantiated due to access restrictions", e);
214         } catch (RuntimeException e) {
215             LOGGER.warn("class [" + clazz + "], level [" + levelName +
216                     "] conversion failed.", e);
217         }
218         return result;
219     }

221     /**
222      * Instantiate an object given a class name. Check that the
223      * <code>className</code> is a subclass of
224      * <code>superClass</code>. If that test fails or the object could
225      * not be instantiated, then <code>defaultValue</code> is returned.
226      *
227      * @param className    The fully qualified class name of the object to instantiate.
228      * @param superClass   The class to which the new object should belong.
229      * @param defaultValue The object to return in case of non-fulfillment
230      */
231     public static Object instantiateByClassName(String className, Class<?> superClass,
232             Object defaultValue) {
233         if (className != null) {
234             try {
235                 Object obj = LoaderUtil.newInstanceOf(className);
236                 if (!superClass.isAssignableFrom(obj.getClass())) {
237                     LOGGER.error("A \"{}\" object is not assignable to a \"{}\" variable", className,
238                             superClass.getName());
239                     return defaultValue;
240                 }
241                 return obj;
242             } catch (ClassNotFoundException | NoSuchMethodException | IllegalAccessException
243                     | InstantiationException | InvocationTargetException e) {
244                 LOGGER.error("Could not instantiate class [" + className + "].", e);
245             }
246         }
247         return defaultValue;
248     }

251     /**
252      * Perform variable substitution in string <code>val</code> from the
253      * values of keys found in the system propeties.
254      *
255      * <p>The variable substitution delimeters are <b>${</b> and <b>}</b>.
256      *
257      * <p>For example, if the System properties contains "key=value", then
258      * the call
259      * <pre>
260      * String s = OptionConverter.substituteVars("Value of key is ${key}.");
261      * </pre>
262      * <p>
263      * will set the variable <code>s</code> to "Value of key is value.".
264      *
265      * <p>If no value could be found for the specified key, then the
266      * <code>props</code> parameter is searched, if the value could not
267      * be found there, then substitution defaults to the empty string.
268      *
269      * <p>For example, if system propeties contains no value for the key
270      * "inexistentKey", then the call
271      *
272      * <pre>
273      * String s = OptionConverter.subsVars("Value of inexistentKey is [${inexistentKey}]");
274      * </pre>
275      * will set <code>s</code> to "Value of inexistentKey is []"
276      *
277      * <p>An {@link IllegalArgumentException} is thrown if
278      * <code>val</code> contains a start delimeter "${" which is not
279      * balanced by a stop delimeter "}". </p>
280      *
281      * <p><b>Author</b> Avy Sharell</a></p>
282      *
283      * @param val The string on which variable substitution is performed.
284      * @throws IllegalArgumentException if <code>val</code> is malformed.
285      */
286     public static String substVars(String val, Properties props) throws IllegalArgumentException {
287 
288         StringBuilder sbuf = new StringBuilder();
289 
290         int i = 0;
291         int j, k;
292 
293         while (true) {
294             j = val.indexOf(DELIM_START, i);
295             if (j == -1) {
296                 // no more variables
297                 if (i == 0) { // this is a simple string
298                     return val;
299                 } else { // add the tail string which contails no variables and return the result.
300                     sbuf.append(val.substring(i, val.length()));
301                     return sbuf.toString();
302                 }
303             } else {
304                 sbuf.append(val.substring(i, j));
305                 k = val.indexOf(DELIM_STOP, j);
306                 if (k == -1) {
307                     throw new IllegalArgumentException('"' + val +
308                             "\" has no closing brace. Opening brace at position " + j
309                             + '.');
310                 } else {
311                     j += DELIM_START_LEN;
312                     String key = val.substring(j, k);
313                     // first try in System properties
314                     String replacement = getSystemProperty(key, null);
315                     // then try props parameter
316                     if (replacement == null && props != null) {
317                         replacement = props.getProperty(key);
318                     }
319 
320                     if (replacement != null) {
321                         // Do variable substitution on the replacement string
322                         // such that we can solve "Hello ${x2}" as "Hello p1"
323                         // the where the properties are
324                         // x1=p1
325                         // x2=${x1}
326                         String recursiveReplacement = substVars(replacement, props);
327                         sbuf.append(recursiveReplacement);
328                     }
329                     i = k + DELIM_STOP_LEN;
330                 }
331             }
332         }
333     }

335     public static org.apache.logging.log4j.Level convertLevel(String level,
336             org.apache.logging.log4j.Level defaultLevel) {
337         Level l = toLevel(level, null);
338         return l != null ? convertLevel(l) : defaultLevel;
339     }

341     public static  org.apache.logging.log4j.Level convertLevel(Level level) {
342         if (level == null) {
343             return org.apache.logging.log4j.Level.ERROR;
344         }
345         if (level.isGreaterOrEqual(Level.FATAL)) {
346             return org.apache.logging.log4j.Level.FATAL;
347         } else if (level.isGreaterOrEqual(Level.ERROR)) {
348             return org.apache.logging.log4j.Level.ERROR;
349         } else if (level.isGreaterOrEqual(Level.WARN)) {
350             return org.apache.logging.log4j.Level.WARN;
351         } else if (level.isGreaterOrEqual(Level.INFO)) {
352             return org.apache.logging.log4j.Level.INFO;
353         } else if (level.isGreaterOrEqual(Level.DEBUG)) {
354             return org.apache.logging.log4j.Level.DEBUG;
355         } else if (level.isGreaterOrEqual(Level.TRACE)) {
356             return org.apache.logging.log4j.Level.TRACE;
357         }
358         return org.apache.logging.log4j.Level.ALL;
359     }
360 
361     public static Level convertLevel(org.apache.logging.log4j.Level level) {
362         if (level == null) {
363             return Level.ERROR;
364         }
365         switch (level.getStandardLevel()) {
366             case FATAL:
367                 return Level.FATAL;
368             case WARN:
369                 return Level.WARN;
370             case INFO:
371                 return Level.INFO;
372             case DEBUG:
373                 return Level.DEBUG;
374             case TRACE:
375                 return Level.TRACE;
376             case ALL:
377                 return Level.ALL;
378             case OFF:
379                 return Level.OFF;
380             default:
381                 return Level.ERROR;
382         }
383     }

385     /**
386      * Find the value corresponding to <code>key</code> in
387      * <code>props</code>. Then perform variable substitution on the
388      * found value.
389      */
390     public static String findAndSubst(String key, Properties props) {
391         String value = props.getProperty(key);
392         if (value == null) {
393             return null;
394         }
395 
396         try {
397             return substVars(value, props);
398         } catch (IllegalArgumentException e) {
399             LOGGER.error("Bad option value [{}].", value, e);
400             return value;
401         }
402     }

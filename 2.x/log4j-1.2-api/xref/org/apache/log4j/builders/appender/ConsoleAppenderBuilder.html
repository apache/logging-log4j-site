49  /**
50   * Build a Console Appender
51   */
52  @Plugin(name = "org.apache.log4j.ConsoleAppender", category = CATEGORY)
53  public class ConsoleAppenderBuilder extends AbstractBuilder implements AppenderBuilder {
54      private static final String SYSTEM_OUT = "System.out";
55      private static final String SYSTEM_ERR = "System.err";
56      private static final String TARGET = "target";
57  
58      private static final Logger LOGGER = StatusLogger.getLogger();
59  
60      public ConsoleAppenderBuilder() {
61      }
62  
63      public ConsoleAppenderBuilder(String prefix, Properties props) {
64          super(prefix, props);
65      }
66  
67      @Override
68      public Appender parseAppender(final Element appenderElement, final XmlConfiguration config) {
69          String name = appenderElement.getAttribute(NAME_ATTR);
70          Holder<String> target = new Holder<>(SYSTEM_OUT);
71          Holder<Layout> layout = new Holder<>();
72          Holder<List<Filter>> filters = new Holder<>(new ArrayList<>());
73          Holder<String> level = new Holder<>();
74          forEachElement(appenderElement.getChildNodes(), (currentElement) -> {
75              switch (currentElement.getTagName()) {
76                  case LAYOUT_TAG:
77                      layout.set(config.parseLayout(currentElement));
78                      break;
79                  case FILTER_TAG:
80                      filters.get().add(config.parseFilters(currentElement));
81                      break;
82                  case PARAM_TAG: {
83                      switch (currentElement.getAttribute(NAME_ATTR)) {
84                          case TARGET: {
85                              String value = currentElement.getAttribute(VALUE_ATTR);
86                              if (value == null) {
87                                  LOGGER.warn("No value supplied for target parameter. Defaulting to System.out.");
88                              } else {
89                                  switch (value) {
90                                      case SYSTEM_OUT:
91                                          target.set(SYSTEM_OUT);
92                                          break;
93                                      case SYSTEM_ERR:
94                                          target.set(SYSTEM_ERR);
95                                          break;
96                                      default:
97                                          LOGGER.warn("Invalid value \"{}\" for target parameter. Using default of System.out",
98                                                  value);
99                                  }
100                             }
101                             break;
102                         }
103                         case THRESHOLD_PARAM: {
104                             String value = currentElement.getAttribute(VALUE_ATTR);
105                             if (value == null) {
106                                 LOGGER.warn("No value supplied for Threshold parameter, ignoring.");
107                             } else {
108                                 level.set(value);
109                             }
110                             break;
111                         }
112                     }
113                     break;
114                 }
115             }
116         });
117         Filter head = null;
118         Filter current = null;
119         for (Filter f : filters.get()) {
120             if (head == null) {
121                 head = f;
122                 current = f;
123             } else {
124                 current.next = f;
125                 current = f;
126             }
127         }
128         return createAppender(name, layout.get(), head, level.get(), target.get(), config);
129     }
130 
131     @Override
132     public Appender parseAppender(final String name, final String appenderPrefix, final String layoutPrefix,
133             final String filterPrefix, final Properties props, final PropertiesConfiguration configuration) {
134         Layout layout = configuration.parseLayout(layoutPrefix, name, props);
135         Filter filter = configuration.parseAppenderFilters(props, filterPrefix, name);
136         String level = getProperty(THRESHOLD_PARAM);
137         String target = getProperty(TARGET);
138         return createAppender(name, layout, filter, level, target, configuration);
139     }
140 
141     private <T extends Log4j1Configuration> Appender createAppender(String name, Layout layout, Filter filter,
142             String level, String target, T configuration) {
143         org.apache.logging.log4j.core.Layout<?> consoleLayout = null;
144 
145         if (layout instanceof LayoutWrapper) {
146             consoleLayout = ((LayoutWrapper) layout).getLayout();
147         } else if (layout != null) {
148             consoleLayout = new LayoutAdapter(layout);
149         }
150         org.apache.logging.log4j.core.Filter consoleFilter = buildFilters(level, filter);
151         ConsoleAppender.Target consoleTarget = SYSTEM_ERR.equals(target)
152                 ? ConsoleAppender.Target.SYSTEM_ERR : ConsoleAppender.Target.SYSTEM_OUT;
153         return new AppenderWrapper(ConsoleAppender.newBuilder()
154                 .setName(name)
155                 .setTarget(consoleTarget)
156                 .setLayout(consoleLayout)
157                 .setFilter(consoleFilter)
158                 .setConfiguration(configuration)
159                 .build());
160     }

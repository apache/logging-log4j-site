017package org.apache.log4j.rewrite;
018
019import org.apache.log4j.bridge.LogEventAdapter;
020import org.apache.log4j.helpers.OptionConverter;
021import org.apache.log4j.spi.LocationInfo;
022import org.apache.log4j.spi.LoggingEvent;
023import org.apache.logging.log4j.core.LogEvent;
024import org.apache.logging.log4j.core.impl.Log4jLogEvent;
025import org.apache.logging.log4j.message.SimpleMessage;
026import org.apache.logging.log4j.util.SortedArrayStringMap;
027
028import java.util.Collections;
029import java.util.HashMap;
030import java.util.Map;
031import java.util.StringTokenizer;
032
033/**
034 * This policy rewrites events by adding
035 * a user-specified list of properties to the event.
036 * Existing properties are not modified.
037 * <p>
038 * The combination of the RewriteAppender and this policy
039 * performs the same actions as the PropertyFilter from log4j 1.3.
040 */
041
042public class PropertyRewritePolicy implements RewritePolicy {
043    private Map<String, String> properties = Collections.EMPTY_MAP;
044
045    public PropertyRewritePolicy() {
046    }
047
048    /**
049     * Set a string representing the property name/value pairs.
050     * <p>
051     * Form: propname1=propvalue1,propname2=propvalue2
052     *
053     * @param props The properties.
054     */
055    public void setProperties(String props) {
056        Map hashTable = new HashMap();
057        StringTokenizer pairs = new StringTokenizer(props, ",");
058        while (pairs.hasMoreTokens()) {
059            StringTokenizer entry = new StringTokenizer(pairs.nextToken(), "=");
060            hashTable.put(entry.nextElement().toString().trim(), entry.nextElement().toString().trim());
061        }
062        synchronized (this) {
063            properties = hashTable;
064        }
065    }
066
067    /**
068     * {@inheritDoc}
069     */
070    public LoggingEvent rewrite(final LoggingEvent source) {
071        if (!properties.isEmpty()) {
072            Map<String, String> rewriteProps = source.getProperties() != null ? new HashMap<>(source.getProperties())
073                    : new HashMap<>();
074            for (Map.Entry<String, String> entry : properties.entrySet()) {
075                if (!rewriteProps.containsKey(entry.getKey())) {
076                    rewriteProps.put(entry.getKey(), entry.getValue());
077                }
078            }
079            LogEvent event;
080            if (source instanceof LogEventAdapter) {
081                event = new Log4jLogEvent.Builder(((LogEventAdapter) source).getEvent())
082                        .setContextData(new SortedArrayStringMap(rewriteProps))
083                        .build();
084            } else {
085                LocationInfo info = source.getLocationInformation();
086                StackTraceElement element = new StackTraceElement(info.getClassName(), info.getMethodName(),
087                        info.getFileName(), Integer.parseInt(info.getLineNumber()));
088                Thread thread = getThread(source.getThreadName());
089                long threadId = thread != null ? thread.getId() : 0;
090                int threadPriority = thread != null ? thread.getPriority() : 0;
091                event = Log4jLogEvent.newBuilder()
092                        .setContextData(new SortedArrayStringMap(rewriteProps))
093                        .setLevel(OptionConverter.convertLevel(source.getLevel()))
094                        .setLoggerFqcn(source.getFQNOfLoggerClass())
095                        .setMarker(null)
096                        .setMessage(new SimpleMessage(source.getRenderedMessage()))
097                        .setSource(element)
098                        .setLoggerName(source.getLoggerName())
099                        .setThreadName(source.getThreadName())
100                        .setThreadId(threadId)
101                        .setThreadPriority(threadPriority)
102                        .setThrown(source.getThrowableInformation().getThrowable())
103                        .setTimeMillis(source.getTimeStamp())
104                        .setNanoTime(0)
105                        .setThrownProxy(null)
106                        .build();
107            }
108            return new LogEventAdapter(event);
109        }
110        return source;
111    }
112
113    private Thread getThread(String name) {
114        for (Thread thread : Thread.getAllStackTraces().keySet()) {
115            if (thread.getName().equals(name)) {
116                return thread;
117            }
118        }
119        return null;
120    }
121}

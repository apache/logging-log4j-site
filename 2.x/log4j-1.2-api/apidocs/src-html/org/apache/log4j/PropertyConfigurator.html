091    class SortedKeyEnumeration implements Enumeration {
092
093        private final Enumeration e;
094
095        public SortedKeyEnumeration(final Hashtable ht) {
096            final Enumeration f = ht.keys();
097            final Vector keys = new Vector(ht.size());
098            for (int i, last = 0; f.hasMoreElements(); ++last) {
099                final String key = (String) f.nextElement();
100                for (i = 0; i < last; ++i) {
101                    final String s = (String) keys.get(i);
102                    if (key.compareTo(s) <= 0) {
103                        break;
104                    }
105                }
106                keys.add(i, key);
107            }
108            e = keys.elements();
109        }
110
111        @Override
112        public boolean hasMoreElements() {
113            return e.hasMoreElements();
114        }
115
116        @Override
117        public Object nextElement() {
118            return e.nextElement();
119        }
120    }

247    void configureRootCategory(final Properties properties, final LoggerRepository loggerRepository) {
248        String effectiveFrefix = ROOT_LOGGER_PREFIX;
249        String value = OptionConverter.findAndSubst(ROOT_LOGGER_PREFIX, properties);
250
251        if (value == null) {
252            value = OptionConverter.findAndSubst(ROOT_CATEGORY_PREFIX, properties);
253            effectiveFrefix = ROOT_CATEGORY_PREFIX;
254        }
255
256        if (value == null) {
257            LogLog.debug("Could not find root logger information. Is this OK?");
258        } else {
259            final Logger root = loggerRepository.getRootLogger();
260            synchronized (root) {
261                parseCategory(properties, root, effectiveFrefix, INTERNAL_ROOT_NAME, value);
262            }
263        }
264    }

360        try (InputStream inputStream = Files.newInputStream(Paths.get(fileName))) {
361            return doConfigure(inputStream, loggerRepository, classLoader);
362        } catch (final Exception e) {
363            if (e instanceof InterruptedIOException || e instanceof InterruptedException) {
364                Thread.currentThread().interrupt();
365            }
366            LogLog.error("Could not read configuration file [" + fileName + "].", e);
367            LogLog.error("Ignoring configuration file [" + fileName + "].");
368            return null;
369        }

383    Configuration doConfigure(final URL url, final LoggerRepository loggerRepository, final ClassLoader classLoader) {
384        LogLog.debug("Reading configuration from URL " + url);
385        try {
386            final URLConnection urlConnection = UrlConnectionFactory.createConnection(url);
387            try (InputStream inputStream = urlConnection.getInputStream()) {
388                return doConfigure(inputStream, loggerRepository, classLoader);
389            }
390        } catch (final IOException e) {
391            LogLog.error("Could not read configuration file from URL [" + url + "].", e);
392            LogLog.error("Ignoring configuration file [" + url + "].");
393            return null;
394        }

398    private Properties loadProperties(final InputStream inputStream) {
399        final Properties loaded = new Properties();
400        try {
401            loaded.load(inputStream);
402        } catch (final IOException | IllegalArgumentException e) {
403            if (e instanceof InterruptedIOException) {
404                Thread.currentThread().interrupt();
405            }
406            LogLog.error("Could not read configuration file from InputStream [" + inputStream + "].", e);
407            LogLog.error("Ignoring configuration InputStream [" + inputStream + "].");
408            return null;
409        }
410        return loaded;
411    }

427    Appender parseAppender(final Properties properties, final String appenderName) {
428        Appender appender = registryGet(appenderName);
429        if ((appender != null)) {
430            LogLog.debug("Appender \"" + appenderName + "\" was already parsed.");
431            return appender;
432        }
433        // Appender was not previously initialized.
434        final String prefix = APPENDER_PREFIX + appenderName;
435        final String layoutPrefix = prefix + ".layout";
436
437        appender = (Appender) OptionConverter.instantiateByKey(properties, prefix, org.apache.log4j.Appender.class, null);
438        if (appender == null) {
439            LogLog.error("Could not instantiate appender named \"" + appenderName + "\".");
440            return null;
441        }
442        appender.setName(appenderName);
443
444        if (appender instanceof OptionHandler) {
445            if (appender.requiresLayout()) {
446                final Layout layout = (Layout) OptionConverter.instantiateByKey(properties, layoutPrefix, Layout.class, null);
447                if (layout != null) {
448                    appender.setLayout(layout);
449                    LogLog.debug("Parsing layout options for \"" + appenderName + "\".");
450                    // configureOptionHandler(layout, layoutPrefix + ".", props);
451                    PropertySetter.setProperties(layout, properties, layoutPrefix + ".");
452                    LogLog.debug("End of parsing for \"" + appenderName + "\".");
453                }
454            }
455            final String errorHandlerPrefix = prefix + ".errorhandler";
456            final String errorHandlerClass = OptionConverter.findAndSubst(errorHandlerPrefix, properties);
457            if (errorHandlerClass != null) {
458                final ErrorHandler eh = (ErrorHandler) OptionConverter.instantiateByKey(properties, errorHandlerPrefix, ErrorHandler.class, null);
459                if (eh != null) {
460                    appender.setErrorHandler(eh);
461                    LogLog.debug("Parsing errorhandler options for \"" + appenderName + "\".");
462                    parseErrorHandler(eh, errorHandlerPrefix, properties, repository);
463                    final Properties edited = new Properties();
464                    final String[] keys = new String[] {errorHandlerPrefix + "." + ROOT_REF, errorHandlerPrefix + "." + LOGGER_REF,
465                        errorHandlerPrefix + "." + APPENDER_REF_TAG};
466                    for (final Object element : properties.entrySet()) {
467                        final Map.Entry entry = (Map.Entry) element;
468                        int i = 0;
469                        for (; i < keys.length; i++) {
470                            if (keys[i].equals(entry.getKey())) {
471                                break;
472                            }
473                        }
474                        if (i == keys.length) {
475                            edited.put(entry.getKey(), entry.getValue());
476                        }
477                    }
478                    PropertySetter.setProperties(eh, edited, errorHandlerPrefix + ".");
479                    LogLog.debug("End of errorhandler parsing for \"" + appenderName + "\".");
480                }
481
482            }
483            // configureOptionHandler((OptionHandler) appender, prefix + ".", props);
484            PropertySetter.setProperties(appender, properties, prefix + ".");
485            LogLog.debug("Parsed \"" + appenderName + "\" options.");
486        }
487        parseAppenderFilters(properties, appenderName, appender);
488        registryPut(appender);
489        return appender;
490    }

492    void parseAppenderFilters(final Properties properties, final String appenderName, final Appender appender) {
493        // extract filters and filter options from props into a hashtable mapping
494        // the property name defining the filter class to a list of pre-parsed
495        // name-value pairs associated to that filter
496        final String filterPrefix = APPENDER_PREFIX + appenderName + ".filter.";
497        final int fIdx = filterPrefix.length();
498        final Hashtable filters = new Hashtable();
499        final Enumeration e = properties.keys();
500        String name = "";
501        while (e.hasMoreElements()) {
502            final String key = (String) e.nextElement();
503            if (key.startsWith(filterPrefix)) {
504                final int dotIdx = key.indexOf('.', fIdx);
505                String filterKey = key;
506                if (dotIdx != -1) {
507                    filterKey = key.substring(0, dotIdx);
508                    name = key.substring(dotIdx + 1);
509                }
510                Vector filterOpts = (Vector) filters.get(filterKey);
511                if (filterOpts == null) {
512                    filterOpts = new Vector();
513                    filters.put(filterKey, filterOpts);
514                }
515                if (dotIdx != -1) {
516                    final String value = OptionConverter.findAndSubst(key, properties);
517                    filterOpts.add(new NameValue(name, value));
518                }
519            }
520        }
521
522        // sort filters by IDs, insantiate filters, set filter options,
523        // add filters to the appender
524        final Enumeration g = new SortedKeyEnumeration(filters);
525        Filter head = null;
526        while (g.hasMoreElements()) {
527            final String key = (String) g.nextElement();
528            final String clazz = properties.getProperty(key);
529            if (clazz != null) {
530                LogLog.debug("Filter key: [" + key + "] class: [" + properties.getProperty(key) + "] props: " + filters.get(key));
531                final Filter filter = (Filter) OptionConverter.instantiateByClassName(clazz, Filter.class, null);
532                if (filter != null) {
533                    final PropertySetter propSetter = new PropertySetter(filter);
534                    final Vector v = (Vector) filters.get(key);
535                    final Enumeration filterProps = v.elements();
536                    while (filterProps.hasMoreElements()) {
537                        final NameValue kv = (NameValue) filterProps.nextElement();
538                        propSetter.setProperty(kv.key, kv.value);
539                    }
540                    propSetter.activate();
541                    LogLog.debug("Adding filter of type [" + filter.getClass() + "] to appender named [" + appender.getName() + "].");
542                    head = FilterAdapter.addFilter(head, filter);
543                }
544            } else {
545                LogLog.warn("Missing class definition for filter: [" + key + "]");
546            }
547        }
548        appender.addFilter(head);
549    }

551    /**
552     * This method must work for the root category as well.
553     */
554    void parseCategory(final Properties properties, final Logger logger, final String optionKey, final String loggerName, final String value) {
555
556        LogLog.debug("Parsing for [" + loggerName + "] with value=[" + value + "].");
557        // We must skip over ',' but not white space
558        final StringTokenizer st = new StringTokenizer(value, ",");
559
560        // If value is not in the form ", appender.." or "", then we should set
561        // the level of the loggeregory.
562
563        if (!(value.startsWith(",") || value.equals(""))) {
564
565            // just to be on the safe side...
566            if (!st.hasMoreTokens()) {
567                return;
568            }
569
570            final String levelStr = st.nextToken();
571            LogLog.debug("Level token is [" + levelStr + "].");
572
573            // If the level value is inherited, set category level value to
574            // null. We also check that the user has not specified inherited for the
575            // root category.
576            if (INHERITED.equalsIgnoreCase(levelStr) || NULL.equalsIgnoreCase(levelStr)) {
577                if (loggerName.equals(INTERNAL_ROOT_NAME)) {
578                    LogLog.warn("The root logger cannot be set to null.");
579                } else {
580                    logger.setLevel(null);
581                }
582            } else {
583                logger.setLevel(OptionConverter.toLevel(levelStr, Log4j1Configuration.DEFAULT_LEVEL));
584            }
585            LogLog.debug("Category " + loggerName + " set to " + logger.getLevel());
586        }
587
588        // Begin by removing all existing appenders.
589        logger.removeAllAppenders();
590
591        Appender appender;
592        String appenderName;
593        while (st.hasMoreTokens()) {
594            appenderName = st.nextToken().trim();
595            if (appenderName == null || appenderName.equals(",")) {
596                continue;
597            }
598            LogLog.debug("Parsing appender named \"" + appenderName + "\".");
599            appender = parseAppender(properties, appenderName);
600            if (appender != null) {
601                logger.addAppender(appender);
602            }
603        }
604    }

606    /**
607     * Parse non-root elements, such non-root categories and renderers.
608     */
609    protected void parseCatsAndRenderers(final Properties properties, final LoggerRepository loggerRepository) {
610        final Enumeration enumeration = properties.propertyNames();
611        while (enumeration.hasMoreElements()) {
612            final String key = (String) enumeration.nextElement();
613            if (key.startsWith(CATEGORY_PREFIX) || key.startsWith(LOGGER_PREFIX)) {
614                String loggerName = null;
615                if (key.startsWith(CATEGORY_PREFIX)) {
616                    loggerName = key.substring(CATEGORY_PREFIX.length());
617                } else if (key.startsWith(LOGGER_PREFIX)) {
618                    loggerName = key.substring(LOGGER_PREFIX.length());
619                }
620                final String value = OptionConverter.findAndSubst(key, properties);
621                final Logger logger = loggerRepository.getLogger(loggerName, loggerFactory);
622                synchronized (logger) {
623                    parseCategory(properties, logger, key, loggerName, value);
624                    parseAdditivityForLogger(properties, logger, loggerName);
625                }
626            } else if (key.startsWith(RENDERER_PREFIX)) {
627                final String renderedClass = key.substring(RENDERER_PREFIX.length());
628                final String renderingClass = OptionConverter.findAndSubst(key, properties);
629                if (loggerRepository instanceof RendererSupport) {
630                    RendererMap.addRenderer((RendererSupport) loggerRepository, renderedClass, renderingClass);
631                }
632            } else if (key.equals(THROWABLE_RENDERER_PREFIX)) {
633                if (loggerRepository instanceof ThrowableRendererSupport) {
634                    final ThrowableRenderer tr = (ThrowableRenderer) OptionConverter.instantiateByKey(properties, THROWABLE_RENDERER_PREFIX,
635                        org.apache.log4j.spi.ThrowableRenderer.class, null);
636                    if (tr == null) {
637                        LogLog.error("Could not instantiate throwableRenderer.");
638                    } else {
639                        final PropertySetter setter = new PropertySetter(tr);
640                        setter.setProperties(properties, THROWABLE_RENDERER_PREFIX + ".");
641                        ((ThrowableRendererSupport) loggerRepository).setThrowableRenderer(tr);
642
643                    }
644                }
645            }
646        }
647    }
648
649    private void parseErrorHandler(final ErrorHandler errorHandler, final String errorHandlerPrefix, final Properties props,
650        final LoggerRepository loggerRepository) {
651        if (errorHandler != null && loggerRepository != null) {
652            final boolean rootRef = OptionConverter.toBoolean(OptionConverter.findAndSubst(errorHandlerPrefix + ROOT_REF, props), false);
653            if (rootRef) {
654                errorHandler.setLogger(loggerRepository.getRootLogger());
655            }
656            final String loggerName = OptionConverter.findAndSubst(errorHandlerPrefix + LOGGER_REF, props);
657            if (loggerName != null) {
658                final Logger logger = loggerFactory == null ? loggerRepository.getLogger(loggerName) : loggerRepository.getLogger(loggerName, loggerFactory);
659                errorHandler.setLogger(logger);
660            }
661            final String appenderName = OptionConverter.findAndSubst(errorHandlerPrefix + APPENDER_REF_TAG, props);
662            if (appenderName != null) {
663                final Appender backup = parseAppender(props, appenderName);
664                if (backup != null) {
665                    errorHandler.setBackupAppender(backup);
666                }
667            }
668        }
669    }

017package org.apache.log4j.bridge;
018
019import org.apache.log4j.Category;
020import org.apache.log4j.Level;
021import org.apache.log4j.spi.LocationInfo;
022import org.apache.log4j.spi.LoggingEvent;
023import org.apache.log4j.spi.ThrowableInformation;
024import org.apache.logging.log4j.core.LogEvent;
025import org.apache.logging.log4j.core.util.Loader;
026import org.apache.logging.log4j.core.util.Throwables;
027import org.apache.logging.log4j.spi.StandardLevel;
028import org.apache.logging.log4j.status.StatusLogger;
029
030import java.lang.reflect.Method;
031import java.util.Map;
032import java.util.Set;
033
034/**
035 * Converts a Log4j 2 LogEvent into the components needed by a Log4j 1.x LoggingEvent.
036 * This class requires Log4j 2.
037 */
038public class LogEventAdapter extends LoggingEvent {
039
040    private static final long JVM_START_TIME = initStartTime();
041
042    private final LogEvent event;
043
044    public LogEventAdapter(LogEvent event) {
045        this.event = event;
046    }
047
048    /**
049     * Returns the time when the application started, in milliseconds
050     * elapsed since 01.01.1970.
051     */
052    public static long getStartTime() {
053        return JVM_START_TIME;
054    }
055
056    /**
057     * Returns the result of {@code ManagementFactory.getRuntimeMXBean().getStartTime()},
058     * or the current system time if JMX is not available.
059     */
060    private static long initStartTime() {
061        // We'd like to call ManagementFactory.getRuntimeMXBean().getStartTime(),
062        // but Google App Engine throws a java.lang.NoClassDefFoundError
063        // "java.lang.management.ManagementFactory is a restricted class".
064        // The reflection is necessary because without it, Google App Engine
065        // will refuse to initialize this class.
066        try {
067            final Class<?> factoryClass = Loader.loadSystemClass("java.lang.management.ManagementFactory");
068            final Method getRuntimeMXBean = factoryClass.getMethod("getRuntimeMXBean");
069            final Object runtimeMXBean = getRuntimeMXBean.invoke(null);
070
071            final Class<?> runtimeMXBeanClass = Loader.loadSystemClass("java.lang.management.RuntimeMXBean");
072            final Method getStartTime = runtimeMXBeanClass.getMethod("getStartTime");
073            return (Long) getStartTime.invoke(runtimeMXBean);
074        } catch (final Throwable t) {
075            StatusLogger.getLogger().error("Unable to call ManagementFactory.getRuntimeMXBean().getStartTime(), "
076                    + "using system time for OnStartupTriggeringPolicy", t);
077            // We have little option but to declare "now" as the beginning of time.
078            return System.currentTimeMillis();
079        }
080    }
081
082    public LogEvent getEvent() {
083        return this.event;
084    }
085
086    /**
087     * Set the location information for this logging event. The collected
088     * information is cached for future use.
089     */
090    @Override
091    public LocationInfo getLocationInformation() {
092        return new LocationInfo(event.getSource());
093    }
094
095    /**
096     * Return the level of this event. Use this form instead of directly
097     * accessing the <code>level</code> field.
098     */
099    @Override
100    public Level getLevel() {
101        switch (StandardLevel.getStandardLevel(event.getLevel().intLevel())) {
102            case TRACE:
103                return Level.TRACE;
104            case DEBUG:
105                return Level.DEBUG;
106            case INFO:
107                return Level.INFO;
108            case WARN:
109                return Level.WARN;
110            case FATAL:
111                return Level.FATAL;
112            case OFF:
113                return Level.OFF;
114            case ALL:
115                return Level.ALL;
116            default:
117                return Level.ERROR;
118        }
119    }
120
121    /**
122     * Return the name of the logger. Use this form instead of directly
123     * accessing the <code>categoryName</code> field.
124     */
125    @Override
126    public String getLoggerName() {
127        return event.getLoggerName();
128    }
129
130    /**
131     * Gets the logger of the event.
132     */
133    @Override
134    public Category getLogger() {
135        return Category.getInstance(event.getLoggerName());
136    }
137
138    /*
139     Return the message for this logging event.
140    */
141    @Override
142    public Object getMessage() {
143        return event.getMessage();
144    }
145
146    /*
147     * This method returns the NDC for this event.
148     */
149    @Override
150    public String getNDC() {
151        return event.getContextStack().toString();
152    }
153
154    /*
155     Returns the the context corresponding to the <code>key</code> parameter.
156     */
157    @Override
158    public Object getMDC(String key) {
159        if (event.getContextData() != null) {
160            return event.getContextData().getValue(key);
161        } else {
162            return null;
163        }
164    }
165
166    /**
167     * Obtain a copy of this thread's MDC prior to serialization or
168     * asynchronous logging.
169     */
170    @Override
171    public void getMDCCopy() {
172    }
173
174    @Override
175    public String getRenderedMessage() {
176        return event.getMessage().getFormattedMessage();
177    }
178
179    @Override
180    public String getThreadName() {
181        return event.getThreadName();
182    }
183
184    /**
185     * Returns the throwable information contained within this
186     * event. May be <code>null</code> if there is no such information.
187     *
188     * <p>Note that the {@link Throwable} object contained within a
189     * {@link ThrowableInformation} does not survive serialization.
190     *
191     * @since 1.1
192     */
193    @Override
194    public ThrowableInformation getThrowableInformation() {
195        if (event.getThrown() != null) {
196            return new ThrowableInformation(event.getThrown());
197        }
198        return null;
199    }
200
201    /**
202     * Return this event's throwable's string[] representaion.
203     */
204    @Override
205    public String[] getThrowableStrRep() {
206        if (event.getThrown() != null) {
207            return Throwables.toStringList(event.getThrown()).toArray(new String[0]);
208        }
209        return null;
210    }
211
212    @Override
213    public String getProperty(final String key) {
214        return event.getContextData().getValue(key);
215    }
216
217    @Override
218    public Set getPropertyKeySet() {
219        return event.getContextData().toMap().keySet();
220    }
221
222    @Override
223    public Map getProperties() {
224        return event.getContextData().toMap();
225    }
226}

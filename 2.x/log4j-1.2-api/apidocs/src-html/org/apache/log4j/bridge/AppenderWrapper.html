017package org.apache.log4j.bridge;
018
019import org.apache.log4j.Appender;
020import org.apache.log4j.Layout;
021import org.apache.log4j.helpers.AppenderAttachableImpl;
022import org.apache.log4j.spi.AppenderAttachable;
023import org.apache.log4j.spi.ErrorHandler;
024import org.apache.log4j.spi.Filter;
025import org.apache.log4j.spi.LoggingEvent;
026import org.apache.logging.log4j.Logger;
027import org.apache.logging.log4j.core.filter.AbstractFilterable;
028import org.apache.logging.log4j.status.StatusLogger;
029
030import java.util.ArrayList;
031import java.util.Enumeration;
032import java.util.List;
033
034/**
035 * Holds a Log4j 2 Appender in an empty Log4j 1 Appender so it can be extracted when constructing the configuration.
036 * Allows a Log4j 1 Appender to reference a Log4j 2 Appender.
037 */
038public class AppenderWrapper extends AppenderAttachableImpl implements Appender {
039
040    private static final Logger LOGGER = StatusLogger.getLogger();
041    private final org.apache.logging.log4j.core.Appender appender;
042
043    public AppenderWrapper(org.apache.logging.log4j.core.Appender appender) {
044        this.appender = appender;
045    }
046
047    public org.apache.logging.log4j.core.Appender getAppender() {
048        return appender;
049    }
050
051    @Override
052    public void addFilter(Filter newFilter) {
053        if (appender instanceof AbstractFilterable) {
054            if (newFilter instanceof FilterWrapper) {
055                ((AbstractFilterable) appender).addFilter(((FilterWrapper) newFilter).getFilter());
056            } else {
057                ((AbstractFilterable) appender).addFilter(new FilterAdapter(newFilter));
058            }
059        } else {
060            LOGGER.warn("Unable to add filter to appender {}, it does not support filters", appender.getName());
061        }
062    }
063
064    @Override
065    public Filter getFilter() {
066        return null;
067    }
068
069    @Override
070    public void clearFilters() {
071
072    }
073
074    @Override
075    public void close() {
076        // Not supported with Log4j 2.
077    }
078
079    @Override
080    public void doAppend(LoggingEvent event) {
081        if (event instanceof LogEventAdapter) {
082            appender.append(((LogEventAdapter) event).getEvent());
083        }
084    }
085
086    @Override
087    public String getName() {
088        return appender.getName();
089    }
090
091    @Override
092    public void setErrorHandler(ErrorHandler errorHandler) {
093        appender.setHandler(new ErrorHandlerAdapter(errorHandler));
094    }
095
096    @Override
097    public ErrorHandler getErrorHandler() {
098        return ((ErrorHandlerAdapter)appender.getHandler()).getHandler();
099    }
100
101    @Override
102    public void setLayout(Layout layout) {
103        // Log4j 2 doesn't support this.
104    }
105
106    @Override
107    public Layout getLayout() {
108        return new LayoutWrapper(appender.getLayout());
109    }
110
111    @Override
112    public void setName(String name) {
113        // Log4j 2 doesn't support this.
114    }
115
116    @Override
117    public boolean requiresLayout() {
118        return false;
119    }
120}

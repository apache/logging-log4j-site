018package org.apache.log4j.helpers;
019
020import org.apache.logging.log4j.status.StatusLogger;
021
022/**
023 * Logs statements from within Log4j.
024 *
025 * <p>
026 * Log4j components cannot make Log4j logging calls. However, it is sometimes useful for the user to learn about what
027 * Log4j is doing. You can enable Log4j internal logging by defining the <b>log4j.configDebug</b> variable.
028 * </p>
029 * <p>
030 * All Log4j internal debug calls go to <code>System.out</code> where as internal error messages are sent to
031 * <code>System.err</code>. All internal messages are prepended with the string "log4j: ".
032 * </p>
033 *
034 * @since 0.8.2
035 */
036public class LogLog {
037
038    private static final StatusLogger LOGGER = StatusLogger.getLogger();
039
040    /**
041     * Makes Log4j print log4j-internal debug statements to <code>System.out</code>.
042     *
043     * <p>
044     * The value of this string is {@value #DEBUG_KEY}
045     * </p>
046     * <p>
047     * Note that the search for all option names is case sensitive.
048     * </p>
049     */
050    public static final String DEBUG_KEY = "log4j.debug";
051
052    /**
053     * Makes Log4j components print log4j-internal debug statements to <code>System.out</code>.
054     *
055     * <p>
056     * The value of this string is {@value #CONFIG_DEBUG_KEY}.
057     * </p>
058     * <p>
059     * Note that the search for all option names is case sensitive.
060     * </p>
061     *
062     * @deprecated Use {@link #DEBUG_KEY} instead.
063     */
064    @Deprecated
065    public static final String CONFIG_DEBUG_KEY = "log4j.configDebug";
066
067    /**
068     * Debug enabled Enable or disable.
069     */
070    protected static boolean debugEnabled = false;
071
072    /**
073     * In quietMode not even errors generate any output.
074     */
075    private static boolean quietMode = false;
076
077    static {
078        String key = OptionConverter.getSystemProperty(DEBUG_KEY, null);
079        if (key == null) {
080            key = OptionConverter.getSystemProperty(CONFIG_DEBUG_KEY, null);
081        }
082        if (key != null) {
083            debugEnabled = OptionConverter.toBoolean(key, true);
084        }
085    }
086
087    /**
088     * Logs Log4j internal debug statements.
089     * 
090     * @param message the message object to log.
091     */
092    public static void debug(final String message) {
093        if (debugEnabled && !quietMode) {
094            LOGGER.debug(message);
095        }
096    }
097
098    /**
099     * Logs Log4j internal debug statements.
100     * 
101     * @param message the message object to log.
102     * @param throwable the {@code Throwable} to log, including its stack trace.
103     */
104    public static void debug(final String message, final Throwable throwable) {
105        if (debugEnabled && !quietMode) {
106            LOGGER.debug(message, throwable);
107        }
108    }
109
110    /**
111     * Logs Log4j internal error statements.
112     * 
113     * @param message the message object to log.
114     */
115    public static void error(final String message) {
116        if (!quietMode) {
117            LOGGER.error(message);
118        }
119    }
120
121    /**
122     * Logs Log4j internal error statements.
123     * 
124     * @param message the message object to log.
125     * @param throwable the {@code Throwable} to log, including its stack trace.
126     */
127    public static void error(final String message, final Throwable throwable) {
128        if (!quietMode) {
129            LOGGER.error(message, throwable);
130        }
131    }
132
133    /**
134     * Enables and disables Log4j internal logging.
135     *
136     * @param enabled Enable or disable.
137     */
138    static public void setInternalDebugging(final boolean enabled) {
139        debugEnabled = enabled;
140    }
141
142    /**
143     * In quite mode no LogLog generates strictly no output, not even for errors.
144     *
145     * @param quietMode A true for not
146     */
147    public static void setQuietMode(final boolean quietMode) {
148        LogLog.quietMode = quietMode;
149    }
150
151    /**
152     * Logs Log4j internal warning statements.
153     * 
154     * @param message the message object to log.
155     */
156    public static void warn(final String message) {
157        if (!quietMode) {
158            LOGGER.warn(message);
159        }
160    }
161
162    /**
163     * Logs Log4j internal warnings.
164     * 
165     * @param message the message object to log.
166     * @param throwable the {@code Throwable} to log, including its stack trace.
167     */
168    public static void warn(final String message, final Throwable throwable) {
169        if (!quietMode) {
170            LOGGER.warn(message, throwable);
171        }
172    }
173}

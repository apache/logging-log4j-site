018package org.apache.log4j.helpers;
019
020import java.text.FieldPosition;
021import java.text.ParsePosition;
022import java.util.Calendar;
023import java.util.Date;
024import java.util.TimeZone;
025
026/**
027 * Formats a {@link Date} in the format "yyyy-MM-dd HH:mm:ss,SSS" for example "1999-11-27 15:49:37,459".
028 *
029 * <p>
030 * Refer to the <a href=http://www.cl.cam.ac.uk/~mgk25/iso-time.html>summary of the International Standard Date and Time
031 * Notation</a> for more information on this format.
032 * </p>
033 *
034 * @since 0.7.5
035 */
036public class ISO8601DateFormat extends AbsoluteTimeDateFormat {
037
038    private static final long serialVersionUID = -759840745298755296L;
039
040    static private long lastTime;
041
042    static private char[] lastTimeString = new char[20];
043
044    public ISO8601DateFormat() {
045    }
046
047    public ISO8601DateFormat(final TimeZone timeZone) {
048        super(timeZone);
049    }
050
051    /**
052     * Appends a date in the format "YYYY-mm-dd HH:mm:ss,SSS" to <code>sbuf</code>. For example: "1999-11-27 15:49:37,459".
053     *
054     * @param sbuf the <code>StringBuffer</code> to write to
055     */
056    @Override
057    public StringBuffer format(final Date date, final StringBuffer sbuf, final FieldPosition fieldPosition) {
058
059        final long now = date.getTime();
060        final int millis = (int) (now % 1000);
061
062        if ((now - millis) != lastTime || lastTimeString[0] == 0) {
063            // We reach this point at most once per second
064            // across all threads instead of each time format()
065            // is called. This saves considerable CPU time.
066
067            calendar.setTime(date);
068
069            final int start = sbuf.length();
070
071            final int year = calendar.get(Calendar.YEAR);
072            sbuf.append(year);
073
074            String month;
075            switch (calendar.get(Calendar.MONTH)) {
076            case Calendar.JANUARY:
077                month = "-01-";
078                break;
079            case Calendar.FEBRUARY:
080                month = "-02-";
081                break;
082            case Calendar.MARCH:
083                month = "-03-";
084                break;
085            case Calendar.APRIL:
086                month = "-04-";
087                break;
088            case Calendar.MAY:
089                month = "-05-";
090                break;
091            case Calendar.JUNE:
092                month = "-06-";
093                break;
094            case Calendar.JULY:
095                month = "-07-";
096                break;
097            case Calendar.AUGUST:
098                month = "-08-";
099                break;
100            case Calendar.SEPTEMBER:
101                month = "-09-";
102                break;
103            case Calendar.OCTOBER:
104                month = "-10-";
105                break;
106            case Calendar.NOVEMBER:
107                month = "-11-";
108                break;
109            case Calendar.DECEMBER:
110                month = "-12-";
111                break;
112            default:
113                month = "-NA-";
114                break;
115            }
116            sbuf.append(month);
117
118            final int day = calendar.get(Calendar.DAY_OF_MONTH);
119            if (day < 10) {
120                sbuf.append('0');
121            }
122            sbuf.append(day);
123
124            sbuf.append(' ');
125
126            final int hour = calendar.get(Calendar.HOUR_OF_DAY);
127            if (hour < 10) {
128                sbuf.append('0');
129            }
130            sbuf.append(hour);
131            sbuf.append(':');
132
133            final int mins = calendar.get(Calendar.MINUTE);
134            if (mins < 10) {
135                sbuf.append('0');
136            }
137            sbuf.append(mins);
138            sbuf.append(':');
139
140            final int secs = calendar.get(Calendar.SECOND);
141            if (secs < 10) {
142                sbuf.append('0');
143            }
144            sbuf.append(secs);
145
146            sbuf.append(',');
147
148            // store the time string for next time to avoid recomputation
149            sbuf.getChars(start, sbuf.length(), lastTimeString, 0);
150            lastTime = now - millis;
151        } else {
152            sbuf.append(lastTimeString);
153        }
154
155        if (millis < 100) {
156            sbuf.append('0');
157        }
158        if (millis < 10) {
159            sbuf.append('0');
160        }
161
162        sbuf.append(millis);
163        return sbuf;
164    }
165
166    /**
167     * Always returns null.
168     */
169    @Override
170    public Date parse(final String s, final ParsePosition pos) {
171        return null;
172    }
173}

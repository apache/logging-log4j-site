060    /**
061     * A Log4j 1.x level above or equal to this value is considered as OFF.
062     */
063    static final int MAX_CUTOFF_LEVEL = Priority.FATAL_INT
064            + 100 * (StandardLevel.FATAL.intLevel() - StandardLevel.OFF.intLevel() - 1) + 1;
065    /**
066     * A Log4j 1.x level below or equal to this value is considered as ALL.
067     * 
068     * Log4j 2.x ALL to TRACE interval is shorter. This is {@link Priority#ALL_INT}
069     * plus the difference.
070     */
071    static final int MIN_CUTOFF_LEVEL = Priority.ALL_INT + Level.TRACE_INT
072            - (Priority.ALL_INT + StandardLevel.ALL.intLevel()) + StandardLevel.TRACE.intLevel();

093    public static String[] concatanateArrays(final String[] l, final String[] r) {
094        final int len = l.length + r.length;
095        final String[] a = new String[len];
096
097        System.arraycopy(l, 0, a, 0, l.length);
098        System.arraycopy(r, 0, a, l.length, r.length);
099
100        return a;
101    }

103    static int toLog4j2Level(final int v1Level) {
104        // I don't believe anyone uses values much bigger than FATAL
105        if (v1Level >= MAX_CUTOFF_LEVEL) {
106            return StandardLevel.OFF.intLevel();
107        }
108        // Linear transformation up to debug: CUTOFF_LEVEL -> OFF, DEBUG -> DEBUG
109        if (v1Level > Priority.DEBUG_INT) {
110            final int offset = Math.round((v1Level - Priority.DEBUG_INT) / 100.0f);
111            return StandardLevel.DEBUG.intLevel() - offset;
112        }
113        // Steeper linear transformation
114        if (v1Level > Level.TRACE_INT) {
115            final int offset = Math.round((v1Level - Level.TRACE_INT) / 50.0f);
116            return StandardLevel.TRACE.intLevel() - offset;
117        }
118        if (v1Level > MIN_CUTOFF_LEVEL) {
119            final int offset = Level.TRACE_INT - v1Level;
120            return StandardLevel.TRACE.intLevel() + offset;
121        }
122        return StandardLevel.ALL.intLevel();
123    }

125    static int toLog4j1Level(int v2Level) {
126        if (v2Level == StandardLevel.ALL.intLevel()) {
127            return Priority.ALL_INT;
128        }
129        if (v2Level > StandardLevel.TRACE.intLevel()) {
130            return MIN_CUTOFF_LEVEL + (StandardLevel.ALL.intLevel() - v2Level);
131        }
132        // Inflating by 50
133        if (v2Level > StandardLevel.DEBUG.intLevel()) {
134            return Level.TRACE_INT + 50 * (StandardLevel.TRACE.intLevel() - v2Level);
135        }
136        // Inflating by 100
137        if (v2Level > StandardLevel.OFF.intLevel()) {
138            return Priority.DEBUG_INT + 100 * (StandardLevel.DEBUG.intLevel() - v2Level);
139        }
140        return Priority.OFF_INT;
141    }

143    static int toSyslogLevel(int v2Level) {
144        if (v2Level <= StandardLevel.FATAL.intLevel()) {
145            return 0;
146        }
147        if (v2Level <= StandardLevel.ERROR.intLevel()) {
148            return 3 - (3 * (StandardLevel.ERROR.intLevel() - v2Level))
149                    / (StandardLevel.ERROR.intLevel() - StandardLevel.FATAL.intLevel());
150        }
151        if (v2Level <= StandardLevel.WARN.intLevel()) {
152            return 4;
153        }
154        if (v2Level <= StandardLevel.INFO.intLevel()) {
155            return 6 - (2 * (StandardLevel.INFO.intLevel() - v2Level))
156                    / (StandardLevel.INFO.intLevel() - StandardLevel.WARN.intLevel());
157        }
158        return 7;
159    }

175        // level is standard or was created by Log4j 1.x custom level
176        Level actualLevel = toLevel(level.name(), null);

190    public static String convertSpecialChars(final String s) {
191        char c;
192        final int len = s.length();
193        final StringBuilder sbuf = new StringBuilder(len);
194
195        int i = 0;
196        while (i < len) {
197            c = s.charAt(i++);
198            if (c == '\\') {
199                c = s.charAt(i++);
200                for (final CharMap entry : charMap) {
201                    if (entry.key == c) {
202                        c = entry.replacement;
203                    }
204                }
205            }
206            sbuf.append(c);
207        }
208        return sbuf.toString();
209    }

213     * <code>props</code>. Then perform variable substitution on the
214     * found value.

216     * @param props The properties to use in the substitution.

219    public static String findAndSubst(final String key, final Properties props) {
220        final String value = props.getProperty(key);
221        if (value == null) {
222            return null;
223        }
224
225        try {
226            return substVars(value, props);
227        } catch (final IllegalArgumentException e) {
228            LOGGER.error("Bad option value [{}].", value, e);
229            return value;
230        }

253    /**
254     * Instantiate an object given a class name. Check that the
255     * <code>className</code> is a subclass of
256     * <code>superClass</code>. If that test fails or the object could
257     * not be instantiated, then <code>defaultValue</code> is returned.
258     *
259     * @param className    The fully qualified class name of the object to instantiate.
260     * @param superClass   The class to which the new object should belong.
261     * @param defaultValue The object to return in case of non-fulfillment
262     * @return The created object.
263     */
264    public static Object instantiateByClassName(final String className, final Class<?> superClass,
265            final Object defaultValue) {
266        if (className != null) {
267            try {
268                final Object obj = LoaderUtil.newInstanceOf(className);
269                if (!superClass.isAssignableFrom(obj.getClass())) {
270                    LOGGER.error("A \"{}\" object is not assignable to a \"{}\" variable", className,
271                            superClass.getName());
272                    return defaultValue;
273                }
274                return obj;
275            } catch (final ReflectiveOperationException e) {
276                LOGGER.error("Could not instantiate class [" + className + "].", e);
277            }
278        }
279        return defaultValue;
280    }
281
282    public static Object instantiateByKey(final Properties props, final String key, final Class superClass, final Object defaultValue) {
283
284        // Get the value of the property in string form
285        final String className = findAndSubst(key, props);
286        if (className == null) {
287            LogLog.error("Could not find value for key " + key);
288            return defaultValue;
289        }
290        // Trim className to avoid trailing spaces that cause problems.
291        return OptionConverter.instantiateByClassName(className.trim(), superClass, defaultValue);
292    }

295     * Configure log4j given an {@link InputStream}.
296     * <p>
297     * The InputStream will be interpreted by a new instance of a log4j configurator.
298     * </p>
299     * <p>
300     * All configurations steps are taken on the <code>hierarchy</code> passed as a parameter.
301     * </p>
302     *
303     * @param inputStream The configuration input stream.
304     * @param clazz The class name, of the log4j configurator which will parse the <code>inputStream</code>. This must be a
305     *        subclass of {@link Configurator}, or null. If this value is null then a default configurator of
306     *        {@link PropertyConfigurator} is used.
307     * @param hierarchy The {@link LoggerRepository} to act on.
308     * @since 1.2.17
309     */
310    static public void selectAndConfigure(final InputStream inputStream, final String clazz, final LoggerRepository hierarchy) {
311        Configurator configurator = null;
312
313        if (clazz != null) {
314            LOGGER.debug("Preferred configurator class: " + clazz);
315            configurator = (Configurator) instantiateByClassName(clazz, Configurator.class, null);
316            if (configurator == null) {
317                LOGGER.error("Could not instantiate configurator [" + clazz + "].");
318                return;
319            }
320        } else {
321            configurator = new PropertyConfigurator();
322        }
323
324        configurator.doConfigure(inputStream, hierarchy);
325    }

367    /**
368     * Perform variable substitution in string <code>val</code> from the
369     * values of keys found in the system propeties.
370     *
371     * <p>The variable substitution delimeters are <b>${</b> and <b>}</b>.
372     *
373     * <p>For example, if the System properties contains "key=value", then
374     * the call
375     * <pre>
376     * String s = OptionConverter.substituteVars("Value of key is ${key}.");
377     * </pre>
378     * <p>
379     * will set the variable <code>s</code> to "Value of key is value.".
380     *
381     * <p>If no value could be found for the specified key, then the
382     * <code>props</code> parameter is searched, if the value could not
383     * be found there, then substitution defaults to the empty string.
384     *
385     * <p>For example, if system propeties contains no value for the key
386     * "inexistentKey", then the call
387     *
388     * <pre>
389     * String s = OptionConverter.subsVars("Value of inexistentKey is [${inexistentKey}]");
390     * </pre>
391     * will set <code>s</code> to "Value of inexistentKey is []"
392     *
393     * <p>An {@link IllegalArgumentException} is thrown if
394     * <code>val</code> contains a start delimeter "${" which is not
395     * balanced by a stop delimeter "}". </p>
396     *
397     * @param val The string on which variable substitution is performed.
398     * @param props The properties to use for the substitution.
399     * @return The substituted string.
400     * @throws IllegalArgumentException if <code>val</code> is malformed.
401     */
402    public static String substVars(final String val, final Properties props) throws IllegalArgumentException {
403        return substVars(val, props, new ArrayList<>());
404    }
405
406    private static String substVars(final String val, final Properties props, final List<String> keys)
407            throws IllegalArgumentException {
408        if (val == null) {
409            return null;
410        }
411        final StringBuilder sbuf = new StringBuilder();
412
413        int i = 0;
414        int j;
415        int k;
416
417        while (true) {
418            j = val.indexOf(DELIM_START, i);
419            if (j == -1) {
420                // no more variables
421                if (i == 0) { // this is a simple string
422                    return val;
423                }
424                // add the tail string which contails no variables and return the result.
425                sbuf.append(val.substring(i));
426                return sbuf.toString();
427            }
428            sbuf.append(val.substring(i, j));
429            k = val.indexOf(DELIM_STOP, j);
430            if (k == -1) {
431                throw new IllegalArgumentException(Strings.dquote(val)
432                        + " has no closing brace. Opening brace at position " + j
433                        + '.');
434            }
435            j += DELIM_START_LEN;
436            final String key = val.substring(j, k);
437            // first try in System properties
438            String replacement = PropertiesUtil.getProperties().getStringProperty(key, null);
439            // then try props parameter
440            if (replacement == null && props != null) {
441                replacement = props.getProperty(key);
442            }
443
444            if (replacement != null) {
445
446                // Do variable substitution on the replacement string
447                // such that we can solve "Hello ${x2}" as "Hello p1"
448                // the where the properties are
449                // x1=p1
450                // x2=${x1}
451                if (!keys.contains(key)) {
452                    final List<String> usedKeys = new ArrayList<>(keys);
453                    usedKeys.add(key);
454                    final String recursiveReplacement = substVars(replacement, props, usedKeys);
455                    sbuf.append(recursiveReplacement);
456                } else {
457                    sbuf.append(replacement);
458                }
459
460            }
461            i = k + DELIM_STOP_LEN;
462        }
463    }

465    /**
466     * If <code>value</code> is "true", then <code>true</code> is
467     * returned. If <code>value</code> is "false", then
468     * <code>true</code> is returned. Otherwise, <code>default</code> is
469     * returned.
470     *
471     * <p>Case of value is unimportant.
472     * @param value The value to convert.
473     * @param dEfault The default value.
474     * @return the value of the result.
475     */
476    public static boolean toBoolean(final String value, final boolean dEfault) {
477        if (value == null) {
478            return dEfault;
479        }
480        final String trimmedVal = value.trim();
481        if ("true".equalsIgnoreCase(trimmedVal)) {
482            return true;
483        }
484        if ("false".equalsIgnoreCase(trimmedVal)) {
485            return false;
486        }
487        return dEfault;
488    }

490    public static long toFileSize(final String value, final long defaultValue) {
491        if (value == null) {
492            return defaultValue;
493        }
494
495        String s = value.trim().toUpperCase();
496        long multiplier = 1;
497        int index;
498
499        if ((index = s.indexOf("KB")) != -1) {
500            multiplier = 1024;
501            s = s.substring(0, index);
502        } else if ((index = s.indexOf("MB")) != -1) {
503            multiplier = 1024 * 1024;
504            s = s.substring(0, index);
505        } else if ((index = s.indexOf("GB")) != -1) {
506            multiplier = 1024 * 1024 * 1024;
507            s = s.substring(0, index);
508        }
509        if (s != null) {
510            try {
511                return Long.valueOf(s).longValue() * multiplier;
512            } catch (final NumberFormatException e) {
513                LogLog.error("[" + s + "] is not in proper int form.");
514                LogLog.error("[" + value + "] not in expected format.", e);
515            }
516        }
517        return defaultValue;
518    }

533    /**
534     * Converts a standard or custom priority level to a Level object.
535     * <p>
536     * If <code>value</code> is of form "level#classname", then the specified class'
537     * toLevel method is called to process the specified level string; if no '#'
538     * character is present, then the default {@link org.apache.log4j.Level} class
539     * is used to process the level value.
540     * </p>
541     * 
542     * <p>
543     * As a special case, if the <code>value</code> parameter is equal to the string
544     * "NULL", then the value <code>null</code> will be returned.
545     * </p>
546     * 
547     * <p>
548     * As a Log4j 2.x extension, a {@code value}
549     * "level#org.apache.logging.log4j.Level" retrieves the corresponding custom
550     * Log4j 2.x level.
551     * </p>
552     *
553     * <p>
554     * If any error occurs while converting the value to a level, the
555     * <code>defaultValue</code> parameter, which may be <code>null</code>, is
556     * returned.
557     * </p>
558     *
559     * <p>
560     * Case of <code>value</code> is insignificant for the level level, but is
561     * significant for the class name part, if present.
562     * </p>
563     * 
564     * @param value        The value to convert.
565     * @param defaultValue The default value.
566     * @return the value of the result.
567     *
568     * @since 1.1
569     */
570    public static Level toLevel(String value, final Level defaultValue) {
571        if (value == null) {
572            return defaultValue;
573        }
574
575        value = value.trim();
576        final Level cached = LEVELS.get(value);
577        if (cached != null) {
578            return cached;
579        }
580
581        final int hashIndex = value.indexOf('#');
582        if (hashIndex == -1) {
583            if ("NULL".equalsIgnoreCase(value)) {
584                return null;
585            }
586            // no class name specified : use standard Level class
587            final Level standardLevel = Level.toLevel(value, defaultValue);
588            if (standardLevel != null && value.equals(standardLevel.toString())) {
589                LEVELS.putIfAbsent(value, standardLevel);
590            }
591            return standardLevel;
592        }
593
594        final String clazz = value.substring(hashIndex + 1);
595        final String levelName = value.substring(0, hashIndex);
596
597        final Level customLevel = toLevel(clazz, levelName, defaultValue);
598        if (customLevel != null && levelName.equals(customLevel.toString())
599                && clazz.equals(customLevel.getClass().getName())) {
600            LEVELS.putIfAbsent(value, customLevel);
601        }
602        return customLevel;
603    }

605    /**
606     * Converts a custom priority level to a Level object.
607     * 
608     * <p>
609     * If {@code clazz} has the special value "org.apache.logging.log4j.Level" a
610     * wrapper of the corresponding Log4j 2.x custom level object is returned.
611     * </p>
612     * 
613     * @param clazz        a custom level class,
614     * @param levelName    the name of the level,
615     * @param defaultValue the value to return in case an error occurs,
616     * @return the value of the result.
617     */
618    public static Level toLevel(final String clazz, final String levelName, final Level defaultValue) {
619
620        // This is degenerate case but you never know.
621        if ("NULL".equalsIgnoreCase(levelName)) {
622            return null;
623        }
624
625        LOGGER.debug("toLevel" + ":class=[" + clazz + "]" + ":pri=[" + levelName + "]");
626
627        // Support for levels defined in Log4j2.
628        if (LOG4J2_LEVEL_CLASS.equals(clazz)) {
629            final org.apache.logging.log4j.Level v2Level = org.apache.logging.log4j.Level.getLevel(levelName.toUpperCase());
630            if (v2Level != null) {
631                return new LevelWrapper(v2Level);
632            }
633            return defaultValue;
634        }
635        try {
636            final Class<?> customLevel = LoaderUtil.loadClass(clazz);
637
638            // get a ref to the specified class' static method
639            // toLevel(String, org.apache.log4j.Level)
640            final Class<?>[] paramTypes = new Class[] { String.class, org.apache.log4j.Level.class };
641            final java.lang.reflect.Method toLevelMethod =
642                    customLevel.getMethod("toLevel", paramTypes);
643
644            // now call the toLevel method, passing level string + default
645            final Object[] params = new Object[]{levelName, defaultValue};
646            final Object o = toLevelMethod.invoke(null, params);
647
648            return (Level) o;
649        } catch (final ClassNotFoundException e) {
650            LOGGER.warn("custom level class [" + clazz + "] not found.");
651        } catch (final NoSuchMethodException e) {
652            LOGGER.warn("custom level class [" + clazz + "]"
653                    + " does not have a class function toLevel(String, Level)", e);
654        } catch (final java.lang.reflect.InvocationTargetException e) {
655            if (e.getTargetException() instanceof InterruptedException
656                    || e.getTargetException() instanceof InterruptedIOException) {
657                Thread.currentThread().interrupt();
658            }
659            LOGGER.warn("custom level class [" + clazz + "]"
660                    + " could not be instantiated", e);
661        } catch (final ClassCastException e) {
662            LOGGER.warn("class [" + clazz
663                    + "] is not a subclass of org.apache.log4j.Level", e);
664        } catch (final IllegalAccessException e) {
665            LOGGER.warn("class [" + clazz +
666                    "] cannot be instantiated due to access restrictions", e);
667        } catch (final RuntimeException e) {
668            LOGGER.warn("class [" + clazz + "], level [" + levelName +
669                    "] conversion failed.", e);
670        }
671        return defaultValue;
672    }

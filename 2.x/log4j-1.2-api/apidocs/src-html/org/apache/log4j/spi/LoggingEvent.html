017package org.apache.log4j.spi;
018
019import java.util.Map;
020import java.util.Set;
021
022import org.apache.log4j.Category;
023import org.apache.log4j.Level;
024import org.apache.log4j.Priority;
025import org.apache.log4j.bridge.LogEventAdapter;
026
027/**
028 *  No-op version of Log4j 1.2 LoggingEvent. This class is not directly used by Log4j 1.x clients but is used by
029 *  the Log4j 2 LogEvent adapter to be compatible with Log4j 1.x components.
030 */
031public class LoggingEvent {
032
033    /**
034     Returns the time when the application started, in milliseconds
035     elapsed since 01.01.1970.
036     @return the JVM start time.
037     */
038    public static long getStartTime() {
039        return LogEventAdapter.getStartTime();
040    }
041
042    /**
043     * The number of milliseconds elapsed from 1/1/1970 until logging event was created.
044     */
045    public final long timeStamp;
046
047    /**
048     * Constructs a new instance.
049     */
050    public LoggingEvent() {
051        timeStamp = System.currentTimeMillis();
052    }
053
054    /**
055     * Create new instance.
056     * 
057     * @since 1.2.15
058     * @param fqnOfCategoryClass Fully qualified class name of Logger implementation.
059     * @param logger The logger generating this event.
060     * @param timeStamp the timestamp of this logging event
061     * @param level The level of this event.
062     * @param message The message of this event.
063     * @param threadName thread name
064     * @param throwable The throwable of this event.
065     * @param ndc Nested diagnostic context
066     * @param info Location info
067     * @param properties MDC properties
068     */
069    public LoggingEvent(final String fqnOfCategoryClass, final Category logger, final long timeStamp, final Level level, final Object message,
070        final String threadName, final ThrowableInformation throwable, final String ndc, final LocationInfo info, final Map properties) {
071        this.timeStamp = timeStamp;
072    }
073
074    /**
075     * Instantiate a LoggingEvent from the supplied parameters.
076     * 
077     * <p>
078     * Except {@link #timeStamp} all the other fields of <code>LoggingEvent</code> are filled when actually needed.
079     * <p>
080     * 
081     * @param logger The logger generating this event.
082     * @param timeStamp the timestamp of this logging event
083     * @param level The level of this event.
084     * @param message The message of this event.
085     * @param throwable The throwable of this event.
086     */
087    public LoggingEvent(String fqnOfCategoryClass, Category logger, long timeStamp, Priority level, Object message, Throwable throwable) {
088        this.timeStamp = timeStamp;
089    }
090
091    /**
092     * Instantiate a LoggingEvent from the supplied parameters.
093     * 
094     * <p>
095     * Except {@link #timeStamp} all the other fields of <code>LoggingEvent</code> are filled when actually needed.
096     * <p>
097     * 
098     * @param logger The logger generating this event.
099     * @param level The level of this event.
100     * @param message The message of this event.
101     * @param throwable The throwable of this event.
102     */
103    public LoggingEvent(String fqnOfCategoryClass, Category logger, Priority level, Object message, Throwable throwable) {
104        timeStamp = System.currentTimeMillis();
105    }
106
107    public String getFQNOfLoggerClass() {
108        return null;
109    }
110
111    /**
112     * Return the level of this event. Use this form instead of directly
113     * accessing the <code>level</code> field.
114     * @return Always returns null.
115     */
116    public Level getLevel() {
117        return null;
118    }
119
120    /**
121     Set the location information for this logging event. The collected
122     information is cached for future use.
123     @return Always returns null.
124     */
125    public LocationInfo getLocationInformation() {
126        return null;
127    }
128
129    /**
130     * Gets the logger of the event.
131     * Use should be restricted to cloning events.
132     * @return Always returns null.
133     * @since 1.2.15
134     */
135    public Category getLogger() {
136        return null;
137    }
138
139    /**
140     * Return the name of the logger. Use this form instead of directly
141     * accessing the <code>categoryName</code> field.
142     * @return Always returns null.
143     */
144    public String getLoggerName() {
145        return null;
146    }
147
148    public
149    Object getMDC(String key) {
150        return null;
151    }
152
153    /**
154     Obtain a copy of this thread's MDC prior to serialization or
155     asynchronous logging.
156     */
157    public void getMDCCopy() {
158    }
159
160    /**
161     Return the message for this logging event.
162
163     <p>Before serialization, the returned object is the message
164     passed by the user to generate the logging event. After
165     serialization, the returned value equals the String form of the
166     message possibly after object rendering.
167     @return Always returns null.
168     @since 1.1 */
169    public
170    Object getMessage() {
171        return null;
172    }
173
174    public
175    String getNDC() {
176        return null;
177    }
178
179    public Map getProperties() {
180        return null;
181    }
182
183    public String getProperty(final String key) {
184        return null;
185    }
186
187    public Set getPropertyKeySet() {
188        return null;
189    }
190
191    public String getRenderedMessage() {
192        return null;
193    }
194
195    public String getThreadName() {
196        return null;
197    }
198
199    /**
200     Returns the throwable information contained within this
201     event. May be <code>null</code> if there is no such information.
202
203     <p>Note that the {@link Throwable} object contained within a
204     {@link ThrowableInformation} does not survive serialization.
205     @return Always returns null.
206     @since 1.1 */
207    public ThrowableInformation getThrowableInformation() {
208        return null;
209    }
210
211    /**
212     Return this event's throwable's string[] representaion.
213     @return Always returns null.
214     */
215    public String[] getThrowableStrRep() {
216        return null;
217    }
218
219    public long getTimeStamp() {
220        return 0;
221    }
222
223    public Object removeProperty(String propName) {
224        return null;
225    }
226
227    public void setProperty(final String propName,
228            final String propValue) {
229
230    }
231}

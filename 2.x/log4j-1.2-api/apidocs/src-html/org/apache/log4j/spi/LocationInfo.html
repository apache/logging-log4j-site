017package org.apache.log4j.spi;
018
019import java.io.Serializable;
020import java.util.Objects;
021
022/**
023 * The internal representation of caller location information.
024 *
025 * @since 0.8.3
026 */
027public class LocationInfo implements Serializable {
028
029    /**
030     * When location information is not available the constant <code>NA</code> is returned. Current value of this string
031     * constant is <b>?</b>.
032     */
033    public final static String NA = "?";
034
035    static final long serialVersionUID = -1325822038990805636L;
036
037    private final StackTraceElement stackTraceElement;
038
039    public String fullInfo;
040
041    /**
042     * Constructs a new instance.
043     */
044    public LocationInfo(final StackTraceElement stackTraceElement) {
045        this.stackTraceElement = Objects.requireNonNull(stackTraceElement, "stackTraceElement");
046        this.fullInfo = stackTraceElement.toString();
047    }
048
049    /**
050     * Constructs a new instance.
051     *
052     * @param file source file name
053     * @param declaringClass class name
054     * @param methodName method
055     * @param line source line number
056     *
057     * @since 1.2.15
058     */
059    public LocationInfo(final String file, final String declaringClass, final String methodName, final String line) {
060        this(new StackTraceElement(declaringClass, methodName, file, Integer.parseInt(line)));
061    }
062
063    /**
064     * Constructs a new instance.
065     */
066    public LocationInfo(final Throwable throwable, final String fqnOfCallingClass) {
067        String declaringClass = null, methodName = null, file = null, line = null;
068        if (throwable != null && fqnOfCallingClass != null) {
069            final StackTraceElement[] elements = throwable.getStackTrace();
070            String prevClass = NA;
071            for (int i = elements.length - 1; i >= 0; i--) {
072                final String thisClass = elements[i].getClassName();
073                if (fqnOfCallingClass.equals(thisClass)) {
074                    final int caller = i + 1;
075                    if (caller < elements.length) {
076                        declaringClass = prevClass;
077                        methodName = elements[caller].getMethodName();
078                        file = elements[caller].getFileName();
079                        if (file == null) {
080                            file = NA;
081                        }
082                        final int lineNo = elements[caller].getLineNumber();
083                        if (lineNo < 0) {
084                            line = NA;
085                        } else {
086                            line = String.valueOf(lineNo);
087                        }
088                        final StringBuilder builder = new StringBuilder();
089                        builder.append(declaringClass);
090                        builder.append(".");
091                        builder.append(methodName);
092                        builder.append("(");
093                        builder.append(file);
094                        builder.append(":");
095                        builder.append(line);
096                        builder.append(")");
097                        this.fullInfo = builder.toString();
098                    }
099                    break;
100                }
101                prevClass = thisClass;
102            }
103        }
104        this.stackTraceElement = new StackTraceElement(declaringClass, methodName, file, Integer.parseInt(line));
105        this.fullInfo = stackTraceElement.toString();
106    }
107
108    /**
109     * Gets the fully qualified class name of the caller making the logging request.
110     */
111    public String getClassName() {
112        return stackTraceElement.getClassName();
113    }
114
115    /**
116     * Gets the file name of the caller.
117     */
118    public String getFileName() {
119        return stackTraceElement.getFileName();
120    }
121
122    /**
123     * Gets the line number of the caller.
124     */
125    public String getLineNumber() {
126        return Integer.toString(stackTraceElement.getLineNumber());
127    }
128
129    /**
130     * Gets the method name of the caller.
131     */
132    public String getMethodName() {
133        return stackTraceElement.getMethodName();
134    }
135}

018package org.apache.log4j.varia;
019
020import org.apache.log4j.Level;
021import org.apache.log4j.helpers.OptionConverter;
022import org.apache.log4j.spi.Filter;
023import org.apache.log4j.spi.LoggingEvent;
024
025/**
026 * Simple filter based on level matching.
027 *
028 * <p>
029 * The filter admits two options <b>LevelToMatch</b> and <b>AcceptOnMatch</b>. If there is an exact match between the
030 * value of the <b>LevelToMatch</b> option and the level of the {@link LoggingEvent}, then the {@link #decide} method
031 * returns {@link Filter#ACCEPT} in case the <b>AcceptOnMatch</b> option value is set to <code>true</code>, if it is
032 * <code>false</code> then {@link Filter#DENY} is returned. If there is no match, {@link Filter#NEUTRAL} is returned.
033 * </p>
034 *
035 * @since 1.2
036 */
037public class LevelMatchFilter extends Filter {
038
039    /**
040     * Do we return ACCEPT when a match occurs. Default is <code>true</code>.
041     */
042    boolean acceptOnMatch = true;
043
044    /**
045     */
046    Level levelToMatch;
047
048    /**
049     * Return the decision of this filter.
050     *
051     * Returns {@link Filter#NEUTRAL} if the <b>LevelToMatch</b> option is not set or if there is not match. Otherwise, if
052     * there is a match, then the returned decision is {@link Filter#ACCEPT} if the <b>AcceptOnMatch</b> property is set to
053     * <code>true</code>. The returned decision is {@link Filter#DENY} if the <b>AcceptOnMatch</b> property is set to false.
054     *
055     */
056    @Override
057    public int decide(final LoggingEvent event) {
058        if (this.levelToMatch == null) {
059            return Filter.NEUTRAL;
060        }
061
062        boolean matchOccured = false;
063        if (this.levelToMatch.equals(event.getLevel())) {
064            matchOccured = true;
065        }
066
067        if (matchOccured) {
068            if (this.acceptOnMatch) {
069                return Filter.ACCEPT;
070            }
071            return Filter.DENY;
072        }
073        return Filter.NEUTRAL;
074    }
075
076    public boolean getAcceptOnMatch() {
077        return acceptOnMatch;
078    }
079
080    public String getLevelToMatch() {
081        return levelToMatch == null ? null : levelToMatch.toString();
082    }
083
084    public void setAcceptOnMatch(final boolean acceptOnMatch) {
085        this.acceptOnMatch = acceptOnMatch;
086    }
087
088    public void setLevelToMatch(final String level) {
089        levelToMatch = OptionConverter.toLevel(level, null);
090    }
091}

017package org.apache.log4j.builders.appender;
018
019import static org.apache.log4j.builders.BuilderManager.CATEGORY;
020import static org.apache.log4j.config.Log4j1Configuration.THRESHOLD_PARAM;
021import static org.apache.log4j.xml.XmlConfiguration.FILTER_TAG;
022import static org.apache.log4j.xml.XmlConfiguration.LAYOUT_TAG;
023import static org.apache.log4j.xml.XmlConfiguration.PARAM_TAG;
024import static org.apache.log4j.xml.XmlConfiguration.forEachElement;
025
026import java.util.Properties;
027import java.util.concurrent.atomic.AtomicBoolean;
028import java.util.concurrent.atomic.AtomicInteger;
029import java.util.concurrent.atomic.AtomicReference;
030
031import org.apache.log4j.Appender;
032import org.apache.log4j.Layout;
033import org.apache.log4j.bridge.AppenderWrapper;
034import org.apache.log4j.bridge.LayoutAdapter;
035import org.apache.log4j.bridge.LayoutWrapper;
036import org.apache.log4j.builders.AbstractBuilder;
037import org.apache.log4j.config.Log4j1Configuration;
038import org.apache.log4j.config.PropertiesConfiguration;
039import org.apache.log4j.spi.Filter;
040import org.apache.log4j.xml.XmlConfiguration;
041import org.apache.logging.log4j.Logger;
042import org.apache.logging.log4j.core.appender.SocketAppender;
043import org.apache.logging.log4j.core.config.plugins.Plugin;
044import org.apache.logging.log4j.status.StatusLogger;
045import org.w3c.dom.Element;
046
047/**
048 * Build a Console Appender
049 */
050@Plugin(name = "org.apache.log4j.net.SocketAppender", category = CATEGORY)
051public class SocketAppenderBuilder extends AbstractBuilder implements AppenderBuilder {
052
053    private static final String HOST_PARAM = "RemoteHost";
054    private static final String PORT_PARAM = "Port";
055    private static final String RECONNECTION_DELAY_PARAM = "ReconnectionDelay";
056    private static final int DEFAULT_PORT = 4560;
057
058    /**
059     * The default reconnection delay (30000 milliseconds or 30 seconds).
060     */
061    private static final int DEFAULT_RECONNECTION_DELAY = 30_000;
062
063    public static final Logger LOGGER = StatusLogger.getLogger();
064
065    public SocketAppenderBuilder() {
066    }
067
068    public SocketAppenderBuilder(final String prefix, final Properties props) {
069        super(prefix, props);
070    }
071
072    private <T extends Log4j1Configuration> Appender createAppender(final String name, final String host, final int port, final Layout layout,
073        final Filter filter, final String level, final boolean immediateFlush, final int reconnectDelayMillis, final T configuration) {
074        org.apache.logging.log4j.core.Layout<?> actualLayout = LayoutAdapter.adapt(layout);
075        final org.apache.logging.log4j.core.Filter actualFilter = buildFilters(level, filter);
076        // @formatter:off
077        return AppenderWrapper.adapt(SocketAppender.newBuilder()
078            .setHost(host)
079            .setPort(port)
080            .setReconnectDelayMillis(reconnectDelayMillis)
081            .setName(name)
082            .setLayout(actualLayout)
083            .setFilter(actualFilter)
084            .setConfiguration(configuration)
085            .setImmediateFlush(immediateFlush)
086            .build());
087        // @formatter:on
088    }
089
090    @Override
091    public Appender parseAppender(final Element appenderElement, final XmlConfiguration config) {
092        final String name = getNameAttribute(appenderElement);
093        final AtomicReference<String> host = new AtomicReference<>("localhost");
094        final AtomicInteger port = new AtomicInteger(DEFAULT_PORT);
095        final AtomicInteger reconnectDelay = new AtomicInteger(DEFAULT_RECONNECTION_DELAY);
096        final AtomicReference<Layout> layout = new AtomicReference<>();
097        final AtomicReference<Filter> filter = new AtomicReference<>();
098        final AtomicReference<String> level = new AtomicReference<>();
099        final AtomicBoolean immediateFlush = new AtomicBoolean(true);
100        forEachElement(appenderElement.getChildNodes(), currentElement -> {
101            switch (currentElement.getTagName()) {
102            case LAYOUT_TAG:
103                layout.set(config.parseLayout(currentElement));
104                break;
105            case FILTER_TAG:
106                config.addFilter(filter, currentElement);
107                break;
108            case PARAM_TAG:
109                switch (getNameAttributeKey(currentElement)) {
110                case HOST_PARAM:
111                    set(HOST_PARAM, currentElement, host);
112                    break;
113                case PORT_PARAM:
114                    set(PORT_PARAM, currentElement, port);
115                    break;
116                case RECONNECTION_DELAY_PARAM:
117                    set(RECONNECTION_DELAY_PARAM, currentElement, reconnectDelay);
118                    break;
119                case THRESHOLD_PARAM:
120                    set(THRESHOLD_PARAM, currentElement, level);
121                    break;
122                case IMMEDIATE_FLUSH_PARAM:
123                    set(IMMEDIATE_FLUSH_PARAM, currentElement, immediateFlush);
124                    break;
125                }
126                break;
127            }
128        });
129        return createAppender(name, host.get(), port.get(), layout.get(), filter.get(), level.get(), immediateFlush.get(), reconnectDelay.get(), config);
130    }
131
132    @Override
133    public Appender parseAppender(final String name, final String appenderPrefix, final String layoutPrefix, final String filterPrefix, final Properties props,
134        final PropertiesConfiguration configuration) {
135        // @formatter:off
136        return createAppender(name,
137            getProperty(HOST_PARAM),
138            getIntegerProperty(PORT_PARAM, DEFAULT_PORT),
139            configuration.parseLayout(layoutPrefix, name, props),
140            configuration.parseAppenderFilters(props, filterPrefix, name),
141            getProperty(THRESHOLD_PARAM),
142            getBooleanProperty(IMMEDIATE_FLUSH_PARAM),
143            getIntegerProperty(RECONNECTION_DELAY_PARAM, DEFAULT_RECONNECTION_DELAY),
144            configuration);
145        // @formatter:on
146    }
147}

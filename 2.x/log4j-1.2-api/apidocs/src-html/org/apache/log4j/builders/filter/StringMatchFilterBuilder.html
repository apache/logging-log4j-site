017package org.apache.log4j.builders.filter;
018
019import org.apache.log4j.bridge.FilterWrapper;
020import org.apache.log4j.builders.AbstractBuilder;
021import org.apache.log4j.builders.BooleanHolder;
022import org.apache.log4j.builders.Holder;
023import org.apache.log4j.config.PropertiesConfiguration;
024import org.apache.log4j.spi.Filter;
025import org.apache.log4j.xml.XmlConfiguration;
026import org.apache.logging.log4j.Logger;
027import org.apache.logging.log4j.core.config.plugins.Plugin;
028import org.apache.logging.log4j.core.filter.StringMatchFilter;
029import org.apache.logging.log4j.status.StatusLogger;
030import org.w3c.dom.Element;
031
032import java.util.Properties;
033
034import static org.apache.log4j.builders.BuilderManager.CATEGORY;
035import static org.apache.log4j.xml.XmlConfiguration.*;
036
037/**
038 * Build a String match filter.
039 */
040@Plugin(name = "org.apache.log4j.varia.StringMatchFilter", category = CATEGORY)
041public class StringMatchFilterBuilder extends AbstractBuilder implements FilterBuilder {
042
043    private static final Logger LOGGER = StatusLogger.getLogger();
044    private static final String STRING_TO_MATCH = "StringToMatch";
045    private static final String ACCEPT_ON_MATCH = "AcceptOnMatch";
046
047    @Override
048    public Filter parseFilter(Element filterElement, XmlConfiguration config) {
049        final Holder<Boolean> acceptOnMatch = new BooleanHolder();
050        final Holder<String> text = new Holder<>();
051        forEachElement(filterElement.getElementsByTagName("param"), (currentElement) -> {
052            if (currentElement.getTagName().equals("param")) {
053                switch (currentElement.getAttribute(NAME_ATTR)) {
054                    case STRING_TO_MATCH:
055                        text.set(currentElement.getAttribute(VALUE_ATTR));
056                        break;
057                    case ACCEPT_ON_MATCH:
058                        acceptOnMatch.set(Boolean.parseBoolean(currentElement.getAttribute(VALUE_ATTR)));
059                        break;
060
061                }
062            }
063        });
064        return createFilter(text.get(), acceptOnMatch.get());
065    }
066
067    @Override
068    public Filter parseFilter(PropertiesConfiguration config) {
069        String text = getProperty(STRING_TO_MATCH);
070        boolean acceptOnMatch = getBooleanProperty(ACCEPT_ON_MATCH);
071        return createFilter(text, acceptOnMatch);
072    }
073
074    private Filter createFilter(String text, boolean acceptOnMatch) {
075        if (text == null) {
076            LOGGER.warn("No text provided for StringMatchFilter");
077            return null;
078        }
079        org.apache.logging.log4j.core.Filter.Result onMatch = acceptOnMatch
080                ? org.apache.logging.log4j.core.Filter.Result.ACCEPT
081                : org.apache.logging.log4j.core.Filter.Result.DENY;
082        return new FilterWrapper(StringMatchFilter.newBuilder()
083                .setMatchString(text)
084                .setOnMatch(onMatch)
085                .setOnMismatch(org.apache.logging.log4j.core.Filter.Result.NEUTRAL)
086                .build());
087    }
088}

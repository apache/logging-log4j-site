043import org.apache.logging.log4j.core.config.plugins.util.PluginType;

045import org.apache.logging.log4j.util.LoaderUtil;

048/**
049 *
050 */
051public class BuilderManager {
052
053    /** Plugin category. */
054    public static final String CATEGORY = "Log4j Builder";
055    public static final Appender INVALID_APPENDER = new AppenderWrapper(null);
056    public static final Filter INVALID_FILTER = new FilterWrapper(null);
057    public static final Layout INVALID_LAYOUT = new LayoutWrapper(null);
058    public static final RewritePolicy INVALID_REWRITE_POLICY = new RewritePolicyWrapper(null);
059
060    private static final Logger LOGGER = StatusLogger.getLogger();
061    private static Class<?>[] CONSTRUCTOR_PARAMS = new Class[] {String.class, Properties.class};
062    private final Map<String, PluginType<?>> plugins;
063
064    /**
065     * Constructs a new instance.
066     */
067    public BuilderManager() {
068        final PluginManager manager = new PluginManager(CATEGORY);
069        manager.collectPlugins();
070        plugins = manager.getPlugins();
071    }
072
073    private <T extends Builder<U>, U> T createBuilder(final PluginType<T> plugin, final String prefix, final Properties props) {
074        if (plugin == null) {
075            return null;
076        }
077        try {
078            final Class<T> clazz = plugin.getPluginClass();
079            if (AbstractBuilder.class.isAssignableFrom(clazz)) {
080                return clazz.getConstructor(CONSTRUCTOR_PARAMS).newInstance(prefix, props);
081            }
082            final T builder = LoaderUtil.newInstanceOf(clazz);
083            // Reasonable message instead of `ClassCastException`
084            if (!Builder.class.isAssignableFrom(clazz)) {
085                LOGGER.warn("Unable to load plugin: builder {} does not implement {}", clazz, Builder.class);
086                return null;
087            }
088            return builder;
089        } catch (final ReflectiveOperationException ex) {
090            LOGGER.warn("Unable to load plugin: {} due to: {}", plugin.getKey(), ex.getMessage());
091            return null;
092        }
093    }
094
095    @SuppressWarnings("unchecked")
096    private <T> PluginType<T> getPlugin(final String className) {
097        Objects.requireNonNull(plugins, "plugins");
098        Objects.requireNonNull(className, "className");
099        final String key = className.toLowerCase(Locale.ROOT).trim();
100        final PluginType<?> pluginType = plugins.get(key);
101        if (pluginType == null) {
102            LOGGER.warn("Unable to load plugin class name {} with key {}", className, key);
103        }
104        return (PluginType<T>) pluginType;
105    }
106
107    private <T extends Builder<U>, U> U newInstance(final PluginType<T> plugin, final Function<T, U> consumer,
108            final U invalidValue) {
109        if (plugin != null) {
110            try {
111                final T builder = LoaderUtil.newInstanceOf(plugin.getPluginClass());
112                if (builder != null) {
113                    U result = consumer.apply(builder);
114                    // returning an empty wrapper is short for "we support this legacy class, but it has validation errors"
115                    return result != null ? result : invalidValue;
116                }
117            } catch (final ReflectiveOperationException ex) {
118                LOGGER.warn("Unable to load plugin: {} due to: {}", plugin.getKey(), ex.getMessage());
119            }
120        }
121        return null;
122    }
123
124    public <P extends Parser<T>, T> T parse(final String className, final String prefix, final Properties props,
125            final PropertiesConfiguration config, T invalidValue) {
126        final P parser = createBuilder(getPlugin(className), prefix, props);
127        if (parser != null) {
128            final T value = parser.parse(config);
129            return value != null ? value : invalidValue;
130        }
131        return null;
132    }
133
134    public Appender parseAppender(final String className, final Element appenderElement,
135            final XmlConfiguration config) {
136        return newInstance(this.<AppenderBuilder<Appender>>getPlugin(className),
137                b -> b.parseAppender(appenderElement, config), INVALID_APPENDER);
138    }
139
140    public Appender parseAppender(final String name, final String className, final String prefix, final String layoutPrefix, final String filterPrefix,
141        final Properties props, final PropertiesConfiguration config) {
142        final AppenderBuilder<Appender> builder = createBuilder(getPlugin(className), prefix, props);
143        if (builder != null) {
144            final Appender appender = builder.parseAppender(name, prefix, layoutPrefix, filterPrefix, props, config);
145            return appender != null ? appender : INVALID_APPENDER;
146        }
147        return null;
148    }
149
150    public Filter parseFilter(final String className, final Element filterElement, final XmlConfiguration config) {
151        return newInstance(this.<FilterBuilder>getPlugin(className), b -> b.parse(filterElement, config),
152                INVALID_FILTER);
153    }
154
155    public Layout parseLayout(final String className, final Element layoutElement, final XmlConfiguration config) {
156        return newInstance(this.<LayoutBuilder>getPlugin(className), b -> b.parse(layoutElement, config),
157                INVALID_LAYOUT);
158    }
159
160    public RewritePolicy parseRewritePolicy(final String className, final Element rewriteElement,
161            final XmlConfiguration config) {
162        return newInstance(this.<RewritePolicyBuilder>getPlugin(className), b -> b.parse(rewriteElement, config),
163                INVALID_REWRITE_POLICY);
164    }
165
166    public TriggeringPolicy parseTriggeringPolicy(final String className, final Element policyElement,
167            final XmlConfiguration config) {
168        return newInstance(this.<TriggeringPolicyBuilder>getPlugin(className), b -> b.parse(policyElement, config),
169                (TriggeringPolicy) null);
170    }
171}

020import static org.apache.log4j.config.Log4j1Configuration.THRESHOLD_PARAM;
021import static org.apache.log4j.xml.XmlConfiguration.FILTER_TAG;
022import static org.apache.log4j.xml.XmlConfiguration.LAYOUT_TAG;
023import static org.apache.log4j.xml.XmlConfiguration.PARAM_TAG;
024import static org.apache.log4j.xml.XmlConfiguration.forEachElement;
025
026import java.util.Properties;
027import java.util.concurrent.atomic.AtomicBoolean;
028import java.util.concurrent.atomic.AtomicInteger;
029import java.util.concurrent.atomic.AtomicReference;
030
031import org.apache.log4j.Appender;
032import org.apache.log4j.Layout;
033import org.apache.log4j.bridge.AppenderWrapper;
034import org.apache.log4j.bridge.LayoutAdapter;

037import org.apache.log4j.config.Log4j1Configuration;

040import org.apache.log4j.xml.XmlConfiguration;

043import org.apache.logging.log4j.core.appender.rolling.CompositeTriggeringPolicy;
044import org.apache.logging.log4j.core.appender.rolling.DefaultRolloverStrategy;
045import org.apache.logging.log4j.core.appender.rolling.RolloverStrategy;

051/**
052 * Build a File Appender
053 */
054@Plugin(name = "org.apache.log4j.RollingFileAppender", category = CATEGORY)
055public class RollingFileAppenderBuilder extends AbstractBuilder implements AppenderBuilder {
056
057    private static final String DEFAULT_MAX_SIZE = "10 MB";
058    private static final String DEFAULT_MAX_BACKUPS = "1";
059
060    private static final Logger LOGGER = StatusLogger.getLogger();
061
062    public RollingFileAppenderBuilder() {
063    }
064
065    public RollingFileAppenderBuilder(final String prefix, final Properties properties) {
066        super(prefix, properties);
067    }
068
069    @Override
070    public Appender parseAppender(final Element appenderElement, final XmlConfiguration config) {
071        final String name = getNameAttribute(appenderElement);
072        final AtomicReference<Layout> layout = new AtomicReference<>();
073        final AtomicReference<Filter> filter = new AtomicReference<>();
074        final AtomicReference<String> fileName = new AtomicReference<>();
075        final AtomicBoolean immediateFlush = new AtomicBoolean(true);
076        final AtomicBoolean append = new AtomicBoolean(true);
077        final AtomicBoolean bufferedIo = new AtomicBoolean();
078        final AtomicInteger bufferSize = new AtomicInteger(8192);
079        final AtomicReference<String> maxSize = new AtomicReference<>(DEFAULT_MAX_SIZE);
080        final AtomicReference<String> maxBackups = new AtomicReference<>(DEFAULT_MAX_BACKUPS);
081        final AtomicReference<String> level = new AtomicReference<>();
082        forEachElement(appenderElement.getChildNodes(), currentElement -> {
083            switch (currentElement.getTagName()) {
084                case LAYOUT_TAG:
085                    layout.set(config.parseLayout(currentElement));
086                    break;
087                case FILTER_TAG:
088                    config.addFilter(filter, currentElement);
089                    break;
090                case PARAM_TAG:
091                    switch (getNameAttributeKey(currentElement)) {
092                        case FILE_PARAM:
093                            set(FILE_PARAM, currentElement, fileName);
094                            break;
095                        case APPEND_PARAM:
096                            set(APPEND_PARAM, currentElement, append);
097                            break;
098                        case BUFFERED_IO_PARAM:
099                            set(BUFFERED_IO_PARAM, currentElement, bufferedIo);
100                            break;
101                        case BUFFER_SIZE_PARAM:
102                            set(BUFFER_SIZE_PARAM, currentElement, bufferSize);
103                            break;
104                        case MAX_BACKUP_INDEX:
105                            set(MAX_BACKUP_INDEX, currentElement, maxBackups);
106                            break;
107                        case MAX_SIZE_PARAM:
108                            set(MAX_SIZE_PARAM, currentElement, maxSize);
109                            break;
110                        case THRESHOLD_PARAM:
111                            set(THRESHOLD_PARAM, currentElement, level);
112                            break;
113                        case IMMEDIATE_FLUSH_PARAM:
114                            set(IMMEDIATE_FLUSH_PARAM, currentElement, immediateFlush);
115                            break;
116                    }
117                    break;
118            }
119        });
120        return createAppender(name, config, layout.get(), filter.get(), append.get(), bufferedIo.get(),
121                bufferSize.get(), immediateFlush.get(), fileName.get(), level.get(), maxSize.get(), maxBackups.get());
122    }
123
124    @Override
125    public Appender parseAppender(final String name, final String appenderPrefix, final String layoutPrefix,
126            final String filterPrefix, final Properties props, final PropertiesConfiguration configuration) {
127        final Layout layout = configuration.parseLayout(layoutPrefix, name, props);
128        final Filter filter = configuration.parseAppenderFilters(props, filterPrefix, name);
129        final String fileName = getProperty(FILE_PARAM);
130        final String level = getProperty(THRESHOLD_PARAM);
131        final boolean append = getBooleanProperty(APPEND_PARAM, true);
132        final boolean immediateFlush = getBooleanProperty(IMMEDIATE_FLUSH_PARAM, true);
133        final boolean bufferedIo = getBooleanProperty(BUFFERED_IO_PARAM, false);
134        final int bufferSize = getIntegerProperty(BUFFER_SIZE_PARAM, 8192);
135        final String maxSize = getProperty(MAX_SIZE_PARAM, DEFAULT_MAX_SIZE);
136        final String maxBackups = getProperty(MAX_BACKUP_INDEX, DEFAULT_MAX_BACKUPS);
137        return createAppender(name, configuration, layout, filter, append, bufferedIo, bufferSize, immediateFlush,
138                fileName, level, maxSize, maxBackups);
139    }
140
141    private Appender createAppender(final String name, final Log4j1Configuration config, final Layout layout,
142            final Filter filter, final boolean append, final boolean bufferedIo, final int bufferSize, boolean immediateFlush,
143            final String fileName, final String level, final String maxSize, final String maxBackups) {
144        org.apache.logging.log4j.core.Layout<?> fileLayout = LayoutAdapter.adapt(layout);
145        if (!bufferedIo) {
146            immediateFlush = false;
147        }
148        final org.apache.logging.log4j.core.Filter fileFilter = buildFilters(level, filter);
149        if (fileName == null) {
150            LOGGER.error("Unable to create RollingFileAppender, no file name provided");
151            return null;
152        }
153        final String filePattern = fileName + ".%i";
154        final SizeBasedTriggeringPolicy sizePolicy = SizeBasedTriggeringPolicy.createPolicy(maxSize);
155        final CompositeTriggeringPolicy policy = CompositeTriggeringPolicy.createPolicy(sizePolicy);
156        final RolloverStrategy strategy = DefaultRolloverStrategy.newBuilder()
157                .withConfig(config)
158                .withMax(maxBackups)
159                .build();
160        return AppenderWrapper.adapt(RollingFileAppender.newBuilder()
161                .setName(name)
162                .setConfiguration(config)
163                .setLayout(fileLayout)
164                .setFilter(fileFilter)
165                .withAppend(append)
166                .setBufferedIo(bufferedIo)
167                .setBufferSize(bufferSize)
168                .setImmediateFlush(immediateFlush)
169                .withFileName(fileName)
170                .withFilePattern(filePattern)
171                .withPolicy(policy)
172                .withStrategy(strategy)
173                .build());
174    }
175}

017package org.apache.log4j.builders;
018
019import static org.apache.log4j.xml.XmlConfiguration.NAME_ATTR;
020import static org.apache.log4j.xml.XmlConfiguration.VALUE_ATTR;
021
022import java.util.HashMap;
023import java.util.Map;
024import java.util.Properties;
025import java.util.concurrent.atomic.AtomicBoolean;
026import java.util.concurrent.atomic.AtomicInteger;
027import java.util.concurrent.atomic.AtomicReference;
028
029import org.apache.log4j.bridge.FilterAdapter;
030import org.apache.log4j.bridge.FilterWrapper;
031import org.apache.log4j.helpers.OptionConverter;
032import org.apache.log4j.spi.Filter;
033import org.apache.logging.log4j.Level;
034import org.apache.logging.log4j.Logger;
035import org.apache.logging.log4j.core.filter.ThresholdFilter;
036import org.apache.logging.log4j.status.StatusLogger;
037import org.apache.logging.log4j.util.Strings;
038import org.w3c.dom.Element;
039
040/**
041 * Base class for Log4j 1 component builders.
042 *
043 * @param <T> The type to build.
044 */
045public abstract class AbstractBuilder<T> implements Builder<T> {
046
047    private static Logger LOGGER = StatusLogger.getLogger();
048    protected static final String FILE_PARAM = "File";
049    protected static final String APPEND_PARAM = "Append";
050    protected static final String BUFFERED_IO_PARAM = "BufferedIO";
051    protected static final String BUFFER_SIZE_PARAM = "BufferSize";
052    protected static final String IMMEDIATE_FLUSH_PARAM = "ImmediateFlush";
053    protected static final String MAX_SIZE_PARAM = "MaxFileSize";
054    protected static final String MAX_BACKUP_INDEX = "MaxBackupIndex";
055    protected static final String RELATIVE = "RELATIVE";
056    protected static final String NULL = "NULL";
057
058    private final String prefix;
059    private final Properties properties;
060
061    public AbstractBuilder() {
062        this(null, new Properties());
063    }
064
065    public AbstractBuilder(final String prefix, final Properties props) {
066        this.prefix = prefix != null ? prefix + "." : null;
067        this.properties = (Properties) props.clone();
068        final Map<String, String> map = new HashMap<>();
069        System.getProperties().forEach((k, v) -> map.put(k.toString(), v.toString()));
070        props.forEach((k, v) -> map.put(k.toString(), v.toString()));
071        // normalize keys to lower case for case-insensitive access.
072        props.forEach((k, v) -> map.put(toBeanKey(k.toString()), v.toString()));
073        props.entrySet().forEach(e -> this.properties.put(toBeanKey(e.getKey().toString()), e.getValue()));
074    }
075
076    protected static org.apache.logging.log4j.core.Filter buildFilters(final String level, final Filter filter) {
077        Filter head = null;
078        if (level != null) {
079            final org.apache.logging.log4j.core.Filter thresholdFilter = ThresholdFilter.createFilter(OptionConverter.convertLevel(level, Level.TRACE),
080                org.apache.logging.log4j.core.Filter.Result.NEUTRAL, org.apache.logging.log4j.core.Filter.Result.DENY);
081            head = new FilterWrapper(thresholdFilter);
082        }
083        if (filter != null) {
084            head = FilterAdapter.addFilter(head, filter);
085        }
086        return FilterAdapter.convertFilter(head);
087    }
088
089    private String capitalize(final String value) {
090        if (Strings.isEmpty(value) || Character.isUpperCase(value.charAt(0))) {
091            return value;
092        }
093        final char[] chars = value.toCharArray();
094        chars[0] = Character.toUpperCase(chars[0]);
095        return new String(chars);
096    }
097
098    public boolean getBooleanProperty(final String key, final boolean defaultValue) {
099        return Boolean.parseBoolean(getProperty(key, Boolean.toString(defaultValue)));
100    }
101
102    public boolean getBooleanProperty(final String key) {
103        return getBooleanProperty(key, false);
104    }
105
106    protected boolean getBooleanValueAttribute(final Element element) {
107        return Boolean.parseBoolean(getValueAttribute(element));
108    }
109
110    public int getIntegerProperty(final String key, final int defaultValue) {
111        String value = null;
112        try {
113            value = getProperty(key);
114            if (value != null) {
115                return Integer.parseInt(value);
116            }
117        } catch (final Exception ex) {
118            LOGGER.warn("Error converting value {} of {} to an integer: {}", value, key, ex.getMessage());
119        }
120        return defaultValue;
121    }
122
123    protected String getNameAttribute(final Element element) {
124        return element.getAttribute(NAME_ATTR);
125    }
126
127    protected String getNameAttributeKey(final Element element) {
128        return toBeanKey(element.getAttribute(NAME_ATTR));
129    }
130
131    public Properties getProperties() {
132        return properties;
133    }
134
135    public String getProperty(final String key) {
136        return getProperty(key, null);
137    }
138
139    public String getProperty(final String key, final String defaultValue) {
140        String value = properties.getProperty(prefix + toJavaKey(key));
141        value = value != null ? value : properties.getProperty(prefix + toBeanKey(key), defaultValue);
142        value = value != null ? substVars(value) : defaultValue;
143        return value != null ? value.trim() : defaultValue;
144    }
145
146    protected String getValueAttribute(final Element element) {
147        return getValueAttribute(element, null);
148    }
149
150    protected String getValueAttribute(final Element element, final String defaultValue) {
151        final String attribute = element.getAttribute(VALUE_ATTR);
152        return substVars(attribute != null ? attribute.trim() : defaultValue);
153    }
154
155    protected String substVars(final String value) {
156        return OptionConverter.substVars(value, properties);
157    }
158
159    String toBeanKey(final String value) {
160        return capitalize(value);
161    }
162
163    String toJavaKey(final String value) {
164        return uncapitalize(value);
165    }
166
167    private String uncapitalize(final String value) {
168        if (Strings.isEmpty(value) || Character.isLowerCase(value.charAt(0))) {
169            return value;
170        }
171        final char[] chars = value.toCharArray();
172        chars[0] = Character.toLowerCase(chars[0]);
173        return new String(chars);
174    }
175
176    protected void set(final String name, final Element element, AtomicBoolean ref) {
177        final String value = getValueAttribute(element);
178        if (value == null) {
179            LOGGER.warn("No value for {} parameter, using default {}", name, ref);
180        } else {
181            ref.set(Boolean.parseBoolean(value));
182        }
183    }
184
185    protected void set(final String name, final Element element, AtomicInteger ref) {
186        final String value = getValueAttribute(element);
187        if (value == null) {
188            LOGGER.warn("No value for {} parameter, using default {}", name, ref);
189        } else {
190            try {
191                ref.set(Integer.parseInt(value));
192            } catch (NumberFormatException e) {
193                LOGGER.warn("{} parsing {} parameter, using default {}: {}", e.getClass().getName(), name, ref, e.getMessage(), e);
194            }
195        }
196    }
197
198    protected void set(final String name, final Element element, AtomicReference<String> ref) {
199        final String value = getValueAttribute(element);
200        if (value == null) {
201            LOGGER.warn("No value for {} parameter, using default {}", name, ref);
202        } else {
203            ref.set(value);
204        }
205    }
206}

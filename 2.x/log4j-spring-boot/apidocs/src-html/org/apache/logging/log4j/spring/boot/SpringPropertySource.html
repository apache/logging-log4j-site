017package org.apache.logging.log4j.spring.boot;
018
019import org.apache.logging.log4j.util.PropertySource;
020import org.springframework.core.env.Environment;
021
022/**
023 * Returns properties from Spring.
024 */
025public class SpringPropertySource extends SpringEnvironmentHolder implements PropertySource {
026
027    /**
028     * System properties take precendence followed by properties in Log4j properties files. Spring properties
029     * follow.
030     * @return This PropertySource's priority.
031     */
032    @Override
033    public int getPriority() {
034        return -50;
035    }
036
037    @Override
038    public String getProperty(String key) {
039        Environment environment = getEnvironment();
040        if (environment != null) {
041            return environment.getProperty(key);
042        }
043        return null;
044    }
045
046    @Override
047    public boolean containsProperty(String key) {
048        Environment environment = getEnvironment();
049        if (environment != null) {
050            return environment.containsProperty(key);
051        }
052        return false;
053    }
054}

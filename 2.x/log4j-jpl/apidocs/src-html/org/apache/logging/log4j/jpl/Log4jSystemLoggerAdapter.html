018package org.apache.logging.log4j.jpl;
019
020import java.lang.System.Logger;
021import java.lang.System.LoggerFinder;
022
023import org.apache.logging.log4j.LogManager;
024import org.apache.logging.log4j.spi.AbstractLoggerAdapter;
025import org.apache.logging.log4j.spi.LoggerContext;
026import org.apache.logging.log4j.util.StackLocatorUtil;
027
028/**
029 * {@link Logger} registry implementation using just log4j-api.
030 *
031 * @since 2.14
032 */
033public class Log4jSystemLoggerAdapter extends AbstractLoggerAdapter<Logger> {
034
035    @Override
036    protected Logger newLogger(String name, LoggerContext context) {
037        return new Log4jSystemLogger(context.getLogger(name));
038    }
039
040    @Override
041    protected LoggerContext getContext() {
042        return getContext(LogManager.getFactory().isClassLoaderDependent()
043                ? StackLocatorUtil.getCallerClass(LoggerFinder.class)
044                : null);
045    }
046}

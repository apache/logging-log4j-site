17  package org.apache.logging.log4j.core.appender.db.jdbc;

19  import java.sql.DriverManager;
20  import java.sql.SQLException;
21  import java.util.Arrays;
22  import java.util.concurrent.TimeUnit;
23  
24  import org.apache.commons.dbcp2.ConnectionFactory;
25  import org.apache.commons.dbcp2.DriverManagerConnectionFactory;
26  import org.apache.commons.dbcp2.PoolableConnection;
27  import org.apache.commons.dbcp2.PoolableConnectionFactory;
28  import org.apache.commons.dbcp2.PoolingDriver;
29  import org.apache.commons.pool2.ObjectPool;
30  import org.apache.commons.pool2.impl.GenericObjectPool;
31  import org.apache.logging.log4j.core.Core;

34  import org.apache.logging.log4j.core.config.plugins.PluginBuilderFactory;

36  /**
37   * A {@link ConnectionSource} that uses a JDBC connection string, a user name, and a password to call
38   * {@link DriverManager#getConnection(String, String, String)}. The connections are served from an
39   * <a href="http://commons.apache.org/proper/commons-dbcp/">Apache Commons DBCP</a> pooling driver.
40   */
41  @Plugin(name = "PoolingDriver", category = Core.CATEGORY_NAME, elementType = "connectionSource", printObject = true)
42  public final class PoolingDriverConnectionSource extends AbstractDriverManagerConnectionSource {
43  
44      /**
45       * Builds PoolingDriverConnectionSource instances.
46       *
47       * @param <B>
48       *            This builder type or a subclass.
49       */
50      public static class Builder<B extends Builder<B>> extends AbstractDriverManagerConnectionSource.Builder<B>
51      implements org.apache.logging.log4j.core.util.Builder<PoolingDriverConnectionSource> {
52  
53          public static final String DEFAULT_POOL_NAME = "example";
54          private String poolName = DEFAULT_POOL_NAME;
55  
56          @Override
57  		public PoolingDriverConnectionSource build() {
58  			try {
59  				return new PoolingDriverConnectionSource(getDriverClassName(), getConnectionString(), getUserName(),
60  						getPassword(), getProperties(), poolName);
61  			} catch (final SQLException e) {
62  				getLogger().error("Exception constructing {} to '{}' with {}", PoolingDriverConnectionSource.class,
63  						getConnectionString(), this, e);
64  				return null;
65  			}
66  		}
67  
68          public B setPoolName(final String poolName) {
69              this.poolName = poolName;
70              return asBuilder();
71          }
72  
73  		@Override
74  		public String toString() {
75  			return "Builder [poolName=" + poolName + ", connectionString=" + connectionString + ", driverClassName="
76  					+ driverClassName + ", properties=" + Arrays.toString(properties) + ", userName="
77  					+ Arrays.toString(userName) + "]";
78  		}
79      }
80  
81      public static final String URL_PREFIX = "jdbc:apache:commons:dbcp:";
82  
83      // This method is not named newBuilder() to make the compiler happy.
84      @PluginBuilderFactory
85      public static <B extends Builder<B>> B newPoolingDriverConnectionSourceBuilder() {
86          return new Builder<B>().asBuilder();
87      }
88  
89      private final String poolingDriverClassName = "org.apache.commons.dbcp2.PoolingDriver";
90  
91      private final String poolName;
92  
93      public PoolingDriverConnectionSource(final String driverClassName, final String connectionString,
94              final char[] userName, final char[] password, final Property[] properties, final String poolName)
95              throws SQLException {
96          super(driverClassName, connectionString, URL_PREFIX + poolName, userName, password, properties);
97          this.poolName = poolName;
98          setupDriver(connectionString);
99      }
100 
101     @Override
102     public String getActualConnectionString() {
103         // TODO Auto-generated method stub
104         return super.getActualConnectionString();
105     }
106 
107     private PoolingDriver getPoolingDriver() throws SQLException {
108         final PoolingDriver driver = (PoolingDriver) DriverManager.getDriver(URL_PREFIX);
109         if (driver == null) {
110             getLogger().error("No JDBC driver for '{}'", URL_PREFIX);
111         }
112         return driver;
113     }
114 
115     private void setupDriver(final String connectionString) throws SQLException {
116         //
117         // First, we'll create a ConnectionFactory that the
118         // pool will use to create Connections.
119         // We'll use the DriverManagerConnectionFactory,
120         // using the connect string passed in the command line
121         // arguments.
122         //
123     	final Property[] properties = getProperties();
124     	final char[] userName = getUserName();
125     	final char[] password = getPassword();
126     	final ConnectionFactory connectionFactory;
127         if (properties != null && properties.length > 0) {
128             if (userName != null || password != null) {
129                 throw new SQLException("Either set the userName and password, or set the Properties, but not both.");
130             }
131             connectionFactory = new DriverManagerConnectionFactory(connectionString, toProperties(properties));
132         } else {
133         	connectionFactory = new DriverManagerConnectionFactory(connectionString, toString(userName), toString(password));
134         }
135 
136         //
137         // Next, we'll create the PoolableConnectionFactory, which wraps
138         // the "real" Connections created by the ConnectionFactory with
139         // the classes that implement the pooling functionality.
140         //
141         final PoolableConnectionFactory poolableConnectionFactory = new PoolableConnectionFactory(connectionFactory,
142                 null);
143 
144         //
145         // Now we'll need a ObjectPool that serves as the
146         // actual pool of connections.
147         //
148         // We'll use a GenericObjectPool instance, although
149         // any ObjectPool implementation will suffice.
150         //
151         @SuppressWarnings("resource")
152         // This GenericObjectPool will be closed on shutown
153         final ObjectPool<PoolableConnection> connectionPool = new GenericObjectPool<>(poolableConnectionFactory);
154 
155         // Set the factory's pool property to the owning pool
156         poolableConnectionFactory.setPool(connectionPool);
157 
158         loadDriver(poolingDriverClassName);
159         final PoolingDriver driver = getPoolingDriver();
160         if (driver != null) {
161             getLogger().debug("Registering DBCP pool '{}'", poolName);
162             driver.registerPool(poolName, connectionPool);
163         }
164         //
165         // Now we can just use the connect string "jdbc:apache:commons:dbcp:example"
166         // to access our pool of Connections.
167         //
168     }
169 
170     @Override
171     public boolean stop(long timeout, TimeUnit timeUnit) {
172         try {
173             final PoolingDriver driver = getPoolingDriver();
174             if (driver != null) {
175                 getLogger().debug("Closing DBCP pool '{}'", poolName);
176                 driver.closePool(poolName);
177             }
178             return true;
179         } catch (Exception e) {
180             getLogger().error("Exception stopping connection source for '{}' → '{}'", getConnectionString(),
181                     getActualConnectionString(), e);
182             return false;
183         }
184     }
185 }

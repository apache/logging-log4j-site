424        public void log(Marker marker, String fqcn, int i, String s1, Object[] objects, Throwable throwable) {
425            if (!logger.isEnabled(getLevel(i), (org.apache.logging.log4j.Marker) marker, s1)) {
426                return;
427            }
428            Message msg;
429            if (eventLogger && marker != null && marker.contains(EVENT_MARKER)) {
430                try {
431                    EventData data = (objects != null && objects[0] instanceof EventData) ? (EventData) objects[0] :
432                        new EventData(s1);
433                    msg = new StructuredDataMessage(data.getEventId(), data.getMessage(), data.getEventType());
434                    for (Map.Entry entry : data.getEventMap().entrySet()) {
435                        String key = entry.getKey().toString();
436                        if (EventData.EVENT_TYPE.equals(key) || EventData.EVENT_ID.equals(key) ||
437                            EventData.EVENT_MESSAGE.equals(key)) {
438                            continue;
439                        }
440                        ((StructuredDataMessage) msg).put(entry.getKey().toString(), entry.getValue().toString());
441                    }
442                } catch (Exception ex) {
443                    msg = new ParameterizedMessage(s1, objects, throwable);
444                }
445    
446           } else if (objects == null) {
447                msg = new SimpleMessage(s1);
448            } else {
449                msg = new ParameterizedMessage(s1, objects, throwable);
450                if (throwable != null) {
451                    throwable = ((ParameterizedMessage) msg).getThrowable();
452                }
453            }
454            logger.log((org.apache.logging.log4j.Marker) marker, fqcn, getLevel(i), msg, throwable);
455        }

457        public String getName() {
458            return name;
459        }
460    
461        private Level getLevel(int i) {
462    
463            switch (i) {
464                case TRACE_INT :
465                    return Level.TRACE;
466                case DEBUG_INT :
467                    return Level.DEBUG;
468                case INFO_INT :
469                    return Level.INFO;
470                case WARN_INT :
471                    return Level.WARN;
472                case ERROR_INT :
473                    return Level.ERROR;
474            }
475            return Level.ERROR;
476        }

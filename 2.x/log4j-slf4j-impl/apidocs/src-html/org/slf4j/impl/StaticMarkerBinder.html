017package org.slf4j.impl;
018
019import org.apache.logging.slf4j.Log4jMarkerFactory;
020import org.slf4j.IMarkerFactory;
021import org.slf4j.spi.MarkerFactoryBinder;
022
023/**
024 * SLF4J MarkerFactoryBinder implementation using Log4j. This class is part of the required classes used to specify an
025 * SLF4J logging provider implementation.
026 */
027public class StaticMarkerBinder implements MarkerFactoryBinder {
028
029    /**
030     * The unique instance of this class.
031     */
032    public static final StaticMarkerBinder SINGLETON = new StaticMarkerBinder();
033
034    private final IMarkerFactory markerFactory = new Log4jMarkerFactory();
035
036    /**
037     * Returns the {@link #SINGLETON} {@link StaticMarkerBinder}.
038     * Added to slf4j-api 1.7.14 via https://github.com/qos-ch/slf4j/commit/ea3cca72cd5a9329a06b788317a17e806ee8acd0
039     * @return the singleton instance
040     */
041    public static StaticMarkerBinder getSingleton() {
042        return SINGLETON;
043    }
044
045    @Override
046    public IMarkerFactory getMarkerFactory() {
047        return markerFactory;
048    }
049
050    @Override
051    public String getMarkerFactoryClassStr() {
052        return Log4jMarkerFactory.class.getName();
053    }
054}

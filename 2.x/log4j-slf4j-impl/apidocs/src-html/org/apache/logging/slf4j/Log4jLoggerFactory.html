017    package org.apache.logging.slf4j;
018    
019    import org.apache.logging.log4j.LogManager;
020    import org.apache.logging.log4j.spi.AbstractLoggerAdapter;
021    import org.apache.logging.log4j.spi.LoggerContext;
022    import org.apache.logging.log4j.util.ReflectionUtil;
023    import org.slf4j.ILoggerFactory;
024    import org.slf4j.Logger;
025    
026    /**
027     * Log4j implementation of SLF4J ILoggerFactory interface.
028     */
029    public class Log4jLoggerFactory extends AbstractLoggerAdapter<Logger> implements ILoggerFactory {
030    
031        private static final String FQCN = Log4jLoggerFactory.class.getName();
032        private static final String PACKAGE = "org.slf4j";
033    
034        @Override
035        protected Logger newLogger(final String name, final LoggerContext context) {
036            final String key = Logger.ROOT_LOGGER_NAME.equals(name) ? LogManager.ROOT_LOGGER_NAME : name;
037            return new Log4jLogger(context.getLogger(key), name);
038        }
039    
040        @Override
041        protected LoggerContext getContext() {
042            final Class<?> anchor = ReflectionUtil.getCallerClass(FQCN, PACKAGE);
043            return anchor == null ? LogManager.getContext() : getContext(ReflectionUtil.getCallerClass(anchor));
044        }
045    
046    }

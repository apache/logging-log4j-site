21  /**
22   * Specifies an interface for initializing and deinitializing Log4j in a Java EE web application. The default and only
23   * implementation is {@link Log4jWebInitializerImpl}. The initializer is based on an interface to improve testability.
24   * The methods here are contained in a package-private sub-interface because general application code should not have
25   * access to them.
26   */
27  interface Log4jWebLifeCycle extends Log4jWebSupport, LifeCycle {
28      
29      /**
30       * Starts up Log4j in the web application. Calls {@link #setLoggerContext()} after initialization is complete.
31       *
32       * @throws IllegalStateException if a JNDI config location is specified but no name is specified.
33       */
34      @Override
35      void start();
36  
37      /**
38       * Shuts down Log4j in the web application. Calls {@link #clearLoggerContext()} immediately before deinitialization
39       * begins.
40       */
41      @Override
42      void stop();
43  }

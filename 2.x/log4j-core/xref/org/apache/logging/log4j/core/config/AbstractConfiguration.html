210     @Override
211     public void initialize() {
212         LOGGER.debug("Initializing configuration {}", this);
213         subst.setConfiguration(this);
214         scriptManager = new ScriptManager(this, watchManager);
215         pluginManager.collectPlugins(pluginPackages);
216         final PluginManager levelPlugins = new PluginManager(Level.CATEGORY);
217         levelPlugins.collectPlugins(pluginPackages);
218         final Map<String, PluginType<?>> plugins = levelPlugins.getPlugins();
219         if (plugins != null) {
220             for (final PluginType<?> type : plugins.values()) {
221                 try {
222                     // Cause the class to be initialized if it isn't already.
223                     Loader.initializeClass(type.getPluginClass().getName(), type.getPluginClass().getClassLoader());
224                 } catch (final Exception e) {
225                     LOGGER.error("Unable to initialize {} due to {}", type.getPluginClass().getName(), e.getClass()
226                             .getSimpleName(), e);
227                 }
228             }
229         }
230         setup();
231         setupAdvertisement();
232         doConfigure();
233         setState(State.INITIALIZED);
234         LOGGER.debug("Configuration {} initialized", this);
235     }

237     /**
238      * Start the configuration.
239      */
240     @Override
241     public void start() {
242         // Preserve the prior behavior of initializing during start if not initialized.
243         if (getState().equals(State.INITIALIZING)) {
244             initialize();
245         }
246         LOGGER.debug("Starting configuration {}", this);
247         this.setStarting();
248         if (watchManager.getIntervalSeconds() > 0) {
249             watchManager.start();
250         }
251         if (hasAsyncLoggers()) {
252             asyncLoggerConfigDisruptor.start();
253         }
254         final Set<LoggerConfig> alreadyStarted = new HashSet<>();
255         for (final LoggerConfig logger : loggerConfigs.values()) {
256             logger.start();
257             alreadyStarted.add(logger);
258         }
259         for (final Appender appender : appenders.values()) {
260             appender.start();
261         }
262         if (!alreadyStarted.contains(root)) { // LOG4J2-392
263             root.start(); // LOG4J2-336
264         }
265         super.start();
266         LOGGER.debug("Started configuration {} OK.", this);
267     }

269     private boolean hasAsyncLoggers() {
270         if (root instanceof AsyncLoggerConfig) {
271             return true;
272         }
273         for (final LoggerConfig logger : loggerConfigs.values()) {
274             if (logger instanceof AsyncLoggerConfig) {
275                 return true;
276             }
277         }
278         return false;
279     }

281     /**
282      * Tear down the configuration.
283      */
284     @Override
285     public boolean stop(final long timeout, final TimeUnit timeUnit) {
286         this.setStopping();
287         super.stop(timeout, timeUnit, false);
288         LOGGER.trace("Stopping {}...", this);
289 
290         // Stop the components that are closest to the application first:
291         // 1. Notify all LoggerConfigs' ReliabilityStrategy that the configuration will be stopped.
292         // 2. Stop the LoggerConfig objects (this may stop nested Filters)
293         // 3. Stop the AsyncLoggerConfigDelegate. This shuts down the AsyncLoggerConfig Disruptor
294         //    and waits until all events in the RingBuffer have been processed.
295         // 4. Stop all AsyncAppenders. This shuts down the associated thread and
296         //    waits until all events in the queue have been processed. (With optional timeout.)
297         // 5. Notify all LoggerConfigs' ReliabilityStrategy that appenders will be stopped.
298         //    This guarantees that any event received by a LoggerConfig before reconfiguration
299         //    are passed on to the Appenders before the Appenders are stopped.
300         // 6. Stop the remaining running Appenders. (It should now be safe to do so.)
301         // 7. Notify all LoggerConfigs that their Appenders can be cleaned up.
302 
303         for (final LoggerConfig loggerConfig : loggerConfigs.values()) {
304             loggerConfig.getReliabilityStrategy().beforeStopConfiguration(this);
305         }
306         root.getReliabilityStrategy().beforeStopConfiguration(this);
307 
308         final String cls = getClass().getSimpleName();
309         LOGGER.trace("{} notified {} ReliabilityStrategies that config will be stopped.", cls, loggerConfigs.size()
310                 + 1);
311 
312         if (!loggerConfigs.isEmpty()) {
313             LOGGER.trace("{} stopping {} LoggerConfigs.", cls, loggerConfigs.size());
314             for (final LoggerConfig logger : loggerConfigs.values()) {
315                 logger.stop(timeout, timeUnit);
316             }
317         }
318         LOGGER.trace("{} stopping root LoggerConfig.", cls);
319         if (!root.isStopped()) {
320             root.stop(timeout, timeUnit);
321         }
322 
323         if (hasAsyncLoggers()) {
324             LOGGER.trace("{} stopping AsyncLoggerConfigDisruptor.", cls);
325             asyncLoggerConfigDisruptor.stop(timeout, timeUnit);
326         }
327 
328         // Stop the appenders in reverse order in case they still have activity.
329         final Appender[] array = appenders.values().toArray(new Appender[appenders.size()]);
330         final List<Appender> async = getAsyncAppenders(array);
331         if (!async.isEmpty()) {
332             // LOG4J2-511, LOG4J2-392 stop AsyncAppenders first
333             LOGGER.trace("{} stopping {} AsyncAppenders.", cls, async.size());
334             for (final Appender appender : async) {
335                 if (appender instanceof LifeCycle2) {
336                     ((LifeCycle2) appender).stop(timeout, timeUnit);
337                 } else {
338                     appender.stop();
339                 }
340             }
341         }
342 
343         LOGGER.trace("{} notifying ReliabilityStrategies that appenders will be stopped.", cls);
344         for (final LoggerConfig loggerConfig : loggerConfigs.values()) {
345             loggerConfig.getReliabilityStrategy().beforeStopAppenders();
346         }
347         root.getReliabilityStrategy().beforeStopAppenders();
348 
349         LOGGER.trace("{} stopping remaining Appenders.", cls);
350         int appenderCount = 0;
351         for (int i = array.length - 1; i >= 0; --i) {
352             if (array[i].isStarted()) { // then stop remaining Appenders
353                 if (array[i] instanceof LifeCycle2) {
354                     ((LifeCycle2) array[i]).stop(timeout, timeUnit);
355                 } else {
356                     array[i].stop();
357                 }
358                 appenderCount++;
359             }
360         }
361         LOGGER.trace("{} stopped {} remaining Appenders.", cls, appenderCount);
362 
363         LOGGER.trace("{} cleaning Appenders from {} LoggerConfigs.", cls, loggerConfigs.size() + 1);
364         for (final LoggerConfig loggerConfig : loggerConfigs.values()) {
365 
366             // LOG4J2-520, LOG4J2-392:
367             // Important: do not clear appenders until after all AsyncLoggerConfigs
368             // have been stopped! Stopping the last AsyncLoggerConfig will
369             // shut down the disruptor and wait for all enqueued events to be processed.
370             // Only *after this* the appenders can be cleared or events will be lost.
371             loggerConfig.clearAppenders();
372         }
373         root.clearAppenders();
374 
375         if (watchManager.isStarted()) {
376             watchManager.stop(timeout, timeUnit);
377         }
378         configurationScheduler.stop(timeout, timeUnit);
379 
380         if (advertiser != null && advertisement != null) {
381             advertiser.unadvertise(advertisement);
382         }
383         setStopped();
384         LOGGER.debug("Stopped {} OK", this);
385         return true;
386     }

388     private List<Appender> getAsyncAppenders(final Appender[] all) {
389         final List<Appender> result = new ArrayList<>();
390         for (int i = all.length - 1; i >= 0; --i) {
391             if (all[i] instanceof AsyncAppender) {
392                 result.add(all[i]);
393             }
394         }
395         return result;
396     }

422     protected void createAdvertiser(final String advertiserString, final ConfigurationSource configSource,
423             final byte[] buffer, final String contentType) {
424         if (advertiserString != null) {
425             final Node node = new Node(null, advertiserString, null);
426             final Map<String, String> attributes = node.getAttributes();
427             attributes.put("content", new String(buffer));
428             attributes.put("contentType", contentType);
429             attributes.put("name", "configuration");
430             if (configSource.getLocation() != null) {
431                 attributes.put("location", configSource.getLocation());
432             }
433             advertiserNode = node;
434         }
435     }

437     private void setupAdvertisement() {
438         if (advertiserNode != null) {
439             final String nodeName = advertiserNode.getName();
440             final PluginType<?> type = pluginManager.getPluginType(nodeName);
441             if (type != null) {
442                 final Class<? extends Advertiser> clazz = type.getPluginClass().asSubclass(Advertiser.class);
443                 try {
444                     advertiser = clazz.newInstance();
445                     advertisement = advertiser.advertise(advertiserNode.getAttributes());
446                 } catch (final InstantiationException e) {
447                     LOGGER.error("InstantiationException attempting to instantiate advertiser: {}", nodeName, e);
448                 } catch (final IllegalAccessException e) {
449                     LOGGER.error("IllegalAccessException attempting to instantiate advertiser: {}", nodeName, e);
450                 }
451             }
452         }
453     }

466     protected void preConfigure(final Node node) {
467         try {
468             for (final Node child : node.getChildren()) {
469                 if (child.getType() == null) {
470                     LOGGER.error("Unable to locate plugin type for " + child.getName());
471                     continue;
472                 }
473                 final Class<?> clazz = child.getType().getPluginClass();
474                 if (clazz.isAnnotationPresent(Scheduled.class)) {
475                     configurationScheduler.incrementScheduledItems();
476                 }
477                 preConfigure(child);
478             }
479         } catch (final Exception ex) {
480             LOGGER.error("Error capturing node data for node " + node.getName(), ex);
481         }
482     }
483 
484     protected void doConfigure() {
485         preConfigure(rootNode);
486         configurationScheduler.start();
487         if (rootNode.hasChildren() && rootNode.getChildren().get(0).getName().equalsIgnoreCase("Properties")) {
488             final Node first = rootNode.getChildren().get(0);
489             createConfiguration(first, null);
490             if (first.getObject() != null) {
491                 subst.setVariableResolver((StrLookup) first.getObject());
492             }
493         } else {
494             final Map<String, String> map = this.getComponent(CONTEXT_PROPERTIES);
495             final StrLookup lookup = map == null ? null : new MapLookup(map);
496             subst.setVariableResolver(new Interpolator(lookup, pluginPackages));
497         }
498 
499         boolean setLoggers = false;
500         boolean setRoot = false;
501         for (final Node child : rootNode.getChildren()) {
502             if (child.getName().equalsIgnoreCase("Properties")) {
503                 if (tempLookup == subst.getVariableResolver()) {
504                     LOGGER.error("Properties declaration must be the first element in the configuration");
505                 }
506                 continue;
507             }
508             createConfiguration(child, null);
509             if (child.getObject() == null) {
510                 continue;
511             }
512             if (child.getName().equalsIgnoreCase("Scripts")) {
513                 for (final AbstractScript script : child.getObject(AbstractScript[].class)) {
514                     if (script instanceof ScriptRef) {
515                         LOGGER.error("Script reference to {} not added. Scripts definition cannot contain script references",
516                                 script.getName());
517                     } else {
518                         scriptManager.addScript(script);
519                     }
520                 }
521             } else if (child.getName().equalsIgnoreCase("Appenders")) {
522                 appenders = child.getObject();
523             } else if (child.isInstanceOf(Filter.class)) {
524                 addFilter(child.getObject(Filter.class));
525             } else if (child.getName().equalsIgnoreCase("Loggers")) {
526                 final Loggers l = child.getObject();
527                 loggerConfigs = l.getMap();
528                 setLoggers = true;
529                 if (l.getRoot() != null) {
530                     root = l.getRoot();
531                     setRoot = true;
532                 }
533             } else if (child.getName().equalsIgnoreCase("CustomLevels")) {
534                 customLevels = child.getObject(CustomLevels.class).getCustomLevels();
535             } else if (child.isInstanceOf(CustomLevelConfig.class)) {
536                 final List<CustomLevelConfig> copy = new ArrayList<>(customLevels);
537                 copy.add(child.getObject(CustomLevelConfig.class));
538                 customLevels = copy;
539             } else {
540                 final List<String> expected = Arrays.asList("\"Appenders\"", "\"Loggers\"", "\"Properties\"",
541                         "\"Scripts\"", "\"CustomLevels\"");
542                 LOGGER.error("Unknown object \"{}\" of type {} is ignored: try nesting it inside one of: {}.",
543                         child.getName(), child.getObject().getClass().getName(), expected);
544             }
545         }
546 
547         if (!setLoggers) {
548             LOGGER.warn("No Loggers were configured, using default. Is the Loggers element missing?");
549             setToDefault();
550             return;
551         } else if (!setRoot) {
552             LOGGER.warn("No Root logger was configured, creating default ERROR-level Root logger with Console appender");
553             setToDefault();
554             // return; // LOG4J2-219: creating default root=ok, but don't exclude configured Loggers
555         }
556 
557         for (final Map.Entry<String, LoggerConfig> entry : loggerConfigs.entrySet()) {
558             final LoggerConfig loggerConfig = entry.getValue();
559             for (final AppenderRef ref : loggerConfig.getAppenderRefs()) {
560                 final Appender app = appenders.get(ref.getRef());
561                 if (app != null) {
562                     loggerConfig.addAppender(app, ref.getLevel(), ref.getFilter());
563                 } else {
564                     LOGGER.error("Unable to locate appender \"{}\" for logger config \"{}\"", ref.getRef(),
565                             loggerConfig);
566                 }
567             }
568 
569         }
570 
571         setParents();
572     }

574     protected void setToDefault() {
575         // LOG4J2-1176 facilitate memory leak investigation
576         setName(DefaultConfiguration.DEFAULT_NAME + "@" + Integer.toHexString(hashCode()));
577         final Layout<? extends Serializable> layout = PatternLayout.newBuilder()
578                 .withPattern(DefaultConfiguration.DEFAULT_PATTERN)
579                 .withConfiguration(this)
580                 .build();
581         final Appender appender = ConsoleAppender.createDefaultAppenderForLayout(layout);
582         appender.start();
583         addAppender(appender);
584         final LoggerConfig rootLoggerConfig = getRootLogger();
585         rootLoggerConfig.addAppender(appender, null, null);
586 
587         final Level defaultLevel = Level.ERROR;
588         final String levelName = PropertiesUtil.getProperties().getStringProperty(DefaultConfiguration.DEFAULT_LEVEL,
589                 defaultLevel.name());
590         final Level level = Level.valueOf(levelName);
591         rootLoggerConfig.setLevel(level != null ? level : defaultLevel);
592     }

702             final Appender appender) {

711             nlc.setParent(lc);
712             loggerConfigs.putIfAbsent(loggerName, nlc);
713             setParents();
714             logger.getContext().updateLoggers();
715         }

795     /**
796      * Locates the appropriate LoggerConfig for a Logger name. This will remove tokens from the package name as
797      * necessary or return the root LoggerConfig if no other matches were found.
798      *
799      * @param loggerName The Logger name.
800      * @return The located LoggerConfig.
801      */
802     @Override
803     public LoggerConfig getLoggerConfig(final String loggerName) {
804         LoggerConfig loggerConfig = loggerConfigs.get(loggerName);
805         if (loggerConfig != null) {
806             return loggerConfig;
807         }
808         String substr = loggerName;
809         while ((substr = NameUtil.getSubName(substr)) != null) {
810             loggerConfig = loggerConfigs.get(substr);
811             if (loggerConfig != null) {
812                 return loggerConfig;
813             }
814         }
815         return root;
816     }

877     @Override
878     public void createConfiguration(final Node node, final LogEvent event) {
879         final PluginType<?> type = node.getType();
880         if (type != null && type.isDeferChildren()) {
881             node.setObject(createPluginObject(type, node, event));
882         } else {
883             for (final Node child : node.getChildren()) {
884                 createConfiguration(child, event);
885             }
886 
887             if (type == null) {
888                 if (node.getParent() != null) {
889                     LOGGER.error("Unable to locate plugin for {}", node.getName());
890                 }
891             } else {
892                 node.setObject(createPluginObject(type, node, event));
893             }
894         }
895     }

897     /**
898      * Invokes a static factory method to either create the desired object or to create a builder object that creates
899      * the desired object. In the case of a factory method, it should be annotated with
900      * {@link org.apache.logging.log4j.core.config.plugins.PluginFactory}, and each parameter should be annotated with
901      * an appropriate plugin annotation depending on what that parameter describes. Parameters annotated with
902      * {@link org.apache.logging.log4j.core.config.plugins.PluginAttribute} must be a type that can be converted from a
903      * string using one of the {@link org.apache.logging.log4j.core.config.plugins.convert.TypeConverter TypeConverters}
904      * . Parameters with {@link org.apache.logging.log4j.core.config.plugins.PluginElement} may be any plugin class or
905      * an array of a plugin class. Collections and Maps are currently not supported, although the factory method that is
906      * called can create these from an array.
907      *
908      * Plugins can also be created using a builder class that implements
909      * {@link org.apache.logging.log4j.core.util.Builder}. In that case, a static method annotated with
910      * {@link org.apache.logging.log4j.core.config.plugins.PluginBuilderAttribute} should create the builder class, and
911      * the various fields in the builder class should be annotated similarly to the method parameters. However, instead
912      * of using PluginAttribute, one should use
913      * {@link org.apache.logging.log4j.core.config.plugins.PluginBuilderAttribute} where the default value can be
914      * specified as the default field value instead of as an additional annotation parameter.
915      *
916      * In either case, there are also annotations for specifying a
917      * {@link org.apache.logging.log4j.core.config.Configuration} (
918      * {@link org.apache.logging.log4j.core.config.plugins.PluginConfiguration}) or a
919      * {@link org.apache.logging.log4j.core.config.Node} (
920      * {@link org.apache.logging.log4j.core.config.plugins.PluginNode}).
921      *
922      * Although the happy path works, more work still needs to be done to log incorrect parameters. These will generally
923      * result in unhelpful InvocationTargetExceptions.
924      *
925      * @param type the type of plugin to create.
926      * @param node the corresponding configuration node for this plugin to create.
927      * @param event the LogEvent that spurred the creation of this plugin
928      * @return the created plugin object or {@code null} if there was an error setting it up.
929      * @see org.apache.logging.log4j.core.config.plugins.util.PluginBuilder
930      * @see org.apache.logging.log4j.core.config.plugins.visitors.PluginVisitor
931      * @see org.apache.logging.log4j.core.config.plugins.convert.TypeConverter
932      */
933     private Object createPluginObject(final PluginType<?> type, final Node node, final LogEvent event) {
934         final Class<?> clazz = type.getPluginClass();
935 
936         if (Map.class.isAssignableFrom(clazz)) {
937             try {
938                 return createPluginMap(node);
939             } catch (final Exception e) {
940                 LOGGER.warn("Unable to create Map for {} of class {}", type.getElementName(), clazz, e);
941             }
942         }
943 
944         if (Collection.class.isAssignableFrom(clazz)) {
945             try {
946                 return createPluginCollection(node);
947             } catch (final Exception e) {
948                 LOGGER.warn("Unable to create List for {} of class {}", type.getElementName(), clazz, e);
949             }
950         }
951 
952         return new PluginBuilder(type).withConfiguration(this).withConfigurationNode(node).forLogEvent(event).build();
953     }
954 
955     private static Map<String, ?> createPluginMap(final Node node) {
956         final Map<String, Object> map = new LinkedHashMap<>();
957         for (final Node child : node.getChildren()) {
958             final Object object = child.getObject();
959             map.put(child.getName(), object);
960         }
961         return map;
962     }
963 
964     private static Collection<?> createPluginCollection(final Node node) {
965         final List<Node> children = node.getChildren();
966         final Collection<Object> list = new ArrayList<>(children.size());
967         for (final Node child : children) {
968             final Object object = child.getObject();
969             list.add(object);
970         }
971         return list;
972     }

974     private void setParents() {
975         for (final Map.Entry<String, LoggerConfig> entry : loggerConfigs.entrySet()) {
976             final LoggerConfig logger = entry.getValue();
977             String key = entry.getKey();
978             if (!key.isEmpty()) {
979                 final int i = key.lastIndexOf('.');
980                 if (i > 0) {
981                     key = key.substring(0, i);
982                     LoggerConfig parent = getLoggerConfig(key);
983                     if (parent == null) {
984                         parent = root;
985                     }
986                     logger.setParent(parent);
987                 } else {
988                     logger.setParent(root);
989                 }
990             }
991         }
992     }

994     /**
995      * Reads an InputStream using buffered reads into a byte array buffer. The given InputStream will remain open after
996      * invocation of this method.
997      *
998      * @param is the InputStream to read into a byte array buffer.
999      * @return a byte array of the InputStream contents.
1000      * @throws IOException if the {@code read} method of the provided InputStream throws this exception.
1001      */
1002     protected static byte[] toByteArray(final InputStream is) throws IOException {
1003         final ByteArrayOutputStream buffer = new ByteArrayOutputStream();
1004 
1005         int nRead;
1006         final byte[] data = new byte[BUF_SIZE];
1007 
1008         while ((nRead = is.read(data, 0, data.length)) != -1) {
1009             buffer.write(data, 0, nRead);
1010         }
1011 
1012         return buffer.toByteArray();
1013     }

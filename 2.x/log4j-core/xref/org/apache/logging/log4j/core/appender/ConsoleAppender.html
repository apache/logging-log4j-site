38  /**
39   * ConsoleAppender appends log events to <code>System.out</code> or
40   * <code>System.err</code> using a layout specified by the user. The
41   * default target is <code>System.out</code>.
42   * @doubt accessing System.out or .err as a byte stream instead of a writer
43   *    bypasses the JVM's knowledge of the proper encoding. (RG) Encoding
44   * is handled within the Layout. Typically, a Layout will generate a String
45   * and then call getBytes which may use a configured encoding or the system
46   * default. OTOH, a Writer cannot print byte streams.
47   *
48   * @param <T> The {@link Layout}'s {@link Serializable} type.
49   */
50  @Plugin(name = "Console", category = "Core", elementType = "appender", printObject = true)
51  public final class ConsoleAppender<T extends Serializable> extends AbstractOutputStreamAppender<T> {
52  
53      private static ConsoleManagerFactory factory = new ConsoleManagerFactory();

55      /**
56       * Enumeration of console destinations.
57       */
58      public enum Target {
59          /** Standard output. */
60          SYSTEM_OUT,
61          /** Standard error output. */
62          SYSTEM_ERR
63      }

65      private ConsoleAppender(final String name, final Layout<T> layout, final Filter filter,
66                              final OutputStreamManager manager,
67                              final boolean handleExceptions) {
68          super(name, layout, filter, handleExceptions, true, manager);
69      }

71      /**
72       * Create a Console Appender.
73       * @param layout The layout to use (required).
74       * @param filter The Filter or null.
75       * @param t The target ("SYSTEM_OUT" or "SYSTEM_ERR"). The default is "SYSTEM_OUT".
76       * @param follow If true will follow changes to the underlying output stream.
77       * @param name The name of the Appender (required).
78       * @param suppress "true" if exceptions should be hidden from the application, "false" otherwise.
79       *                 The default is "true".
80       * @param <S> The {@link Layout}'s {@link Serializable} type.
81       * @return The ConsoleAppender.
82       */
83      @PluginFactory
84      public static <S extends Serializable> ConsoleAppender<S> createAppender(@PluginElement("layout") Layout<S> layout,
85                                                   @PluginElement("filters") final Filter filter,
86                                                   @PluginAttr("target") final String t,
87                                                   @PluginAttr("name") final String name,
88                                                   @PluginAttr("follow") final String follow,
89                                                   @PluginAttr("suppressExceptions") final String suppress) {
90          if (name == null) {
91              LOGGER.error("No name provided for ConsoleAppender");
92              return null;
93          }
94          if (layout == null) {
95              @SuppressWarnings({ "unchecked", "UnnecessaryLocalVariable" })
96              final
97              Layout<S> l = (Layout<S>) PatternLayout.createLayout(null, null, null, null, null);
98              layout = l;
99          }
100         final boolean isFollow = Boolean.parseBoolean(follow);
101         final boolean handleExceptions = Booleans.parseBoolean(suppress, true);
102         final Target target = t == null ? Target.SYSTEM_OUT : Target.valueOf(t);
103         return new ConsoleAppender<S>(name, layout, filter, getManager(isFollow, target, layout), handleExceptions);
104     }

106     private static OutputStreamManager getManager(final boolean follow, final Target target, final Layout layout) {
107         final String type = target.name();
108         final OutputStream os = getOutputStream(follow, target);
109         return OutputStreamManager.getManager(target.name() + "." + follow, new FactoryData(os, type, layout), factory);
110     }
111 
112     private static OutputStream getOutputStream(final boolean follow, final Target target) {
113         final String enc = Charset.defaultCharset().name();
114         PrintStream printStream = null;
115         try {
116             printStream = target == Target.SYSTEM_OUT ?
117             follow ? new PrintStream(new SystemOutStream(), true, enc) : System.out :
118             follow ? new PrintStream(new SystemErrStream(), true, enc) : System.err;
119         } catch (final UnsupportedEncodingException ex) { // should never happen
120             throw new IllegalStateException("Unsupported default encoding " + enc, ex);
121         }
122         final PropertiesUtil propsUtil = PropertiesUtil.getProperties();
123         if (!propsUtil.getStringProperty("os.name").startsWith("Windows") ||
124             propsUtil.getBooleanProperty("log4j.skipJansi")) {
125             return printStream;
126         } else {
127             try {
128                 final ClassLoader loader = Loader.getClassLoader();
129                 // We type the parameter as a wildcard to avoid a hard reference to Jansi.
130                 final Class<?> clazz = loader.loadClass("org.fusesource.jansi.WindowsAnsiOutputStream");
131                 final Constructor<?> constructor = clazz.getConstructor(OutputStream.class);
132                 return (OutputStream) constructor.newInstance(printStream);
133             } catch (final ClassNotFoundException cnfe) {
134                 LOGGER.debug("Jansi is not installed");
135             } catch (final NoSuchMethodException nsme) {
136                 LOGGER.warn("WindowsAnsiOutputStream is missing the proper constructor");
137             } catch (final Exception ex) {
138                 LOGGER.warn("Unable to instantiate WindowsAnsiOutputStream");
139             }
140             return printStream;
141         }
142     }

113     private RFC5424Layout(final Configuration config, final Facility facility, final String id, final int ein,
114                           final boolean includeMDC, final boolean includeNL, final String escapeNL, final String mdcId,
115                           final String mdcPrefix, final String eventPrefix,
116                           final String appName, final String messageId, final String excludes, final String includes,
117                           final String required, final Charset charset, final String exceptionPattern,
118                           final Map<String, String> loggerFields) {
119         super(charset);
120         final PatternParser exceptionParser = createPatternParser(config, ThrowablePatternConverter.class);
121         exceptionFormatters = exceptionPattern == null ? null : exceptionParser.parse(exceptionPattern, false);
122         this.facility = facility;
123         this.defaultId = id == null ? DEFAULT_ID : id;
124         this.enterpriseNumber = ein;
125         this.includeMDC = includeMDC;
126         this.includeNewLine = includeNL;
127         this.escapeNewLine = escapeNL == null ? null : Matcher.quoteReplacement(escapeNL);
128         this.mdcId = mdcId;
129         this.mdcPrefix = mdcPrefix;
130         this.eventPrefix = eventPrefix;
131         this.appName = appName;
132         this.messageId = messageId;
133         this.localHostName = NetUtils.getLocalHostname();
134         ListChecker c = null;
135         if (excludes != null) {
136             final String[] array = excludes.split(",");
137             if (array.length > 0) {
138                 c = new ExcludeChecker();
139                 mdcExcludes = new ArrayList<String>(array.length);
140                 for (final String str : array) {
141                     mdcExcludes.add(str.trim());
142                 }
143             } else {
144                 mdcExcludes = null;
145             }
146         } else {
147             mdcExcludes = null;
148         }
149         if (includes != null) {
150             final String[] array = includes.split(",");
151             if (array.length > 0) {
152                 c = new IncludeChecker();
153                 mdcIncludes = new ArrayList<String>(array.length);
154                 for (final String str : array) {
155                     mdcIncludes.add(str.trim());
156                 }
157             } else {
158                 mdcIncludes = null;
159             }
160         } else {
161             mdcIncludes = null;
162         }
163         if (required != null) {
164             final String[] array = required.split(",");
165             if (array.length > 0) {
166                 mdcRequired = new ArrayList<String>(array.length);
167                 for (final String str : array) {
168                     mdcRequired.add(str.trim());
169                 }
170             } else {
171                 mdcRequired = null;
172             }
173 
174         } else {
175             mdcRequired = null;
176         }
177         this.checker = c != null ? c : noopChecker;
178         final String name = config == null ? null : config.getName();
179         configName = name != null && name.length() > 0 ? name : null;
180         if (loggerFields != null && !loggerFields.isEmpty()) {
181             final PatternParser fieldParser = createPatternParser(config, null);
182 
183             final Map<String, List<PatternFormatter>> map = new HashMap<String, List<PatternFormatter>>();
184             for (final Map.Entry<String, String> entry : loggerFields.entrySet()) {
185                 final List<PatternFormatter> formatters = fieldParser.parse(entry.getValue(), false);
186                 map.put(entry.getKey(), formatters);
187             }
188             this.fieldFormatters = map;
189         } else {
190             this.fieldFormatters = null;
191         }
192     }

200     public static PatternParser createPatternParser(final Configuration config,
201                                                     final Class<? extends PatternConverter> filterClass) {
202         if (config == null) {
203             return new PatternParser(config, PatternLayout.KEY, LogEventPatternConverter.class,
204                 filterClass);
205         }
206         PatternParser parser = (PatternParser) config.getComponent(COMPONENT_KEY);
207         if (parser == null) {
208             parser = new PatternParser(config, PatternLayout.KEY, ThrowablePatternConverter.class);
209             config.addComponent(COMPONENT_KEY, parser);
210             parser = (PatternParser) config.getComponent(COMPONENT_KEY);
211         }
212         return parser;
213     }

230     /**
231      * Formats a {@link org.apache.logging.log4j.core.LogEvent} in conformance with the RFC 5424 Syslog specification.
232      *
233      * @param event The LogEvent.
234      * @return The RFC 5424 String representation of the LogEvent.
235      */
236     @Override
237     public String toSerializable(final LogEvent event) {
238         final Message msg = event.getMessage();
239         final boolean isStructured = msg instanceof StructuredDataMessage;
240         final StringBuilder buf = new StringBuilder();
241 
242         buf.append("<");
243         buf.append(Priority.getPriority(facility, event.getLevel()));
244         buf.append(">1 ");
245         buf.append(computeTimeStampString(event.getMillis()));
246         buf.append(' ');
247         buf.append(localHostName);
248         buf.append(' ');
249         if (appName != null) {
250             buf.append(appName);
251         } else if (configName != null) {
252             buf.append(configName);
253         } else {
254             buf.append("-");
255         }
256         buf.append(" ");
257         buf.append(getProcId());
258         buf.append(" ");
259         final String type = isStructured ? ((StructuredDataMessage) msg).getType() : null;
260         if (type != null) {
261             buf.append(type);
262         } else if (messageId != null) {
263             buf.append(messageId);
264         } else {
265             buf.append("-");
266         }
267         buf.append(" ");
268         if (isStructured || includeMDC) {
269             StructuredDataId id = null;
270             String text;
271             if (isStructured) {
272                 final StructuredDataMessage data = (StructuredDataMessage) msg;
273                 final Map<String, String> map = data.getData();
274                 id = data.getId();
275                 formatStructuredElement(id, eventPrefix, map, buf, noopChecker);
276                 text = data.getFormat();
277             } else {
278                 text = msg.getFormattedMessage();
279             }
280             if (includeMDC) {
281                 Map<String, String> map = event.getContextMap();
282                 if (mdcRequired != null) {
283                     checkRequired(map);
284                 }
285                 final int ein = id == null || id.getEnterpriseNumber() < 0 ?
286                     enterpriseNumber : id.getEnterpriseNumber();
287                 final StructuredDataId mdcSDID = new StructuredDataId(mdcId, ein, null, null);
288                 if (fieldFormatters != null) {
289                     map = new HashMap<String, String>(map);
290                     for (final Map.Entry<String, List<PatternFormatter>> entry : fieldFormatters.entrySet()) {
291                         final StringBuilder value = new StringBuilder();
292                         for (final PatternFormatter formatter : entry.getValue()) {
293                             formatter.format(event, value);
294                         }
295                         map.put(entry.getKey(), value.toString());
296                     }
297                 }
298                 formatStructuredElement(mdcSDID, mdcPrefix, map, buf, checker);
299             }
300             if (text != null && text.length() > 0) {
301                 buf.append(" ").append(escapeNewlines(text, escapeNewLine));
302             }
303         } else {
304             buf.append("- ");
305             buf.append(escapeNewlines(msg.getFormattedMessage(), escapeNewLine));
306         }
307         if (exceptionFormatters != null && event.getThrown() != null) {
308             final StringBuilder exception = new StringBuilder("\n");
309             for (final PatternFormatter formatter : exceptionFormatters) {
310                 formatter.format(event, exception);
311             }
312             buf.append(escapeNewlines(exception.toString(), escapeNewLine));
313         }
314         if (includeNewLine) {
315             buf.append("\n");
316         }
317         return buf.toString();
318     }

320     private String escapeNewlines(final String text, final String escapeNewLine) {
321         if (null == escapeNewLine) {
322             return text;
323         }
324         return NEWLINE_PATTERN.matcher(text).replaceAll(escapeNewLine);
325     }

339     private String computeTimeStampString(final long now) {
340         long last;
341         synchronized (this) {
342             last = lastTimestamp;
343             if (now == lastTimestamp) {
344                 return timestamppStr;
345             }
346         }
347 
348         final StringBuilder buf = new StringBuilder();
349         final Calendar cal = new GregorianCalendar();
350         cal.setTimeInMillis(now);
351         buf.append(Integer.toString(cal.get(Calendar.YEAR)));
352         buf.append("-");
353         pad(cal.get(Calendar.MONTH) + 1, TWO_DIGITS, buf);
354         buf.append("-");
355         pad(cal.get(Calendar.DAY_OF_MONTH), TWO_DIGITS, buf);
356         buf.append("T");
357         pad(cal.get(Calendar.HOUR_OF_DAY), TWO_DIGITS, buf);
358         buf.append(":");
359         pad(cal.get(Calendar.MINUTE), TWO_DIGITS, buf);
360         buf.append(":");
361         pad(cal.get(Calendar.SECOND), TWO_DIGITS, buf);
362 
363         final int millis = cal.get(Calendar.MILLISECOND);
364         if (millis != 0) {
365             buf.append('.');
366             pad(millis, THREE_DIGITS, buf);
367         }
368 
369         int tzmin = (cal.get(Calendar.ZONE_OFFSET) + cal.get(Calendar.DST_OFFSET)) / MILLIS_PER_MINUTE;
370         if (tzmin == 0) {
371             buf.append("Z");
372         } else {
373             if (tzmin < 0) {
374                 tzmin = -tzmin;
375                 buf.append("-");
376             } else {
377                 buf.append("+");
378             }
379             final int tzhour = tzmin / MINUTES_PER_HOUR;
380             tzmin -= tzhour * MINUTES_PER_HOUR;
381             pad(tzhour, TWO_DIGITS, buf);
382             buf.append(":");
383             pad(tzmin, TWO_DIGITS, buf);
384         }
385         synchronized (this) {
386             if (last == lastTimestamp) {
387                 lastTimestamp = now;
388                 timestamppStr = buf.toString();
389             }
390         }
391         return buf.toString();
392     }

394     private void pad(final int val, int max, final StringBuilder buf) {
395         while (max > 1) {
396             if (val < max) {
397                 buf.append("0");
398             }
399             max = max / TWO_DIGITS;
400         }
401         buf.append(Integer.toString(val));
402     }

404     private void formatStructuredElement(final StructuredDataId id, final String prefix, final Map<String, String> data,
405                                          final StringBuilder sb, final ListChecker checker) {
406         if (id == null && defaultId == null) {
407             return;
408         }
409         sb.append("[");
410         sb.append(getId(id));
411         appendMap(prefix, data, sb, checker);
412         sb.append("]");
413     }

415     private String getId(final StructuredDataId id) {
416         final StringBuilder sb = new StringBuilder();
417         if (id == null || id.getName() == null) {
418             sb.append(defaultId);
419         } else {
420             sb.append(id.getName());
421         }
422         int ein = id != null ? id.getEnterpriseNumber() : enterpriseNumber;
423         if (ein < 0) {
424             ein = enterpriseNumber;
425         }
426         if (ein >= 0) {
427             sb.append("@").append(ein);
428         }
429         return sb.toString();
430     }

432     private void checkRequired(final Map<String, String> map) {
433         for (final String key : mdcRequired) {
434             final String value = map.get(key);
435             if (value == null) {
436                 throw new LoggingException("Required key " + key + " is missing from the " + mdcId);
437             }
438         }
439     }

441     private void appendMap(final String prefix, final Map<String, String> map, final StringBuilder sb,
442                            final ListChecker checker) {
443         final SortedMap<String, String> sorted = new TreeMap<String, String>(map);
444         for (final Map.Entry<String, String> entry : sorted.entrySet()) {
445             if (checker.check(entry.getKey()) && entry.getValue() != null) {
446                 sb.append(" ");
447                 if (prefix != null) {
448                     sb.append(prefix);
449                 }
450                 sb.append(escapeNewlines(escapeSDParams(entry.getKey()), escapeNewLine)).append("=\"")
451                     .append(escapeNewlines(escapeSDParams(entry.getValue()), escapeNewLine)).append("\"");
452             }
453         }
454     }

456     private String escapeSDParams(final String value) {
457         return PARAM_VALUE_ESCAPE_PATTERN.matcher(value).replaceAll("\\\\$0");
458     }

510     /**
511      * Create the RFC 5424 Layout.
512      *
513      * @param facility         The Facility is used to try to classify the message.
514      * @param id               The default structured data id to use when formatting according to RFC 5424.
515      * @param ein              The IANA enterprise number.
516      * @param includeMDC       Indicates whether data from the ThreadContextMap will be included in the RFC 5424 Syslog
517      *                         record. Defaults to "true:.
518      * @param mdcId            The id to use for the MDC Structured Data Element.
519      * @param mdcPrefix        The prefix to add to MDC key names.
520      * @param eventPrefix      The prefix to add to event key names.
521      * @param includeNL        If true, a newline will be appended to the end of the syslog record. The default is false.
522      * @param escapeNL         String that should be used to replace newlines within the message text.
523      * @param appName          The value to use as the APP-NAME in the RFC 5424 syslog record.
524      * @param msgId            The default value to be used in the MSGID field of RFC 5424 syslog records.
525      * @param excludes         A comma separated list of mdc keys that should be excluded from the LogEvent.
526      * @param includes         A comma separated list of mdc keys that should be included in the FlumeEvent.
527      * @param required         A comma separated list of mdc keys that must be present in the MDC.
528      * @param exceptionPattern The pattern for formatting exceptions.
529      * @param loggerFields     Container for the KeyValuePairs containing the patterns
530      * @param config           The Configuration. Some Converters require access to the Interpolator.
531      * @return An RFC5424Layout.
532      */
533     @PluginFactory
534     public static RFC5424Layout createLayout(@PluginAttr("facility") final String facility,
535                                              @PluginAttr("id") final String id,
536                                              @PluginAttr("enterpriseNumber") final String ein,
537                                              @PluginAttr("includeMDC") final String includeMDC,
538                                              @PluginAttr("mdcId") String mdcId,
539                                              @PluginAttr("mdcPrefix") final String mdcPrefix,
540                                              @PluginAttr("eventPrefix") final String eventPrefix,
541                                              @PluginAttr("newLine") final String includeNL,
542                                              @PluginAttr("newLineEscape") final String escapeNL,
543                                              @PluginAttr("appName") final String appName,
544                                              @PluginAttr("messageId") final String msgId,
545                                              @PluginAttr("mdcExcludes") final String excludes,
546                                              @PluginAttr("mdcIncludes") String includes,
547                                              @PluginAttr("mdcRequired") final String required,
548                                              @PluginAttr("exceptionPattern") final String exceptionPattern,
549                                              @PluginElement("loggerFields") final LoggerFields loggerFields,
550                                              @PluginConfiguration final Configuration config) {
551         final Charset charset = Charsets.UTF_8;
552         if (includes != null && excludes != null) {
553             LOGGER.error("mdcIncludes and mdcExcludes are mutually exclusive. Includes wil be ignored");
554             includes = null;
555         }
556         final Facility f = Facility.toFacility(facility, Facility.LOCAL0);
557         final int enterpriseNumber = Integers.parseInt(ein, DEFAULT_ENTERPRISE_NUMBER);
558         final boolean isMdc = Booleans.parseBoolean(includeMDC, true);
559         final boolean includeNewLine = Boolean.parseBoolean(includeNL);
560         final Map<String, String> loggerFieldValues = loggerFields == null ? null : loggerFields.getMap();
561         if (mdcId == null) {
562             mdcId = DEFAULT_MDCID;
563         }
564 
565         return new RFC5424Layout(config, f, id, enterpriseNumber, isMdc, includeNewLine, escapeNL, mdcId, mdcPrefix,
566             eventPrefix, appName, msgId, excludes, includes, required, charset, exceptionPattern,
567             loggerFieldValues);
568     }

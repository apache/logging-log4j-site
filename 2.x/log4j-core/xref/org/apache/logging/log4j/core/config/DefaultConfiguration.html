17  package org.apache.logging.log4j.core.config;
18  
19  import org.apache.logging.log4j.Level;
20  import org.apache.logging.log4j.core.Appender;
21  import org.apache.logging.log4j.core.Layout;
22  import org.apache.logging.log4j.core.appender.ConsoleAppender;
23  import org.apache.logging.log4j.core.layout.PatternLayout;
24  
25  /**
26   * The default configuration writes all output to the Console using the default logging level. You configure default logging level by setting the
27   * system property "org.apache.logging.log4j.level" to a level name. If you do not specify the property, Log4J uses the ERROR Level. Log
28   * Events will be printed using the basic formatting provided by each Message.
29   */
30  public class DefaultConfiguration extends BaseConfiguration {
31  
32      /**
33       * The name of the default configuration.
34       */
35      public static final String DEFAULT_NAME = "Default";
36      /**
37       * The System Proerty used to specify the logging level.
38       */
39      public static final String DEFAULT_LEVEL = "org.apache.logging.log4j.level";
40  
41      /**
42       * Constructor to create the default configuration.
43       */
44      public DefaultConfiguration() {
45  
46          setName(DEFAULT_NAME);
47          Layout layout = PatternLayout.createLayout("%d{HH:mm:ss.SSS} [%thread] %-5level %logger{36} - %msg%n",
48              null, null, null);
49          Appender appender = ConsoleAppender.createAppender(layout, null, "SYSTEM_OUT", "Console", "true");
50          appender.start();
51          addAppender(appender);
52          LoggerConfig root = getRootLogger();
53          root.addAppender(appender, null, null);
54  
55          String levelName = System.getProperty(DEFAULT_LEVEL);
56          Level level = levelName != null && Level.valueOf(levelName) != null ? Level.valueOf(levelName) : Level.ERROR;
57          root.setLevel(level);
58      }
59  
60      @Override
61      protected void doConfigure() {
62      }
63  }

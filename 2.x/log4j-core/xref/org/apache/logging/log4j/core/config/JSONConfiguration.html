60      public JSONConfiguration(InputSource source, File configFile) {
61          this.configFile = configFile;
62          byte[] buffer;
63  
64          try {
65              buffer = toByteArray(source.getByteStream());
66              InputStream is = new ByteArrayInputStream(buffer);
67              source = new InputSource(is);
68              ObjectMapper mapper = new ObjectMapper().configure(JsonParser.Feature.ALLOW_COMMENTS, true);
69              root = mapper.readTree(is);
70              if (root.size() == 1) {
71                  Iterator<JsonNode> i = root.getElements();
72                  root = i.next();
73              }
74              processAttributes(rootNode, root);
75              Level status = Level.OFF;
76              boolean verbose = false;
77              for (Map.Entry<String, String> entry : rootNode.getAttributes().entrySet()) {
78                  if ("status".equalsIgnoreCase(entry.getKey())) {
79                      status = Level.toLevel(getSubst().replace(entry.getValue()).toUpperCase(), null);
80                      if (status == null) {
81                          status = Level.ERROR;
82                          messages.add("Invalid status specified: " + entry.getValue() + ". Defaulting to ERROR");
83                      }
84                  } else if ("verbose".equalsIgnoreCase(entry.getKey())) {
85                      verbose = Boolean.parseBoolean(getSubst().replace(entry.getValue()));
86                  } else if ("packages".equalsIgnoreCase(entry.getKey())) {
87                      String[] packages = getSubst().replace(entry.getValue()).split(",");
88                      for (String p : packages) {
89                          PluginManager.addPackage(p);
90                      }
91                  } else if ("name".equalsIgnoreCase(entry.getKey())) {
92                      setName(getSubst().replace(entry.getValue()));
93                  } else if ("monitorInterval".equalsIgnoreCase(entry.getKey())) {
94                      int interval = Integer.parseInt(getSubst().replace(entry.getValue()));
95                      if (interval > 0 && configFile != null) {
96                          monitor = new FileConfigurationMonitor(this, configFile, listeners, interval);
97                      }
98                  }
99              }
100 
101             Iterator<StatusListener> statusIter = ((StatusLogger) LOGGER).getListeners();
102             boolean found = false;
103             while (statusIter.hasNext()) {
104                 StatusListener listener = statusIter.next();
105                 if (listener instanceof StatusConsoleListener) {
106                     found = true;
107                     ((StatusConsoleListener) listener).setLevel(status);
108                     if (!verbose) {
109                         ((StatusConsoleListener) listener).setFilters(VERBOSE_CLASSES);
110                     }
111                 }
112             }
113             if (!found && status != Level.OFF) {
114                 StatusConsoleListener listener = new StatusConsoleListener(status);
115                 if (!verbose) {
116                     listener.setFilters(VERBOSE_CLASSES);
117                 }
118                 ((StatusLogger) LOGGER).registerListener(listener);
119                 for (String msg : messages) {
120                     LOGGER.error(msg);
121                 }
122             }
123             if (getName() == null) {
124                 setName(source.getSystemId());
125             }
126         } catch (Exception ex) {
127             LOGGER.error("Error parsing " + source.getSystemId(), ex);
128             ex.printStackTrace();
129         }
130     }

132      public void setup() {
133         Iterator<Map.Entry<String, JsonNode>> iter = root.getFields();
134         List<Node> children = rootNode.getChildren();
135         while (iter.hasNext()) {
136             Map.Entry<String, JsonNode> entry = iter.next();
137             JsonNode n = entry.getValue();
138             if (n.isObject()) {
139                 LOGGER.debug("Processing node for object " + entry.getKey());
140                 children.add(constructNode(entry.getKey(), rootNode, n));
141             } else if (n.isArray()) {
142                 LOGGER.error("Arrays are not supported at the root configuration.");
143             }
144         }
145         LOGGER.debug("Completed parsing configuration");
146         if (status.size() > 0) {
147             for (Status s : status) {
148                 LOGGER.error("Error processing element " + s.name + ": " + s.errorType);
149             }
150             return;
151         }
152     }

154     public Configuration reconfigure() {
155         if (configFile != null) {
156             try {
157                 InputSource source = new InputSource(new FileInputStream(configFile));
158                 source.setSystemId(configFile.getAbsolutePath());
159                 return new JSONConfiguration(source, configFile);
160             } catch (FileNotFoundException ex) {
161                 LOGGER.error("Cannot locate file " + configFile, ex);
162             }
163         }
164         return null;
165     }

167     private Node constructNode(String name, Node parent, JsonNode jsonNode) {
168         PluginType type = getPluginManager().getPluginType(name);
169         Node node = new Node(parent, name, type);
170         processAttributes(node, jsonNode);
171         Iterator<Map.Entry<String, JsonNode>> iter = jsonNode.getFields();
172         List<Node> children = node.getChildren();
173         while (iter.hasNext()) {
174             Map.Entry<String, JsonNode> entry = iter.next();
175             JsonNode n = entry.getValue();
176             if (n.isArray() || n.isObject()) {
177                 if (type == null) {
178                     status.add(new Status(name, n, ErrorType.CLASS_NOT_FOUND));
179                 }
180                 if (n.isArray()) {
181                     LOGGER.debug("Processing node for array " + entry.getKey());
182                     for (int i = 0; i < n.size(); ++i) {
183                         String pluginType = getType(n.get(i), entry.getKey());
184                         PluginType entryType = getPluginManager().getPluginType(pluginType);
185                         Node item = new Node(node, entry.getKey(), entryType);
186                         processAttributes(item, n.get(i));
187                         if (pluginType.equals(entry.getKey())) {
188                             LOGGER.debug("Processing " + entry.getKey() + "[" + i + "]");
189                         } else {
190                             LOGGER.debug("Processing " + pluginType + " " + entry.getKey() + "[" + i + "]");
191                         }
192                         Iterator<Map.Entry<String, JsonNode>> itemIter = n.get(i).getFields();
193                         List<Node> itemChildren = item.getChildren();
194                         while (itemIter.hasNext()) {
195                             Map.Entry<String, JsonNode> itemEntry = itemIter.next();
196                             if (itemEntry.getValue().isObject()) {
197                                 LOGGER.debug("Processing node for object " + itemEntry.getKey());
198                                 itemChildren.add(constructNode(itemEntry.getKey(), item, itemEntry.getValue()));
199                             }
200                         }
201                         children.add(item);
202                     }
203                 } else {
204                     LOGGER.debug("Processing node for object " + entry.getKey());
205                     children.add(constructNode(entry.getKey(), node, n));
206                 }
207             }
208         }
209 
210         String t;
211         if (type == null) {
212             t = "null";
213         } else {
214             t = type.getElementName() + ":" + type.getPluginClass();
215         }
216 
217         String p = node.getParent() == null ? "null" : node.getParent().getName() == null ?
218             "root" : node.getParent().getName();
219         LOGGER.debug("Returning " + node.getName() + " with parent " + p + " of type " +  t);
220         return node;
221     }

223     private String getType(JsonNode node, String name) {
224         Iterator<Map.Entry<String, JsonNode>> iter = node.getFields();
225         while (iter.hasNext()) {
226             Map.Entry<String, JsonNode> entry = iter.next();
227             if (entry.getKey().equalsIgnoreCase("type")) {
228                 JsonNode n = entry.getValue();
229                 if (n.isValueNode()) {
230                     return n.asText();
231                 }
232             }
233         }
234         return name;
235     }
236 
237     private void processAttributes(Node parent, JsonNode node) {
238         Map<String, String> attrs = parent.getAttributes();
239         Iterator<Map.Entry<String, JsonNode>> iter = node.getFields();
240         while (iter.hasNext()) {
241             Map.Entry<String, JsonNode> entry = iter.next();
242             if (!entry.getKey().equalsIgnoreCase("type")) {
243                 JsonNode n = entry.getValue();
244                 if (n.isValueNode()) {
245                     attrs.put(entry.getKey(), n.asText());
246                 }
247             }
248         }
249     }

251     protected byte[] toByteArray(InputStream is) throws IOException {
252         ByteArrayOutputStream buffer = new ByteArrayOutputStream();
253 
254         int nRead;
255         byte[] data = new byte[BUF_SIZE];
256 
257         while ((nRead = is.read(data, 0, data.length)) != -1) {
258             buffer.write(data, 0, nRead);
259         }
260 
261         return buffer.toByteArray();
262     }

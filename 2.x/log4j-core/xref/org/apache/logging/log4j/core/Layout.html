17  package org.apache.logging.log4j.core;
18  
19  import java.io.Serializable;
20  
21  /**
22   * @doubt There is still a need for a character-based layout for character based event sinks (databases, etc).
23   *  Would introduce an EventEncoder, EventRenderer or something similar for the logging event to byte encoding.
24   * (RG) A layout can be configured with a Charset and then Strings can be converted to byte arrays. OTOH, it isn't
25   * possible to write byte arrays as character streams.
26   * @param <T> The Object type that will be returned on the formatAs call.
27   */
28  public interface Layout<T extends Serializable> {
29      /**
30       * Formats the event suitable for display.
31       * @param event The Logging Event.
32       * @return The formatted event.
33       * @doubt Likely better to write to a OutputStream instead of return a byte[]. (RG) That limits how the
34       * Appender can use the Layout. For example, it might concatenate information in front or behind the
35       * data and then write it all to the OutputStream in one call.
36       */
37      byte[] format(LogEvent event);
38  
39      /**
40       * Formats the event as an Object that can be serialized.
41       * @param event The Logging Event.
42       * @return The formatted event.
43       */
44      T formatAs(LogEvent event);
45  
46      /**
47       * Returns the header for the layout format.
48       * @return The header.
49       * @doubt the concept of header and footer is not universal, should not be on the base interface.
50       * (RG) I agree with this.
51       */
52      byte[] getHeader();
53  
54      /**
55       * Returns the format for the layout format.
56       * @return The footer.
57       * @doubt the concept of header and footer is not universal, should not be on the base interface.
58       * (RG) I agree with this.
59       */
60      byte[] getFooter();

63  }

17  package org.apache.logging.log4j.core.appender.rolling.helper;
18  
19  import java.io.IOException;
20  import java.util.List;
21  
22  
23  /**
24   * A group of Actions to be executed in sequence.
25   */
26  public class CompositeAction extends AbstractAction {
27      /**
28       * Actions to perform.
29       */
30      private final Action[] actions;
31  
32      /**
33       * Stop on error.
34       */
35      private final boolean stopOnError;
36  
37      /**
38       * Construct a new composite action.
39       *
40       * @param actions     list of actions, may not be null.
41       * @param stopOnError if true, stop on the first false return value or exception.
42       */
43      public CompositeAction(final List<Action> actions,
44                             final boolean stopOnError) {
45          this.actions = new Action[actions.size()];
46          actions.toArray(this.actions);
47          this.stopOnError = stopOnError;
48      }
49  
50      /**
51       * {@inheritDoc}
52       */
53      @Override
54      public void run() {
55          try {
56              execute();
57          } catch (final IOException ex) {
58              LOGGER.warn("Exception during file rollover.", ex);
59          }
60      }
61  
62      /**
63       * Execute sequence of actions.
64       *
65       * @return true if all actions were successful.
66       * @throws IOException on IO error.
67       */
68      @Override
69      public boolean execute() throws IOException {
70          if (stopOnError) {
71              for (final Action action : actions) {
72                  if (!action.execute()) {
73                      return false;
74                  }
75              }
76  
77              return true;
78          } else {
79              boolean status = true;
80              IOException exception = null;
81  
82              for (final Action action : actions) {
83                  try {
84                      status &= action.execute();
85                  } catch (final IOException ex) {
86                      status = false;
87  
88                      if (exception == null) {
89                          exception = ex;
90                      }
91                  }
92              }
93  
94              if (exception != null) {
95                  throw exception;
96              }
97  
98              return status;
99          }
100     }
101 }

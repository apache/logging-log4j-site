17  package org.apache.logging.log4j.core.pattern;
18  
19  import org.apache.logging.log4j.Level;
20  import org.apache.logging.log4j.core.LogEvent;
21  import org.apache.logging.log4j.core.config.Configuration;
22  import org.apache.logging.log4j.core.config.plugins.Plugin;
23  import org.apache.logging.log4j.core.layout.PatternLayout;
24  
25  import java.util.EnumMap;
26  import java.util.List;
27  
28  /**
29   * Highlight pattern converter. Formats the result of a pattern using a color appropriate
30   * for the Level in the LogEvent.
31   */
32  @Plugin(name = "highlight", type = "Converter")
33  @ConverterKeys({"highlight" })
34  public final class HighlightConverter extends LogEventPatternConverter {
35  
36      private static final EnumMap<Level, String> LEVEL_STYLES = new EnumMap<Level, String>(Level.class);
37  
38      private static final String[] FATAL = new String[]{"BLINK", "BRIGHT", "RED"};
39      private static final String[] ERROR = new String[] {"BRIGHT", "RED"};
40      private static final String[] WARN = new String[] {"RED"};
41      private static final String[] INFO = new String[] {"BLUE"};
42      private static final String[] DEBUG = null;
43      private static final String[] TRACE = null;
44  
45      private List<PatternFormatter> formatters;
46  
47      static {
48          LEVEL_STYLES.put(Level.FATAL, AnsiEscape.createSequence(FATAL));
49          LEVEL_STYLES.put(Level.ERROR, AnsiEscape.createSequence(ERROR));
50          LEVEL_STYLES.put(Level.WARN, AnsiEscape.createSequence(WARN));
51          LEVEL_STYLES.put(Level.INFO, AnsiEscape.createSequence(INFO));
52          LEVEL_STYLES.put(Level.DEBUG, AnsiEscape.createSequence(DEBUG));
53          LEVEL_STYLES.put(Level.TRACE, AnsiEscape.createSequence(TRACE));
54      }
55  
56      /**
57       * Construct the converter.
58       * @param formatters The PatternFormatters to generate the text to manipulate.
59       */
60      private HighlightConverter(List<PatternFormatter> formatters) {
61          super("style", "style");
62          this.formatters = formatters;
63      }
64  
65      /**
66       * Gets an instance of the class.
67       *
68       * @param config The current Configuration.
69       * @param options pattern options, may be null.  If first element is "short",
70       *                only the first line of the throwable will be formatted.
71       * @return instance of class.
72       */
73      public static HighlightConverter newInstance(Configuration config, final String[] options) {
74          if (options.length < 1) {
75              LOGGER.error("Incorrect number of options on style. Expected at least 1, received " + options.length);
76              return null;
77          }
78          if (options[0] == null) {
79              LOGGER.error("No pattern supplied on style");
80              return null;
81          }
82  
83          PatternParser parser = PatternLayout.createPatternParser(config);
84          List<PatternFormatter> formatters = parser.parse(options[0]);
85          return new HighlightConverter(formatters);
86      }
87  
88  
89      /**
90       * {@inheritDoc}
91       */
92      @Override
93      public void format(final LogEvent event, final StringBuilder toAppendTo) {
94          StringBuilder buf = new StringBuilder();
95          for (PatternFormatter formatter : formatters) {
96              formatter.format(event, buf);
97          }
98  
99          if (buf.length() > 0) {
100             toAppendTo.append(LEVEL_STYLES.get(event.getLevel())).append(buf.toString()).
101                 append(AnsiEscape.getDefaultStyle());
102         }
103     }
104 }

17  package org.apache.logging.log4j.core.time;
18  
19  import org.apache.logging.log4j.core.util.Clock;
20  
21  /**
22   * Extension of the {@link Clock} interface that is able to provide more accurate time information than milliseconds
23   * since the epoch. {@code PreciseClock} implementations are free to return millisecond-precision time
24   * if that is the most accurate time information available on this platform.
25   * @since 2.11
26   */
27  public interface PreciseClock extends Clock {
28  
29      /**
30       * Initializes the specified instant with time information as accurate as available on this platform.
31       * @param mutableInstant the container to be initialized with the accurate time information
32       * @since 2.11
33       */
34      void init(final MutableInstant mutableInstant);
35  }

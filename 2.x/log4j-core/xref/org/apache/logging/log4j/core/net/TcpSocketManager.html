133     public TcpSocketManager(final String name, final OutputStream os, final Socket socket,
134             final InetAddress inetAddress, final String host, final int port, final int connectTimeoutMillis,
135             final int delay, final boolean immediateFail, final Layout<? extends Serializable> layout,
136             final int bufferSize, final SocketOptions socketOptions) {
137         super(name, os, inetAddress, host, port, layout, true, bufferSize);
138         this.connectTimeoutMillis = connectTimeoutMillis;
139         this.reconnectionDelay = delay;
140         this.socket = socket;
141         this.immediateFail = immediateFail;
142         retry = delay > 0;
143         if (socket == null) {
144             reconnector = createReconnector();
145             reconnector.start();
146         }
147         this.socketOptions = socketOptions;
148     }

174     /**
175      * Obtains a TcpSocketManager.
176      * 
177      * @param host
178      *            The host to connect to.
179      * @param port
180      *            The port on the host.
181      * @param connectTimeoutMillis
182      *            the connect timeout in milliseconds
183      * @param reconnectDelayMillis
184      *            The interval to pause between retries.
185      * @param bufferSize
186      *            The buffer size.
187      * @return A TcpSocketManager.
188      */
189     public static TcpSocketManager getSocketManager(final String host, int port, final int connectTimeoutMillis,
190             int reconnectDelayMillis, final boolean immediateFail, final Layout<? extends Serializable> layout,
191             final int bufferSize, final SocketOptions socketOptions) {
192         if (Strings.isEmpty(host)) {
193             throw new IllegalArgumentException("A host name is required");
194         }
195         if (port <= 0) {
196             port = DEFAULT_PORT;
197         }
198         if (reconnectDelayMillis == 0) {
199             reconnectDelayMillis = DEFAULT_RECONNECTION_DELAY_MILLIS;
200         }
201         return (TcpSocketManager) getManager("TCP:" + host + ':' + port, new FactoryData(host, port,
202                 connectTimeoutMillis, reconnectDelayMillis, immediateFail, layout, bufferSize, socketOptions), FACTORY);
203     }

205     @Override
206     protected void write(final byte[] bytes, final int offset, final int length, final boolean immediateFlush) {
207         if (socket == null) {
208             if (reconnector != null && !immediateFail) {
209                 reconnector.latch();
210             }
211             if (socket == null) {
212                 final String msg = "Error writing to " + getName() + " socket not available";
213                 throw new AppenderLoggingException(msg);
214             }
215         }
216         synchronized (this) {
217             try {
218                 final OutputStream outputStream = getOutputStream();
219                 outputStream.write(bytes, offset, length);
220                 if (immediateFlush) {
221                     outputStream.flush();
222                 }
223             } catch (final IOException ex) {
224                 if (retry && reconnector == null) {
225                     reconnector = createReconnector();
226                     reconnector.start();
227                 }
228                 final String msg = "Error writing to " + getName();
229                 throw new AppenderLoggingException(msg, ex);
230             }
231         }
232     }
233 
234     @Override
235     protected synchronized boolean closeOutputStream() {
236         final boolean closed = super.closeOutputStream();
237         if (reconnector != null) {
238             reconnector.shutdown();
239             reconnector.interrupt();
240             reconnector = null;
241         }
242         final Socket oldSocket = socket;
243         socket = null;
244         if (oldSocket != null) {
245             try {
246                 oldSocket.close();
247             } catch (final IOException e) {
248                 LOGGER.error("Could not close socket {}", socket);
249                 return false;
250             }
251         }
252         return closed;
253     }

276     /**
277      * Handles reconnecting to a Thread.
278      */
279     private class Reconnector extends Log4jThread {
280 
281         private final CountDownLatch latch = new CountDownLatch(1);
282 
283         private boolean shutdown = false;
284 
285         private final Object owner;
286 
287         public Reconnector(final OutputStreamManager owner) {
288             super("TcpSocketManager-Reconnector");
289             this.owner = owner;
290         }
291 
292         public void latch() {
293             try {
294                 latch.await();
295             } catch (final InterruptedException ex) {
296                 // Ignore the exception.
297             }
298         }
299 
300         public void shutdown() {
301             shutdown = true;
302         }
303 
304         @Override
305         public void run() {
306             while (!shutdown) {
307                 try {
308                     sleep(reconnectionDelay);
309                     final Socket sock = createSocket(inetAddress, port);
310                     final OutputStream newOS = sock.getOutputStream();
311                     synchronized (owner) {
312                         try {
313                             getOutputStream().close();
314                         } catch (final IOException ioe) {
315                             // Ignore this.
316                         }
317 
318                         setOutputStream(newOS);
319                         socket = sock;
320                         reconnector = null;
321                         shutdown = true;
322                     }
323                     LOGGER.debug("Connection to " + host + ':' + port + " reestablished.");
324                 } catch (final InterruptedException ie) {
325                     LOGGER.debug("Reconnection interrupted.");
326                 } catch (final ConnectException ex) {
327                     LOGGER.debug(host + ':' + port + " refused connection");
328                 } catch (final IOException ioe) {
329                     LOGGER.debug("Unable to reconnect to " + host + ':' + port);
330                 } finally {
331                     latch.countDown();
332                 }
333             }
334         }
335     }
336 
337     private Reconnector createReconnector() {
338         final Reconnector recon = new Reconnector(this);
339         recon.setDaemon(true);
340         recon.setPriority(Thread.MIN_PRIORITY);
341         return recon;
342     }
343 
344     protected Socket createSocket(final InetAddress host, final int port) throws IOException {
345         return createSocket(host.getHostName(), port);
346     }
347 
348     protected Socket createSocket(final String host, final int port) throws IOException {
349         final Socket newSocket = new Socket();
350         newSocket.connect(new InetSocketAddress(host, port), connectTimeoutMillis);
351         if (socketOptions != null) {
352             socketOptions.apply(newSocket);
353         }
354         return newSocket;
355     }

384     /**
385      * Factory to create a TcpSocketManager.
386      */
387     protected static class TcpSocketManagerFactory implements ManagerFactory<TcpSocketManager, FactoryData> {
388         
389         @SuppressWarnings("resource")
390         @Override
391         public TcpSocketManager createManager(final String name, final FactoryData data) {
392 
393             InetAddress inetAddress;
394             OutputStream os;
395             try {
396                 inetAddress = InetAddress.getByName(data.host);
397             } catch (final UnknownHostException ex) {
398                 LOGGER.error("Could not find address of " + data.host, ex, ex);
399                 return null;
400             }
401             Socket socket = null;
402             try {
403                 // LOG4J2-1042
404                 socket = createSocket(data);
405                 os = socket.getOutputStream();
406                 return new TcpSocketManager(name, os, socket, inetAddress, data.host, data.port,
407                         data.connectTimeoutMillis, data.reconnectDelayMillis, data.immediateFail, data.layout,
408                         data.bufferSize, data.socketOptions);
409             } catch (final IOException ex) {
410                 LOGGER.error("TcpSocketManager (" + name + ") " + ex, ex);
411                 os = NullOutputStream.getInstance();
412             }
413             if (data.reconnectDelayMillis == 0) {
414                 Closer.closeSilently(socket);
415                 return null;
416             }
417             return new TcpSocketManager(name, os, null, inetAddress, data.host, data.port, data.connectTimeoutMillis,
418                     data.reconnectDelayMillis, data.immediateFail, data.layout, data.bufferSize, data.socketOptions);
419         }
420 
421         static Socket createSocket(final FactoryData data) throws IOException, SocketException {
422             final Socket socket = new Socket();
423             socket.connect(new InetSocketAddress(data.host, data.port), data.connectTimeoutMillis);
424             final SocketOptions socketOptions = data.socketOptions;
425             if (socketOptions != null) {
426                 socketOptions.apply(socket);
427             }
428             return socket;
429         }

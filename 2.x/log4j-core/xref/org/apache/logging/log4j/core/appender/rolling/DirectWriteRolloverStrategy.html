40  /**
41   * When rolling over, <code>DirectWriteRolloverStrategy</code> writes directly to the file as resolved by the file
42   * pattern. Files will be renamed files according to an algorithm as described below.
43   *
44   * <p>
45   * The DirectWriteRolloverStrategy uses similar logic as DefaultRolloverStrategy to determine the file name based
46   * on the file pattern, however the DirectWriteRolloverStrategy writes directly to a file and does not rename it
47   * during rollover, except if it is compressed, in which case it will add the appropriate file extension.
48   * </p>
49   *
50   * @since 2.8
51   */
52  @Plugin(name = "DirectWriteRolloverStrategy", category = Core.CATEGORY_NAME, printObject = true)
53  public class DirectWriteRolloverStrategy extends AbstractRolloverStrategy implements DirectFileRolloverStrategy {
54  
55      private static final int DEFAULT_MAX_FILES = 7;
56  
57      /**
58       * Creates the DirectWriteRolloverStrategy.
59       *
60       * @param maxFiles The maximum number of files that match the date portion of the pattern to keep.
61       * @param compressionLevelStr The compression level, 0 (less) through 9 (more); applies only to ZIP files.
62       * @param customActions custom actions to perform asynchronously after rollover
63       * @param stopCustomActionsOnError whether to stop executing asynchronous actions if an error occurs
64       * @param config The Configuration.
65       * @return A DefaultRolloverStrategy.
66       */
67      @PluginFactory
68      public static DirectWriteRolloverStrategy createStrategy(
69              // @formatter:off
70              @PluginAttribute("maxFiles") final String maxFiles,
71              @PluginAttribute("compressionLevel") final String compressionLevelStr,
72              @PluginElement("Actions") final Action[] customActions,
73              @PluginAttribute(value = "stopCustomActionsOnError", defaultBoolean = true)
74                      final boolean stopCustomActionsOnError,
75              @PluginConfiguration final Configuration config) {
76              // @formatter:on
77          int maxIndex = Integer.MAX_VALUE;
78          if (maxFiles != null) {
79              maxIndex = Integer.parseInt(maxFiles);
80              if (maxIndex < 0) {
81                  maxIndex = Integer.MAX_VALUE;
82              } else if (maxIndex < 2) {
83                  LOGGER.error("Maximum files too small. Limited to " + DEFAULT_MAX_FILES);
84                  maxIndex = DEFAULT_MAX_FILES;
85              }
86          }
87          final int compressionLevel = Integers.parseInt(compressionLevelStr, Deflater.DEFAULT_COMPRESSION);
88          return new DirectWriteRolloverStrategy(maxIndex, compressionLevel, config.getStrSubstitutor(),
89                  customActions, stopCustomActionsOnError);
90      }
91  
92      /**
93       * Index for most recent log file.
94       */
95      private final int maxFiles;
96      private final int compressionLevel;
97      private final List<Action> customActions;
98      private final boolean stopCustomActionsOnError;
99      private volatile String currentFileName;
100     private int nextIndex = -1;
101 
102     /**
103      * Constructs a new instance.
104      *
105      * @param maxFiles The minimum index.
106      * @param customActions custom actions to perform asynchronously after rollover
107      * @param stopCustomActionsOnError whether to stop executing asynchronous actions if an error occurs
108      */
109     protected DirectWriteRolloverStrategy(final int maxFiles, final int compressionLevel,
110                                           final StrSubstitutor strSubstitutor, final Action[] customActions,
111                                           final boolean stopCustomActionsOnError) {
112         super(strSubstitutor);
113         this.maxFiles = maxFiles;
114         this.compressionLevel = compressionLevel;
115         this.stopCustomActionsOnError = stopCustomActionsOnError;
116         this.customActions = customActions == null ? Collections.<Action> emptyList() : Arrays.asList(customActions);
117     }
118 
119     public int getCompressionLevel() {
120         return this.compressionLevel;
121     }
122 
123     public List<Action> getCustomActions() {
124         return customActions;
125     }
126 
127     public int getMaxFiles() {
128         return this.maxFiles;
129     }
130 
131     public boolean isStopCustomActionsOnError() {
132         return stopCustomActionsOnError;
133     }
134 
135     private int purge(final RollingFileManager manager) {
136         SortedMap<Integer, Path> eligibleFiles = getEligibleFiles(manager);
137         LOGGER.debug("Found {} eligible files, max is  {}", eligibleFiles.size(), maxFiles);
138         while (eligibleFiles.size() >= maxFiles) {
139             try {
140                 Integer key = eligibleFiles.firstKey();
141                 Files.delete(eligibleFiles.get(key));
142                 eligibleFiles.remove(key);
143             } catch (IOException ioe) {
144                 LOGGER.error("Unable to delete {}", eligibleFiles.firstKey(), ioe);
145                 break;
146             }
147         }
148         return eligibleFiles.size() > 0 ? eligibleFiles.lastKey() : 1;
149     }
150 
151     public String getCurrentFileName(final RollingFileManager manager) {
152         if (currentFileName == null) {
153             SortedMap<Integer, Path> eligibleFiles = getEligibleFiles(manager);
154             final int fileIndex = eligibleFiles.size() > 0 ? (nextIndex > 0 ? nextIndex : eligibleFiles.size()) : 1;
155             final StringBuilder buf = new StringBuilder(255);
156             manager.getPatternProcessor().formatFileName(strSubstitutor, buf, true, fileIndex);
157             int suffixLength = suffixLength(buf.toString());
158             String name = suffixLength > 0 ? buf.substring(0, buf.length() - suffixLength) : buf.toString();
159             currentFileName = name;
160         }
161         return currentFileName;
162     }
163 
164     /**
165      * Performs the rollover.
166      *
167      * @param manager The RollingFileManager name for current active log file.
168      * @return A RolloverDescription.
169      * @throws SecurityException if an error occurs.
170      */
171     @Override
172     public RolloverDescription rollover(final RollingFileManager manager) throws SecurityException {
173         LOGGER.debug("Rolling " + currentFileName);
174         if (maxFiles < 0) {
175             return null;
176         }
177         final long startNanos = System.nanoTime();
178         final int fileIndex = purge(manager);
179         if (LOGGER.isTraceEnabled()) {
180             final double durationMillis = TimeUnit.NANOSECONDS.toMillis(System.nanoTime() - startNanos);
181             LOGGER.trace("DirectWriteRolloverStrategy.purge() took {} milliseconds", durationMillis);
182         }
183         Action compressAction = null;
184         final String sourceName = currentFileName;
185         currentFileName = null;
186         nextIndex = fileIndex + 1;
187         FileExtension fileExtension = manager.getFileExtension();
188         if (fileExtension != null) {
189             compressAction = fileExtension.createCompressAction(sourceName, sourceName + fileExtension.getExtension(),
190                     true, compressionLevel);
191         }
192 
193         final Action asyncAction = merge(compressAction, customActions, stopCustomActionsOnError);
194         return new RolloverDescriptionImpl(sourceName, false, null, asyncAction);
195     }
196 
197     @Override
198     public String toString() {
199         return "DirectWriteRolloverStrategy(maxFiles=" + maxFiles + ')';
200     }

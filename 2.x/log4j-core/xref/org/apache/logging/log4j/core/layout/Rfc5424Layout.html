134     private Rfc5424Layout(final Configuration config, final Facility facility, final String id, final String ein,
135               final boolean includeMDC, final boolean includeNL, final String escapeNL, final String mdcId,
136               final String mdcPrefix, final String eventPrefix, final String appName, final String messageId,
137               final String excludes, final String includes, final String required, final Charset charset,
138               final String exceptionPattern, final boolean useTLSMessageFormat, final LoggerFields[] loggerFields) {
139         super(charset);
140         final PatternParser exceptionParser = createPatternParser(config, ThrowablePatternConverter.class);
141         exceptionFormatters = exceptionPattern == null ? null : exceptionParser.parse(exceptionPattern);
142         this.facility = facility;
143         this.defaultId = id == null ? DEFAULT_ID : id;
144         this.enterpriseNumber = ein;
145         this.includeMdc = includeMDC;
146         this.includeNewLine = includeNL;
147         this.escapeNewLine = escapeNL == null ? null : Matcher.quoteReplacement(escapeNL);
148         this.mdcId = mdcId != null ? mdcId : id == null ? DEFAULT_MDCID : id;
149         this.mdcSdId = new StructuredDataId(this.mdcId, enterpriseNumber, null, null);
150         this.mdcPrefix = mdcPrefix;
151         this.eventPrefix = eventPrefix;
152         this.appName = appName;
153         this.messageId = messageId;
154         this.useTlsMessageFormat = useTLSMessageFormat;
155         this.localHostName = NetUtils.getLocalHostname();
156         ListChecker checker = null;
157         if (excludes != null) {
158             final String[] array = excludes.split(Patterns.COMMA_SEPARATOR);
159             if (array.length > 0) {
160                 mdcExcludes = new ArrayList<>(array.length);
161                 for (final String str : array) {
162                     mdcExcludes.add(str.trim());
163                 }
164                 checker = new ExcludeChecker(mdcExcludes);
165             } else {
166                 mdcExcludes = null;
167             }
168         } else {
169             mdcExcludes = null;
170         }
171         if (includes != null) {
172             final String[] array = includes.split(Patterns.COMMA_SEPARATOR);
173             if (array.length > 0) {
174                 mdcIncludes = new ArrayList<>(array.length);
175                 for (final String str : array) {
176                     mdcIncludes.add(str.trim());
177                 }
178                 checker = new IncludeChecker(mdcIncludes);
179             } else {
180                 mdcIncludes = null;
181             }
182         } else {
183             mdcIncludes = null;
184         }
185         if (required != null) {
186             final String[] array = required.split(Patterns.COMMA_SEPARATOR);
187             if (array.length > 0) {
188                 mdcRequired = new ArrayList<>(array.length);
189                 for (final String str : array) {
190                     mdcRequired.add(str.trim());
191                 }
192             } else {
193                 mdcRequired = null;
194             }
195 
196         } else {
197             mdcRequired = null;
198         }
199         this.listChecker = checker != null ? checker : ListChecker.NOOP_CHECKER;
200         final String name = config == null ? null : config.getName();
201         configName = Strings.isNotEmpty(name) ? name : null;
202         this.fieldFormatters = createFieldFormatters(loggerFields, config);
203         this.procId = ProcessIdUtil.getProcessId();
204     }

206     private Map<String, FieldFormatter> createFieldFormatters(final LoggerFields[] loggerFields,
207             final Configuration config) {
208         final Map<String, FieldFormatter> sdIdMap = new HashMap<>(loggerFields == null ? 0 : loggerFields.length);
209         if (loggerFields != null) {
210             for (final LoggerFields loggerField : loggerFields) {
211                 final StructuredDataId key = loggerField.getSdId() == null ? mdcSdId : loggerField.getSdId();
212                 final Map<String, List<PatternFormatter>> sdParams = new HashMap<>();
213                 final Map<String, String> fields = loggerField.getMap();
214                 if (!fields.isEmpty()) {
215                     final PatternParser fieldParser = createPatternParser(config, null);
216 
217                     for (final Map.Entry<String, String> entry : fields.entrySet()) {
218                         final List<PatternFormatter> formatters = fieldParser.parse(entry.getValue());
219                         sdParams.put(entry.getKey(), formatters);
220                     }
221                     final FieldFormatter fieldFormatter = new FieldFormatter(sdParams,
222                             loggerField.getDiscardIfAllFieldsAreEmpty());
223                     sdIdMap.put(key.toString(), fieldFormatter);
224                 }
225             }
226         }
227         return sdIdMap.size() > 0 ? sdIdMap : null;
228     }

230     /**
231      * Create a PatternParser.
232      *
233      * @param config The Configuration.
234      * @param filterClass Filter the returned plugins after calling the plugin manager.
235      * @return The PatternParser.
236      */
237     private static PatternParser createPatternParser(final Configuration config,
238             final Class<? extends PatternConverter> filterClass) {
239         if (config == null) {
240             return new PatternParser(config, PatternLayout.KEY, LogEventPatternConverter.class, filterClass);
241         }
242         PatternParser parser = config.getComponent(COMPONENT_KEY);
243         if (parser == null) {
244             parser = new PatternParser(config, PatternLayout.KEY, ThrowablePatternConverter.class);
245             config.addComponent(COMPONENT_KEY, parser);
246             parser = config.getComponent(COMPONENT_KEY);
247         }
248         return parser;
249     }

268     /**
269      * Formats a {@link org.apache.logging.log4j.core.LogEvent} in conformance with the RFC 5424 Syslog specification.
270      *
271      * @param event The LogEvent.
272      * @return The RFC 5424 String representation of the LogEvent.
273      */
274     @Override
275     public String toSerializable(final LogEvent event) {
276         final StringBuilder buf = getStringBuilder();
277         appendPriority(buf, event.getLevel());
278         appendTimestamp(buf, event.getTimeMillis());
279         appendSpace(buf);
280         appendHostName(buf);
281         appendSpace(buf);
282         appendAppName(buf);
283         appendSpace(buf);
284         appendProcessId(buf);
285         appendSpace(buf);
286         appendMessageId(buf, event.getMessage());
287         appendSpace(buf);
288         appendStructuredElements(buf, event);
289         appendMessage(buf, event);
290         if (useTlsMessageFormat) {
291             return new TlsSyslogFrame(buf.toString()).toString();
292         }
293         return buf.toString();
294     }

328     private void appendMessageId(final StringBuilder buffer, final Message message) {
329         final boolean isStructured = message instanceof StructuredDataMessage;
330         final String type = isStructured ? ((StructuredDataMessage) message).getType() : null;
331         if (type != null) {
332             buffer.append(type);
333         } else if (messageId != null) {
334             buffer.append(messageId);
335         } else {
336             buffer.append('-');
337         }

340     private void appendMessage(final StringBuilder buffer, final LogEvent event) {
341         final Message message = event.getMessage();
342         // This layout formats StructuredDataMessages instead of delegating to the Message itself.
343         final String text = (message instanceof StructuredDataMessage || message instanceof MessageCollectionMessage)
344                 ? message.getFormat() : message.getFormattedMessage();
345 
346         if (text != null && text.length() > 0) {
347             buffer.append(' ').append(escapeNewlines(text, escapeNewLine));
348         }
349 
350         if (exceptionFormatters != null && event.getThrown() != null) {
351             final StringBuilder exception = new StringBuilder(LF);
352             for (final PatternFormatter formatter : exceptionFormatters) {
353                 formatter.format(event, exception);
354             }
355             buffer.append(escapeNewlines(exception.toString(), escapeNewLine));
356         }
357         if (includeNewLine) {
358             buffer.append(LF);
359         }
360     }

362     private void appendStructuredElements(final StringBuilder buffer, final LogEvent event) {
363         final Message message = event.getMessage();
364         final boolean isStructured = message instanceof StructuredDataMessage ||
365                 message instanceof StructuredDataCollectionMessage;
366 
367         if (!isStructured && (fieldFormatters != null && fieldFormatters.isEmpty()) && !includeMdc) {
368             buffer.append('-');
369             return;
370         }
371 
372         final Map<String, StructuredDataElement> sdElements = new HashMap<>();
373         final Map<String, String> contextMap = event.getContextData().toMap();
374 
375         if (mdcRequired != null) {
376             checkRequired(contextMap);
377         }
378 
379         if (fieldFormatters != null) {
380             for (final Map.Entry<String, FieldFormatter> sdElement : fieldFormatters.entrySet()) {
381                 final String sdId = sdElement.getKey();
382                 final StructuredDataElement elem = sdElement.getValue().format(event);
383                 sdElements.put(sdId, elem);
384             }
385         }
386 
387         if (includeMdc && contextMap.size() > 0) {
388             final String mdcSdIdStr = mdcSdId.toString();
389             final StructuredDataElement union = sdElements.get(mdcSdIdStr);
390             if (union != null) {
391                 union.union(contextMap);
392                 sdElements.put(mdcSdIdStr, union);
393             } else {
394                 final StructuredDataElement formattedContextMap = new StructuredDataElement(contextMap, mdcPrefix, false);
395                 sdElements.put(mdcSdIdStr, formattedContextMap);
396             }
397         }
398 
399         if (isStructured) {
400             if (message instanceof MessageCollectionMessage) {
401                 for (final StructuredDataMessage data : ((StructuredDataCollectionMessage)message)) {
402                     addStructuredData(sdElements, data);
403                 }
404             } else {
405                 addStructuredData(sdElements, (StructuredDataMessage) message);
406             }
407         }
408 
409         if (sdElements.isEmpty()) {
410             buffer.append('-');
411             return;
412         }
413 
414         for (final Map.Entry<String, StructuredDataElement> entry : sdElements.entrySet()) {
415             formatStructuredElement(entry.getKey(), entry.getValue(), buffer, listChecker);
416         }
417     }
418 
419     private void addStructuredData(final Map<String, StructuredDataElement> sdElements, final StructuredDataMessage data) {
420         final Map<String, String> map = data.getData();
421         final StructuredDataId id = data.getId();
422         final String sdId = getId(id);
423 
424         if (sdElements.containsKey(sdId)) {
425             final StructuredDataElement union = sdElements.get(id.toString());
426             union.union(map);
427             sdElements.put(sdId, union);
428         } else {
429             final StructuredDataElement formattedData = new StructuredDataElement(map, eventPrefix, false);
430             sdElements.put(sdId, formattedData);
431         }
432     }

434     private String escapeNewlines(final String text, final String replacement) {
435         if (null == replacement) {
436             return text;
437         }
438         return NEWLINE_PATTERN.matcher(text).replaceAll(replacement);
439     }

453     private String computeTimeStampString(final long now) {
454         long last;
455         synchronized (this) {
456             last = lastTimestamp;
457             if (now == lastTimestamp) {
458                 return timestamppStr;
459             }
460         }
461 
462         final StringBuilder buffer = new StringBuilder();
463         final Calendar cal = new GregorianCalendar();
464         cal.setTimeInMillis(now);
465         buffer.append(Integer.toString(cal.get(Calendar.YEAR)));
466         buffer.append('-');
467         pad(cal.get(Calendar.MONTH) + 1, TWO_DIGITS, buffer);
468         buffer.append('-');
469         pad(cal.get(Calendar.DAY_OF_MONTH), TWO_DIGITS, buffer);
470         buffer.append('T');
471         pad(cal.get(Calendar.HOUR_OF_DAY), TWO_DIGITS, buffer);
472         buffer.append(':');
473         pad(cal.get(Calendar.MINUTE), TWO_DIGITS, buffer);
474         buffer.append(':');
475         pad(cal.get(Calendar.SECOND), TWO_DIGITS, buffer);
476         buffer.append('.');
477         pad(cal.get(Calendar.MILLISECOND), THREE_DIGITS, buffer);
478 
479         int tzmin = (cal.get(Calendar.ZONE_OFFSET) + cal.get(Calendar.DST_OFFSET)) / MILLIS_PER_MINUTE;
480         if (tzmin == 0) {
481             buffer.append('Z');
482         } else {
483             if (tzmin < 0) {
484                 tzmin = -tzmin;
485                 buffer.append('-');
486             } else {
487                 buffer.append('+');
488             }
489             final int tzhour = tzmin / MINUTES_PER_HOUR;
490             tzmin -= tzhour * MINUTES_PER_HOUR;
491             pad(tzhour, TWO_DIGITS, buffer);
492             buffer.append(':');
493             pad(tzmin, TWO_DIGITS, buffer);
494         }
495         synchronized (this) {
496             if (last == lastTimestamp) {
497                 lastTimestamp = now;
498                 timestamppStr = buffer.toString();
499             }
500         }
501         return buffer.toString();
502     }

504     private void pad(final int val, int max, final StringBuilder buf) {
505         while (max > 1) {
506             if (val < max) {
507                 buf.append('0');
508             }
509             max = max / TWO_DIGITS;
510         }
511         buf.append(Integer.toString(val));
512     }

514     private void formatStructuredElement(final String id, final StructuredDataElement data,
515             final StringBuilder sb, final ListChecker checker) {
516         if ((id == null && defaultId == null) || data.discard()) {
517             return;
518         }
519 
520         sb.append('[');
521         sb.append(id);
522         if (!mdcSdId.toString().equals(id)) {
523             appendMap(data.getPrefix(), data.getFields(), sb, ListChecker.NOOP_CHECKER);
524         } else {
525             appendMap(data.getPrefix(), data.getFields(), sb, checker);
526         }
527         sb.append(']');
528     }

530     private String getId(final StructuredDataId id) {
531         final StringBuilder sb = new StringBuilder();
532         if (id == null || id.getName() == null) {
533             sb.append(defaultId);
534         } else {
535             sb.append(id.getName());
536         }
537         String ein = id != null ? id.getEnterpriseNumber() : enterpriseNumber;
538         if (StructuredDataId.RESERVED.equals(ein)) {
539             ein = enterpriseNumber;
540         }
541         if (!StructuredDataId.RESERVED.equals(ein)) {
542             sb.append('@').append(ein);
543         }
544         return sb.toString();
545     }

547     private void checkRequired(final Map<String, String> map) {
548         for (final String key : mdcRequired) {
549             final String value = map.get(key);
550             if (value == null) {
551                 throw new LoggingException("Required key " + key + " is missing from the " + mdcId);
552             }
553         }
554     }

556     private void appendMap(final String prefix, final Map<String, String> map, final StringBuilder sb,
557             final ListChecker checker) {
558         final SortedMap<String, String> sorted = new TreeMap<>(map);
559         for (final Map.Entry<String, String> entry : sorted.entrySet()) {
560             if (checker.check(entry.getKey()) && entry.getValue() != null) {
561                 sb.append(' ');
562                 if (prefix != null) {
563                     sb.append(prefix);
564                 }
565                 final String safeKey = escapeNewlines(escapeSDParams(entry.getKey()), escapeNewLine);
566                 final String safeValue = escapeNewlines(escapeSDParams(entry.getValue()), escapeNewLine);
567                 StringBuilders.appendKeyDqValue(sb, safeKey, safeValue);
568             }
569         }
570     }

572     private String escapeSDParams(final String value) {
573         return PARAM_VALUE_ESCAPE_PATTERN.matcher(value).replaceAll("\\\\$0");
574     }

589     /**
590      * Create the RFC 5424 Layout.
591      *
592      * @param facility The Facility is used to try to classify the message.
593      * @param id The default structured data id to use when formatting according to RFC 5424.
594      * @param enterpriseNumber The IANA enterprise number.
595      * @param includeMDC Indicates whether data from the ThreadContextMap will be included in the RFC 5424 Syslog
596      *            record. Defaults to "true:.
597      * @param mdcId The id to use for the MDC Structured Data Element.
598      * @param mdcPrefix The prefix to add to MDC key names.
599      * @param eventPrefix The prefix to add to event key names.
600      * @param newLine If true, a newline will be appended to the end of the syslog record. The default is false.
601      * @param escapeNL String that should be used to replace newlines within the message text.
602      * @param appName The value to use as the APP-NAME in the RFC 5424 syslog record.
603      * @param msgId The default value to be used in the MSGID field of RFC 5424 syslog records.
604      * @param excludes A comma separated list of MDC keys that should be excluded from the LogEvent.
605      * @param includes A comma separated list of MDC keys that should be included in the FlumeEvent.
606      * @param required A comma separated list of MDC keys that must be present in the MDC.
607      * @param exceptionPattern The pattern for formatting exceptions.
608      * @param useTlsMessageFormat If true the message will be formatted according to RFC 5425.
609      * @param loggerFields Container for the KeyValuePairs containing the patterns
610      * @param config The Configuration. Some Converters require access to the Interpolator.
611      * @return An Rfc5424Layout.
612      * @deprecated Use {@link Rfc5424LayoutBuilder instead}
613      */
614     @PluginFactory
615     public static Rfc5424Layout createLayout(
616             // @formatter:off
617             @PluginAttribute(value = "facility", defaultString = "LOCAL0") final Facility facility,
618             @PluginAttribute("id") final String id,
619             @PluginAttribute(value = "enterpriseNumber", defaultInt = -1)
620             final int enterpriseNumber,
621             @PluginAttribute(value = "includeMDC", defaultBoolean = true) final boolean includeMDC,
622             @PluginAttribute(value = "mdcId", defaultString = DEFAULT_MDCID) final String mdcId,

779     private class FieldFormatter {
780 
781         private final Map<String, List<PatternFormatter>> delegateMap;
782         private final boolean discardIfEmpty;
783 
784         public FieldFormatter(final Map<String, List<PatternFormatter>> fieldMap, final boolean discardIfEmpty) {
785             this.discardIfEmpty = discardIfEmpty;
786             this.delegateMap = fieldMap;
787         }
788 
789         public StructuredDataElement format(final LogEvent event) {
790             final Map<String, String> map = new HashMap<>(delegateMap.size());
791 
792             for (final Map.Entry<String, List<PatternFormatter>> entry : delegateMap.entrySet()) {
793                 final StringBuilder buffer = new StringBuilder();
794                 for (final PatternFormatter formatter : entry.getValue()) {
795                     formatter.format(event, buffer);
796                 }
797                 map.put(entry.getKey(), buffer.toString());
798             }
799             return new StructuredDataElement(map, eventPrefix, discardIfEmpty);
800         }
801     }

803     private class StructuredDataElement {
804 
805         private final Map<String, String> fields;
806         private final boolean discardIfEmpty;
807         private final String prefix;
808 
809         public StructuredDataElement(final Map<String, String> fields, final String prefix,
810                                      final boolean discardIfEmpty) {
811             this.discardIfEmpty = discardIfEmpty;
812             this.fields = fields;
813             this.prefix = prefix;
814         }
815 
816         boolean discard() {
817             if (discardIfEmpty == false) {
818                 return false;
819             }
820             boolean foundNotEmptyValue = false;
821             for (final Map.Entry<String, String> entry : fields.entrySet()) {
822                 if (Strings.isNotEmpty(entry.getValue())) {
823                     foundNotEmptyValue = true;
824                     break;
825                 }
826             }
827             return !foundNotEmptyValue;
828         }
829 
830         void union(final Map<String, String> addFields) {
831             this.fields.putAll(addFields);
832         }
833 
834         Map<String, String> getFields() {
835             return this.fields;
836         }
837 
838         String getPrefix() {
839             return prefix;
840         }
841     }

18  package org.apache.logging.log4j.core.util;
19  
20  import java.lang.reflect.Array;
21  
22  /**
23   * Copied from Apache Commons Lang (including the {@code @since} tags.)
24   */
25  public class ArrayUtils {
26  
27      /**
28       * Checks if an array of Objects is empty or {@code null}.
29       *
30       * @param array  the array to test
31       * @return {@code true} if the array is empty or {@code null}
32       * @since 2.1
33       */
34      public static boolean isEmpty(final byte[] array) {
35          return getLength(array) == 0;
36      }
37      
38      /**
39       * <p>Returns the length of the specified array.
40       * This method can deal with {@code Object} arrays and with primitive arrays.</p>
41       *
42       * <p>If the input array is {@code null}, {@code 0} is returned.</p>
43       *
44       * <pre>
45       * ArrayUtils.getLength(null)            = 0
46       * ArrayUtils.getLength([])              = 0
47       * ArrayUtils.getLength([null])          = 1
48       * ArrayUtils.getLength([true, false])   = 2
49       * ArrayUtils.getLength([1, 2, 3])       = 3
50       * ArrayUtils.getLength(["a", "b", "c"]) = 3
51       * </pre>
52       *
53       * @param array  the array to retrieve the length from, may be null
54       * @return The length of the array, or {@code 0} if the array is {@code null}
55       * @throws IllegalArgumentException if the object argument is not an array.
56       * @since 2.1
57       */
58      public static int getLength(final Object array) {
59          if (array == null) {
60              return 0;
61          }
62          return Array.getLength(array);
63      }
64  
65      /**
66       * <p>Removes the element at the specified position from the specified array.
67       * All subsequent elements are shifted to the left (subtracts one from
68       * their indices).</p>
69       *
70       * <p>This method returns a new array with the same elements of the input
71       * array except the element on the specified position. The component
72       * type of the returned array is always the same as that of the input
73       * array.</p>
74       *
75       * <p>If the input array is {@code null}, an IndexOutOfBoundsException
76       * will be thrown, because in that case no valid index can be specified.</p>
77       *
78       * @param array  the array to remove the element from, may not be {@code null}
79       * @param index  the position of the element to be removed
80       * @return A new array containing the existing elements except the element
81       *         at the specified position.
82       * @throws IndexOutOfBoundsException if the index is out of range
83       * (index &lt; 0 || index &gt;= array.length), or if the array is {@code null}.
84       * @since 2.1
85       */
86      private static Object remove(final Object array, final int index) {
87          final int length = getLength(array);
88          if (index < 0 || index >= length) {
89              throw new IndexOutOfBoundsException("Index: " + index + ", Length: " + length);
90          }
91  
92          final Object result = Array.newInstance(array.getClass().getComponentType(), length - 1);
93          System.arraycopy(array, 0, result, 0, index);
94          if (index < length - 1) {
95              System.arraycopy(array, index + 1, result, index, length - index - 1);
96          }
97  
98          return result;
99      }
100 
101     /**
102      * <p>Removes the element at the specified position from the specified array.
103      * All subsequent elements are shifted to the left (subtracts one from
104      * their indices).</p>
105      *
106      * <p>This method returns a new array with the same elements of the input
107      * array except the element on the specified position. The component
108      * type of the returned array is always the same as that of the input
109      * array.</p>
110      *
111      * <p>If the input array is {@code null}, an IndexOutOfBoundsException
112      * will be thrown, because in that case no valid index can be specified.</p>
113      *
114      * <pre>
115      * ArrayUtils.remove(["a"], 0)           = []
116      * ArrayUtils.remove(["a", "b"], 0)      = ["b"]
117      * ArrayUtils.remove(["a", "b"], 1)      = ["a"]
118      * ArrayUtils.remove(["a", "b", "c"], 1) = ["a", "c"]
119      * </pre>
120      *
121      * @param <T> the component type of the array
122      * @param array  the array to remove the element from, may not be {@code null}
123      * @param index  the position of the element to be removed
124      * @return A new array containing the existing elements except the element
125      *         at the specified position.
126      * @throws IndexOutOfBoundsException if the index is out of range
127      * (index &lt; 0 || index &gt;= array.length), or if the array is {@code null}.
128      * @since 2.1
129      */
130     @SuppressWarnings("unchecked") // remove() always creates an array of the same type as its input
131     public static <T> T[] remove(final T[] array, final int index) {
132         return (T[]) remove((Object) array, index);
133     }

91      /**
92       * Returns the FileManager.
93       * @param fileName The name of the file to manage.
94       * @param append true if the file should be appended to, false if it should be overwritten.
95       * @param locking true if the file should be locked while writing, false otherwise.
96       * @param bufferedIo true if the contents should be buffered as they are written.
97       * @param createOnDemand true if you want to lazy-create the file (a.k.a. on-demand.)
98       * @param advertiseUri the URI to use when advertising the file
99       * @param layout The layout
100      * @param bufferSize buffer size for buffered IO
101      * @param configuration The configuration.
102      * @return A FileManager for the File.
103      */
104     public static FileManager getFileManager(final String fileName, final boolean append, boolean locking,
105             final boolean bufferedIo, final boolean createOnDemand, final String advertiseUri,
106             final Layout<? extends Serializable> layout, final int bufferSize, final Configuration configuration) {
107 
108         if (locking && bufferedIo) {
109             locking = false;
110         }
111         return (FileManager) getManager(fileName, new FactoryData(append, locking, bufferedIo, bufferSize,
112                 createOnDemand, advertiseUri, layout, configuration), FACTORY);
113     }

115     @Override
116     protected OutputStream createOutputStream() throws FileNotFoundException {
117         String filename = getFileName();
118         LOGGER.debug("Now writing to {} at {}", filename, new Date());
119         return new FileOutputStream(filename, isAppend);
120     }

122     @Override
123     protected synchronized void write(final byte[] bytes, final int offset, final int length,
124             final boolean immediateFlush) {
125         if (isLocking) {
126             try {
127                 @SuppressWarnings("resource")
128                 final FileChannel channel = ((FileOutputStream) getOutputStream()).getChannel();
129                 /*
130                  * Lock the whole file. This could be optimized to only lock from the current file position. Note that
131                  * locking may be advisory on some systems and mandatory on others, so locking just from the current
132                  * position would allow reading on systems where locking is mandatory. Also, Java 6 will throw an
133                  * exception if the region of the file is already locked by another FileChannel in the same JVM.
134                  * Hopefully, that will be avoided since every file should have a single file manager - unless two
135                  * different files strings are configured that somehow map to the same file.
136                  */
137                 try (final FileLock lock = channel.lock(0, Long.MAX_VALUE, false)) {
138                     super.write(bytes, offset, length, immediateFlush);
139                 }
140             } catch (final IOException ex) {
141                 throw new AppenderLoggingException("Unable to obtain lock on " + getName(), ex);
142             }
143         } else {
144             super.write(bytes, offset, length, immediateFlush);
145         }
146     }

148     /**
149      * Overrides {@link OutputStreamManager#writeToDestination(byte[], int, int)} to add support for file locking.
150      *
151      * @param bytes the array containing data
152      * @param offset from where to write
153      * @param length how many bytes to write
154      * @since 2.8
155      */
156     protected synchronized void writeToDestination(final byte[] bytes, final int offset, final int length) {
157         if (isLocking) {
158             try {
159                 @SuppressWarnings("resource")
160                 final FileChannel channel = ((FileOutputStream) getOutputStream()).getChannel();
161                 /*
162                  * Lock the whole file. This could be optimized to only lock from the current file position. Note that
163                  * locking may be advisory on some systems and mandatory on others, so locking just from the current
164                  * position would allow reading on systems where locking is mandatory. Also, Java 6 will throw an
165                  * exception if the region of the file is already locked by another FileChannel in the same JVM.
166                  * Hopefully, that will be avoided since every file should have a single file manager - unless two
167                  * different files strings are configured that somehow map to the same file.
168                  */
169                 try (final FileLock lock = channel.lock(0, Long.MAX_VALUE, false)) {
170                     super.writeToDestination(bytes, offset, length);
171                 }
172             } catch (final IOException ex) {
173                 throw new AppenderLoggingException("Unable to obtain lock on " + getName(), ex);
174             }
175         } else {
176             super.writeToDestination(bytes, offset, length);
177         }
178     }

233     /**
234      * Factory Data.
235      */
236     private static class FactoryData extends ConfigurationFactoryData {
237         private final boolean append;
238         private final boolean locking;
239         private final boolean bufferedIo;
240         private final int bufferSize;
241         private final boolean createOnDemand;
242         private final String advertiseURI;
243         private final Layout<? extends Serializable> layout;
244 
245         /**
246          * Constructor.
247          * @param append Append status.
248          * @param locking Locking status.
249          * @param bufferedIo Buffering flag.
250          * @param bufferSize Buffer size.
251          * @param createOnDemand if you want to lazy-create the file (a.k.a. on-demand.)
252          * @param advertiseURI the URI to use when advertising the file
253          * @param layout The layout
254          * @param configuration the configuration
255          */
256         public FactoryData(final boolean append, final boolean locking, final boolean bufferedIo, final int bufferSize,
257                 final boolean createOnDemand, final String advertiseURI, final Layout<? extends Serializable> layout,
258                 final Configuration configuration) {
259             super(configuration);
260             this.append = append;
261             this.locking = locking;
262             this.bufferedIo = bufferedIo;
263             this.bufferSize = bufferSize;
264             this.createOnDemand = createOnDemand;
265             this.advertiseURI = advertiseURI;
266             this.layout = layout;
267         }
268     }

270     /**
271      * Factory to create a FileManager.
272      */
273     private static class FileManagerFactory implements ManagerFactory<FileManager, FactoryData> {
274 
275         /**
276          * Creates a FileManager.
277          * @param name The name of the File.
278          * @param data The FactoryData
279          * @return The FileManager for the File.
280          */
281         @Override
282         public FileManager createManager(final String name, final FactoryData data) {
283             final File file = new File(name);
284             final File parent = file.getParentFile();
285             if (null != parent && !parent.exists()) {
286                 parent.mkdirs();
287             }
288 
289             final boolean writeHeader = !data.append || !file.exists();
290             try {
291                 final int actualSize = data.bufferedIo ? data.bufferSize : Constants.ENCODER_BYTE_BUFFER_SIZE;
292                 final ByteBuffer byteBuffer = ByteBuffer.wrap(new byte[actualSize]);
293                 final FileOutputStream fos = data.createOnDemand ? null : new FileOutputStream(file, data.append);
294                 return new FileManager(data.getLoggerContext(), name, fos, data.append, data.locking,
295                         data.createOnDemand, data.advertiseURI, data.layout, writeHeader, byteBuffer);
296             } catch (final IOException ex) {
297                 LOGGER.error("FileManager (" + name + ") " + ex, ex);
298             }
299             return null;
300         }
301     }

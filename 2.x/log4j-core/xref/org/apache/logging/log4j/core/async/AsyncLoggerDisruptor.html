40  /**
41   * Helper class for async loggers: AsyncLoggerDisruptor handles the mechanics of working with the LMAX Disruptor, and
42   * works with its associated AsyncLoggerContext to synchronize the life cycle of the Disruptor and its thread with the
43   * life cycle of the context. The AsyncLoggerDisruptor of the context is shared by all AsyncLogger objects created by
44   * that AsyncLoggerContext.
45   */
46  class AsyncLoggerDisruptor extends AbstractLifeCycle {
47      private static final int SLEEP_MILLIS_BETWEEN_DRAIN_ATTEMPTS = 50;
48      private static final int MAX_DRAIN_ATTEMPTS_BEFORE_SHUTDOWN = 200;
49  
50      private final Object queueFullEnqueueLock = new Object();
51  
52      private volatile Disruptor<RingBufferLogEvent> disruptor;
53      private String contextName;
54  
55      private boolean useThreadLocalTranslator = true;
56      private long backgroundThreadId;
57      private AsyncQueueFullPolicy asyncQueueFullPolicy;
58      private int ringBufferSize;
59  
60      AsyncLoggerDisruptor(final String contextName) {
61          this.contextName = contextName;
62      }
63  
64      public String getContextName() {
65          return contextName;
66      }
67  
68      public void setContextName(final String name) {
69          contextName = name;
70      }

72      Disruptor<RingBufferLogEvent> getDisruptor() {
73          return disruptor;

76      /**
77       * Creates and starts a new Disruptor and associated thread if none currently exists.
78       *
79       * @see #stop()
80       */
81      @Override
82      public synchronized void start() {
83          if (disruptor != null) {
84              LOGGER.trace(
85                      "[{}] AsyncLoggerDisruptor not starting new disruptor for this context, using existing object.",
86                      contextName);
87              return;
88          }
89          if (isStarting()) {
90              LOGGER.trace("[{}] AsyncLoggerDisruptor is already starting.", contextName);
91              return;
92          }
93          setStarting();
94          LOGGER.trace("[{}] AsyncLoggerDisruptor creating new disruptor for this context.", contextName);
95          ringBufferSize = DisruptorUtil.calculateRingBufferSize("AsyncLogger.RingBufferSize");
96          final WaitStrategy waitStrategy = DisruptorUtil.createWaitStrategy("AsyncLogger.WaitStrategy");
97  
98          final ThreadFactory threadFactory = new Log4jThreadFactory("AsyncLogger[" + contextName + "]", true, Thread.NORM_PRIORITY) {
99              @Override
100             public Thread newThread(final Runnable r) {
101                 final Thread result = super.newThread(r);
102                 backgroundThreadId = result.getId();
103                 return result;
104             }
105         };
106         asyncQueueFullPolicy = AsyncQueueFullPolicyFactory.create();
107 
108         disruptor = new Disruptor<>(RingBufferLogEvent.FACTORY, ringBufferSize, threadFactory, ProducerType.MULTI,
109                 waitStrategy);
110 
111         final ExceptionHandler<RingBufferLogEvent> errorHandler = DisruptorUtil.getAsyncLoggerExceptionHandler();
112         disruptor.setDefaultExceptionHandler(errorHandler);
113 
114         final RingBufferLogEventHandler[] handlers = {new RingBufferLogEventHandler()};
115         disruptor.handleEventsWith(handlers);
116 
117         LOGGER.debug("[{}] Starting AsyncLogger disruptor for this context with ringbufferSize={}, waitStrategy={}, "
118                 + "exceptionHandler={}...", contextName, disruptor.getRingBuffer().getBufferSize(), waitStrategy
119                 .getClass().getSimpleName(), errorHandler);
120         disruptor.start();
121 
122         LOGGER.trace("[{}] AsyncLoggers use a {} translator", contextName, useThreadLocalTranslator ? "threadlocal"
123                 : "vararg");
124         super.start();
125     }

127     /**
128      * Decreases the reference count. If the reference count reached zero, the Disruptor and its associated thread are
129      * shut down and their references set to {@code null}.
130      */
131     @Override
132     public boolean stop(final long timeout, final TimeUnit timeUnit) {
133         final Disruptor<RingBufferLogEvent> temp = getDisruptor();
134         if (temp == null) {
135             LOGGER.trace("[{}] AsyncLoggerDisruptor: disruptor for this context already shut down.", contextName);
136             return true; // disruptor was already shut down by another thread
137         }
138         setStopping();
139         LOGGER.debug("[{}] AsyncLoggerDisruptor: shutting down disruptor for this context.", contextName);
140 
141         // We must guarantee that publishing to the RingBuffer has stopped before we call disruptor.shutdown().
142         disruptor = null; // client code fails with NPE if log after stop. This is by design.
143 
144         // Calling Disruptor.shutdown() will wait until all enqueued events are fully processed,
145         // but this waiting happens in a busy-spin. To avoid (postpone) wasting CPU,
146         // we sleep in short chunks, up to 10 seconds, waiting for the ringbuffer to drain.
147         for (int i = 0; hasBacklog(temp) && i < MAX_DRAIN_ATTEMPTS_BEFORE_SHUTDOWN; i++) {
148             try {
149                 Thread.sleep(SLEEP_MILLIS_BETWEEN_DRAIN_ATTEMPTS); // give up the CPU for a while
150             } catch (final InterruptedException e) { // ignored
151             }
152         }
153         try {
154             // busy-spins until all events currently in the disruptor have been processed, or timeout
155             temp.shutdown(timeout, timeUnit);
156         } catch (final TimeoutException e) {
157             LOGGER.warn("[{}] AsyncLoggerDisruptor: shutdown timed out after {} {}", contextName, timeout, timeUnit);
158             temp.halt(); // give up on remaining log events, if any
159         }
160 
161         LOGGER.trace("[{}] AsyncLoggerDisruptor: disruptor has been shut down.", contextName);
162 
163         if (DiscardingAsyncQueueFullPolicy.getDiscardCount(asyncQueueFullPolicy) > 0) {
164             LOGGER.trace("AsyncLoggerDisruptor: {} discarded {} events.", asyncQueueFullPolicy,
165                     DiscardingAsyncQueueFullPolicy.getDiscardCount(asyncQueueFullPolicy));
166         }
167         setStopped();
168         return true;
169     }

171     /**
172      * Returns {@code true} if the specified disruptor still has unprocessed events.
173      */
174     private static boolean hasBacklog(final Disruptor<?> theDisruptor) {
175         final RingBuffer<?> ringBuffer = theDisruptor.getRingBuffer();
176         return !ringBuffer.hasAvailableCapacity(ringBuffer.getBufferSize());
177     }

179     /**
180      * Creates and returns a new {@code RingBufferAdmin} that instruments the ringbuffer of the {@code AsyncLogger}.
181      *
182      * @param jmxContextName name of the {@code AsyncLoggerContext}
183      * @return a new {@code RingBufferAdmin} that instruments the ringbuffer
184      */
185     public RingBufferAdmin createRingBufferAdmin(final String jmxContextName) {
186         final RingBuffer<RingBufferLogEvent> ring = disruptor == null ? null : disruptor.getRingBuffer();
187         return RingBufferAdmin.forAsyncLogger(ring, jmxContextName);
188     }

190     EventRoute getEventRoute(final Level logLevel) {
191         final int remainingCapacity = remainingDisruptorCapacity();
192         if (remainingCapacity < 0) {
193             return EventRoute.DISCARD;
194         }
195         return asyncQueueFullPolicy.getRoute(backgroundThreadId, logLevel);
196     }
197 
198     private int remainingDisruptorCapacity() {
199         final Disruptor<RingBufferLogEvent> temp = disruptor;
200         if (hasLog4jBeenShutDown(temp)) {
201             return -1;
202         }
203         return (int) temp.getRingBuffer().remainingCapacity();
204     }
205         /**
206          * Returns {@code true} if the specified disruptor is null.
207          */
208     private boolean hasLog4jBeenShutDown(final Disruptor<RingBufferLogEvent> aDisruptor) {
209         if (aDisruptor == null) { // LOG4J2-639
210             LOGGER.warn("Ignoring log event after log4j was shut down");
211             return true;
212         }
213         return false;
214     }

216     boolean tryPublish(final RingBufferLogEventTranslator translator) {
217         try {
218             // Note: we deliberately access the volatile disruptor field afresh here.
219             // Avoiding this and using an older reference could result in adding a log event to the disruptor after it
220             // was shut down, which could cause the publishEvent method to hang and never return.
221             return disruptor.getRingBuffer().tryPublishEvent(translator);
222         } catch (final NullPointerException npe) {
223             // LOG4J2-639: catch NPE if disruptor field was set to null in stop()
224             logWarningOnNpeFromDisruptorPublish(translator);
225             return false;
226         }
227     }
228 
229     void enqueueLogMessageWhenQueueFull(final RingBufferLogEventTranslator translator) {
230         try {
231             // Note: we deliberately access the volatile disruptor field afresh here.
232             // Avoiding this and using an older reference could result in adding a log event to the disruptor after it
233             // was shut down, which could cause the publishEvent method to hang and never return.
234             if (synchronizeEnqueueWhenQueueFull()) {
235                 synchronized (queueFullEnqueueLock) {
236                     disruptor.publishEvent(translator);
237                 }
238             } else {
239                 disruptor.publishEvent(translator);
240             }
241         } catch (final NullPointerException npe) {
242             // LOG4J2-639: catch NPE if disruptor field was set to null in stop()
243             logWarningOnNpeFromDisruptorPublish(translator);
244         }
245     }

247     void enqueueLogMessageWhenQueueFull(
248             final EventTranslatorVararg<RingBufferLogEvent> translator,
249             final AsyncLogger asyncLogger,
250             final StackTraceElement location,
251             final String fqcn,
252             final Level level,
253             final Marker marker,
254             final Message msg,
255             final Throwable thrown) {
256         try {
257             // Note: we deliberately access the volatile disruptor field afresh here.
258             // Avoiding this and using an older reference could result in adding a log event to the disruptor after it
259             // was shut down, which could cause the publishEvent method to hang and never return.
260             if (synchronizeEnqueueWhenQueueFull()) {
261                 synchronized (queueFullEnqueueLock) {
262                     disruptor.getRingBuffer().publishEvent(translator,
263                             asyncLogger, // asyncLogger: 0
264                             location, // location: 1
265                             fqcn, // 2
266                             level, // 3
267                             marker, // 4
268                             msg, // 5
269                             thrown); // 6
270                 }
271             } else {
272                 disruptor.getRingBuffer().publishEvent(translator,
273                         asyncLogger, // asyncLogger: 0
274                         location, // location: 1
275                         fqcn, // 2
276                         level, // 3
277                         marker, // 4
278                         msg, // 5
279                         thrown); // 6
280             }
281         } catch (final NullPointerException npe) {
282             // LOG4J2-639: catch NPE if disruptor field was set to null in stop()
283             logWarningOnNpeFromDisruptorPublish(level, fqcn, msg, thrown);
284         }
285     }
286 
287     private boolean synchronizeEnqueueWhenQueueFull() {
288         return DisruptorUtil.ASYNC_LOGGER_SYNCHRONIZE_ENQUEUE_WHEN_QUEUE_FULL
289                 // Background thread must never block
290                 && backgroundThreadId != Thread.currentThread().getId()
291                 // Threads owned by log4j are most likely to result in
292                 // deadlocks because they generally consume events.
293                 // This prevents deadlocks between AsyncLoggerContext
294                 // disruptors.
295                 && !(Thread.currentThread() instanceof Log4jThread);
296     }
297 
298     private void logWarningOnNpeFromDisruptorPublish(final RingBufferLogEventTranslator translator) {
299         logWarningOnNpeFromDisruptorPublish(
300                 translator.level, translator.loggerName, translator.message, translator.thrown);
301     }
302 
303     private void logWarningOnNpeFromDisruptorPublish(
304             final Level level, final String fqcn, final Message msg, final Throwable thrown) {
305         LOGGER.warn("[{}] Ignoring log event after log4j was shut down: {} [{}] {}{}", contextName,
306                 level, fqcn, msg.getFormattedMessage(), thrown == null ? "" : Throwables.toStringList(thrown));
307     }

320     /**
321      * Signals this AsyncLoggerDisruptor whether it is allowed to store non-JDK classes in ThreadLocal objects for
322      * efficiency.
323      * <p>
324      * This property may be modified after the {@link #start()} method has been called.
325      * </p>
326      *
327      * @param allow whether AsyncLoggers are allowed to use ThreadLocal objects
328      * @since 2.5
329      * @see <a href="https://issues.apache.org/jira/browse/LOG4J2-1172">LOG4J2-1172</a>
330      */
331     public void setUseThreadLocals(final boolean allow) {
332         useThreadLocalTranslator = allow;
333         LOGGER.trace("[{}] AsyncLoggers have been modified to use a {} translator", contextName,
334                 useThreadLocalTranslator ? "threadlocal" : "vararg");
335     }

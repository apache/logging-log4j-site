40  /**
41   * The Syslog Appender.
42   */
43  @Plugin(name = "Syslog", category = "Core", elementType = "appender", printObject = true)
44  public class SyslogAppender extends SocketAppender {
45  
46      protected static final String RFC5424 = "RFC5424";
47  
48      protected SyslogAppender(final String name, final Layout<? extends Serializable> layout, final Filter filter,
49                               final boolean ignoreExceptions, final boolean immediateFlush,
50                               final AbstractSocketManager manager, final Advertiser advertiser) {
51          super(name, layout, filter, manager, ignoreExceptions, immediateFlush, advertiser);
52  
53      }

55      /**
56       * Create a SyslogAppender.
57       * @param host The name of the host to connect to.
58       * @param port The port to connect to on the target host.
59       * @param protocolStr The Protocol to use.
60       * @param sslConfig TODO
61       * @param reconnectionDelay The interval in which failed writes should be retried.
62       * @param immediateFail True if the write should fail if no socket is immediately available.
63       * @param name The name of the Appender.
64       * @param immediateFlush "true" if data should be flushed on each write.
65       * @param ignoreExceptions If {@code "true"} (default) exceptions encountered when appending events are logged;
66       *                         otherwise they are propagated to the caller.
67       * @param facility The Facility is used to try to classify the message.
68       * @param id The default structured data id to use when formatting according to RFC 5424.
69       * @param enterpriseNumber The IANA enterprise number.
70       * @param includeMdc Indicates whether data from the ThreadContextMap will be included in the RFC 5424 Syslog
71       * record. Defaults to "true:.
72       * @param mdcId The id to use for the MDC Structured Data Element.
73       * @param mdcPrefix The prefix to add to MDC key names.
74       * @param eventPrefix The prefix to add to event key names.
75       * @param newLine If true, a newline will be appended to the end of the syslog record. The default is false.
76       * @param escapeNL String that should be used to replace newlines within the message text.
77       * @param appName The value to use as the APP-NAME in the RFC 5424 syslog record.
78       * @param msgId The default value to be used in the MSGID field of RFC 5424 syslog records.
79       * @param excludes A comma separated list of mdc keys that should be excluded from the LogEvent.
80       * @param includes A comma separated list of mdc keys that should be included in the FlumeEvent.
81       * @param required A comma separated list of mdc keys that must be present in the MDC.
82       * @param format If set to "RFC5424" the data will be formatted in accordance with RFC 5424. Otherwise,
83       * it will be formatted as a BSD Syslog record.
84       * @param filter A Filter to determine if the event should be handled by this Appender.
85       * @param config The Configuration.
86       * @param charsetName The character set to use when converting the syslog String to a byte array.
87       * @param exceptionPattern The converter pattern to use for formatting exceptions.
88       * @param loggerFields The logger fields
89       * @param advertise Whether to advertise
90       * @return A SyslogAppender.
91       */
92      @PluginFactory
93      public static SyslogAppender createAppender(
94              // @formatter:off
95              @PluginAttribute("host") final String host,
96              @PluginAttribute(value = "port", defaultInt = 0) final int port,
97              @PluginAttribute("protocol") final String protocolStr,
98              @PluginElement("SSL") final SslConfiguration sslConfig,
99              @PluginAttribute(value = "reconnectionDelay", defaultInt = 0) final int reconnectionDelay,
100             @PluginAttribute(value = "immediateFail", defaultBoolean = true) final boolean immediateFail,
101             @PluginAttribute("name") final String name,
102             @PluginAttribute(value = "immediateFlush", defaultBoolean = true) final boolean immediateFlush,
103             @PluginAttribute(value = "ignoreExceptions", defaultBoolean = true) final boolean ignoreExceptions,
104             @PluginAttribute(value = "facility", defaultString = "LOCAL0") final Facility facility,
105             @PluginAttribute("id") final String id,
106             @PluginAttribute(value = "enterpriseNumber", defaultInt = Rfc5424Layout.DEFAULT_ENTERPRISE_NUMBER) final int enterpriseNumber,
107             @PluginAttribute(value = "includeMdc", defaultBoolean = true) final boolean includeMdc,
108             @PluginAttribute("mdcId") final String mdcId,
109             @PluginAttribute("mdcPrefix") final String mdcPrefix,
110             @PluginAttribute("eventPrefix") final String eventPrefix,
111             @PluginAttribute(value = "newLine", defaultBoolean = false) final boolean newLine,
112             @PluginAttribute("newLineEscape") final String escapeNL,
113             @PluginAttribute("appName") final String appName,
114             @PluginAttribute("messageId") final String msgId,
115             @PluginAttribute("mdcExcludes") final String excludes,
116             @PluginAttribute("mdcIncludes") final String includes,
117             @PluginAttribute("mdcRequired") final String required,
118             @PluginAttribute("format") final String format,
119             @PluginElement("Filter") final Filter filter,
120             @PluginConfiguration final Configuration config,
121             @PluginAttribute(value = "charset", defaultString = "UTF-8") final Charset charsetName,
122             @PluginAttribute("exceptionPattern") final String exceptionPattern,
123             @PluginElement("LoggerFields") final LoggerFields[] loggerFields,
124             @PluginAttribute(value = "advertise", defaultBoolean = false) final boolean advertise) {
125         // @formatter:on
126 
127         // TODO: add Protocol to TypeConverters
128         final Protocol protocol = EnglishEnums.valueOf(Protocol.class, protocolStr);
129         final boolean useTlsMessageFormat = sslConfig != null || protocol == Protocol.SSL;
130         final Layout<? extends Serializable> layout = RFC5424.equalsIgnoreCase(format) ?
131             Rfc5424Layout.createLayout(facility, id, enterpriseNumber, includeMdc, mdcId, mdcPrefix, eventPrefix, newLine,
132                 escapeNL, appName, msgId, excludes, includes, required, exceptionPattern, useTlsMessageFormat, loggerFields,
133                 config) :
134             SyslogLayout.createLayout(facility, newLine, escapeNL, charsetName);
135 
136         if (name == null) {
137             LOGGER.error("No name provided for SyslogAppender");
138             return null;
139         }
140         final AbstractSocketManager manager = createSocketManager(name, protocol, host, port, sslConfig,
141             reconnectionDelay, immediateFail, layout);
142 
143         return new SyslogAppender(name, layout, filter, ignoreExceptions, immediateFlush, manager,
144                 advertise ? config.getAdvertiser() : null);
145     }

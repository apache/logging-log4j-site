038    private static final String DELIM_START = "${";
039    private static final char DELIM_STOP = '}';
040    private static final int DELIM_START_LEN = 2;
041    private static final int DELIM_STOP_LEN = 1;
042    private static final int ONE_K = 1024;

050    public static String[] concatenateArrays(final String[] l, final String[] r) {
051        final int len = l.length + r.length;
052        final String[] a = new String[len];
053
054        System.arraycopy(l, 0, a, 0, l.length);
055        System.arraycopy(r, 0, a, l.length, r.length);
056
057        return a;
058    }

060    public static String convertSpecialChars(final String s) {
061        char c;
062        final int len = s.length();
063        final StringBuilder sbuf = new StringBuilder(len);
064
065        int i = 0;
066        while (i < len) {
067            c = s.charAt(i++);
068            if (c == '\\') {
069                c = s.charAt(i++);
070                switch (c) {
071                case 'n':
072                    c = '\n';
073                    break;
074                case 'r':
075                    c = '\r';
076                    break;
077                case 't':
078                    c = '\t';
079                    break;
080                case 'f':
081                    c = '\f';
082                    break;
083                case 'b':
084                    c = '\b';
085                    break;
086                case '"':
087                    c = '\"';
088                    break;
089                case '\'':
090                    c = '\'';
091                    break;
092                case '\\':
093                    c = '\\';
094                    break;
095                default:
096                    // there is no default case.
097                }
098            }
099            sbuf.append(c);
100        }
101        return sbuf.toString();
102    }

104    public static Object instantiateByKey(final Properties props, final String key, final Class<?> superClass,
105                                   final Object defaultValue) {
106
107        // Get the value of the property in string form
108        final String className = findAndSubst(key, props);
109        if (className == null) {
110            LOGGER.error("Could not find value for key {}", key);
111            return defaultValue;
112        }
113        // Trim className to avoid trailing spaces that cause problems.
114        return OptionConverter.instantiateByClassName(className.trim(), superClass,
115            defaultValue);
116    }

118    /**
119     * If <code>value</code> is "true", then {@code true} is
120     * returned. If <code>value</code> is "false", then
121     * {@code false} is returned. Otherwise, <code>default</code> is
122     * returned.
123     *
124     * <p>Case of value is unimportant.</p>
125     * @param value The value to convert.
126     * @param defaultValue The default value.
127     * @return true or false, depending on the value and/or default.
128     */
129    public static boolean toBoolean(final String value, final boolean defaultValue) {
130        if (value == null) {
131            return defaultValue;
132        }
133        final String trimmedVal = value.trim();
134        if ("true".equalsIgnoreCase(trimmedVal)) {
135            return true;
136        }
137        if ("false".equalsIgnoreCase(trimmedVal)) {
138            return false;
139        }
140        return defaultValue;
141    }

143    /**
144     * Convert the String value to an int.
145     * @param value The value as a String.
146     * @param defaultValue The default value.
147     * @return The value as an int.
148     */
149    public static int toInt(final String value, final int defaultValue) {
150        if (value != null) {
151            final String s = value;
152            try {
153                return Integers.parseInt(s);
154            } catch (final NumberFormatException e) {
155                LOGGER.error("[{}] is not in proper int form.", s, e);
156            }
157        }
158        return defaultValue;
159    }

161    public static Level toLevel(String value, Level defaultValue) {
162        if(value == null) {
163            return defaultValue;
164        }
165
166        value = value.trim();
167
168        int hashIndex = value.indexOf('#');
169        if (hashIndex == -1) {
170            if("NULL".equalsIgnoreCase(value)) {
171                return null;
172            } else {
173                // no class name specified : use standard Level class
174                return Level.toLevel(value, defaultValue);
175            }
176        }
177
178        Level result = defaultValue;
179
180        String clazz = value.substring(hashIndex+1);
181        String levelName = value.substring(0, hashIndex);
182
183        // This is degenerate case but you never know.
184        if("NULL".equalsIgnoreCase(levelName)) {
185            return null;
186        }
187
188        LOGGER.debug("toLevel" + ":class=[" + clazz + "]"
189            + ":pri=[" + levelName + "]");
190
191        try {
192            Class<?> customLevel = Loader.loadClass(clazz);
193
194            // get a ref to the specified class' static method
195            // toLevel(String, org.apache.log4j.Level)
196            Class<?>[] paramTypes = new Class[] { String.class, Level.class };
197            java.lang.reflect.Method toLevelMethod =
198                customLevel.getMethod("toLevel", paramTypes);
199
200            // now call the toLevel method, passing level string + default
201            Object[] params = new Object[] {levelName, defaultValue};
202            Object o = toLevelMethod.invoke(null, params);
203
204            result = (Level) o;
205        } catch(ClassNotFoundException e) {
206            LOGGER.warn("custom level class [" + clazz + "] not found.");
207        } catch(NoSuchMethodException e) {
208            LOGGER.warn("custom level class [" + clazz + "]"
209                + " does not have a class function toLevel(String, Level)", e);
210        } catch(java.lang.reflect.InvocationTargetException e) {
211            if (e.getTargetException() instanceof InterruptedException
212                || e.getTargetException() instanceof InterruptedIOException) {
213                Thread.currentThread().interrupt();
214            }
215            LOGGER.warn("custom level class [" + clazz + "]" + " could not be instantiated", e);
216        } catch(ClassCastException e) {
217            LOGGER.warn("class [" + clazz + "] is not a subclass of org.apache.log4j.Level", e);
218        } catch(IllegalAccessException e) {
219            LOGGER.warn("class ["+clazz+ "] cannot be instantiated due to access restrictions", e);
220        } catch(RuntimeException e) {
221            LOGGER.warn("class ["+clazz+"], level [" + levelName + "] conversion failed.", e);
222        }
223        return result;
224    }

226    /**
227     *
228     * @param value The size of the file as a String.
229     * @param defaultValue The default value.
230     * @return The size of the file as a long.
231     */
232    public static long toFileSize(final String value, final long defaultValue) {
233        if (value == null) {
234            return defaultValue;
235        }
236
237        String str = value.trim().toUpperCase(Locale.ENGLISH);
238        long multiplier = 1;
239        int index;
240
241        if ((index = str.indexOf("KB")) != -1) {
242            multiplier = ONE_K;
243            str = str.substring(0, index);
244        } else if ((index = str.indexOf("MB")) != -1) {
245            multiplier = ONE_K * ONE_K;
246            str = str.substring(0, index);
247        } else if ((index = str.indexOf("GB")) != -1) {
248            multiplier = ONE_K * ONE_K * ONE_K;
249            str = str.substring(0, index);
250        }
251        try {
252            return Long.parseLong(str) * multiplier;
253        } catch (final NumberFormatException e) {
254            LOGGER.error("[{}] is not in proper int form.", str);
255            LOGGER.error("[{}] not in expected format.", value, e);
256        }
257        return defaultValue;
258    }

260    /**
261     * Find the value corresponding to <code>key</code> in
262     * <code>props</code>. Then perform variable substitution on the
263     * found value.
264     * @param key The key to locate.
265     * @param props The properties.
266     * @return The String after substitution.
267     */
268    public static String findAndSubst(final String key, final Properties props) {
269        final String value = props.getProperty(key);
270        if (value == null) {
271            return null;
272        }
273
274        try {
275            return substVars(value, props);
276        } catch (final IllegalArgumentException e) {
277            LOGGER.error("Bad option value [{}].", value, e);
278            return value;
279        }
280    }

282    /**
283     * Instantiate an object given a class name. Check that the
284     * <code>className</code> is a subclass of
285     * <code>superClass</code>. If that test fails or the object could
286     * not be instantiated, then <code>defaultValue</code> is returned.
287     *
288     * @param className    The fully qualified class name of the object to instantiate.
289     * @param superClass   The class to which the new object should belong.
290     * @param defaultValue The object to return in case of non-fulfillment
291     * @return The created object.
292     */
293    public static Object instantiateByClassName(final String className, final Class<?> superClass,
294                                         final Object defaultValue) {
295        if (className != null) {
296            try {
297                final Class<?> classObj = Loader.loadClass(className);
298                if (!superClass.isAssignableFrom(classObj)) {
299                    LOGGER.error("A \"{}\" object is not assignable to a \"{}\" variable.", className,
300                        superClass.getName());
301                    LOGGER.error("The class \"{}\" was loaded by [{}] whereas object of type [{}] was loaded by [{}].",
302                        superClass.getName(), superClass.getClassLoader(), classObj.getTypeName(), classObj.getName());
303                    return defaultValue;
304                }
305                return classObj.newInstance();
306            } catch (final Exception e) {
307                LOGGER.error("Could not instantiate class [{}].", className, e);
308            }
309        }
310        return defaultValue;
311    }

314    /**
315     * Perform variable substitution in string <code>val</code> from the
316     * values of keys found in the system propeties.
317     *
318     * <p>The variable substitution delimiters are <b>${</b> and <b>}</b>.</p>
319     *
320     * <p>For example, if the System properties contains "key=value", then
321     * the call</p>
322     * <pre>
323     * String s = OptionConverter.substituteVars("Value of key is ${key}.");
324     * </pre>
325     * <p>
326     * will set the variable <code>s</code> to "Value of key is value.".
327     * </p>
328     * <p>If no value could be found for the specified key, then the
329     * <code>props</code> parameter is searched, if the value could not
330     * be found there, then substitution defaults to the empty string.</p>
331     *
332     * <p>For example, if system properties contains no value for the key
333     * "inexistentKey", then the call
334     * </p>
335     * <pre>
336     * String s = OptionConverter.subsVars("Value of inexistentKey is [${inexistentKey}]");
337     * </pre>
338     * <p>
339     * will set <code>s</code> to "Value of inexistentKey is []"
340     * </p>
341     * <p>An {@link java.lang.IllegalArgumentException} is thrown if
342     * <code>val</code> contains a start delimeter "${" which is not
343     * balanced by a stop delimeter "}". </p>
344     *
345     * @param val The string on which variable substitution is performed.
346     * @param props The properties to use for substitution.
347     * @return The String after substitution.
348     * @throws IllegalArgumentException if <code>val</code> is malformed.
349     */
350    public static String substVars(final String val, final Properties props) throws
351        IllegalArgumentException {
352        return substVars(val, props, new ArrayList<>());
353    }
354
355    private static String substVars(final String val, final Properties props, List<String> keys)
356            throws IllegalArgumentException {
357
358        final StringBuilder sbuf = new StringBuilder();
359
360        int i = 0;
361        int j;
362        int k;
363
364        while (true) {
365            j = val.indexOf(DELIM_START, i);
366            if (j == -1) {
367                // no more variables
368                if (i == 0) { // this is a simple string
369                    return val;
370                }
371                // add the tail string which contails no variables and return the result.
372                sbuf.append(val.substring(i, val.length()));
373                return sbuf.toString();
374            }
375            sbuf.append(val.substring(i, j));
376            k = val.indexOf(DELIM_STOP, j);
377            if (k == -1) {
378                throw new IllegalArgumentException(Strings.dquote(val)
379                        + " has no closing brace. Opening brace at position " + j
380                        + '.');
381            }
382            j += DELIM_START_LEN;
383            final String key = val.substring(j, k);
384            // first try in System properties
385            String replacement = PropertiesUtil.getProperties().getStringProperty(key, null);
386            // then try props parameter
387            if (replacement == null && props != null) {
388                replacement = props.getProperty(key);
389            }
390
391            if (replacement != null) {
392
393                // Do variable substitution on the replacement string
394                // such that we can solve "Hello ${x2}" as "Hello p1"
395                // the where the properties are
396                // x1=p1
397                // x2=${x1}
398                if (!keys.contains(key)) {
399                    List<String> usedKeys = new ArrayList<>(keys);
400                    usedKeys.add(key);
401                    final String recursiveReplacement = substVars(replacement, props, usedKeys);
402                    sbuf.append(recursiveReplacement);
403                } else {
404                    sbuf.append(replacement);
405                }
406
407            }
408            i = k + DELIM_STOP_LEN;
409        }
410    }

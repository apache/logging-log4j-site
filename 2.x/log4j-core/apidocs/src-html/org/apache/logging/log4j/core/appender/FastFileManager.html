017    package org.apache.logging.log4j.core.appender;
018    
019    import java.io.File;
020    import java.io.IOException;
021    import java.io.OutputStream;
022    import java.io.RandomAccessFile;
023    import java.nio.ByteBuffer;
024    import java.util.HashMap;
025    import java.util.Map;
026    
027    import org.apache.logging.log4j.core.Layout;
028    
029    /**
030     * Extends OutputStreamManager but instead of using a buffered output stream,
031     * this class uses a {@code ByteBuffer} and a {@code RandomAccessFile} to do the
032     * I/O.
033     */
034    public class FastFileManager extends OutputStreamManager {
035        static final int DEFAULT_BUFFER_SIZE = 256 * 1024;
036    
037        private static final FastFileManagerFactory FACTORY = new FastFileManagerFactory();
038    
039        private final boolean isImmediateFlush;
040        private final String advertiseURI;
041        private final RandomAccessFile randomAccessFile;
042        private final ByteBuffer buffer;
043        private final ThreadLocal<Boolean> isEndOfBatch = new ThreadLocal<Boolean>();
044    
045        protected FastFileManager(final RandomAccessFile file,
046                final String fileName, final OutputStream os,
047                final boolean immediateFlush, final String advertiseURI,
048                final Layout layout) {
049            super(os, fileName, layout);
050            this.isImmediateFlush = immediateFlush;
051            this.randomAccessFile = file;
052            this.advertiseURI = advertiseURI;
053            isEndOfBatch.set(Boolean.FALSE);
054    
055            // TODO make buffer size configurable?
056            buffer = ByteBuffer.allocate(DEFAULT_BUFFER_SIZE);
057        }
058    
059        /**
060         * Returns the FastFileManager.
061         * 
062         * @param fileName The name of the file to manage.
063         * @param append true if the file should be appended to, false if it should
064         *            be overwritten.
065         * @param isFlush true if the contents should be flushed to disk on every
066         *            write
067         * @param advertiseURI the URI to use when advertising the file
068         * @param layout The layout.
069         * @return A FastFileManager for the File.
070         */
071        public static FastFileManager getFileManager(final String fileName,
072                final boolean append, final boolean isFlush,
073                final String advertiseURI, final Layout layout) {
074            return (FastFileManager) getManager(fileName, new FactoryData(append,
075                    isFlush, advertiseURI, layout), FACTORY);
076        }
077    
078        public Boolean isEndOfBatch() {
079            return isEndOfBatch.get();
080        }
081    
082        public void setEndOfBatch(final boolean isEndOfBatch) {
083            this.isEndOfBatch.set(Boolean.valueOf(isEndOfBatch));
084        }
085    
086        @Override
087        protected synchronized void write(final byte[] bytes, int offset, int length) {
088            super.write(bytes, offset, length); // writes to dummy output stream
089    
090            int chunk = 0;
091            do {
092                if (length > buffer.remaining()) {
093                    flush();
094                }
095                chunk = Math.min(length, buffer.remaining());
096                buffer.put(bytes, offset, chunk);
097                offset += chunk;
098                length -= chunk;
099            } while (length > 0);
100    
101            if (isImmediateFlush || isEndOfBatch.get() == Boolean.TRUE) {
102                flush();
103            }
104        }
105    
106        @Override
107        public void flush() {
108            buffer.flip();
109            try {
110                randomAccessFile.write(buffer.array(), 0, buffer.limit());
111            } catch (final IOException ex) {
112                final String msg = "Error writing to RandomAccessFile " + getName();
113                throw new AppenderRuntimeException(msg, ex);
114            }
115            buffer.clear();
116        }
117    
118        @Override
119        public void close() {
120            flush();
121            try {
122                randomAccessFile.close();
123            } catch (final IOException ex) {
124                LOGGER.error("Unable to close RandomAccessFile " + getName() + ". "
125                        + ex);
126            }
127        }
128    
129        /**
130         * Returns the name of the File being managed.
131         * 
132         * @return The name of the File being managed.
133         */
134        public String getFileName() {
135            return getName();
136        }
137    
138        /** {@code OutputStream} subclass that does not write anything. */
139        static class DummyOutputStream extends OutputStream {
140            @Override
141            public void write(final int b) throws IOException {
142            }
143    
144            @Override
145            public void write(final byte[] b, final int off, final int len) throws IOException {
146            }
147        }
148    
149        /**
150         * FileManager's content format is specified by:
151         * <p/>
152         * Key: "fileURI" Value: provided "advertiseURI" param.
153         * 
154         * @return Map of content format keys supporting FileManager
155         */
156        @Override
157        public Map<String, String> getContentFormat() {
158            final Map<String, String> result = new HashMap<String, String>(
159                    super.getContentFormat());
160            result.put("fileURI", advertiseURI);
161            return result;
162        }
163    
164        /**
165         * Factory Data.
166         */
167        private static class FactoryData {
168            private final boolean append;
169            private final boolean immediateFlush;
170            private final String advertiseURI;
171            private final Layout layout;
172    
173            /**
174             * Constructor.
175             * 
176             * @param append Append status.
177             */
178            public FactoryData(final boolean append, final boolean immediateFlush,
179                    final String advertiseURI, final Layout layout) {
180                this.append = append;
181                this.immediateFlush = immediateFlush;
182                this.advertiseURI = advertiseURI;
183                this.layout = layout;
184            }
185        }
186    
187        /**
188         * Factory to create a FastFileManager.
189         */
190        private static class FastFileManagerFactory implements
191                ManagerFactory<FastFileManager, FactoryData> {
192    
193            /**
194             * Create a FastFileManager.
195             * 
196             * @param name The name of the File.
197             * @param data The FactoryData
198             * @return The FastFileManager for the File.
199             */
200            @Override
201            public FastFileManager createManager(final String name, final FactoryData data) {
202                final File file = new File(name);
203                final File parent = file.getParentFile();
204                if (null != parent && !parent.exists()) {
205                    parent.mkdirs();
206                }
207                if (!data.append) {
208                    file.delete();
209                }
210    
211                final OutputStream os = new DummyOutputStream();
212                RandomAccessFile raf;
213                try {
214                    raf = new RandomAccessFile(name, "rw");
215                    if (data.append) {
216                        raf.seek(raf.length());
217                    } else {
218                        raf.setLength(0);
219                    }
220                    return new FastFileManager(raf, name, os, data.immediateFlush,
221                            data.advertiseURI, data.layout);
222                } catch (final Exception ex) {
223                    LOGGER.error("FastFileManager (" + name + ") " + ex);
224                }
225                return null;
226            }
227        }

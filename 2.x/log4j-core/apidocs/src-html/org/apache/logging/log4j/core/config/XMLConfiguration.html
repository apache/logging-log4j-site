054    /**
055     * Creates a Node hierarchy from an XML file.
056     */
057    public class XMLConfiguration extends BaseConfiguration {
058    
059        private static final String[] VERBOSE_CLASSES = new String[] {ResolverUtil.class.getName()};
060    
061        private static final String LOG4J_XSD = "Log4J-V2.0.xsd";
062    
063        private static final int BUF_SIZE = 16384;
064    
065        private List<Status> status = new ArrayList<Status>();
066    
067        private Element rootElement = null;
068    
069        private boolean strict = false;
070    
071        private String schema = null;
072    
073        private Validator validator;
074    
075        private List<String> messages = new ArrayList<String>();
076    
077        public XMLConfiguration(InputSource source, File configFile) {
078            byte[] buffer = null;
079    
080            try {
081                buffer = toByteArray(source.getByteStream());
082                source = new InputSource(new ByteArrayInputStream(buffer));
083                DocumentBuilder builder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
084                Document document = builder.parse(source);
085                rootElement = document.getDocumentElement();
086                Map<String, String> attrs = processAttributes(rootNode, rootElement);
087                Level status = Level.OFF;
088                boolean verbose = false;
089    
090                for (Map.Entry<String, String> entry : attrs.entrySet()) {
091                    if ("status".equalsIgnoreCase(entry.getKey())) {
092                        status = Level.toLevel(getSubst().replace(entry.getValue()).toUpperCase(), null);
093                        if (status == null) {
094                            status = Level.ERROR;
095                            messages.add("Invalid status specified: " + entry.getValue() + ". Defaulting to ERROR");
096                        }
097                    } else if ("verbose".equalsIgnoreCase(entry.getKey())) {
098                        verbose = Boolean.parseBoolean(getSubst().replace(entry.getValue()));
099                    } else if ("packages".equalsIgnoreCase(getSubst().replace(entry.getKey()))) {
100                        String[] packages = entry.getValue().split(",");
101                        for (String p : packages) {
102                            PluginManager.addPackage(p);
103                        }
104                    } else if ("name".equalsIgnoreCase(entry.getKey())) {
105                        setName(getSubst().replace(entry.getValue()));
106                    } else if ("strict".equalsIgnoreCase(entry.getKey())) {
107                        strict = Boolean.parseBoolean(getSubst().replace(entry.getValue()));
108                    } else if ("schema".equalsIgnoreCase(entry.getKey())) {
109                        schema = getSubst().replace(entry.getValue());
110                    } else if ("monitorInterval".equalsIgnoreCase(entry.getKey())) {
111                        int interval = Integer.parseInt(getSubst().replace(entry.getValue()));
112                        if (interval > 0 && configFile != null) {
113                            monitor = new FileConfigurationMonitor(configFile, listeners, interval);
114                        }
115                    }
116                }
117                Iterator<StatusListener> iter = ((StatusLogger) LOGGER).getListeners();
118                boolean found = false;
119                while (iter.hasNext()) {
120                    StatusListener listener = iter.next();
121                    if (listener instanceof StatusConsoleListener) {
122                        found = true;
123                        ((StatusConsoleListener) listener).setLevel(status);
124                        if (!verbose) {
125                            ((StatusConsoleListener) listener).setFilters(VERBOSE_CLASSES);
126                        }
127                    }
128                }
129                if (!found && status != Level.OFF) {
130                    StatusConsoleListener listener = new StatusConsoleListener(status);
131                    if (!verbose) {
132                        listener.setFilters(VERBOSE_CLASSES);
133                    }
134                    ((StatusLogger) LOGGER).registerListener(listener);
135                    for (String msg : messages) {
136                        LOGGER.error(msg);
137                    }
138                }
139    
140            } catch (SAXException domEx) {
141                LOGGER.error("Error parsing " + source.getSystemId(), domEx);
142            } catch (IOException ioe) {
143                LOGGER.error("Error parsing " + source.getSystemId(), ioe);
144            } catch (ParserConfigurationException pex) {
145                LOGGER.error("Error parsing " + source.getSystemId(), pex);
146            }
147            if (strict && schema != null && buffer != null) {
148                InputStream is = null;
149                try {
150                    is = getClass().getClassLoader().getResourceAsStream(schema);
151                } catch (Exception ex) {
152                    LOGGER.error("Unable to access schema " + schema);
153                }
154                if (is != null) {
155                    Source src = new StreamSource(is, LOG4J_XSD);
156                    SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
157                    Schema schema = null;
158                    try {
159                        schema = factory.newSchema(src);
160                    } catch (SAXException ex) {
161                        LOGGER.error("Error parsing Log4j schema", ex);
162                    }
163                    if (schema != null) {
164                        validator = schema.newValidator();
165                        try {
166                            validator.validate(new StreamSource(new ByteArrayInputStream(buffer)));
167                        } catch (IOException ioe) {
168                            LOGGER.error("Error reading configuration for validation", ioe);
169                        } catch (SAXException ex) {
170                            LOGGER.error("Error validating configuration", ex);
171                        }
172                    }
173                }
174            }
175    
176            if (getName() == null) {
177                setName(source.getSystemId());
178            }
179        }
180    
181        public void setup() {
182            constructHierarchy(rootNode, rootElement);
183            if (status.size() > 0) {
184                for (Status s : status) {
185                    LOGGER.error("Error processing element " + s.name + ": " + s.errorType);
186                }
187                return;
188            }
189            rootElement = null;
190        }
191    
192        private void constructHierarchy(Node node, Element element) {
193            processAttributes(node, element);
194            StringBuffer buffer = new StringBuffer();
195            NodeList list = element.getChildNodes();
196            List<Node> children = node.getChildren();
197            for (int i = 0; i < list.getLength(); i++) {
198                org.w3c.dom.Node w3cNode = list.item(i);
199                if (w3cNode instanceof Element) {
200                    Element child = (Element) w3cNode;
201                    String name = getType(child);
202                    PluginType type = getPluginManager().getPluginType(name);
203                    Node childNode = new Node(node, name, type);
204                    constructHierarchy(childNode, child);
205                    if (type == null) {
206                        String value = childNode.getValue();
207                        if (!childNode.hasChildren() && value != null) {
208                            node.getAttributes().put(name, value);
209                        } else {
210                            status.add(new Status(name, element, ErrorType.CLASS_NOT_FOUND));
211                        }
212                    } else {
213                        children.add(childNode);
214                    }
215                } else if (w3cNode instanceof Text) {
216                    Text data = (Text) w3cNode;
217                    buffer.append(data.getData());
218                }
219            }
220    
221            String text = buffer.toString().trim();
222            if (text.length() > 0 || (!node.hasChildren() && !node.isRoot())) {
223                node.setValue(text);
224            }
225        }
226    
227        private String getType(Element element) {
228            if (strict) {
229                NamedNodeMap attrs = element.getAttributes();
230                for (int i = 0; i < attrs.getLength(); ++i) {
231                    org.w3c.dom.Node w3cNode = attrs.item(i);
232                    if (w3cNode instanceof Attr) {
233                        Attr attr = (Attr) w3cNode;
234                        if (attr.getName().equalsIgnoreCase("type")) {
235                            String type = attr.getValue();
236                            attrs.removeNamedItem(attr.getName());
237                            return type;
238                        }
239                    }
240                }
241            }
242            return element.getTagName();
243        }
244    
245        private byte[] toByteArray(InputStream is) throws IOException {
246            ByteArrayOutputStream buffer = new ByteArrayOutputStream();
247    
248            int nRead;
249            byte[] data = new byte[BUF_SIZE];
250    
251            while ((nRead = is.read(data, 0, data.length)) != -1) {
252                buffer.write(data, 0, nRead);
253            }
254    
255            return buffer.toByteArray();
256        }
257    
258        private Map<String, String> processAttributes(Node node, Element element) {
259            NamedNodeMap attrs = element.getAttributes();
260            Map<String, String> attributes = node.getAttributes();
261    
262            for (int i = 0; i < attrs.getLength(); ++i) {
263                org.w3c.dom.Node w3cNode = attrs.item(i);
264                if (w3cNode instanceof Attr) {
265                    Attr attr = (Attr) w3cNode;
266                    attributes.put(attr.getName(), attr.getValue());
267                }
268            }
269            return attributes;
270        }

272        /**
273         * The error that occurred.
274         */
275        private enum ErrorType {
276            CLASS_NOT_FOUND
277        }
278    
279        /**
280         * Status for recording errors.
281         */
282        private class Status {
283            private Element element;
284            private String name;
285            private ErrorType errorType;
286    
287            public Status(String name, Element element, ErrorType errorType) {
288                this.name = name;
289                this.element = element;
290                this.errorType = errorType;
291            }
292        }

017package org.apache.logging.log4j.core.appender.nosql;
018
019/**
020 * Implementations of this class are plugins for configuring the {@link NoSqlAppender} with the proper provider
021 * (MongoDB, etc.).
022 *
023 * @param <C> Specifies which implementation of {@link NoSqlConnection} this provider provides.
024 */
025public interface NoSqlProvider<C extends NoSqlConnection<?, ? extends NoSqlObject<?>>> {
026
027    /**
028     * Obtains a connection from this provider. The concept of a connection in this case is not strictly an active
029     * duplex UDP or TCP connection to the underlying database. It can be thought of more as a gateway, a path for
030     * inserting objects that may use a persistent connection or may use HTTP web service calls, etc.
031     * <p>
032     * Where applicable, this method should return a connection from the connection pool as opposed to opening a
033     * brand new connection every time.
034     * </p>
035     *
036     * @return a connection that can be used to create and persist objects to this database.
037     * @see NoSqlConnection
038     */
039    C getConnection();
040
041    /**
042     * All implementations must override {@link Object#toString()} to provide information about the provider
043     * configuration (obscuring passwords with one-way hashes).
044     *
045     * @return the string representation of this NoSQL provider.
046     */
047    @Override
048    String toString();
049}

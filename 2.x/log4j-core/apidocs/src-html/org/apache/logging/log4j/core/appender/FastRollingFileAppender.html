024    import org.apache.logging.log4j.core.Layout;
025    import org.apache.logging.log4j.core.LogEvent;

028    import org.apache.logging.log4j.core.appender.rolling.RollingFileManager;
029    import org.apache.logging.log4j.core.appender.rolling.RolloverStrategy;
030    import org.apache.logging.log4j.core.appender.rolling.TriggeringPolicy;
031    import org.apache.logging.log4j.core.config.Configuration;
032    import org.apache.logging.log4j.core.config.plugins.Plugin;

041    /**
042     * An appender that writes to random access files and can roll over at
043     * intervals.
044     *
045     * @param <T> The {@link Layout}'s {@link Serializable} type.
046     */
047    @Plugin(name = "FastRollingFile", category = "Core", elementType = "appender", printObject = true)
048    public final class FastRollingFileAppender<T extends Serializable> extends AbstractOutputStreamAppender<T> {
049    
050        private final String fileName;
051        private final String filePattern;
052        private Object advertisement;
053        private final Advertiser advertiser;
054    
055        private FastRollingFileAppender(final String name, final Layout<T> layout,
056                final Filter filter, final RollingFileManager manager, final String fileName,
057                final String filePattern, final boolean handleException,
058                final boolean immediateFlush, final Advertiser advertiser) {
059            super(name, layout, filter, handleException, immediateFlush, manager);
060            if (advertiser != null) {
061                final Map<String, String> configuration = new HashMap<String, String>(
062                        layout.getContentFormat());
063                configuration.put("contentType", layout.getContentType());
064                configuration.put("name", name);
065                advertisement = advertiser.advertise(configuration);
066            }
067            this.fileName = fileName;
068            this.filePattern = filePattern;
069            this.advertiser = advertiser;
070        }
071    
072        @Override
073        public void stop() {
074            super.stop();
075            if (advertiser != null) {
076                advertiser.unadvertise(advertisement);
077            }
078        }
079    
080        /**
081         * Write the log entry rolling over the file when required.
082         *
083         * @param event The LogEvent.
084         */
085        @Override
086        public void append(final LogEvent event) {
087            final FastRollingFileManager manager = (FastRollingFileManager) getManager();
088            manager.checkRollover(event);
089    
090            // Leverage the nice batching behaviour of async Loggers/Appenders:
091            // we can signal the file manager that it needs to flush the buffer
092            // to disk at the end of a batch.
093            // From a user's point of view, this means that all log events are
094            // _always_ available in the log file, without incurring the overhead
095            // of immediateFlush=true.
096            manager.setEndOfBatch(event.isEndOfBatch());
097            super.append(event);
098        }
099    
100        /**
101         * Returns the File name for the Appender.
102         *
103         * @return The file name.
104         */
105        public String getFileName() {
106            return fileName;
107        }
108    
109        /**
110         * Returns the file pattern used when rolling over.
111         *
112         * @return The file pattern.
113         */
114        public String getFilePattern() {
115            return filePattern;
116        }
117    
118        /**
119         * Create a FastRollingFileAppender.
120         *
121         * @param fileName The name of the file that is actively written to.
122         *            (required).
123         * @param filePattern The pattern of the file name to use on rollover.
124         *            (required).
125         * @param append If true, events are appended to the file. If false, the
126         *            file is overwritten when opened. Defaults to "true"
127         * @param name The name of the Appender (required).
128         * @param immediateFlush When true, events are immediately flushed. Defaults
129         *            to "true".
130         * @param policy The triggering policy. (required).
131         * @param strategy The rollover strategy. Defaults to
132         *            DefaultRolloverStrategy.
133         * @param layout The layout to use (defaults to the default PatternLayout).
134         * @param filter The Filter or null.
135         * @param suppress "true" if exceptions should be hidden from the
136         *            application, "false" otherwise. The default is "true".
137         * @param advertise "true" if the appender configuration should be
138         *            advertised, "false" otherwise.
139         * @param advertiseURI The advertised URI which can be used to retrieve the
140         *            file contents.
141         * @param config The Configuration.
142         * @param <S> The {@link Layout}'s {@link Serializable} type.
143         * @return A FastRollingFileAppender.
144         */
145        @PluginFactory
146        public static <S extends Serializable> FastRollingFileAppender<S> createAppender(
147                @PluginAttr("fileName") final String fileName,
148                @PluginAttr("filePattern") final String filePattern,
149                @PluginAttr("append") final String append,
150                @PluginAttr("name") final String name,
151                @PluginAttr("immediateFlush") final String immediateFlush,
152                @PluginElement("policy") final TriggeringPolicy policy,
153                @PluginElement("strategy") RolloverStrategy strategy,
154                @PluginElement("layout") Layout<S> layout,
155                @PluginElement("filter") final Filter filter,
156                @PluginAttr("suppressExceptions") final String suppress,
157                @PluginAttr("advertise") final String advertise,
158                @PluginAttr("advertiseURI") final String advertiseURI,
159                @PluginConfiguration final Configuration config) {
160    
161            final boolean isAppend = Booleans.parseBoolean(append, true);
162            final boolean handleExceptions = Booleans.parseBoolean(suppress, true);
163            final boolean isFlush = Booleans.parseBoolean(immediateFlush, true);
164            final boolean isAdvertise = Boolean.parseBoolean(advertise);
165    
166            if (name == null) {
167                LOGGER.error("No name provided for FileAppender");
168                return null;
169            }
170    
171            if (fileName == null) {
172                LOGGER.error("No filename was provided for FileAppender with name "
173                        + name);
174                return null;
175            }
176    
177            if (filePattern == null) {
178                LOGGER.error("No filename pattern provided for FileAppender with name "
179                        + name);
180                return null;
181            }
182    
183            if (policy == null) {
184                LOGGER.error("A TriggeringPolicy must be provided");
185                return null;
186            }
187    
188            if (strategy == null) {
189                strategy = DefaultRolloverStrategy.createStrategy(null, null,
190                        "true", config);
191            }
192    
193            if (layout == null) {
194                @SuppressWarnings({ "unchecked", "UnnecessaryLocalVariable" })
195                final
196                Layout<S> l = (Layout<S>) PatternLayout.createLayout(null, null, null, null, null);
197                layout = l;
198            }
199    
200    
201            final FastRollingFileManager manager = FastRollingFileManager.getFastRollingFileManager(fileName, filePattern,
202                isAppend, isFlush, policy, strategy, advertiseURI, layout);
203            if (manager == null) {
204                return null;
205            }
206    
207            return new FastRollingFileAppender<S>(name, layout, filter, manager,
208                    fileName, filePattern, handleExceptions, isFlush,
209                    isAdvertise ? config.getAdvertiser() : null);
210        }

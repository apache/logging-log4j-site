017    package org.apache.logging.log4j.core.config.plugins;
018    
019    import java.util.HashMap;
020    import java.util.Map;
021    
022    import org.apache.logging.log4j.core.config.Configuration;
023    import org.apache.logging.log4j.core.config.Property;
024    import org.apache.logging.log4j.core.lookup.Interpolator;
025    import org.apache.logging.log4j.core.lookup.MapLookup;
026    import org.apache.logging.log4j.core.lookup.StrLookup;
027    
028    /**
029     * Handles properties defined in the configuration.
030     */
031    @Plugin(name = "properties", category = "Core", printObject = true)
032    public final class PropertiesPlugin {
033    
034        private PropertiesPlugin() {
035        }
036    
037        /**
038         * Create the Properties component.
039         * @param properties An array of Property elements.
040         * @param config The Configuration.
041         * @return An Interpolator that includes the configuration properties.
042         */
043        @PluginFactory
044        public static StrLookup configureSubstitutor(@PluginElement("properties") final Property[] properties,
045                                                     @PluginConfiguration final Configuration config) {
046            if (properties == null) {
047                return new Interpolator(null);
048            }
049            final Map<String, String> map = new HashMap<String, String>(config.getProperties());
050    
051            for (final Property prop : properties) {
052                map.put(prop.getName(), prop.getValue());
053            }
054    
055            return new Interpolator(new MapLookup(map));
056        }
057    }

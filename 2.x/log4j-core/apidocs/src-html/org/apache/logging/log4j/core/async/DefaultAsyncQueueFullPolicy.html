017package org.apache.logging.log4j.core.async;
018
019import org.apache.logging.log4j.Level;
020
021/**
022 * Default router: enqueue the event for asynchronous logging in the background thread, unless the current thread is the
023 * background thread and the queue is full (enqueueing would cause a deadlock). In that case send the event directly to
024 * the appender (in the current thread).
025 */
026public class DefaultAsyncQueueFullPolicy implements AsyncQueueFullPolicy {
027    @Override
028    public EventRoute getRoute(final long backgroundThreadId, final Level level) {
029
030        // LOG4J2-471: prevent deadlock when RingBuffer is full and object
031        // being logged calls Logger.log() from its toString() method
032        if (Thread.currentThread().getId() == backgroundThreadId) {
033            return EventRoute.SYNCHRONOUS;
034        }
035        return EventRoute.ENQUEUE;
036    }
037}

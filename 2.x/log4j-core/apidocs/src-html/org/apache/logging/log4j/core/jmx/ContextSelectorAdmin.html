017    package org.apache.logging.log4j.core.jmx;
018    
019    import javax.management.ObjectName;
020    
021    import org.apache.logging.log4j.core.selector.ContextSelector;
022    
023    /**
024     * Implementation of the {@code ContextSelectorAdminMBean} interface.
025     */
026    public class ContextSelectorAdmin implements ContextSelectorAdminMBean {
027    
028        private final ObjectName objectName;
029        private final ContextSelector selector;
030    
031        /**
032         * Constructs a new {@code ContextSelectorAdmin}.
033         * 
034         * @param selector the instrumented object
035         */
036        public ContextSelectorAdmin(ContextSelector selector) {
037            super();
038            this.selector = Assert.isNotNull(selector, "ContextSelector");
039            try {
040                objectName = new ObjectName(NAME);
041            } catch (Exception e) {
042                throw new IllegalStateException(e);
043            }
044        }
045    
046        /** @see ContextSelectorAdminMBean#NAME */
047        public ObjectName getObjectName() {
048            return objectName;
049        }
050    
051        @Override
052        public String getImplementationClassName() {
053            return selector.getClass().getName();
054        }
055    }

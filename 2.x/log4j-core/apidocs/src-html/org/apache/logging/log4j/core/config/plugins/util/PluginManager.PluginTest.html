084         * @param p The package name.

086        public static void addPackage(final String p) {
087            if (PACKAGES.addIfAbsent(p)) {
088                // set of available plugins could have changed, reset plugin cache for newly-retrieved managers
089                REGISTRY.clear(); // TODO confirm if this is correct
090            }

119        /**
120         * Collects plugins, optionally obtaining them from a preload map.
121         * 
122         * @param preLoad if true, plugins will be obtained from the preload map.
123         *
124         */
125        public void collectPlugins(boolean preLoad) {
126            if (REGISTRY.hasCategory(category)) {
127                plugins = REGISTRY.getCategory(category);
128                preLoad = false;
129            }
130            long start = System.nanoTime();
131            if (preLoad) {
132                final ResourceLoader loader = new ClassLoaderResourceLoader(Loader.getClassLoader());
133                loadPlugins(loader);
134            }
135            plugins = REGISTRY.getCategory(category);
136            loadFromPackages(start, preLoad);
137    
138            long elapsed = System.nanoTime() - start;
139            reportPluginLoadDuration(preLoad, elapsed);
140        }
141        
142        @SuppressWarnings({ "unchecked", "rawtypes" })
143        private void loadFromPackages(final long start, final boolean preLoad) {
144            if (plugins == null || plugins.size() == 0) {
145                if (!PACKAGES.contains(LOG4J_PACKAGES)) {
146                    PACKAGES.add(LOG4J_PACKAGES);
147                }
148            }
149            final ResolverUtil resolver = new ResolverUtil();
150            final ClassLoader classLoader = Loader.getClassLoader();
151            if (classLoader != null) {
152                resolver.setClassLoader(classLoader);
153            }
154            final Class<?> cls = null;
155            final ResolverUtil.Test test = new PluginTest(cls);
156            for (final String pkg : PACKAGES) {
157                resolver.findInPackage(test, pkg);
158            }
159            for (final Class<?> clazz : resolver.getClasses()) {
160                final Plugin plugin = clazz.getAnnotation(Plugin.class);
161                final String pluginCategory = plugin.category();
162                final Map<String, PluginType<?>> map = REGISTRY.getCategory(pluginCategory);
163                String type = plugin.elementType().equals(Plugin.EMPTY) ? plugin.name() : plugin.elementType();
164                PluginType<?> pluginType = new PluginType(clazz, type, plugin.printObject(), plugin.deferChildren());
165                map.put(plugin.name().toLowerCase(), pluginType);
166                final PluginAliases pluginAliases = clazz.getAnnotation(PluginAliases.class);
167                if (pluginAliases != null) {
168                    for (String alias : pluginAliases.value()) {
169                        type =  plugin.elementType().equals(Plugin.EMPTY) ? alias : plugin.elementType();
170                        pluginType = new PluginType(clazz, type, plugin.printObject(), plugin.deferChildren());
171                        map.put(alias.trim().toLowerCase(), pluginType);
172                    }
173                }
174            }
175            plugins = REGISTRY.getCategory(category);
176        }
177    
178        private void reportPluginLoadDuration(final boolean preLoad, long elapsed) {
179            final StringBuilder sb = new StringBuilder("Generated plugins in ");
180            DecimalFormat numFormat = new DecimalFormat("#0.000000");
181            final double seconds = elapsed / (1000.0 * 1000.0 * 1000.0);
182            sb.append(numFormat.format(seconds)).append(" seconds, packages: ");
183            sb.append(PACKAGES);
184            sb.append(", preload: ");
185            sb.append(preLoad);
186            sb.append(".");
187            LOGGER.debug(sb.toString());
188        }

190        public static void loadPlugins(final ResourceLoader loader) {
191            final PluginRegistry<PluginType<?>> registry = decode(loader);
192            if (registry != null) {
193                for (final Map.Entry<String, ConcurrentMap<String, PluginType<?>>> entry : registry.getCategories()) {
194                    REGISTRY.getCategory(entry.getKey()).putAll(entry.getValue());
195                }
196            } else {
197                LOGGER.info("Plugin preloads not available from class loader {}", loader);
198            }
199        }
200    
201        private static PluginRegistry<PluginType<?>> decode(final ResourceLoader loader) {
202            final Enumeration<URL> resources;
203            try {
204                resources = loader.getResources(PluginProcessor.PLUGIN_CACHE_FILE);
205                if (resources == null) {
206                    return null;
207                }
208            } catch (final IOException ioe) {
209                LOGGER.warn("Unable to preload plugins", ioe);
210                return null;
211            }
212            final PluginRegistry<PluginType<?>> map = new PluginRegistry<PluginType<?>>();
213            while (resources.hasMoreElements()) {
214                final URL url = resources.nextElement();
215                LOGGER.debug("Found Plugin Map at {}", url.toExternalForm());
216                final InputStream is;
217                try {
218                    is = url.openStream();
219                } catch (final IOException e) {
220                    LOGGER.warn("Unable to open {}", url.toExternalForm(), e);
221                    continue;
222                }
223                final DataInputStream dis = new DataInputStream(new BufferedInputStream(is));
224                try {
225                    final int count = dis.readInt();
226                    for (int j = 0; j < count; ++j) {
227                        final String category = dis.readUTF();
228                        final int entries = dis.readInt();
229                        final Map<String, PluginType<?>> types = map.getCategory(category);
230                        for (int i = 0; i < entries; ++i) {
231                            final String key = dis.readUTF();
232                            final String className = dis.readUTF();
233                            final String name = dis.readUTF();
234                            final boolean printable = dis.readBoolean();
235                            final boolean defer = dis.readBoolean();
236                            try {
237                                final Class<?> clazz = loader.loadClass(className);
238                                @SuppressWarnings({ "unchecked", "rawtypes" })
239                                final PluginType<?> pluginType = new PluginType(clazz, name, printable, defer);
240                                types.put(key, pluginType);
241                            } catch (final ClassNotFoundException e) {
242                                LOGGER.info("Plugin [{}] could not be loaded due to missing classes.", className, e);
243                            } catch (final VerifyError e) {
244                                LOGGER.info("Plugin [{}] could not be loaded due to verification error.", className, e);
245                            }
246                        }
247                    }
248                } catch (final IOException ex) {
249                    LOGGER.warn("Unable to preload plugins", ex);
250                } finally {
251                    Closer.closeSilently(dis);
252                }
253            }
254            return map.isEmpty() ? null : map;
255        }

257        /**
258         * A Test that checks to see if each class is annotated with a specific annotation. If it
259         * is, then the test returns true, otherwise false.
260         */
261        public static class PluginTest implements ResolverUtil.Test {
262            private final Class<?> isA;
263    
264            /**
265             * Constructs an AnnotatedWith test for the specified annotation type.
266             * @param isA The class to compare against.
267             */
268            public PluginTest(final Class<?> isA) {
269                this.isA = isA;
270            }
271    
272            /**
273             * Returns true if the type is annotated with the class provided to the constructor.
274             * @param type The type to check for.
275             * @return true if the Class is of the specified type.
276             */
277            @Override
278            public boolean matches(final Class<?> type) {
279                return type != null && type.isAnnotationPresent(Plugin.class) &&
280                    (isA == null || isA.isAssignableFrom(type));
281            }
282    
283            @Override
284            public String toString() {
285                final StringBuilder msg = new StringBuilder("annotated with @" + Plugin.class.getSimpleName());
286                if (isA != null) {
287                    msg.append(" is assignable to " + isA.getSimpleName());
288                }
289                return msg.toString();
290            }
291    
292            @Override
293            public boolean matches(final URI resource) {
294                throw new UnsupportedOperationException();
295            }
296    
297            @Override
298            public boolean doesMatchClass() {
299                return true;
300            }
301    
302            @Override
303            public boolean doesMatchResource() {
304                return false;
305            }
306        }

017    package org.apache.logging.log4j.core.appender.db.jpa.converter;
018    
019    import javax.persistence.AttributeConverter;
020    import javax.persistence.Converter;
021    
022    import org.apache.logging.log4j.ThreadContext;
023    
024    /**
025     * A JPA 2.1 attribute converter for {@link ThreadContext.ContextStack}s in
026     * {@link org.apache.logging.log4j.core.LogEvent}s. This converter is only capable of converting to {@link String}s. The
027     * {@link #convertToEntityAttribute(String)} method throws an {@link UnsupportedOperationException}. If you need to
028     * support converting to an entity attribute, you should use the {@link ContextStackJsonAttributeConverter} for
029     * conversion both ways.
030     */
031    @Converter(autoApply = false)
032    public class ContextStackAttributeConverter implements AttributeConverter<ThreadContext.ContextStack, String> {
033        @Override
034        public String convertToDatabaseColumn(final ThreadContext.ContextStack contextStack) {
035            StringBuilder builder = new StringBuilder();
036            for (String value : contextStack.asList()) {
037                if (builder.length() > 0) {
038                    builder.append('\n');
039                }
040                builder.append(value);
041            }
042            return builder.toString();
043        }
044    
045        @Override
046        public ThreadContext.ContextStack convertToEntityAttribute(final String s) {
047            throw new UnsupportedOperationException("Log events can only be persisted, not extracted.");
048        }
049    }

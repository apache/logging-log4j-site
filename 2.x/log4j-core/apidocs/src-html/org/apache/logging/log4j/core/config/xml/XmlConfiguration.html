073        /**
074         * Creates a new DocumentBuilder suitable for parsing a configuration file.
075         *
076         * @return a new DocumentBuilder
077         * @throws ParserConfigurationException
078         */
079        static DocumentBuilder newDocumentBuilder() throws ParserConfigurationException {
080            final DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
081            factory.setNamespaceAware(true);
082            enableXInclude(factory);
083            return factory.newDocumentBuilder();
084        }
085    
086        /**
087         * Enables XInclude for the given DocumentBuilderFactory
088         *
089         * @param factory a DocumentBuilderFactory
090         */
091        private static void enableXInclude(final DocumentBuilderFactory factory) {
092            try {
093                // Alternative: We set if a system property on the command line is set, for example:
094                // -DLog4j.XInclude=true
095                factory.setXIncludeAware(true);
096            } catch (final UnsupportedOperationException e) {
097                LOGGER.warn("The DocumentBuilderFactory does not support XInclude: {}", factory, e);
098            } catch (@SuppressWarnings("ErrorNotRethrown") final AbstractMethodError err) {
099                LOGGER.warn("The DocumentBuilderFactory is out of date and does not support XInclude: {}", factory, err);
100            }
101            try {
102                // Alternative: We could specify all features and values with system properties like:
103                // -DLog4j.DocumentBuilderFactory.Feature="http://apache.org/xml/features/xinclude/fixup-base-uris true"
104                factory.setFeature(XINCLUDE_FIXUP_BASE_URIS, true);
105            } catch (final ParserConfigurationException e) {
106                LOGGER.warn("The DocumentBuilderFactory [{}] does not support the feature [{}].", factory,
107                        XINCLUDE_FIXUP_BASE_URIS, e);
108            } catch (@SuppressWarnings("ErrorNotRethrown") final AbstractMethodError err) {
109                LOGGER.warn("The DocumentBuilderFactory is out of date and does not support setFeature: {}", factory, err);
110            }
111            try {
112                factory.setFeature(XINCLUDE_FIXUP_LANGUAGE, true);
113            } catch (final ParserConfigurationException e) {
114                LOGGER.warn("The DocumentBuilderFactory [{}] does not support the feature [{}].", factory,
115                        XINCLUDE_FIXUP_LANGUAGE, e);
116            } catch (@SuppressWarnings("ErrorNotRethrown") final AbstractMethodError err) {
117                LOGGER.warn("The DocumentBuilderFactory is out of date and does not support setFeature: {}", factory, err);
118            }
119        }

121        public XmlConfiguration(final ConfigurationSource configSource) {
122            super(configSource);
123            final File configFile = configSource.getFile();
124            byte[] buffer = null;
125    
126            try {
127                final InputStream configStream = configSource.getInputStream();
128                try {
129                    buffer = toByteArray(configStream);
130                } finally {
131                    configStream.close();
132                }
133                final InputSource source = new InputSource(new ByteArrayInputStream(buffer));
134                source.setSystemId(configSource.getLocation());
135                final Document document = newDocumentBuilder().parse(source);
136                rootElement = document.getDocumentElement();
137                final Map<String, String> attrs = processAttributes(rootNode, rootElement);
138                final StatusConfiguration statusConfig = new StatusConfiguration().withVerboseClasses(VERBOSE_CLASSES)
139                        .withStatus(getDefaultStatus());
140                for (final Map.Entry<String, String> entry : attrs.entrySet()) {
141                    final String key = entry.getKey();
142                    final String value = getStrSubstitutor().replace(entry.getValue());
143                    if ("status".equalsIgnoreCase(key)) {
144                        statusConfig.withStatus(value);
145                    } else if ("dest".equalsIgnoreCase(key)) {
146                        statusConfig.withDestination(value);
147                    } else if ("shutdownHook".equalsIgnoreCase(key)) {
148                        isShutdownHookEnabled = !"disable".equalsIgnoreCase(value);
149                    } else if ("verbose".equalsIgnoreCase(key)) {
150                        statusConfig.withVerbosity(value);
151                    } else if ("packages".equalsIgnoreCase(key)) {
152                        final String[] packages = value.split(Patterns.COMMA_SEPARATOR);
153                        for (final String p : packages) {
154                            PluginManager.addPackage(p);
155                        }
156                    } else if ("name".equalsIgnoreCase(key)) {
157                        setName(value);
158                    } else if ("strict".equalsIgnoreCase(key)) {
159                        strict = Boolean.parseBoolean(value);
160                    } else if ("schema".equalsIgnoreCase(key)) {
161                        schema = value;
162                    } else if ("monitorInterval".equalsIgnoreCase(key)) {
163                        final int interval = Integer.parseInt(value);
164                        if (interval > 0 && configFile != null) {
165                            monitor = new FileConfigurationMonitor(this, configFile, listeners, interval);
166                        }
167                    } else if ("advertiser".equalsIgnoreCase(key)) {
168                        createAdvertiser(value, configSource, buffer, "text/xml");
169                    }
170                }
171                statusConfig.initialize();
172            } catch (final SAXException domEx) {
173                LOGGER.error("Error parsing " + configSource.getLocation(), domEx);
174            } catch (final IOException ioe) {
175                LOGGER.error("Error parsing " + configSource.getLocation(), ioe);
176            } catch (final ParserConfigurationException pex) {
177                LOGGER.error("Error parsing " + configSource.getLocation(), pex);
178            }
179            if (strict && schema != null && buffer != null) {
180                InputStream is = null;
181                try {
182                    is = Loader.getResourceAsStream(schema, XmlConfiguration.class.getClassLoader());
183                } catch (final Exception ex) {
184                    LOGGER.error("Unable to access schema {}", this.schema, ex);
185                }
186                if (is != null) {
187                    final Source src = new StreamSource(is, LOG4J_XSD);
188                    final SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
189                    Schema schema = null;
190                    try {
191                        schema = factory.newSchema(src);
192                    } catch (final SAXException ex) {
193                        LOGGER.error("Error parsing Log4j schema", ex);
194                    }
195                    if (schema != null) {
196                        final Validator validator = schema.newValidator();
197                        try {
198                            validator.validate(new StreamSource(new ByteArrayInputStream(buffer)));
199                        } catch (final IOException ioe) {
200                            LOGGER.error("Error reading configuration for validation", ioe);
201                        } catch (final SAXException ex) {
202                            LOGGER.error("Error validating configuration", ex);
203                        }
204                    }
205                }
206            }
207    
208            if (getName() == null) {
209                setName(configSource.getLocation());
210            }
211        }

213        @Override
214        public void setup() {
215            if (rootElement == null) {
216                LOGGER.error("No logging configuration");
217                return;
218            }
219            constructHierarchy(rootNode, rootElement);
220            if (status.size() > 0) {
221                for (final Status s : status) {
222                    LOGGER.error("Error processing element " + s.name + ": " + s.errorType);
223                }
224                return;
225            }
226            rootElement = null;
227        }
228    
229        @Override
230        public Configuration reconfigure() {
231            try {
232                final ConfigurationSource source = getConfigurationSource().resetInputStream();
233                if (source == null) {
234                    return null;
235                }
236                final XmlConfiguration config = new XmlConfiguration(source);
237                return (config.rootElement == null) ? null : config;
238            } catch (final IOException ex) {
239                LOGGER.error("Cannot locate file " + getConfigurationSource(), ex);
240            }
241            return null;
242        }
243    
244        private void constructHierarchy(final Node node, final Element element) {
245            processAttributes(node, element);
246            final StringBuilder buffer = new StringBuilder();
247            final NodeList list = element.getChildNodes();
248            final List<Node> children = node.getChildren();
249            for (int i = 0; i < list.getLength(); i++) {
250                final org.w3c.dom.Node w3cNode = list.item(i);
251                if (w3cNode instanceof Element) {
252                    final Element child = (Element) w3cNode;
253                    final String name = getType(child);
254                    final PluginType<?> type = pluginManager.getPluginType(name);
255                    final Node childNode = new Node(node, name, type);
256                    constructHierarchy(childNode, child);
257                    if (type == null) {
258                        final String value = childNode.getValue();
259                        if (!childNode.hasChildren() && value != null) {
260                            node.getAttributes().put(name, value);
261                        } else {
262                            status.add(new Status(name, element, ErrorType.CLASS_NOT_FOUND));
263                        }
264                    } else {
265                        children.add(childNode);
266                    }
267                } else if (w3cNode instanceof Text) {
268                    final Text data = (Text) w3cNode;
269                    buffer.append(data.getData());
270                }
271            }
272    
273            final String text = buffer.toString().trim();
274            if (text.length() > 0 || (!node.hasChildren() && !node.isRoot())) {
275                node.setValue(text);
276            }
277        }
278    
279        private String getType(final Element element) {
280            if (strict) {
281                final NamedNodeMap attrs = element.getAttributes();
282                for (int i = 0; i < attrs.getLength(); ++i) {
283                    final org.w3c.dom.Node w3cNode = attrs.item(i);
284                    if (w3cNode instanceof Attr) {
285                        final Attr attr = (Attr) w3cNode;
286                        if (attr.getName().equalsIgnoreCase("type")) {
287                            final String type = attr.getValue();
288                            attrs.removeNamedItem(attr.getName());
289                            return type;
290                        }
291                    }
292                }
293            }
294            return element.getTagName();
295        }
296    
297        private Map<String, String> processAttributes(final Node node, final Element element) {
298            final NamedNodeMap attrs = element.getAttributes();
299            final Map<String, String> attributes = node.getAttributes();
300    
301            for (int i = 0; i < attrs.getLength(); ++i) {
302                final org.w3c.dom.Node w3cNode = attrs.item(i);
303                if (w3cNode instanceof Attr) {
304                    final Attr attr = (Attr) w3cNode;
305                    if (attr.getName().equals("xml:base")) {
306                        continue;
307                    }
308                    attributes.put(attr.getName(), attr.getValue());
309                }
310            }
311            return attributes;
312        }

314        @Override
315        public String toString() {
316            return getClass().getSimpleName() + "[location=" + getConfigurationSource() + "]";
317        }

322        private enum ErrorType {
323            CLASS_NOT_FOUND
324        }

326        /**
327         * Status for recording errors.
328         */
329        private static class Status {
330            private final Element element;
331            private final String name;
332            private final ErrorType errorType;
333    
334            public Status(final String name, final Element element, final ErrorType errorType) {
335                this.name = name;
336                this.element = element;
337                this.errorType = errorType;
338            }

034/**
035 * <p>
036 * Users of the JPA appender MUST extend this class, using JPA annotations on the concrete class and all of its
037 * accessor methods (as needed) to map them to the proper table and columns. Accessors you do not want persisted should
038 * be annotated with {@link Transient @Transient}. All accessors should call {@link #getWrappedEvent()} and delegate the
039 * call to the underlying event. Users may want to instead extend {@link BasicLogEventEntity}, which takes care of all
040 * of this for you.
041 * </p>
042 * <p>
043 * The concrete class must have two constructors: a public no-arg constructor to convince the JPA provider that it's a
044 * valid entity, and a public constructor that takes a single {@link LogEvent event} and passes it to the parent class
045 * with {@link #AbstractLogEventWrapperEntity(LogEvent) super(event)}. Furthermore, the concrete class must be annotated
046 * {@link javax.persistence.Entity @Entity} and {@link javax.persistence.Table @Table} and must implement a fully
047 * mutable ID property annotated with {@link javax.persistence.Id @Id} and
048 * {@link javax.persistence.GeneratedValue @GeneratedValue} to tell the JPA provider how to calculate an ID for new
049 * events.
050 * </p>
051 * <p>
052 * Many of the return types of {@link LogEvent} methods (e.g., {@link StackTraceElement}, {@link Message},
053 * {@link Marker}, {@link Throwable},
054 * {@link org.apache.logging.log4j.ThreadContext.ContextStack ThreadContext.ContextStack}, and
055 * {@link Map Map&lt;String, String&gt;}) will not be recognized by the JPA provider. In conjunction with
056 * {@link javax.persistence.Convert @Convert}, you can use the converters in the
057 * {@link org.apache.logging.log4j.core.appender.db.jpa.converter} package to convert these types to database columns.
058 * If you want to retrieve log events from the database, you can create a true POJO entity and also use these
059 * converters for extracting persisted values.<br>
060 * </p>
061 * <p>
062 * The mutator methods in this class not specified in {@link LogEvent} are no-op methods, implemented to satisfy the JPA
063 * requirement that accessor methods have matching mutator methods. If you create additional accessor methods, you must
064 * likewise create matching no-op mutator methods.
065 * </p>
066 *
067 * @see BasicLogEventEntity
068 */
069@MappedSuperclass
070@Inheritance(strategy = InheritanceType.SINGLE_TABLE)
071public abstract class AbstractLogEventWrapperEntity implements LogEvent {
072    private static final long serialVersionUID = 1L;
073
074    private final LogEvent wrappedEvent;
075
076    /**
077     * Instantiates this base class. All concrete implementations must have a constructor matching this constructor's
078     * signature. The no-argument constructor is required for a standards-compliant JPA provider to accept this as an
079     * entity.
080     */
081    @SuppressWarnings("unused")
082    protected AbstractLogEventWrapperEntity() {
083        this(new NullLogEvent());
084    }
085
086    /**
087     * Instantiates this base class. All concrete implementations must have a constructor matching this constructor's
088     * signature. This constructor is used for wrapping this entity around a logged event.
089     *
090     * @param wrappedEvent The underlying event from which information is obtained.
091     */
092    protected AbstractLogEventWrapperEntity(final LogEvent wrappedEvent) {
093        if (wrappedEvent == null) {
094            throw new IllegalArgumentException("The wrapped event cannot be null.");
095        }
096        this.wrappedEvent = wrappedEvent;
097    }
098
099    /**
100     * All eventual accessor methods must call this method and delegate the method call to the underlying wrapped event.
101     * Annotated {@link Transient} so as not to be included in the persisted entity.
102     *
103     * @return The underlying event from which information is obtained.
104     */
105    @Transient
106    protected final LogEvent getWrappedEvent() {
107        return this.wrappedEvent;
108    }

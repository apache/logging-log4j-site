017    package org.apache.logging.log4j.core.appender.rolling.helper;
018    
019    import java.io.IOException;
020    
021    import java.util.List;
022    
023    
024    /**
025     * A group of Actions to be executed in sequence.
026     */
027    public class CompositeAction extends ActionBase {
028        /**
029         * Actions to perform.
030         */
031        private final Action[] actions;
032    
033        /**
034         * Stop on error.
035         */
036        private final boolean stopOnError;
037    
038        /**
039         * Construct a new composite action.
040         *
041         * @param actions     list of actions, may not be null.
042         * @param stopOnError if true, stop on the first false return value or exception.
043         */
044        public CompositeAction(final List actions,
045                               final boolean stopOnError) {
046            this.actions = new Action[actions.size()];
047            actions.toArray(this.actions);
048            this.stopOnError = stopOnError;
049        }
050    
051        /**
052         * {@inheritDoc}
053         */
054        public void run() {
055            try {
056                execute();
057            } catch (IOException ex) {
058                LOGGER.warn("Exception during file rollover.", ex);
059            }
060        }
061    
062        /**
063         * Execute sequence of actions.
064         *
065         * @return true if all actions were successful.
066         * @throws IOException on IO error.
067         */
068        public boolean execute() throws IOException {
069            if (stopOnError) {
070                for (Action action : actions) {
071                    if (!action.execute()) {
072                        return false;
073                    }
074                }
075    
076                return true;
077            } else {
078                boolean status = true;
079                IOException exception = null;
080    
081                for (Action action : actions) {
082                    try {
083                        status &= action.execute();
084                    } catch (IOException ex) {
085                        status = false;
086    
087                        if (exception == null) {
088                            exception = ex;
089                        }
090                    }
091                }
092    
093                if (exception != null) {
094                    throw exception;
095                }
096    
097                return status;
098            }
099        }
100    }

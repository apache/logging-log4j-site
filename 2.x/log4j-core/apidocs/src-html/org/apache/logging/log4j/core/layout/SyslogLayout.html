017    package org.apache.logging.log4j.core.layout;
018    
019    import org.apache.logging.log4j.core.LogEvent;
020    import org.apache.logging.log4j.core.config.plugins.Plugin;
021    import org.apache.logging.log4j.core.config.plugins.PluginAttr;
022    import org.apache.logging.log4j.core.config.plugins.PluginFactory;
023    import org.apache.logging.log4j.core.net.Facility;
024    import org.apache.logging.log4j.core.net.Priority;
025    
026    import java.net.InetAddress;
027    import java.net.UnknownHostException;
028    import java.nio.charset.Charset;
029    import java.text.SimpleDateFormat;
030    import java.util.Date;
031    import java.util.Locale;
032    
033    
034    /**
035     * Formats a log event as a BSD Log record.
036     */
037    @Plugin(name = "SyslogLayout", type = "Core", elementType = "layout", printObject = true)
038    public class SyslogLayout extends AbstractStringLayout {
039    
040        private final Facility facility;
041        private final boolean includeNewLine;
042    
043        /**
044         * Date format used if header = true.
045         */
046        private final SimpleDateFormat dateFormat = new SimpleDateFormat("MMM dd HH:mm:ss ", Locale.ENGLISH);
047        /**
048         * Host name used to identify messages from this appender.
049         */
050        private final String localHostname = getLocalHostname();
051    
052    
053        protected SyslogLayout(Facility facility, boolean includeNL, Charset c) {
054            super(c);
055            this.facility = facility;
056            this.includeNewLine = includeNL;
057        }
058    
059        /**
060         * Formats a {@link org.apache.logging.log4j.core.LogEvent} in conformance with the log4j.dtd.
061         * @param event The LogEvent
062         * @return the event formatted as a String.
063         */
064        public String formatAs(final LogEvent event) {
065            StringBuilder buf = new StringBuilder();
066    
067            buf.append("<");
068            buf.append(Priority.getPriority(facility, event.getLevel()));
069            buf.append(">");
070            addDate(event.getMillis(), buf);
071            buf.append(" ");
072            buf.append(localHostname);
073            buf.append(" ");
074            buf.append(event.getMessage().getFormattedMessage());
075            if (includeNewLine) {
076                buf.append("\n");
077            }
078            return buf.toString();
079        }
080    
081        /**
082         * This method gets the network name of the machine we are running on.
083         * Returns "UNKNOWN_LOCALHOST" in the unlikely case where the host name
084         * cannot be found.
085         *
086         * @return String the name of the local host
087         */
088        private String getLocalHostname() {
089            try {
090                InetAddress addr = InetAddress.getLocalHost();
091                return addr.getHostName();
092            } catch (UnknownHostException uhe) {
093                LOGGER.error("Could not determine local host name", uhe);
094                return "UNKNOWN_LOCALHOST";
095            }
096        }
097    
098        private synchronized void addDate(final long timestamp, StringBuilder buf) {
099            int index = buf.length() + 4;
100            buf.append(dateFormat.format(new Date(timestamp)));
101            //  RFC 3164 says leading space, not leading zero on days 1-9
102            if (buf.charAt(index) == '0') {
103                buf.setCharAt(index, ' ');
104            }
105        }
106    
107        /**
108         * Create a SyslogLayout.
109         * @param facility The Facility is used to try to classify the message.
110         * @param includeNL If true a newline will be appended to the result.
111         * @param charset The character set.
112         * @return A SyslogLayout.
113         */
114        @PluginFactory
115        public static SyslogLayout createLayout(@PluginAttr("facility") String facility,
116                                                @PluginAttr("newLine") String includeNL,
117                                                @PluginAttr("charset") String charset) {
118    
119            Charset c = Charset.isSupported("UTF-8") ? Charset.forName("UTF-8") : Charset.defaultCharset();
120            if (charset != null) {
121                if (Charset.isSupported(charset)) {
122                    c = Charset.forName(charset);
123                } else {
124                    LOGGER.error("Charset " + charset + " is not supported for layout, using " + c.displayName());
125                }
126            }
127            boolean includeNewLine = includeNL == null ? false : Boolean.valueOf(includeNL);
128            Facility f = facility != null ? Facility.valueOf(facility.toUpperCase()) : Facility.LOCAL0;
129            return new SyslogLayout(f, includeNewLine, c);
130        }
131    }

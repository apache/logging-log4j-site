067    /**
068     * Constructs a new {@code ConfigurationSource} with the specified input stream that originated from the specified
069     * file.
070     *
071     * @param stream the input stream
072     * @param file the file where the input stream originated
073     */
074    public ConfigurationSource(final InputStream stream, final File file) {
075        this.stream = Objects.requireNonNull(stream, "stream is null");
076        this.file = Objects.requireNonNull(file, "file is null");
077        this.location = file.getAbsolutePath();
078        this.url = null;
079        this.data = null;
080        long modified = 0;
081        try {
082            modified = file.lastModified();
083        } catch (Exception ex) {
084            // There is a problem with the file. It will be handled somewhere else.
085        }
086        this.lastModified = modified;
087    }

133    public ConfigurationSource(final Source source, final byte[] data, long lastModified) throws IOException {
134        Objects.requireNonNull(source, "source is null");
135        this.data = Objects.requireNonNull(data, "data is null");
136        this.stream = new ByteArrayInputStream(data);
137        this.file = source.getFile();
138        this.url = source.getURI().toURL();
139        this.location = source.getLocation();
140        this.lastModified = lastModified;
141    }

143    private ConfigurationSource(final byte[] data, final URL url, long lastModified) {
144        Objects.requireNonNull(data, "data is null");
145        this.stream = new ByteArrayInputStream(data);
146        this.file = null;
147        this.url = url;
148        this.location = null;
149        this.lastModified = lastModified;
150    }

152    /**
153     * Returns the contents of the specified {@code InputStream} as a byte array.
154     *
155     * @param inputStream the stream to read
156     * @return the contents of the specified stream
157     * @throws IOException if a problem occurred reading from the stream
158     */
159    private static byte[] toByteArray(final InputStream inputStream) throws IOException {
160        final int buffSize = Math.max(4096, inputStream.available());
161        final ByteArrayOutputStream contents = new ByteArrayOutputStream(buffSize);
162        final byte[] buff = new byte[buffSize];
163
164        int length = inputStream.read(buff);
165        while (length > 0) {
166            contents.write(buff, 0, length);
167            length = inputStream.read(buff);
168        }
169        return contents.toByteArray();
170    }

204    /**
205     * Returns a URI representing the configuration resource or null if it cannot be determined.
206     * @return The URI.
207     */
208    public URI getURI() {
209        URI sourceURI = null;
210        if (url != null) {
211            try {
212                sourceURI = url.toURI();
213            } catch (final URISyntaxException ex) {
214                    /* Ignore the exception */
215            }
216        }
217        if (sourceURI == null && file != null) {
218            sourceURI = file.toURI();
219        }
220        if (sourceURI == null && location != null) {
221            try {
222                sourceURI = new URI(location);
223            } catch (final URISyntaxException ex) {
224                // Assume the scheme was missing.
225                try {
226                    sourceURI = new URI("file://" + location);
227                } catch (final URISyntaxException uriEx) {
228                    /* Ignore the exception */
229                }
230            }
231        }
232        return sourceURI;
233    }

262    /**
263     * Returns a new {@code ConfigurationSource} whose input stream is reset to the beginning.
264     *
265     * @return a new {@code ConfigurationSource}
266     * @throws IOException if a problem occurred while opening the new input stream
267     */
268    public ConfigurationSource resetInputStream() throws IOException {
269        if (source != null) {
270            return new ConfigurationSource(source, data, this.lastModified);
271        } else if (file != null) {
272            return new ConfigurationSource(new FileInputStream(file), file);
273        } else if (url != null && data != null) {
274            // Creates a ConfigurationSource without accessing the URL since the data was provided.
275            return new ConfigurationSource(data, url, modifiedMillis == 0 ? lastModified : modifiedMillis);
276        } else if (url != null) {
277            return fromUri(getURI());
278        } else if (data != null) {
279            return new ConfigurationSource(data, null, lastModified);
280        }
281        return null;
282    }

284    @Override
285    public String toString() {
286        if (location != null) {
287            return location;
288        }
289        if (this == NULL_SOURCE) {
290            return "NULL_SOURCE";
291        }
292        final int length = data == null ? -1 : data.length;
293        return "stream (" + length + " bytes, unknown location)";
294    }

296    /**
297     * Loads the configuration from a URI.
298     * @param configLocation A URI representing the location of the configuration.
299     * @return The ConfigurationSource for the configuration.
300     */
301    public static ConfigurationSource fromUri(final URI configLocation) {
302        final File configFile = FileUtils.fileFromUri(configLocation);
303        if (configFile != null && configFile.exists() && configFile.canRead()) {
304            try {
305                return new ConfigurationSource(new FileInputStream(configFile), configFile);
306            } catch (final FileNotFoundException ex) {
307                ConfigurationFactory.LOGGER.error("Cannot locate file {}", configLocation.getPath(), ex);
308            }
309        }
310        if (ConfigurationFactory.isClassLoaderUri(configLocation)) {
311            final ClassLoader loader = LoaderUtil.getThreadContextClassLoader();
312            final String path = ConfigurationFactory.extractClassLoaderUriPath(configLocation);
313            final ConfigurationSource source = fromResource(path, loader);
314            if (source != null) {
315                return source;
316            }
317        }
318        if (!configLocation.isAbsolute()) { // LOG4J2-704 avoid confusing error message thrown by uri.toURL()
319            ConfigurationFactory.LOGGER.error("File not found in file system or classpath: {}", configLocation.toString());
320            return null;
321        }
322        try {
323            URL url = configLocation.toURL();
324            URLConnection urlConnection = UrlConnectionFactory.createConnection(url);
325            InputStream is = urlConnection.getInputStream();
326            long lastModified = urlConnection.getLastModified();
327            return new ConfigurationSource(is, configLocation.toURL(), lastModified);
328        } catch (final MalformedURLException ex) {
329            ConfigurationFactory.LOGGER.error("Invalid URL {}", configLocation.toString(), ex);
330        } catch (final Exception ex) {
331            ConfigurationFactory.LOGGER.error("Unable to access {}", configLocation.toString(), ex);
332        }
333        return null;
334    }

336    /**
337     * Retrieves the configuration via the ClassLoader.
338     * @param resource The resource to load.
339     * @param loader The default ClassLoader to use.
340     * @return The ConfigurationSource for the configuration.
341     */
342    public static ConfigurationSource fromResource(final String resource, final ClassLoader loader) {
343        final URL url = Loader.getResource(resource, loader);
344        if (url == null) {
345            return null;
346        }
347        InputStream is = null;
348        try {
349            is = url.openStream();
350        } catch (final IOException ioe) {
351            ConfigurationFactory.LOGGER.catching(Level.DEBUG, ioe);
352            return null;
353        }
354        if (is == null) {
355            return null;
356        }
357
358        if (FileUtils.isFile(url)) {
359            try {
360                return new ConfigurationSource(is, FileUtils.fileFromUri(url.toURI()));
361            } catch (final URISyntaxException ex) {
362                // Just ignore the exception.
363                ConfigurationFactory.LOGGER.catching(Level.DEBUG, ex);
364            }
365        }
366        return new ConfigurationSource(is, url);
367    }

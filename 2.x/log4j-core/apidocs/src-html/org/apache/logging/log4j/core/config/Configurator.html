017    package org.apache.logging.log4j.core.config;
018    
019    import org.apache.logging.log4j.LogManager;
020    import org.apache.logging.log4j.core.LoggerContext;
021    
022    import java.net.URI;
023    import java.net.URISyntaxException;
024    
025    /**
026     * Class that can be used to initialize and configure the Logging system.
027     */
028    public final class Configurator {
029    
030        private Configurator() {
031        }
032    
033        /**
034         * Initialize the Logging Context.
035         * @param name The Context name.
036         * @param loader The ClassLoader for the Context (or null).
037         * @param configLocation The configuration for the logging context.
038         * @return The LoggerContext.
039         */
040        public static LoggerContext initialize(String name, ClassLoader loader, String configLocation) {
041    
042            try {
043                URI uri = configLocation == null ? null : new URI(configLocation);
044                return initialize(name, loader, uri);
045            } catch (URISyntaxException ex) {
046                ex.printStackTrace();
047            }
048            return null;
049        }
050    
051        /**
052         * Initialize the Logging Context.
053         * @param name The Context name.
054         * @param loader The ClassLoader for the Context (or null).
055         * @param configLocation The configuration for the logging context.
056         * @return The LoggerContext.
057         */
058        public static LoggerContext initialize(String name, ClassLoader loader, URI configLocation) {
059    
060            try {
061                LoggerContext ctx = (LoggerContext) LogManager.getContext(loader, false);
062                Configuration config = ConfigurationFactory.getInstance().getConfiguration(name, configLocation);
063                ctx.setConfiguration(config);
064                return ctx;
065            } catch (Exception ex) {
066                ex.printStackTrace();
067            }
068            return null;
069        }
070    
071        public static void shutdown(LoggerContext ctx) {
072            if (ctx != null) {
073                ctx.setConfiguration(new DefaultConfiguration());
074            }
075        }
076    
077    }

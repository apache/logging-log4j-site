017    package org.apache.logging.log4j.core.impl;
018    
019    import org.apache.logging.log4j.Level;
020    import org.apache.logging.log4j.Marker;
021    import org.apache.logging.log4j.core.LogEvent;
022    import org.apache.logging.log4j.core.config.Property;
023    import org.apache.logging.log4j.message.Message;
024    
025    import java.util.List;
026    
027    /**
028     *
029     */
030    public class DefaultLogEventFactory implements LogEventFactory {
031    
032    
033        /**
034         * Creates a log event.
035         *
036         * @param loggerName The name of the Logger.
037         * @param marker An optional Marker.
038         * @param fqcn The fully qualified class name of the caller.
039         * @param level The event Level.
040         * @param data The Message.
041         * @param properties Properties to be added to the log event.
042         * @param t An optional Throwable.
043         * @return The LogEvent.
044         */
045        @Override
046        public LogEvent createEvent(final String loggerName, final Marker marker,
047                                    final String fqcn, final Level level, final Message data,
048                                    final List<Property> properties, final Throwable t) {
049            return new Log4jLogEvent(loggerName, marker, fqcn, level, data, properties, t);
050        }
051    }

017package org.apache.logging.log4j.core.appender.db.jpa.converter;
018
019import java.io.IOException;
020import java.util.Map;
021
022import javax.persistence.AttributeConverter;
023import javax.persistence.Converter;
024import javax.persistence.PersistenceException;
025
026import org.apache.logging.log4j.util.Strings;
027
028import com.fasterxml.jackson.core.type.TypeReference;
029import com.fasterxml.jackson.databind.ObjectMapper;
030
031/**
032 * A JPA 2.1 attribute converter for {@link Map Map&lt;String, String&gt;}s in
033 * {@link org.apache.logging.log4j.core.LogEvent}s. This converter is capable of converting both to and from
034 * {@link String}s.
035 *
036 * In addition to other optional dependencies required by the JPA appender, this converter requires the Jackson Data
037 * Processor.
038 */
039@Converter(autoApply = false)
040public class ContextMapJsonAttributeConverter implements AttributeConverter<Map<String, String>, String> {
041    static final ObjectMapper OBJECT_MAPPER = new ObjectMapper();
042
043    @Override
044    public String convertToDatabaseColumn(final Map<String, String> contextMap) {
045        if (contextMap == null) {
046            return null;
047        }
048
049        try {
050            return OBJECT_MAPPER.writeValueAsString(contextMap);
051        } catch (final IOException e) {
052            throw new PersistenceException("Failed to convert map to JSON string.", e);
053        }
054    }
055
056    @Override
057    public Map<String, String> convertToEntityAttribute(final String s) {
058        if (Strings.isEmpty(s)) {
059            return null;
060        }
061        try {
062            return OBJECT_MAPPER.readValue(s, new TypeReference<Map<String, String>>() { });
063        } catch (final IOException e) {
064            throw new PersistenceException("Failed to convert JSON string to map.", e);
065        }
066    }
067}

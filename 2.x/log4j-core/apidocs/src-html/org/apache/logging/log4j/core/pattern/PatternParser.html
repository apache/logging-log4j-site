032    /**
033     * Most of the work of the {@link org.apache.logging.log4j.core.layout.PatternLayout} class
034     * is delegated to the PatternParser class.
035     * <p>It is this class that parses conversion patterns and creates
036     * a chained list of {@link PatternConverter PatternConverters}.
037     */
038    public final class PatternParser {
039        /**
040         * Escape character for format specifier.
041         */
042        private static final char ESCAPE_CHAR = '%';
043    
044        /**
045         * The states the parser can be in while parsing the pattern.
046         */
047        private enum ParserState {
048            /**
049             * Literal state.
050             */
051            LITERAL_STATE,
052    
053            /**
054             * In converter name state.
055             */
056            CONVERTER_STATE,
057    
058            /**
059             * Dot state.
060             */
061            DOT_STATE,
062    
063            /**
064             * Min state.
065             */
066            MIN_STATE,
067    
068            /**
069             * Max state.
070             */
071            MAX_STATE;
072        }
073    
074        private static final Logger LOGGER = StatusLogger.getLogger();
075    
076        private static final int BUF_SIZE = 32;
077    
078        private static final int DECIMAL = 10;
079    
080        /**
081         * Does pattern process exceptions.
082         */
083        private boolean handlesExceptions;
084    
085        private Configuration config;
086    
087        private final Map<String, Class<PatternConverter>> converterRules;
088    
089    
090        /**
091         * Constructor.
092         * @param converterKey The type of converters that will be used.
093         */
094        public PatternParser(String converterKey) {
095            this(null, converterKey, null);
096        }
097    
098        /**
099         * Constructor.
100         * @param config The current Configuration.
101         * @param converterKey The key to lookup the converters.
102         * @param expected The expected base Class of each Converter.
103         */
104        public PatternParser(Configuration config, String converterKey, Class expected) {
105            this.config = config;
106            PluginManager manager = new PluginManager(converterKey, expected);
107            manager.collectPlugins();
108            Map<String, PluginType> plugins = manager.getPlugins();
109            Map<String, Class<PatternConverter>> converters = new HashMap<String, Class<PatternConverter>>();
110    
111            for (PluginType type : plugins.values()) {
112                try {
113                    Class<PatternConverter> clazz = type.getPluginClass();
114                    ConverterKeys keys = clazz.getAnnotation(ConverterKeys.class);
115                    if (keys != null) {
116                        for (String key : keys.value()) {
117                            converters.put(key, clazz);
118                        }
119                    }
120                } catch (Exception ex) {
121                    LOGGER.error("Error processing plugin " + type.getElementName(), ex);
122                }
123            }
124            converterRules = converters;
125        }

128        public List<PatternFormatter> parse(String pattern) {
129            List<PatternFormatter> list = new ArrayList<PatternFormatter>();
130            List<PatternConverter> converters = new ArrayList<PatternConverter>();
131            List<FormattingInfo> fields = new ArrayList<FormattingInfo>();
132    
133            parse(pattern, converters, fields);
134    
135            Iterator fieldIter = fields.iterator();
136    
137            for (PatternConverter converter : converters) {
138                LogEventPatternConverter pc;
139                if (converter instanceof LogEventPatternConverter) {
140                    pc = (LogEventPatternConverter) converter;
141                    handlesExceptions |= pc.handlesThrowable();
142                } else {
143                    pc = new LiteralPatternConverter(config, "");
144                }
145    
146                FormattingInfo field;
147                if (fieldIter.hasNext()) {
148                    field = (FormattingInfo) fieldIter.next();
149                } else {
150                    field = FormattingInfo.getDefault();
151                }
152                list.add(new PatternFormatter(pc, field));
153            }
154            return list;
155        }
156    
157        public boolean handlesExceptions() {
158            return handlesExceptions;
159        }

161        /**
162         * Extract the converter identifier found at position i.
163         * <p/>
164         * After this function returns, the variable i will point to the
165         * first char after the end of the converter identifier.
166         * <p/>
167         * If i points to a char which is not a character acceptable at the
168         * start of a unicode identifier, the value null is returned.
169         *
170         * @param lastChar       last processed character.
171         * @param pattern        format string.
172         * @param i              current index into pattern format.
173         * @param convBuf        buffer to receive conversion specifier.
174         * @param currentLiteral literal to be output in case format specifier in unrecognized.
175         * @return position in pattern after converter.
176         */
177        private static int extractConverter(
178            char lastChar, final String pattern, int i, final StringBuilder convBuf,
179            final StringBuilder currentLiteral) {
180            convBuf.setLength(0);
181    
182            // When this method is called, lastChar points to the first character of the
183            // conversion word. For example:
184            // For "%hello"     lastChar = 'h'
185            // For "%-5hello"   lastChar = 'h'
186            //System.out.println("lastchar is "+lastChar);
187            if (!Character.isUnicodeIdentifierStart(lastChar)) {
188                return i;
189            }
190    
191            convBuf.append(lastChar);
192    
193            while ((i < pattern.length()) && Character.isUnicodeIdentifierPart(pattern.charAt(i))) {
194                convBuf.append(pattern.charAt(i));
195                currentLiteral.append(pattern.charAt(i));
196                i++;
197            }
198    
199            return i;
200        }
201    
202        /**
203         * Extract options.
204         *
205         * @param pattern conversion pattern.
206         * @param i       start of options.
207         * @param options array to receive extracted options
208         * @return position in pattern after options.
209         */
210        private static int extractOptions(String pattern, int i, List<String> options) {
211            while ((i < pattern.length()) && (pattern.charAt(i) == '{')) {
212                int begin = i;
213                int end;
214                int depth = 0;
215                do {
216                    end = pattern.indexOf('}', i);
217                    if (end != -1) {
218                        int next = pattern.indexOf("{", i + 1);
219                        if (next != -1 && next < end) {
220                            i = end + 1;
221                            ++depth;
222                        } else if (depth > 0) {
223                            --depth;
224                        }
225                    }
226                } while (depth > 0);
227    
228                if (end == -1) {
229                    break;
230                }
231    
232                String r = pattern.substring(begin + 1, end);
233                options.add(r);
234                i = end + 1;
235            }
236    
237            return i;
238        }

240        /**
241         * Parse a format specifier.
242         *
243         * @param pattern           pattern to parse.
244         * @param patternConverters list to receive pattern converters.
245         * @param formattingInfos   list to receive field specifiers corresponding to pattern converters.
246         */
247        public void parse(final String pattern, final List<PatternConverter> patternConverters,
248                final List<FormattingInfo> formattingInfos) {
249            if (pattern == null) {
250                throw new NullPointerException("pattern");
251            }
252    
253            StringBuilder currentLiteral = new StringBuilder(BUF_SIZE);
254    
255            int patternLength = pattern.length();
256            ParserState state = ParserState.LITERAL_STATE;
257            char c;
258            int i = 0;
259            FormattingInfo formattingInfo = FormattingInfo.getDefault();
260    
261            while (i < patternLength) {
262                c = pattern.charAt(i++);
263    
264                switch (state) {
265                    case LITERAL_STATE:
266    
267                        // In literal state, the last char is always a literal.
268                        if (i == patternLength) {
269                            currentLiteral.append(c);
270    
271                            continue;
272                        }
273    
274                        if (c == ESCAPE_CHAR) {
275                            // peek at the next char.
276                            switch (pattern.charAt(i)) {
277                                case ESCAPE_CHAR:
278                                    currentLiteral.append(c);
279                                    i++; // move pointer
280    
281                                    break;
282    
283                                default:
284    
285                                    if (currentLiteral.length() != 0) {
286                                        patternConverters.add(new LiteralPatternConverter(config,
287                                            currentLiteral.toString()));
288                                        formattingInfos.add(FormattingInfo.getDefault());
289                                    }
290    
291                                    currentLiteral.setLength(0);
292                                    currentLiteral.append(c); // append %
293                                    state = ParserState.CONVERTER_STATE;
294                                    formattingInfo = FormattingInfo.getDefault();
295                            }
296                        } else {
297                            currentLiteral.append(c);
298                        }
299    
300                        break;
301    
302                    case CONVERTER_STATE:
303                        currentLiteral.append(c);
304    
305                        switch (c) {
306                            case '-':
307                                formattingInfo =
308                                    new FormattingInfo(true, formattingInfo.getMinLength(),
309                                        formattingInfo.getMaxLength());
310                                break;
311    
312                            case '.':
313                                state = ParserState.DOT_STATE;
314                                break;
315    
316                            default:
317    
318                                if ((c >= '0') && (c <= '9')) {
319                                    formattingInfo = new FormattingInfo(formattingInfo.isLeftAligned(), c - '0',
320                                            formattingInfo.getMaxLength());
321                                    state = ParserState.MIN_STATE;
322                                } else {
323                                    i = finalizeConverter(c, pattern, i, currentLiteral, formattingInfo,
324                                            converterRules, patternConverters, formattingInfos);
325    
326                                    // Next pattern is assumed to be a literal.
327                                    state = ParserState.LITERAL_STATE;
328                                    formattingInfo = FormattingInfo.getDefault();
329                                    currentLiteral.setLength(0);
330                                }
331                        } // switch
332    
333                        break;
334    
335                    case MIN_STATE:
336                        currentLiteral.append(c);
337    
338                        if ((c >= '0') && (c <= '9')) {
339                            // Multiply the existing value and add the value of the number just encountered.
340                            formattingInfo = new FormattingInfo(formattingInfo.isLeftAligned(),
341                                    (formattingInfo.getMinLength() * DECIMAL) + (c - '0'),
342                                    formattingInfo.getMaxLength());
343                        } else if (c == '.') {
344                            state = ParserState.DOT_STATE;
345                        } else {
346                            i = finalizeConverter(c, pattern, i, currentLiteral, formattingInfo,
347                                    converterRules, patternConverters, formattingInfos);
348                            state = ParserState.LITERAL_STATE;
349                            formattingInfo = FormattingInfo.getDefault();
350                            currentLiteral.setLength(0);
351                        }
352    
353                        break;
354    
355                    case DOT_STATE:
356                        currentLiteral.append(c);
357    
358                        if ((c >= '0') && (c <= '9')) {
359                            formattingInfo = new FormattingInfo(formattingInfo.isLeftAligned(),
360                                formattingInfo.getMinLength(), c - '0');
361                            state = ParserState.MAX_STATE;
362                        } else {
363                            LOGGER.error("Error occurred in position " + i
364                                    + ".\n Was expecting digit, instead got char \"" + c + "\".");
365    
366                            state = ParserState.LITERAL_STATE;
367                        }
368    
369                        break;
370    
371                    case MAX_STATE:
372                        currentLiteral.append(c);
373    
374                        if ((c >= '0') && (c <= '9')) {
375                            // Multiply the existing value and add the value of the number just encountered.
376                            formattingInfo = new FormattingInfo(
377                                    formattingInfo.isLeftAligned(), formattingInfo.getMinLength(),
378                                    (formattingInfo.getMaxLength() * DECIMAL) + (c - '0'));
379                        } else {
380                            i = finalizeConverter(c, pattern, i, currentLiteral, formattingInfo,
381                                    converterRules, patternConverters, formattingInfos);
382                            state = ParserState.LITERAL_STATE;
383                            formattingInfo = FormattingInfo.getDefault();
384                            currentLiteral.setLength(0);
385                        }
386    
387                        break;
388                } // switch
389            }
390    
391            // while
392            if (currentLiteral.length() != 0) {
393                patternConverters.add(new LiteralPatternConverter(config, currentLiteral.toString()));
394                formattingInfos.add(FormattingInfo.getDefault());
395            }
396        }

398        /**
399         * Creates a new PatternConverter.
400         *
401         * @param converterId       converterId.
402         * @param currentLiteral    literal to be used if converter is unrecognized or following converter
403         *                          if converterId contains extra characters.
404         * @param rules             map of stock pattern converters keyed by format specifier.
405         * @param options           converter options.
406         * @return converter or null.
407         */
408        private PatternConverter createConverter(final String converterId, final StringBuilder currentLiteral,
409                final Map<String, Class<PatternConverter>> rules,
410                final List<String> options) {
411            String converterName = converterId;
412            Class<PatternConverter> converterClass = null;
413    
414            for (int i = converterId.length(); (i > 0) && (converterClass == null); i--) {
415                converterName = converterName.substring(0, i);
416    
417                if ((converterClass == null) && (rules != null)) {
418                    converterClass = rules.get(converterName);
419                }
420            }
421    
422            if (converterClass == null) {
423                LOGGER.error("Unrecognized format specifier [" + converterId + "]");
424    
425                return null;
426            }
427    
428            Method[] methods = converterClass.getDeclaredMethods();
429            Method newInstance = null;
430            for (Method method : methods) {
431                if (method.getName().equals("newInstance")) {
432                    if (newInstance == null) {
433                        newInstance = method;
434                    } else {
435                        LOGGER.error("Class " + converterClass + " cannot contain multiple newInstance methods");
436                        return null;
437                    }
438                }
439            }
440            if (newInstance == null) {
441                LOGGER.error("Class " + converterClass + " does not contain a newInstance method");
442                return null;
443            }
444    
445            Class[] parmTypes = newInstance.getParameterTypes();
446            Object [] parms = parmTypes.length > 0 ? new Object[parmTypes.length] : null;
447    
448            if (parms != null) {
449                int i = 0;
450                boolean errors = false;
451                for (Class clazz : parmTypes) {
452                    if (clazz.isArray() && clazz.getName().equals("[Ljava.lang.String;")) {
453                        String[] optionsArray = options.toArray(new String[options.size()]);
454                        parms[i] = optionsArray;
455                    } else if (clazz.isAssignableFrom(Configuration.class)) {
456                        parms[i] = config;
457                    } else {
458                        LOGGER.error("Unknown parameter type " + clazz.getName() + " for newInstance method of " +
459                            converterClass.getName());
460                        errors = true;
461                    }
462                    ++i;
463                }
464                if (errors) {
465                    return null;
466                }
467            }
468    
469            try {
470                Object newObj = newInstance.invoke(null, parms);
471    
472                if (newObj instanceof PatternConverter) {
473                    currentLiteral.delete(0, currentLiteral.length()
474                            - (converterId.length() - converterName.length()));
475    
476                    return (PatternConverter) newObj;
477                } else {
478                    LOGGER.warn("Class " + converterClass.getName() + " does not extend PatternConverter.");
479                }
480            } catch (Exception ex) {
481                LOGGER.error("Error creating converter for " + converterId, ex);
482            }
483    
484            return null;
485        }

487        /**
488         * Processes a format specifier sequence.
489         *
490         * @param c                 initial character of format specifier.
491         * @param pattern           conversion pattern
492         * @param i                 current position in conversion pattern.
493         * @param currentLiteral    current literal.
494         * @param formattingInfo    current field specifier.
495         * @param rules             map of stock pattern converters keyed by format specifier.
496         * @param patternConverters list to receive parsed pattern converter.
497         * @param formattingInfos   list to receive corresponding field specifier.
498         * @return position after format specifier sequence.
499         */
500        private int finalizeConverter(char c, String pattern, int i,
501                final StringBuilder currentLiteral, final FormattingInfo formattingInfo,
502                final Map<String, Class<PatternConverter>> rules,
503                final List<PatternConverter> patternConverters, final List<FormattingInfo> formattingInfos) {
504            StringBuilder convBuf = new StringBuilder();
505            i = extractConverter(c, pattern, i, convBuf, currentLiteral);
506    
507            String converterId = convBuf.toString();
508    
509            List<String> options = new ArrayList<String>();
510            i = extractOptions(pattern, i, options);
511    
512            PatternConverter pc = createConverter(converterId, currentLiteral, rules, options);
513    
514            if (pc == null) {
515                StringBuilder msg;
516    
517                if ((converterId == null) || (converterId.length() == 0)) {
518                    msg =
519                        new StringBuilder("Empty conversion specifier starting at position ");
520                } else {
521                    msg = new StringBuilder("Unrecognized conversion specifier [");
522                    msg.append(converterId);
523                    msg.append("] starting at position ");
524                }
525    
526                msg.append(Integer.toString(i));
527                msg.append(" in conversion pattern.");
528    
529                LOGGER.error(msg.toString());
530    
531                patternConverters.add(new LiteralPatternConverter(config, currentLiteral.toString()));
532                formattingInfos.add(FormattingInfo.getDefault());
533            } else {
534                patternConverters.add(pc);
535                formattingInfos.add(formattingInfo);
536    
537                if (currentLiteral.length() > 0) {
538                    patternConverters.add(new LiteralPatternConverter(config, currentLiteral.toString()));
539                    formattingInfos.add(FormattingInfo.getDefault());
540                }
541            }
542    
543            currentLiteral.setLength(0);
544    
545            return i;
546        }

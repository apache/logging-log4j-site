018package org.apache.logging.log4j.core.util;
019
020import java.io.File;
021import java.io.IOException;
022import java.net.MalformedURLException;
023import java.net.URI;
024import java.net.URISyntaxException;
025import java.net.URL;
026import java.nio.file.Path;
027import java.nio.file.Paths;
028import java.util.Objects;
029
030import org.apache.logging.log4j.Logger;
031import org.apache.logging.log4j.core.config.ConfigurationSource;
032import org.apache.logging.log4j.status.StatusLogger;
033import org.apache.logging.log4j.util.Strings;
034
035/**
036 * Represents the source for the logging configuration as an immutable object.
037 */
038public class Source {
039    private static final Logger LOGGER = StatusLogger.getLogger();
040
041    private static String normalize(final File file) {
042        try {
043            return file.getCanonicalFile().getAbsolutePath();
044        } catch (final IOException e) {
045            throw new IllegalArgumentException(e);
046        }
047    }
048
049    private static File toFile(final Path path) {
050        try {
051            return Objects.requireNonNull(path, "path").toFile();
052        } catch (final UnsupportedOperationException e) {
053            return null;
054        }
055    }
056
057    // LOG4J2-3527 - Don't use Paths.get().
058    private static File toFile(final URI uri) {
059        try {
060            final String scheme = Objects.requireNonNull(uri, "uri").getScheme();
061            if (Strings.isBlank(scheme) || scheme.equals("file")) {
062                return new File(uri.getPath());
063            } else {
064                LOGGER.debug("uri does not represent a local file: " + uri);
065                return null;
066            }
067        } catch (final Exception e) {
068            LOGGER.debug("uri is malformed: " + uri.toString());
069            return null;
070        }
071    }
072
073    private static URI toURI(final URL url) {
074        try {
075            return Objects.requireNonNull(url, "url").toURI();
076        } catch (final URISyntaxException e) {
077            throw new IllegalArgumentException(e);
078        }
079    }
080
081    private final File file;
082    private final URI uri;
083    private final String location;
084
085    /**
086     * Constructs a Source from a ConfigurationSource.
087     *
088     * @param source The ConfigurationSource.
089     */
090    public Source(final ConfigurationSource source) {
091        this.file = source.getFile();
092        this.uri = source.getURI();
093        this.location = source.getLocation();
094    }
095
096    /**
097     * Constructs a new {@code Source} with the specified file.
098     * file.
099     *
100     * @param file the file where the input stream originated
101     */
102    public Source(final File file) {
103        this.file = Objects.requireNonNull(file, "file");
104        this.location = normalize(file);
105        this.uri = file.toURI();
106    }
107
108    /**
109     * Constructs a new {@code Source} from the specified Path.
110     *
111     * @param path the Path where the input stream originated
112     */
113    public Source(final Path path) {
114        final Path normPath = Objects.requireNonNull(path, "path").normalize();
115        this.file = toFile(normPath);
116        this.uri = normPath.toUri();
117        this.location = normPath.toString();
118    }
119
120    /**
121     * Constructs a new {@code Source} from the specified URI.
122     *
123     * @param uri the URI where the input stream originated
124     */
125    public Source(final URI uri) {
126        final URI normUri = Objects.requireNonNull(uri, "uri").normalize();
127        this.uri = normUri;
128        this.location = normUri.toString();
129        this.file = toFile(normUri);
130    }
131
132    /**
133     * Constructs a new {@code Source} from the specified URI.
134     *
135     * @param uri the URI where the input stream originated
136     * @param lastModified Not used.
137     * @deprecated Use {@link Source#Source(URI)}.
138     */
139    @Deprecated
140    public Source(final URI uri, final long lastModified) {
141        this(uri);
142    }
143
144    /**
145     * Constructs a new {@code Source} from the specified URL.
146     *
147     * @param url the URL where the input stream originated
148     * @throws IllegalArgumentException if this URL is not formatted strictly according to RFC2396 and cannot be
149     *         converted to a URI.
150     */
151    public Source(final URL url) {
152        this.uri = toURI(url);
153        this.location = uri.toString();
154        this.file = toFile(uri);
155    }
156
157    @Override
158    public boolean equals(Object obj) {
159        if (this == obj) {
160            return true;
161        }
162        if (!(obj instanceof Source)) {
163            return false;
164        }
165        Source other = (Source) obj;
166        return Objects.equals(location, other.location);
167    }
168
169    /**
170     * Gets the file configuration source, or {@code null} if this configuration source is based on an URL or has
171     * neither a file nor an URL.
172     *
173     * @return the configuration source file, or {@code null}
174     */
175    public File getFile() {
176        return file;
177    }
178
179    /**
180     * Gets a string describing the configuration source file or URI, or {@code null} if this configuration source
181     * has neither a file nor an URI.
182     *
183     * @return a string describing the configuration source file or URI, or {@code null}
184     */
185    public String getLocation() {
186        return location;
187    }
188
189    /**
190     * Gets this source as a Path.
191     *
192     * @return this source as a Path.
193     */
194    public Path getPath() {
195        return file != null ? file.toPath() : uri != null ? Paths.get(uri) : Paths.get(location);
196    }
197
198    /**
199     * Gets the configuration source URI, or {@code null} if this configuration source is based on a file or has
200     * neither a file nor an URI.
201     *
202     * @return the configuration source URI, or {@code null}
203     */
204    public URI getURI() {
205        return uri;
206    }
207
208    /**
209     * Gets the configuration source URL.
210     *
211     * @return the configuration source URI, or {@code null}
212     */
213    public URL getURL() {
214        try {
215            return uri.toURL();
216        } catch (final MalformedURLException e) {
217            throw new IllegalStateException(e);
218        }
219    }
220
221    @Override
222    public int hashCode() {
223        return Objects.hash(location);
224    }
225
226    @Override
227    public String toString() {
228        return location;
229    }
230}

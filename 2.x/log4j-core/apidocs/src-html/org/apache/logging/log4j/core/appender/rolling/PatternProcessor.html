017    package org.apache.logging.log4j.core.appender.rolling;
018    
019    import org.apache.logging.log4j.core.pattern.ArrayPatternConverter;
020    import org.apache.logging.log4j.core.pattern.DatePatternConverter;
021    import org.apache.logging.log4j.core.pattern.FormattingInfo;
022    import org.apache.logging.log4j.core.pattern.PatternConverter;
023    import org.apache.logging.log4j.core.pattern.PatternParser;
024    
025    import java.util.ArrayList;
026    import java.util.Calendar;
027    import java.util.Date;
028    import java.util.List;
029    
030    /**
031     * Parse the rollover pattern.
032     */
033    public class PatternProcessor {
034    
035        private static final String KEY = "FileConverter";
036    
037        private static final char YEAR_CHAR = 'y';
038        private static final char MONTH_CHAR = 'M';
039        private static final char[] WEEK_CHARS = {'w', 'W'};
040        private static final char[] DAY_CHARS = {'D', 'd', 'F', 'E'};
041        private static final char[] HOUR_CHARS = {'H', 'K', 'h', 'k'};
042        private static final char MINUTE_CHAR = 'm';
043        private static final char SECOND_CHAR = 's';
044        private static final char MILLIS_CHAR = 'S';
045    
046        private final ArrayPatternConverter[] patternConverters;
047        private final FormattingInfo[] patternFields;
048    
049        private RolloverFrequency frequency = null;
050    
051        /**
052         * Constructor.
053         * @param pattern The file pattern.
054         */
055        public PatternProcessor(String pattern) {
056            PatternParser parser = createPatternParser();
057            List<PatternConverter> converters = new ArrayList<PatternConverter>();
058            List<FormattingInfo> fields = new ArrayList<FormattingInfo>();
059            parser.parse(pattern, converters, fields);
060            FormattingInfo[] infoArray = new FormattingInfo[fields.size()];
061            patternFields = fields.toArray(infoArray);
062            ArrayPatternConverter[] converterArray = new ArrayPatternConverter[converters.size()];
063            patternConverters = converters.toArray(converterArray);
064    
065            for (ArrayPatternConverter converter : patternConverters) {
066                if (converter instanceof DatePatternConverter) {
067                    DatePatternConverter dateConverter = (DatePatternConverter) converter;
068                    frequency = calculateFrequency(dateConverter.getPattern());
069                }
070            }
071        }
072    
073        /**
074         * Return the next expire time.
075         * @param current The current time.
076         * @return The next expire time.
077         */
078        public long getNextTime(long current) {
079            return getNextTime(current, 1);
080        }
081    
082        /**
083         * Return the next potential rollover time.
084         * @param current The current time.
085         * @param increment The increment to the next time.
086         * @return the next potential rollover time.
087         */
088        public long getNextTime(long current, int increment) {
089            if (frequency == null) {
090                throw new IllegalStateException("Pattern does not contain a date");
091            }
092            Calendar currentCal = Calendar.getInstance();
093            currentCal.setTimeInMillis(current);
094            Calendar cal = Calendar.getInstance();
095            cal.set(currentCal.get(Calendar.YEAR), 0, 1, 0, 0, 0);
096            cal.set(Calendar.MILLISECOND, 0);
097            if (frequency == RolloverFrequency.ANNUALLY) {
098                cal.add(Calendar.YEAR, increment);
099                return cal.getTimeInMillis();
100            }
101            if (frequency == RolloverFrequency.MONTHLY) {
102                cal.add(Calendar.MONTH, increment);
103                return cal.getTimeInMillis();
104            }
105            if (frequency == RolloverFrequency.WEEKLY) {
106                cal.set(Calendar.WEEK_OF_YEAR, currentCal.get(Calendar.WEEK_OF_YEAR) + increment);
107                return cal.getTimeInMillis();
108            }
109            cal.set(Calendar.DAY_OF_YEAR, currentCal.get(Calendar.DAY_OF_YEAR));
110            if (frequency == RolloverFrequency.DAILY) {
111                cal.add(Calendar.DAY_OF_YEAR, increment);
112                return cal.getTimeInMillis();
113            }
114            cal.set(Calendar.HOUR, currentCal.get(Calendar.HOUR));
115            if (frequency == RolloverFrequency.HOURLY) {
116                cal.add(Calendar.HOUR, increment);
117                return cal.getTimeInMillis();
118            }
119            cal.set(Calendar.MINUTE, currentCal.get(Calendar.MINUTE));
120            if (frequency == RolloverFrequency.EVERY_MINUTE) {
121                cal.add(Calendar.MINUTE, increment);
122                return cal.getTimeInMillis();
123            }
124            cal.set(Calendar.SECOND, currentCal.get(Calendar.SECOND));
125            if (frequency == RolloverFrequency.EVERY_SECOND) {
126                cal.add(Calendar.SECOND, increment);
127                return cal.getTimeInMillis();
128            }
129            cal.set(Calendar.MILLISECOND, currentCal.get(Calendar.MILLISECOND) + increment);
130            return cal.getTimeInMillis();
131        }
132    
133        /**
134         * Format file name.
135         *
136         * @param obj object to be evaluated in formatting, may not be null.
137         * @param buf string buffer to which formatted file name is appended, may not be null.
138         */
139        protected final void formatFileName(final Object obj, final StringBuilder buf) {
140            Object[] objects = new Object[] {new Date(System.currentTimeMillis()), obj};
141            formatFileName(objects, buf);
142        }
143    
144        /**
145         * Format file name.
146         *
147         * @param objects objects to be evaluated in formatting, may not be null.
148         * @param buf string buffer to which formatted file name is appended, may not be null.
149         */
150        protected final void formatFileName(final Object[] objects, final StringBuilder buf) {
151            for (int i = 0; i < patternConverters.length; i++) {
152                int fieldStart = buf.length();
153                patternConverters[i].format(objects, buf);
154    
155                if (patternFields[i] != null) {
156                    patternFields[i].format(fieldStart, buf);
157                }
158            }
159        }
160    
161        private RolloverFrequency calculateFrequency(String pattern) {
162            if (patternContains(pattern, MILLIS_CHAR)) {
163                return RolloverFrequency.EVERY_MILLISECOND;
164            }
165            if (patternContains(pattern, SECOND_CHAR)) {
166                return RolloverFrequency.EVERY_SECOND;
167            }
168            if (patternContains(pattern, MINUTE_CHAR)) {
169                return RolloverFrequency.EVERY_MINUTE;
170            }
171            if (patternContains(pattern, HOUR_CHARS)) {
172                return RolloverFrequency.HOURLY;
173            }
174            if (patternContains(pattern, DAY_CHARS)) {
175                return RolloverFrequency.DAILY;
176            }
177            if (patternContains(pattern, WEEK_CHARS)) {
178                return RolloverFrequency.WEEKLY;
179            }
180            if (patternContains(pattern, MONTH_CHAR)) {
181                return RolloverFrequency.MONTHLY;
182            }
183            if (patternContains(pattern, YEAR_CHAR)) {
184                return RolloverFrequency.ANNUALLY;
185            }
186            return null;
187        }
188    
189        private PatternParser createPatternParser() {
190    
191            return new PatternParser(null, KEY, null);
192        }
193    
194        private boolean patternContains(String pattern, char[] chars) {
195            for (char character : chars) {
196                if (patternContains(pattern, character)) {
197                    return true;
198                }
199            }
200            return false;
201        }
202    
203        private boolean patternContains(String pattern, char character) {
204            return pattern.indexOf(character) >= 0;
205        }
206    }

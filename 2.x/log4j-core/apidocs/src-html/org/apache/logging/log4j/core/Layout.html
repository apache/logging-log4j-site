017    package org.apache.logging.log4j.core;
018    
019    import java.io.Serializable;
020    
021    /**
022     * @doubt There is still a need for a character-based layout for character based event sinks (databases, etc).
023     *  Would introduce an EventEncoder, EventRenderer or something similar for the logging event to byte encoding.
024     * (RG) A layout can be configured with a Charset and then Strings can be converted to byte arrays. OTOH, it isn't
025     * possible to write byte arrays as character streams.
026     * @param <T> The Object type that will be returned on the formatAs call.
027     */
028    public interface Layout<T extends Serializable> {
029        /**
030         * Formats the event suitable for display.
031         * @param event The Logging Event.
032         * @return The formatted event.
033         * @doubt Likely better to write to a OutputStream instead of return a byte[]. (RG) That limits how the
034         * Appender can use the Layout. For example, it might concatenate information in front or behind the
035         * data and then write it all to the OutputStream in one call.
036         */
037        byte[] format(LogEvent event);
038    
039        /**
040         * Formats the event as an Object that can be serialized.
041         * @param event The Logging Event.
042         * @return The formatted event.
043         */
044        T formatAs(LogEvent event);
045    
046        /**
047         * Returns the header for the layout format.
048         * @return The header.
049         * @doubt the concept of header and footer is not universal, should not be on the base interface.
050         * (RG) I agree with this.
051         */
052        byte[] getHeader();
053    
054        /**
055         * Returns the format for the layout format.
056         * @return The footer.
057         * @doubt the concept of header and footer is not universal, should not be on the base interface.
058         * (RG) I agree with this.
059         */
060        byte[] getFooter();
061    
062    
063    }

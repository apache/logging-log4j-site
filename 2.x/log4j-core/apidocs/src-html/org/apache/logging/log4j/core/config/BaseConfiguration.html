140            setup();
141            setupAdvertisement();
142            doConfigure();
143            for (final LoggerConfig logger : loggers.values()) {
144                logger.startFilter();
145            }
146            for (final Appender appender : appenders.values()) {
147                appender.start();
148            }
149            root.startFilter(); // LOG4J2-336
150            startFilter();

154         * Tear down the configuration.
155         */
156        @Override
157        public void stop() {
158            // Stop the appenders in reverse order in case they still have activity.
159            final Appender[] array = appenders.values().toArray(new Appender[appenders.size()]);
160            for (int i = array.length - 1; i >= 0; --i) {
161                array[i].stop();
162            }
163            for (final LoggerConfig logger : loggers.values()) {
164                logger.clearAppenders();
165                logger.stopFilter();
166            }
167            root.stopFilter();
168            stopFilter();
169            if (advertiser != null && advertisement != null)
170            {
171                advertiser.unadvertise(advertisement);
172            }
173        }

198                attributes.put("content", new String(buffer));

208        private void setupAdvertisement() {
209            if (advertiserNode != null)
210            {
211                String name = advertiserNode.getName();
212                @SuppressWarnings("unchecked")
213                final PluginType<Advertiser> type = (PluginType<Advertiser>) pluginManager.getPluginType(name);
214                if (type != null)
215                {
216                    final Class<Advertiser> clazz = type.getPluginClass();
217                    try {
218                        advertiser = clazz.newInstance();
219                        advertisement = advertiser.advertise(advertiserNode.getAttributes());
220                    } catch (final InstantiationException e) {
221                        System.err.println("InstantiationException attempting to instantiate advertiser: " + name);
222                    } catch (final IllegalAccessException e) {
223                        System.err.println("IllegalAccessException attempting to instantiate advertiser: " + name);
224                    }
225                }
226            }
227        }

239        @SuppressWarnings("unchecked")
240        protected void doConfigure() {
241            boolean setRoot = false;
242            boolean setLoggers = false;
243            for (final Node child : rootNode.getChildren()) {
244                createConfiguration(child, null);
245                if (child.getObject() == null) {
246                    continue;
247                }
248                if (child.getName().equalsIgnoreCase("Properties")) {
249                    if (tempLookup == subst.getVariableResolver()) {
250                        subst.setVariableResolver((StrLookup) child.getObject());
251                    } else {
252                        LOGGER.error("Properties declaration must be the first element in the configuration");
253                    }
254                    continue;
255                } else if (tempLookup == subst.getVariableResolver()) {
256                    final Map<String, String> map = (Map<String, String>) componentMap.get(CONTEXT_PROPERTIES);
257                    final StrLookup lookup = map == null ? null : new MapLookup(map);
258                    subst.setVariableResolver(new Interpolator(lookup));
259                }
260                if (child.getName().equalsIgnoreCase("Appenders")) {
261                    appenders = (ConcurrentMap<String, Appender>) child.getObject();
262                } else if (child.getObject() instanceof Filter) {
263                    addFilter((Filter) child.getObject());
264                } else if (child.getName().equalsIgnoreCase("Loggers")) {
265                    final Loggers l = (Loggers) child.getObject();
266                    loggers = l.getMap();
267                    setLoggers = true;
268                    if (l.getRoot() != null) {
269                        root = l.getRoot();
270                        setRoot = true;
271                    }
272                } else {
273                    LOGGER.error("Unknown object \"" + child.getName() + "\" of type " +
274                        child.getObject().getClass().getName() + " is ignored");
275                }
276            }
277    
278            if (!setLoggers) {
279                LOGGER.warn("No Loggers were configured, using default. Is the Loggers element missing?");
280                setToDefault();
281                return;
282            } else if (!setRoot) {
283                LOGGER.warn("No Root logger was configured, creating default ERROR-level Root logger with Console appender");
284                setToDefault();
285                // return; // LOG4J2-219: creating default root=ok, but don't exclude configured Loggers
286            }
287    
288            for (final Map.Entry<String, LoggerConfig> entry : loggers.entrySet()) {
289                final LoggerConfig l = entry.getValue();
290                for (final AppenderRef ref : l.getAppenderRefs()) {
291                    final Appender app = appenders.get(ref.getRef());
292                    if (app != null) {
293                        l.addAppender(app, ref.getLevel(), ref.getFilter());
294                    } else {
295                        LOGGER.error("Unable to locate appender " + ref.getRef() + " for logger " + l.getName());
296                    }
297                }
298    
299            }
300    
301            setParents();
302        }

304        private void setToDefault() {
305            setName(DefaultConfiguration.DEFAULT_NAME);
306            final Layout<? extends Serializable> layout =
307                    PatternLayout.createLayout("%d{HH:mm:ss.SSS} [%thread] %-5level %logger{36} - %msg%n",
308                            null, null, null, null);
309            final Appender appender = ConsoleAppender.createAppender(layout, null, "SYSTEM_OUT", "Console", "false",
310                "true");
311            appender.start();
312            addAppender(appender);
313            final LoggerConfig root = getRootLogger();
314            root.addAppender(appender, null, null);
315    
316            final String levelName = PropertiesUtil.getProperties().getStringProperty(DefaultConfiguration.DEFAULT_LEVEL);
317            final Level level = levelName != null && Level.valueOf(levelName) != null ?
318                Level.valueOf(levelName) : Level.ERROR;
319            root.setLevel(level);
320        }

408        /**
409         * Associates an Appender with a LoggerConfig. This method is synchronized in case a Logger with the
410         * same name is being updated at the same time.
411         *
412         * Note: This method is not used when configuring via configuration. It is primarily used by
413         * unit tests.
414         * @param logger The Logger the Appender will be associated with.
415         * @param appender The Appender.
416         */
417        @Override
418        public synchronized void addLoggerAppender(final org.apache.logging.log4j.core.Logger logger,
419                                                   final Appender appender) {
420            final String name = logger.getName();
421            appenders.putIfAbsent(appender.getName(), appender);
422            final LoggerConfig lc = getLoggerConfig(name);
423            if (lc.getName().equals(name)) {
424                lc.addAppender(appender, null, null);
425            } else {
426                final LoggerConfig nlc = new LoggerConfig(name, lc.getLevel(), lc.isAdditive());
427                nlc.addAppender(appender, null, null);
428                nlc.setParent(lc);
429                loggers.putIfAbsent(name, nlc);
430                setParents();
431                logger.getContext().updateLoggers();
432            }
433        }

485         * Remove an Appender. First removes any associations between LoggerConfigs and the Appender, removes

494            final Appender app = appenders.remove(name);

496            if (app != null) {

501        /**
502         * Locates the appropriate LoggerConfig for a Logger name. This will remove tokens from the
503         * package name as necessary or return the root LoggerConfig if no other matches were found.
504         * @param name The Logger name.
505         * @return The located LoggerConfig.
506         */
507        @Override
508        public LoggerConfig getLoggerConfig(final String name) {
509            if (loggers.containsKey(name)) {
510                return loggers.get(name);
511            }
512            String substr = name;
513            while ((substr = NameUtil.getSubName(substr)) != null) {
514                if (loggers.containsKey(substr)) {
515                    return loggers.get(substr);
516                }
517            }
518            return root;
519        }

580        @Override
581        public void createConfiguration(final Node node, final LogEvent event) {
582            final PluginType<?> type = node.getType();
583            if (type != null && type.isDeferChildren()) {
584                node.setObject(createPluginObject(type, node, event));
585            } else {
586                for (final Node child : node.getChildren()) {
587                    createConfiguration(child, event);
588                }
589    
590                if (type == null) {
591                    if (node.getParent() != null) {
592                        LOGGER.error("Unable to locate plugin for " + node.getName());
593                    }
594                } else {
595                    node.setObject(createPluginObject(type, node, event));
596                }
597            }
598        }

600       /*
601        * Retrieve a static public 'method to create the desired object. Every parameter
602        * will be annotated to identify the appropriate attribute or element to use to
603        * set the value of the parameter.
604        * Parameters annotated with PluginAttribute will always be set as Strings.
605        * Parameters annotated with PluginElement may be Objects or arrays. Collections
606        * and Maps are currently not supported, although the factory method that is called
607        * can create these from an array.
608        *
609        * Although the happy path works, more work still needs to be done to log incorrect
610        * parameters. These will generally result in unhelpful InvocationTargetExceptions.
611        * @param classClass the class.
612        * @return the instantiate method or null if there is none by that
613        * description.
614        */
615        private <T> Object createPluginObject(final PluginType<T> type, final Node node, final LogEvent event)
616        {
617            final Class<T> clazz = type.getPluginClass();
618    
619            if (Map.class.isAssignableFrom(clazz)) {
620                try {
621                    @SuppressWarnings("unchecked")
622                    final Map<String, Object> map = (Map<String, Object>) clazz.newInstance();
623                    for (final Node child : node.getChildren()) {
624                        map.put(child.getName(), child.getObject());
625                    }
626                    return map;
627                } catch (final Exception ex) {
628                    LOGGER.warn("Unable to create Map for " + type.getElementName() + " of class " +
629                        clazz);
630                }
631            }
632    
633            if (List.class.isAssignableFrom(clazz)) {
634                try {
635                    @SuppressWarnings("unchecked")
636                    final List<Object> list = (List<Object>) clazz.newInstance();
637                    for (final Node child : node.getChildren()) {
638                        list.add(child.getObject());
639                    }
640                    return list;
641                } catch (final Exception ex) {
642                    LOGGER.warn("Unable to create List for " + type.getElementName() + " of class " +
643                        clazz);
644                }
645            }
646    
647            Method factoryMethod = null;
648    
649            for (final Method method : clazz.getMethods()) {
650                if (method.isAnnotationPresent(PluginFactory.class)) {
651                    factoryMethod = method;
652                    break;
653                }
654            }
655            if (factoryMethod == null) {
656                return null;
657            }
658    
659            final Annotation[][] parmArray = factoryMethod.getParameterAnnotations();
660            final Class<?>[] parmClasses = factoryMethod.getParameterTypes();
661            if (parmArray.length != parmClasses.length) {
662                LOGGER.error("Number of parameter annotations does not equal the number of paramters");
663            }
664            final Object[] parms = new Object[parmClasses.length];
665    
666            int index = 0;
667            final Map<String, String> attrs = node.getAttributes();
668            final List<Node> children = node.getChildren();
669            final StringBuilder sb = new StringBuilder();
670            final List<Node> used = new ArrayList<Node>();
671    
672            /*
673             * For each parameter:
674             * If the parameter is an attribute store the value of the attribute in the parameter array.
675             * If the parameter is an element:
676             *   Determine if the required parameter is an array.
677             *     If so, if a child contains the array, use it,
678             *      otherwise create the array from all child nodes of the correct type.
679             *     Store the array into the parameter array.
680             *   If not an array, store the object in the child node into the parameter array.
681             */
682            for (final Annotation[] parmTypes : parmArray) {
683                String[] aliases = null;
684                for (final Annotation a: parmTypes) {
685                    if (a instanceof PluginAliases) {
686                        aliases = ((PluginAliases) a).value();
687                    }
688                }
689                for (final Annotation a : parmTypes) {
690                    if (a instanceof PluginAliases) {
691                        continue;
692                    }
693                    if (sb.length() == 0) {
694                        sb.append(" with params(");
695                    } else {
696                        sb.append(", ");
697                    }
698                    if (a instanceof PluginNode) {
699                        parms[index] = node;
700                        sb.append("Node=").append(node.getName());
701                    } else if (a instanceof PluginConfiguration) {
702                        parms[index] = this;
703                        if (this.name != null) {
704                            sb.append("Configuration(").append(name).append(")");
705                        } else {
706                            sb.append("Configuration");
707                        }
708                    } else if (a instanceof PluginValue) {
709                        final String name = ((PluginValue) a).value();
710                        String v = node.getValue();
711                        if (v == null) {
712                            v = getAttrValue("value", null, attrs);
713                        }
714                        final String value = subst.replace(event, v);
715                        sb.append(name).append("=\"").append(value).append("\"");
716                        parms[index] = value;
717                    } else if (a instanceof PluginAttribute) {
718                        PluginAttribute attr = (PluginAttribute) a;
719                        final String name = attr.value();
720                        final String value = subst.replace(event, getAttrValue(name, aliases, attrs));
721                        sb.append(name).append("=\"").append(value).append("\"");
722                        parms[index] = value;
723                    } else if (a instanceof PluginElement) {
724                        final PluginElement elem = (PluginElement) a;
725                        final String name = elem.value();
726                        if (parmClasses[index].isArray()) {
727                            final Class<?> parmClass = parmClasses[index].getComponentType();
728                            final List<Object> list = new ArrayList<Object>();
729                            sb.append(name).append("={");
730                            boolean first = true;
731                            for (final Node child : children) {
732                                final PluginType<?> childType = child.getType();
733                                if (elem.value().equalsIgnoreCase(childType.getElementName()) ||
734                                    parmClass.isAssignableFrom(childType.getPluginClass())) {
735                                    used.add(child);
736                                    if (!first) {
737                                        sb.append(", ");
738                                    }
739                                    first = false;
740                                    final Object obj = child.getObject();
741                                    if (obj == null) {
742                                        LOGGER.error("Null object returned for " + child.getName() + " in " +
743                                            node.getName());
744                                        continue;
745                                    }
746                                    if (obj.getClass().isArray()) {
747                                        printArray(sb, (Object[]) obj);
748                                        parms[index] = obj;
749                                        break;
750                                    }
751                                    sb.append(child.toString());
752                                    list.add(obj);
753                                }
754                            }
755                            sb.append("}");
756                            if (parms[index] != null) {
757                                break;
758                            }
759                            if (list.size() > 0 && !parmClass.isAssignableFrom(list.get(0).getClass())) {
760                                LOGGER.error("Attempted to assign List containing class " +
761                                    list.get(0).getClass().getName() + " to array of type " + parmClass +
762                                    " for attribute " + name);
763                                break;
764                            }
765                            final Object[] array = (Object[]) Array.newInstance(parmClass, list.size());
766                            int i = 0;
767                            for (final Object obj : list) {
768                                array[i] = obj;
769                                ++i;
770                            }
771                            parms[index] = array;
772                        } else {
773                            final Class<?> parmClass = parmClasses[index];
774                            boolean present = false;
775                            for (final Node child : children) {
776                                final PluginType<?> childType = child.getType();
777                                if (elem.value().equals(childType.getElementName()) ||
778                                    parmClass.isAssignableFrom(childType.getPluginClass())) {
779                                    sb.append(child.getName()).append("(").append(child.toString()).append(")");
780                                    present = true;
781                                    used.add(child);
782                                    parms[index] = child.getObject();
783                                    break;
784                                }
785                            }
786                            if (!present) {
787                                sb.append("null");
788                            }
789                        }
790                    }
791                }
792                ++index;
793            }
794            if (sb.length() > 0) {
795                sb.append(")");
796            }
797    
798            if (attrs.size() > 0) {
799                final StringBuilder eb = new StringBuilder();
800                for (final String key : attrs.keySet()) {
801                    if (eb.length() == 0) {
802                        eb.append(node.getName());
803                        eb.append(" contains ");
804                        if (attrs.size() == 1) {
805                            eb.append("an invalid element or attribute ");
806                        } else {
807                            eb.append("invalid attributes ");
808                        }
809                    } else {
810                        eb.append(", ");
811                    }
812                    eb.append("\"");
813                    eb.append(key);
814                    eb.append("\"");
815    
816                }
817                LOGGER.error(eb.toString());
818            }
819    
820            if (!type.isDeferChildren() && used.size() != children.size()) {
821                for (final Node child : children) {
822                    if (used.contains(child)) {
823                        continue;
824                    }
825                    final String nodeType = node.getType().getElementName();
826                    final String start = nodeType.equals(node.getName()) ? node.getName() : nodeType + " " + node.getName();
827                    LOGGER.error(start + " has no parameter that matches element " + child.getName());
828                }
829            }
830    
831            try {
832                final int mod = factoryMethod.getModifiers();
833                if (!Modifier.isStatic(mod)) {
834                    LOGGER.error(factoryMethod.getName() + " method is not static on class " +
835                        clazz.getName() + " for element " + node.getName());
836                    return null;
837                }
838                LOGGER.debug("Calling {} on class {} for element {}{}", factoryMethod.getName(), clazz.getName(),
839                    node.getName(), sb.toString());
840                //if (parms.length > 0) {
841                    return factoryMethod.invoke(null, parms);
842                //}
843                //return factoryMethod.invoke(null, node);
844            } catch (final Exception e) {
845                LOGGER.error("Unable to invoke method " + factoryMethod.getName() + " in class " +
846                    clazz.getName() + " for element " + node.getName(), e);
847            }
848            return null;
849        }

851        private void printArray(final StringBuilder sb, final Object... array) {
852            boolean first = true;
853            for (final Object obj : array) {
854                if (!first) {
855                    sb.append(", ");
856                }
857                sb.append(obj.toString());
858                first = false;
859            }
860        }

862        private String getAttrValue(final String name, final String[] aliases, final Map<String, String> attrs) {
863            for (final String key : attrs.keySet()) {
864                if (key.equalsIgnoreCase(name)) {
865                    final String attr = attrs.get(key);
866                    attrs.remove(key);
867                    return attr;
868                }
869                if (aliases != null) {
870                    for (String alias : aliases) {
871                        if (key.equalsIgnoreCase(alias)) {
872                            final String attr = attrs.get(key);
873                            attrs.remove(key);
874                            return attr;
875                        }
876                    }
877                }
878            }
879            return null;
880        }

882        private void setParents() {
883             for (final Map.Entry<String, LoggerConfig> entry : loggers.entrySet()) {
884                final LoggerConfig logger = entry.getValue();
885                String name = entry.getKey();
886                if (!name.equals("")) {
887                    final int i = name.lastIndexOf('.');
888                    if (i > 0) {
889                        name = name.substring(0, i);
890                        LoggerConfig parent = getLoggerConfig(name);
891                        if (parent == null) {
892                            parent = root;
893                        }
894                        logger.setParent(parent);
895                    } else {
896                        logger.setParent(root);
897                    }
898                }
899            }
900        }

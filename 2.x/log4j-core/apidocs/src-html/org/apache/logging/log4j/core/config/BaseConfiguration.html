104        /**
105         * Initialize the configuration.
106         */
107        public void start() {
108            pluginManager.collectPlugins();
109            setup();
110            doConfigure();
111            for (LoggerConfig logger : loggers.values()) {
112                logger.startFilter();
113            }
114            for (Appender appender : appenders.values()) {
115                appender.start();
116            }
117    
118            startFilter();
119        }

121        /**
122         * Tear down the configuration.
123         */
124        public void stop() {
125            for (LoggerConfig logger : loggers.values()) {
126                logger.clearAppenders();
127                logger.stopFilter();
128            }
129            for (Appender appender : appenders.values()) {
130                appender.stop();
131            }
132            stopFilter();
133        }

146        protected void doConfigure() {
147            boolean setRoot = false;
148            boolean setLoggers = false;
149            for (Node child : rootNode.getChildren()) {
150                createConfiguration(child, null);
151                if (child.getObject() == null) {
152                    continue;
153                }
154                if (child.getName().equalsIgnoreCase("properties")) {
155                    if (tempLookup == subst.getVariableResolver()) {
156                        subst.setVariableResolver((StrLookup) child.getObject());
157                    } else {
158                        LOGGER.error("Properties declaration must be the first element in the configuration");
159                    }
160                    continue;
161                } else if (tempLookup == subst.getVariableResolver()) {
162                    subst.setVariableResolver(new Interpolator(null));
163                }
164                if (child.getName().equalsIgnoreCase("appenders")) {
165                    appenders = (ConcurrentMap<String, Appender>) child.getObject();
166                } else if (child.getObject() instanceof Filter) {
167                    addFilter((Filter) child.getObject());
168                } else if (child.getName().equalsIgnoreCase("loggers")) {
169                    Loggers l = (Loggers) child.getObject();
170                    loggers = l.getMap();
171                    setLoggers = true;
172                    if (l.getRoot() != null) {
173                        root = l.getRoot();
174                        setRoot = true;
175                    }
176                } else {
177                    LOGGER.error("Unknown object \"" + child.getName() + "\" of type " +
178                        child.getObject().getClass().getName() + " is ignored");
179                }
180            }
181    
182            if (!setLoggers) {
183                LOGGER.warn("No Loggers were configured, using default");
184            } else if (!setRoot) {
185                LOGGER.warn("No Root logger was configured, using default");
186            }
187    
188            root.setConfigurationMonitor(monitor);
189    
190            for (Map.Entry<String, LoggerConfig> entry : loggers.entrySet()) {
191                LoggerConfig l = entry.getValue();
192                l.setConfigurationMonitor(monitor);
193                for (AppenderRef ref : l.getAppenderRefs()) {
194                    Appender app = appenders.get(ref.getRef());
195                    if (app != null) {
196                        l.addAppender(app, ref.getLevel(), ref.getFilter());
197                    } else {
198                        LOGGER.error("Unable to locate appender " + ref.getRef() + " for logger " + l.getName());
199                    }
200                }
201    
202            }
203    
204            setParents();
205        }

272        /**
273         * Associates an Appender with a LoggerConfig. This method is synchronized in case a Logger with the
274         * same name is being updated at the same time.
275         *
276         * Note: This method is not used when configuring via configuration. It is primarily used by
277         * unit tests.
278         * @param logger The Logger the Appender will be associated with.
279         * @param appender The Appender.
280         */
281        public synchronized void addLoggerAppender(org.apache.logging.log4j.core.Logger logger, Appender appender) {
282            String name = logger.getName();
283            appenders.putIfAbsent(name, appender);
284            LoggerConfig lc = getLoggerConfig(name);
285            if (lc.getName().equals(name)) {
286                lc.addAppender(appender, null, null);
287            } else {
288                LoggerConfig nlc = new LoggerConfig(name, lc.getLevel(), lc.isAdditive());
289                nlc.setConfigurationMonitor(monitor);
290                nlc.addAppender(appender, null, null);
291                nlc.setParent(lc);
292                loggers.putIfAbsent(name, nlc);
293                setParents();
294                logger.getContext().updateLoggers();
295            }
296        }
297        /**
298         * Associates a Filter with a LoggerConfig. This method is synchronized in case a Logger with the
299         * same name is being updated at the same time.
300         *
301         * Note: This method is not used when configuring via configuration. It is primarily used by
302         * unit tests.
303         * @param logger The Logger the Fo;ter will be associated with.
304         * @param filter The Filter.
305         */
306        public synchronized void addLoggerFilter(org.apache.logging.log4j.core.Logger logger, Filter filter) {
307            String name = logger.getName();
308            LoggerConfig lc = getLoggerConfig(name);
309            if (lc.getName().equals(name)) {
310    
311                lc.addFilter(filter);
312            } else {
313                LoggerConfig nlc = new LoggerConfig(name, lc.getLevel(), lc.isAdditive());
314                nlc.setConfigurationMonitor(monitor);
315                nlc.addFilter(filter);
316                nlc.setParent(lc);
317                loggers.putIfAbsent(name, nlc);
318                setParents();
319                logger.getContext().updateLoggers();
320            }
321        }
322        /**
323         * Marks a LoggerConfig as additive. This method is synchronized in case a Logger with the
324         * same name is being updated at the same time.
325         *
326         * Note: This method is not used when configuring via configuration. It is primarily used by
327         * unit tests.
328         * @param logger The Logger the Appender will be associated with.
329         * @param additive True if the LoggerConfig should be additive, false otherwise.
330         */
331        public synchronized void setLoggerAdditive(org.apache.logging.log4j.core.Logger logger, boolean additive) {
332            String name = logger.getName();
333            LoggerConfig lc = getLoggerConfig(name);
334            if (lc.getName().equals(name)) {
335                lc.setAdditive(additive);
336            } else {
337                LoggerConfig nlc = new LoggerConfig(name, lc.getLevel(), additive);
338                nlc.setConfigurationMonitor(monitor);
339                nlc.setParent(lc);
340                loggers.putIfAbsent(name, nlc);
341                setParents();
342                logger.getContext().updateLoggers();
343            }
344        }

346        /**
347         * Remove an Appender. First removes any associations between LoggerContigs and the Appender, removes
348         * the Appender from this appender list and then stops the appender. This method is synchronized in
349         * case an Appender with the same name is being added during the removal.
350         * @param name the name of the appender to remove.
351         */
352        public synchronized void removeAppender(String name) {
353            for (LoggerConfig logger : loggers.values()) {
354                logger.removeAppender(name);
355            }
356            Appender app = appenders.remove(name);
357    
358            if (app != null) {
359                app.stop();
360            }
361        }

363        /**
364         * Locates the appropriate LoggerConfig for a Logger name. This will remove tokens from the
365         * package name as necessary or return the root LoggerConfig if no other matches were found.
366         * @param name The Logger name.
367         * @return The located LoggerConfig.
368         */
369        public LoggerConfig getLoggerConfig(String name) {
370            if (loggers.containsKey(name)) {
371                return loggers.get(name);
372            }
373            String substr = name;
374            while ((substr = NameUtil.getSubName(substr)) != null) {
375                if (loggers.containsKey(substr)) {
376                    return loggers.get(substr);
377                }
378            }
379            return root;
380        }

407        /**
408         * Adding a logger cannot be done atomically so is not allowed in an active configuration. Adding
409         * or removing a Logger requires creating a new configuration and then switching.
410         *
411         * @param name The name of the Logger.
412         * @param loggerConfig The LoggerConfig.
413         */
414        public void addLogger(String name, LoggerConfig loggerConfig) {
415            if (started) {
416                String msg = "Cannot add logger " + name + " to an active configuration";
417                LOGGER.warn(msg);
418                throw new IllegalStateException(msg);
419            }
420            loggers.put(name, loggerConfig);
421            setParents();
422        }

435            }

440        public void createConfiguration(Node node, LogEvent event) {
441            PluginType type = node.getType();
442            if (type != null && type.isDeferChildren()) {
443                node.setObject(createPluginObject(type, node, event));
444            } else {
445                for (Node child : node.getChildren()) {
446                    createConfiguration(child, event);
447                }
448    
449                if (type == null) {
450                    if (node.getParent() != null) {
451                        LOGGER.error("Unable to locate plugin for " + node.getName());
452                    }
453                } else {
454                    node.setObject(createPluginObject(type, node, event));
455                }
456            }
457        }

459       /*
460        * Retrieve a static public 'method to create the desired object. Every parameter
461        * will be annotated to identify the appropriate attribute or element to use to
462        * set the value of the paraemter.
463        * Parameters annotated with PluginAttr will always be set as Strings.
464        * Parameters annotated with PluginElement may be Objects or arrays. Collections
465        * and Maps are currently not supported, although the factory method that is called
466        * can create these from an array.
467        *
468        * Although the happy path works, more work still needs to be done to log incorrect
469        * parameters. These will generally result in unhelpful InvocationTargetExceptions.
470        * @param classClass the class.
471        * @return the instantiate method or null if there is none by that
472        * description.
473        */
474        private Object createPluginObject(PluginType type, Node node, LogEvent event)
475        {
476            Class clazz = type.getPluginClass();
477    
478            if (Map.class.isAssignableFrom(clazz)) {
479                try {
480                    Map<String, Object> map = (Map<String, Object>) clazz.newInstance();
481                    for (Node child : node.getChildren()) {
482                        map.put(child.getName(), child.getObject());
483                    }
484                    return map;
485                } catch (Exception ex) {
486                    LOGGER.warn("Unable to create Map for " + type.getElementName() + " of class " +
487                        clazz);
488                }
489            }
490    
491            if (List.class.isAssignableFrom(clazz)) {
492                try {
493                    List<Object> list = (List<Object>) clazz.newInstance();
494                    for (Node child : node.getChildren()) {
495                        list.add(child.getObject());
496                    }
497                    return list;
498                } catch (Exception ex) {
499                    LOGGER.warn("Unable to create List for " + type.getElementName() + " of class " +
500                        clazz);
501                }
502            }
503    
504            Method factoryMethod = null;
505    
506            for (Method method : clazz.getMethods()) {
507                if (method.isAnnotationPresent(PluginFactory.class)) {
508                    factoryMethod = method;
509                    break;
510                }
511            }
512            if (factoryMethod == null) {
513                return null;
514            }
515    
516            Annotation[][] parmArray = factoryMethod.getParameterAnnotations();
517            Class[] parmClasses = factoryMethod.getParameterTypes();
518            if (parmArray.length != parmClasses.length) {
519                LOGGER.error("Number of parameter annotations does not equal the number of paramters");
520            }
521            Object[] parms = new Object[parmClasses.length];
522    
523            int index = 0;
524            Map<String, String> attrs = node.getAttributes();
525            List<Node> children = node.getChildren();
526            StringBuilder sb = new StringBuilder();
527            List<Node> used = new ArrayList<Node>();
528    
529            /*
530             * For each parameter:
531             * If the parameter is an attribute store the value of the attribute in the parameter array.
532             * If the parameter is an element:
533             *   Determine if the required parameter is an array.
534             *     If so, if a child contains the array, use it,
535             *      otherwise create the array from all child nodes of the correct type.
536             *     Store the array into the parameter array.
537             *   If not an array, store the object in the child node into the parameter array.
538             */
539            for (Annotation[] parmTypes : parmArray) {
540                for (Annotation a : parmTypes) {
541                    if (sb.length() == 0) {
542                        sb.append(" with params(");
543                    } else {
544                        sb.append(", ");
545                    }
546                    if (a instanceof PluginNode) {
547                        parms[index] = node;
548                        sb.append("Node=").append(node.getName());
549                    } else if (a instanceof PluginConfiguration) {
550                        parms[index] = this;
551                        if (this.name != null) {
552                            sb.append("Configuration(").append(name).append(")");
553                        } else {
554                            sb.append("Configuration");
555                        }
556                    } else if (a instanceof PluginValue) {
557                        String name = ((PluginValue) a).value();
558                        String v = node.getValue();
559                        if (v == null) {
560                            v = getAttrValue("value", attrs);
561                        }
562                        String value = subst.replace(event, v);
563                        sb.append(name).append("=\"").append(value).append("\"");
564                        parms[index] = value;
565                    } else if (a instanceof PluginAttr) {
566                        String name = ((PluginAttr) a).value();
567                        String value = subst.replace(event, getAttrValue(name, attrs));
568                        sb.append(name).append("=\"").append(value).append("\"");
569                        parms[index] = value;
570                    } else if (a instanceof PluginElement) {
571                        PluginElement elem = (PluginElement) a;
572                        String name = elem.value();
573                        if (parmClasses[index].isArray()) {
574                            Class parmClass = parmClasses[index].getComponentType();
575                            List<Object> list = new ArrayList<Object>();
576                            sb.append(name).append("={");
577                            boolean first = true;
578                            for (Node child : children) {
579                                PluginType childType = child.getType();
580                                if (elem.value().equalsIgnoreCase(childType.getElementName()) ||
581                                    parmClass.isAssignableFrom(childType.getPluginClass())) {
582                                    used.add(child);
583                                    if (!first) {
584                                        sb.append(", ");
585                                    }
586                                    first = false;
587                                    Object obj = child.getObject();
588                                    if (obj == null) {
589                                        System.out.println("Null object returned for " + child.getName());
590                                    }
591                                    if (obj.getClass().isArray()) {
592                                        printArray(sb, (Object[]) obj);
593                                        parms[index] = obj;
594                                        break;
595                                    }
596                                    sb.append(child.toString());
597                                    list.add(obj);
598                                }
599                            }
600                            sb.append("}");
601                            if (parms[index] != null) {
602                                break;
603                            }
604                            if (list.size() > 0 && !parmClass.isAssignableFrom(list.get(0).getClass())) {
605                                LOGGER.error("Attempted to assign List containing class " +
606                                    list.get(0).getClass().getName() + " to array of type " + parmClass +
607                                    " for attribute " + name);
608                                break;
609                            }
610                            Object[] array = (Object[]) Array.newInstance(parmClass, list.size());
611                            int i = 0;
612                            for (Object obj : list) {
613                                array[i] = obj;
614                                ++i;
615                            }
616                            parms[index] = array;
617                        } else {
618                            Class parmClass = parmClasses[index];
619                            boolean present = false;
620                            for (Node child : children) {
621                                PluginType childType = child.getType();
622                                if (elem.value().equals(childType.getElementName()) ||
623                                    parmClass.isAssignableFrom(childType.getPluginClass())) {
624                                    sb.append(child.getName()).append("(").append(child.toString()).append(")");
625                                    present = true;
626                                    used.add(child);
627                                    parms[index] = child.getObject();
628                                    break;
629                                }
630                            }
631                            if (!present) {
632                                sb.append("null");
633                            }
634                        }
635                    }
636                }
637                ++index;
638            }
639            if (sb.length() > 0) {
640                sb.append(")");
641            }
642    
643            if (attrs.size() > 0) {
644                StringBuilder eb = new StringBuilder();
645                for (String key : attrs.keySet()) {
646                    if (eb.length() == 0) {
647                        eb.append(node.getName());
648                        eb.append(" contains ");
649                        if (attrs.size() == 1) {
650                            eb.append("an invalid element or attribute ");
651                        } else {
652                            eb.append("invalid attributes ");
653                        }
654                    } else {
655                        eb.append(", ");
656                    }
657                    eb.append("\"");
658                    eb.append(key);
659                    eb.append("\"");
660    
661                }
662                LOGGER.error(eb.toString());
663            }
664    
665            if (!type.isDeferChildren() && used.size() != children.size()) {
666                for (Node child : children) {
667                    if (used.contains(child)) {
668                        continue;
669                    }
670                    String nodeType = node.getType().getElementName();
671                    String start = nodeType.equals(node.getName()) ? node.getName() : nodeType + " " + node.getName();
672                    LOGGER.error(start + " has no parameter that matches element " + child.getName());
673                }
674            }
675    
676            try {
677                int mod = factoryMethod.getModifiers();
678                if (!Modifier.isStatic(mod)) {
679                    LOGGER.error(factoryMethod.getName() + " method is not static on class " +
680                        clazz.getName() + " for element " + node.getName());
681                    return null;
682                }
683                LOGGER.debug("Calling {} on class {} for element {}", factoryMethod.getName(), clazz.getName(),
684                    node.getName(), sb.toString());
685                //if (parms.length > 0) {
686                    return factoryMethod.invoke(null, parms);
687                //}
688                //return factoryMethod.invoke(null, node);
689            } catch (Exception e) {
690                LOGGER.error("Unable to invoke method " + factoryMethod.getName() + " in class " +
691                    clazz.getName() + " for element " + node.getName(), e);
692            }
693            return null;
694        }

696        private void printArray(StringBuilder sb, Object[] array) {
697            boolean first = true;
698            for (Object obj : array) {
699                if (!first) {
700                    sb.append(", ");
701                }
702                sb.append(obj.toString());
703                first = false;
704            }
705        }

707        private String getAttrValue(String name, Map<String, String> attrs) {
708            for (String key : attrs.keySet()) {
709                if (key.equalsIgnoreCase(name)) {
710                    String attr = attrs.get(key);
711                    attrs.remove(key);
712                    return attr;
713                }
714            }
715            return null;
716        }

718        private void setParents() {
719             for (Map.Entry<String, LoggerConfig> entry : loggers.entrySet()) {
720                LoggerConfig logger = entry.getValue();
721                String name = entry.getKey();
722                if (!name.equals("")) {
723                    int i = name.lastIndexOf(".");
724                    if (i > 0) {
725                        name = name.substring(0, i);
726                        LoggerConfig parent = getLoggerConfig(name);
727                        if (parent == null) {
728                            parent = root;
729                        }
730                        logger.setParent(parent);
731                    }
732                }
733            }
734        }

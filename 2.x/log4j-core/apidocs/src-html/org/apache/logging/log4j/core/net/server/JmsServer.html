018package org.apache.logging.log4j.core.net.server;
019
020import java.io.BufferedReader;
021import java.io.IOException;
022import java.io.InputStreamReader;
023import java.nio.charset.Charset;
024import java.util.concurrent.TimeUnit;
025import java.util.concurrent.atomic.AtomicReference;
026import javax.jms.JMSException;
027import javax.jms.Message;
028import javax.jms.MessageConsumer;
029import javax.jms.MessageListener;
030import javax.jms.ObjectMessage;
031
032import org.apache.logging.log4j.LoggingException;
033import org.apache.logging.log4j.core.AbstractLifeCycle;
034import org.apache.logging.log4j.core.LifeCycle2;
035import org.apache.logging.log4j.core.LogEvent;
036import org.apache.logging.log4j.core.LogEventListener;
037import org.apache.logging.log4j.core.appender.mom.JmsManager;
038import org.apache.logging.log4j.core.net.JndiManager;
039
040/**
041 * LogEventListener server that receives LogEvents over a JMS {@link javax.jms.Destination}.
042 *
043 * @since 2.1
044 */
045public class JmsServer extends LogEventListener implements MessageListener, LifeCycle2 {
046
047    private final AtomicReference<State> state = new AtomicReference<>(State.INITIALIZED);
048    private final JmsManager jmsManager;
049    private MessageConsumer messageConsumer;
050
051    public JmsServer(final String connectionFactoryBindingName,
052                     final String destinationBindingName,
053                     final String username,
054                     final String password) {
055        final String managerName = JmsServer.class.getName() + '@' + JmsServer.class.hashCode();
056        final JndiManager jndiManager = JndiManager.getDefaultManager(managerName);
057        jmsManager = JmsManager.getJmsManager(managerName, jndiManager, connectionFactoryBindingName,
058            destinationBindingName, username, password);
059    }
060
061    @Override
062    public State getState() {
063        return state.get();
064    }
065
066    @Override
067    public void onMessage(final Message message) {
068        try {
069            if (message instanceof ObjectMessage) {
070                final Object body = ((ObjectMessage) message).getObject();
071                if (body instanceof LogEvent) {
072                    log((LogEvent) body);
073                } else {
074                    LOGGER.warn("Expected ObjectMessage to contain LogEvent. Got type {} instead.", body.getClass());
075                }
076            } else {
077                LOGGER.warn("Received message of type {} and JMSType {} which cannot be handled.", message.getClass(),
078                    message.getJMSType());
079            }
080        } catch (final JMSException e) {
081            LOGGER.catching(e);
082        }
083    }
084
085    @Override
086    public void initialize() {
087    }
088
089    @Override
090    public void start() {
091        if (state.compareAndSet(State.INITIALIZED, State.STARTING)) {
092            try {
093                messageConsumer = jmsManager.createMessageConsumer();
094                messageConsumer.setMessageListener(this);
095            } catch (final JMSException e) {
096                throw new LoggingException(e);
097            }
098        }
099    }
100
101    @Override
102    public void stop() {
103        stop(AbstractLifeCycle.DEFAULT_STOP_TIMEOUT, AbstractLifeCycle.DEFAULT_STOP_TIMEUNIT);
104    }
105
106    @Override
107    public boolean stop(final long timeout, final TimeUnit timeUnit) {
108        boolean stopped = true;
109        try {
110            messageConsumer.close();
111        } catch (final JMSException e) {
112            LOGGER.debug("Exception closing {}", messageConsumer, e);
113            stopped = false;
114        }
115        return stopped && jmsManager.stop(timeout, timeUnit);
116    }
117
118    @Override
119    public boolean isStarted() {
120        return state.get() == State.STARTED;
121    }
122
123    @Override
124    public boolean isStopped() {
125        return state.get() == State.STOPPED;
126    }
127
128    /**
129     * Starts and runs this server until the user types "exit" into standard input.
130     *
131     * @throws IOException
132     * @since 2.6
133     */
134    public void run() throws IOException {
135        this.start();
136        System.out.println("Type \"exit\" to quit.");
137        final BufferedReader stdin = new BufferedReader(new InputStreamReader(System.in, Charset.defaultCharset()));
138        while (true) {
139            final String line = stdin.readLine();
140            if (line == null || line.equalsIgnoreCase("exit")) {
141                System.out.println("Exiting. Kill the application if it does not exit due to daemon threads.");
142                this.stop();
143                return;
144            }
145        }
146    }
147
148}

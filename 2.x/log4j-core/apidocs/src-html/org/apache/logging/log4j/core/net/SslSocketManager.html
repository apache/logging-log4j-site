017package org.apache.logging.log4j.core.net;
018
019import java.io.ByteArrayOutputStream;
020import java.io.IOException;
021import java.io.OutputStream;
022import java.io.Serializable;
023import java.net.InetAddress;
024import java.net.InetSocketAddress;
025import java.net.Socket;
026import java.net.UnknownHostException;
027
028import javax.net.ssl.SSLSocket;
029import javax.net.ssl.SSLSocketFactory;
030
031import org.apache.logging.log4j.Level;
032import org.apache.logging.log4j.core.Layout;
033import org.apache.logging.log4j.core.appender.ManagerFactory;
034import org.apache.logging.log4j.core.net.ssl.SslConfiguration;
035import org.apache.logging.log4j.util.Strings;
036
037/**
038 *
039 */
040public class SslSocketManager extends TcpSocketManager {
041    public static final int DEFAULT_PORT = 6514;
042    private static final SslSocketManagerFactory FACTORY = new SslSocketManagerFactory();
043    private final SslConfiguration sslConfig;
044
045    /**
046     *
047     *
048     * @param name          The unique name of this connection.
049     * @param os            The OutputStream.
050     * @param sock          The Socket.
051     * @param inetAddress          The Internet address of the host.
052     * @param host          The name of the host.
053     * @param port          The port number on the host.
054     * @param connectTimeoutMillis the connect timeout in milliseconds.
055     * @param delay         Reconnection interval.
056     * @param immediateFail
057     * @param layout        The Layout.
058     * @param bufferSize The buffer size.
059     */
060    public SslSocketManager(final String name, final OutputStream os, final Socket sock,
061            final SslConfiguration sslConfig, final InetAddress inetAddress, final String host, final int port,
062            final int connectTimeoutMillis, final int delay, final boolean immediateFail,
063            final Layout<? extends Serializable> layout, final int bufferSize) {
064        super(name, os, sock, inetAddress, host, port, connectTimeoutMillis, delay, immediateFail, layout, bufferSize);
065        this.sslConfig = sslConfig;
066    }
067
068    private static class SslFactoryData {
069        protected SslConfiguration sslConfiguration;
070        private final String host;
071        private final int port;
072        private final int connectTimeoutMillis;
073        private final int delayMillis;
074        private final boolean immediateFail;
075        private final Layout<? extends Serializable> layout;
076        private final int bufferSize;
077
078        public SslFactoryData(final SslConfiguration sslConfiguration, final String host, final int port,
079                final int connectTimeoutMillis, final int delayMillis, final boolean immediateFail,
080                final Layout<? extends Serializable> layout, final int bufferSize) {
081            this.host = host;
082            this.port = port;
083            this.connectTimeoutMillis = connectTimeoutMillis;
084            this.delayMillis = delayMillis;
085            this.immediateFail = immediateFail;
086            this.layout = layout;
087            this.sslConfiguration = sslConfiguration;
088            this.bufferSize = bufferSize;
089        }
090    }
091
092    public static SslSocketManager getSocketManager(final SslConfiguration sslConfig, final String host, int port,
093            final int connectTimeoutMillis, int reconnectDelayMillis, final boolean immediateFail,
094            final Layout<? extends Serializable> layout, final int bufferSize) {
095        if (Strings.isEmpty(host)) {
096            throw new IllegalArgumentException("A host name is required");
097        }
098        if (port <= 0) {
099            port = DEFAULT_PORT;
100        }
101        if (reconnectDelayMillis == 0) {
102            reconnectDelayMillis = DEFAULT_RECONNECTION_DELAY_MILLIS;
103        }
104        return (SslSocketManager) getManager("TLS:" + host + ':' + port, new SslFactoryData(sslConfig, host, port,
105                connectTimeoutMillis, reconnectDelayMillis, immediateFail, layout, bufferSize), FACTORY);
106    }
107
108    @Override
109    protected Socket createSocket(final String host, final int port) throws IOException {
110        final SSLSocketFactory socketFactory = createSslSocketFactory(sslConfig);
111        final InetSocketAddress address = new InetSocketAddress(host, port);
112        final Socket newSocket = socketFactory.createSocket();
113        newSocket.connect(address, getConnectTimeoutMillis());
114        return newSocket;
115    }
116
117    private static SSLSocketFactory createSslSocketFactory(final SslConfiguration sslConf) {
118        SSLSocketFactory socketFactory;
119
120        if (sslConf != null) {
121            socketFactory = sslConf.getSslSocketFactory();
122        } else {
123            socketFactory = (SSLSocketFactory) SSLSocketFactory.getDefault();
124        }
125
126        return socketFactory;
127    }
128
129
130    private static class SslSocketManagerFactory implements ManagerFactory<SslSocketManager, SslFactoryData> {
131
132        private static class TlsSocketManagerFactoryException extends Exception {
133
134            private static final long serialVersionUID = 1L;
135        }
136
137        @Override
138        public SslSocketManager createManager(final String name, final SslFactoryData data) {
139            InetAddress inetAddress = null;
140            OutputStream os = null;
141            Socket socket = null;
142
143            try {
144                inetAddress = resolveAddress(data.host);
145                socket = createSocket(data);
146                os = socket.getOutputStream();
147                checkDelay(data.delayMillis, os);
148            }
149            catch (final IOException e) {
150                LOGGER.error("SslSocketManager ({})", name, e);
151                os = new ByteArrayOutputStream();
152            }
153            catch (final TlsSocketManagerFactoryException e) {
154                LOGGER.catching(Level.DEBUG, e);
155                return null;
156            }
157            return new SslSocketManager(name, os, socket, data.sslConfiguration, inetAddress, data.host, data.port, 0,
158                    data.delayMillis, data.immediateFail, data.layout, data.bufferSize);
159        }
160
161        private InetAddress resolveAddress(final String hostName) throws TlsSocketManagerFactoryException {
162            InetAddress address;
163
164            try {
165                address = InetAddress.getByName(hostName);
166            } catch (final UnknownHostException ex) {
167                LOGGER.error("Could not find address of {}", hostName, ex);
168                throw new TlsSocketManagerFactoryException();
169            }
170
171            return address;
172        }
173
174        private void checkDelay(final int delay, final OutputStream os) throws TlsSocketManagerFactoryException {
175            if (delay == 0 && os == null) {
176                throw new TlsSocketManagerFactoryException();
177            }
178        }
179
180        private Socket createSocket(final SslFactoryData data) throws IOException {
181            SSLSocketFactory socketFactory;
182            SSLSocket socket;
183
184            socketFactory = createSslSocketFactory(data.sslConfiguration);
185            socket = (SSLSocket) socketFactory.createSocket(data.host, data.port);
186            return socket;
187        }
188    }
189}

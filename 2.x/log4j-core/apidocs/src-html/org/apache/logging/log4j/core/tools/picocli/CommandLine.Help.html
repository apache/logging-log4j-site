075/**
076 * <p>
077 * CommandLine interpreter that uses reflection to initialize an annotated domain object with values obtained from the
078 * command line arguments.
079 * </p><h2>Example</h2>
080 * <pre>import static picocli.CommandLine.*;
081 *
082 * &#064;Command(header = "Encrypt FILE(s), or standard input, to standard output or to the output file.")
083 * public class Encrypt {
084 *
085 *     &#064;Parameters(type = File.class, description = "Any number of input files")
086 *     private List&lt;File&gt; files = new ArrayList&lt;File&gt;();
087 *
088 *     &#064;Option(names = { "-o", "--out" }, description = "Output file (default: print to console)")
089 *     private File outputFile;
090 *
091 *     &#064;Option(names = { "-v", "--verbose"}, description = "Verbosely list files processed")
092 *     private boolean verbose;
093 *
094 *     &#064;Option(names = { "-h", "--help", "-?", "-help"}, help = true, description = "Display this help and exit")
095 *     private boolean help;
096 * }
097 * </pre>
098 * <p>
099 * Use {@code CommandLine} to initialize a domain object as follows:
100 * </p><pre>
101 * public static void main(String... args) {
102 *     try {
103 *         Encrypt encrypt = CommandLine.populateCommand(new Encrypt(), args);
104 *         if (encrypt.help) {
105 *             CommandLine.usage(encrypt, System.out);
106 *         } else {
107 *             runProgram(encrypt);
108 *         }
109 *     } catch (ParameterException ex) { // command line arguments could not be parsed
110 *         System.err.println(ex.getMessage());
111 *         CommandLine.usage(new Encrypt(), System.err);
112 *     }
113 * }
114 * </pre><p>
115 * Invoke the above program with some command line arguments. The below are all equivalent:
116 * </p>
117 * <pre>
118 * --verbose --out=outfile in1 in2
119 * --verbose --out outfile in1 in2
120 * -v --out=outfile in1 in2
121 * -v -o outfile in1 in2
122 * -v -o=outfile in1 in2
123 * -vo outfile in1 in2
124 * -vo=outfile in1 in2
125 * -v -ooutfile in1 in2
126 * -vooutfile in1 in2
127 * </pre>
128 *
129 * <p>
130 * Copied and modified from <a href="http://github.com/remkop/picocli/">picocli</a>.
131 * </p>
132 *
133 * @since 2.9
134 */

159    /** Registers a subcommand with the specified name. For example:
160     * <pre>
161     * CommandLine commandLine = new CommandLine(new Git())
162     *         .addSubcommand("status",   new GitStatus())
163     *         .addSubcommand("commit",   new GitCommit();
164     *         .addSubcommand("add",      new GitAdd())
165     *         .addSubcommand("branch",   new GitBranch())
166     *         .addSubcommand("checkout", new GitCheckout())
167     *         //...
168     *         ;
169     * </pre>
170     *
171     * <p>The specified object can be an annotated object or a
172     * {@code CommandLine} instance with its own nested subcommands. For example:</p>
173     * <pre>
174     * CommandLine commandLine = new CommandLine(new MainCommand())
175     *         .addSubcommand("cmd1",                 new ChildCommand1()) // subcommand
176     *         .addSubcommand("cmd2",                 new ChildCommand2())
177     *         .addSubcommand("cmd3", new CommandLine(new ChildCommand3()) // subcommand with nested sub-subcommands
178     *                 .addSubcommand("cmd3sub1",                 new GrandChild3Command1())
179     *                 .addSubcommand("cmd3sub2",                 new GrandChild3Command2())
180     *                 .addSubcommand("cmd3sub3", new CommandLine(new GrandChild3Command3()) // deeper nesting
181     *                         .addSubcommand("cmd3sub3sub1", new GreatGrandChild3Command3_1())
182     *                         .addSubcommand("cmd3sub3sub2", new GreatGrandChild3Command3_2())
183     *                 )
184     *         );
185     * </pre>
186     * <p>The default type converters are available on all subcommands and nested sub-subcommands, but custom type
187     * converters are registered only with the subcommand hierarchy as it existed when the custom type was registered.
188     * To ensure a custom type converter is available to all subcommands, register the type converter last, after
189     * adding subcommands.</p>
190     *
191     * @param name the string to recognize on the command line as a subcommand
192     * @param command the object to initialize with command line arguments following the subcommand name.
193     *          This may be a {@code CommandLine} instance with its own (nested) subcommands
194     * @return this CommandLine object, to allow method chaining
195     * @see #registerConverter(Class, ITypeConverter)
196     * @since 0.9.7
197     */

199        CommandLine commandLine = toCommandLine(command);
200        commandLine.parent = this;
201        interpreter.commands.put(name, commandLine);
202        return this;
203    }

396    /**
397     * Prints a usage help message for the annotated command class to the specified {@code PrintStream}.
398     * Delegates construction of the usage help message to the {@link Help} inner class and is equivalent to:
399     * <pre>
400     * Help help = new Help(command).addAllSubcommands(getSubcommands());
401     * StringBuilder sb = new StringBuilder()
402     *         .append(help.headerHeading())
403     *         .append(help.header())
404     *         .append(help.synopsisHeading())      //e.g. Usage:
405     *         .append(help.synopsis())             //e.g. &lt;main class&gt; [OPTIONS] &lt;command&gt; [COMMAND-OPTIONS] [ARGUMENTS]
406     *         .append(help.descriptionHeading())   //e.g. %nDescription:%n%n
407     *         .append(help.description())          //e.g. {"Converts foos to bars.", "Use options to control conversion mode."}
408     *         .append(help.parameterListHeading()) //e.g. %nPositional parameters:%n%n
409     *         .append(help.parameterList())        //e.g. [FILE...] the files to convert
410     *         .append(help.optionListHeading())    //e.g. %nOptions:%n%n
411     *         .append(help.optionList())           //e.g. -h, --help   displays this help and exits
412     *         .append(help.commandListHeading())   //e.g. %nCommands:%n%n
413     *         .append(help.commandList())          //e.g.    add       adds the frup to the frooble
414     *         .append(help.footerHeading())
415     *         .append(help.footer());
416     * out.print(sb);
417     * </pre>
418     * <p>Annotate your class with {@link Command} to control many aspects of the usage help message, including
419     * the program name, text of section headings and section contents, and some aspects of the auto-generated sections
420     * of the usage help message.
421     * <p>To customize the auto-generated sections of the usage help message, like how option details are displayed,
422     * instantiate a {@link Help} object and use a {@link Help.TextTable} with more of fewer columns, a custom
423     * {@linkplain Help.Layout layout}, and/or a custom option {@linkplain Help.IOptionRenderer renderer}
424     * for ultimate control over which aspects of an Option or Field are displayed where.</p>
425     * @param out the {@code PrintStream} to print the usage help message to
426     * @param colorScheme the {@code ColorScheme} defining the styles for options, parameters and commands when ANSI is enabled
427     */
428    public void usage(PrintStream out, Help.ColorScheme colorScheme) {
429        Help help = new Help(interpreter.command, colorScheme).addAllSubcommands(getSubcommands());
430        StringBuilder sb = new StringBuilder()
431                .append(help.headerHeading())
432                .append(help.header())
433                .append(help.synopsisHeading())      //e.g. Usage:
434                .append(help.synopsis(help.synopsisHeadingLength())) //e.g. &lt;main class&gt; [OPTIONS] &lt;command&gt; [COMMAND-OPTIONS] [ARGUMENTS]
435                .append(help.descriptionHeading())   //e.g. %nDescription:%n%n
436                .append(help.description())          //e.g. {"Converts foos to bars.", "Use options to control conversion mode."}
437                .append(help.parameterListHeading()) //e.g. %nPositional parameters:%n%n
438                .append(help.parameterList())        //e.g. [FILE...] the files to convert
439                .append(help.optionListHeading())    //e.g. %nOptions:%n%n
440                .append(help.optionList())           //e.g. -h, --help   displays this help and exits
441                .append(help.commandListHeading())   //e.g. %nCommands:%n%n
442                .append(help.commandList())          //e.g.    add       adds the frup to the frooble
443                .append(help.footerHeading())
444                .append(help.footer());
445        out.print(sb);
446    }

483    /**
484     * Convenience method to allow command line application authors to avoid some boilerplate code in their application.
485     * The annotated object needs to implement {@link Runnable}. Calling this method is equivalent to:
486     * <pre>
487     * CommandLine cmd = new CommandLine(command);
488     * try {
489     *     cmd.parse(args);
490     * } catch (Exception ex) {
491     *     System.err.println(ex.getMessage());
492     *     cmd.usage(out, ansi);
493     *     return;
494     * }
495     * command.run();
496     * </pre>
497     * Note that this method is not suitable for commands with subcommands.
498     * @param command the command to run when {@linkplain #populateCommand(Object, String...) parsing} succeeds.
499     * @param out the printStream to print to
500     * @param ansi whether the usage message should include ANSI escape codes or not
501     * @param args the command line arguments to parse
502     * @param <R> the annotated object must implement Runnable
503     * @throws IllegalArgumentException if the specified command object does not have a {@link Command}, {@link Option} or {@link Parameters} annotation
504     */
505    public static <R extends Runnable> void run(R command, PrintStream out, Help.Ansi ansi, String... args) {
506        CommandLine cmd = new CommandLine(command); // validate command outside of try-catch
507        try {
508            cmd.parse(args);
509        } catch (Exception ex) {
510            out.println(ex.getMessage());
511            cmd.usage(out, ansi);
512            return;
513        }
514        command.run();
515    }

517    /**
518     * Registers the specified type converter for the specified class. When initializing fields annotated with
519     * {@link Option}, the field's type is used as a lookup key to find the associated type converter, and this
520     * type converter converts the original command line argument string value to the correct type.
521     * <p>
522     * Java 8 lambdas make it easy to register custom type converters:
523     * </p>
524     * <pre>
525     * commandLine.registerConverter(java.nio.file.Path.class, s -&gt; java.nio.file.Paths.get(s));
526     * commandLine.registerConverter(java.time.Duration.class, s -&gt; java.time.Duration.parse(s));</pre>
527     * <p>
528     * Built-in type converters are pre-registered for the following java 1.5 types:
529     * </p>
530     * <ul>
531     *   <li>all primitive types</li>
532     *   <li>all primitive wrapper types: Boolean, Byte, Character, Double, Float, Integer, Long, Short</li>
533     *   <li>any enum</li>
534     *   <li>java.io.File</li>
535     *   <li>java.math.BigDecimal</li>
536     *   <li>java.math.BigInteger</li>
537     *   <li>java.net.InetAddress</li>

558     * @see #addSubcommand(String, Object)

619        /**
620         * One or more option names. At least one option name is required.
621         * <p>
622         * Different environments have different conventions for naming options, but usually options have a prefix
623         * that sets them apart from parameters.
624         * Picocli supports all of the below styles. The default separator is {@code '='}, but this can be configured.
625         * </p><p>
626         * <b>*nix</b>
627         * </p><p>
628         * In Unix and Linux, options have a short (single-character) name, a long name or both.
629         * Short options
630         * (<a href="http://pubs.opengroup.org/onlinepubs/9699919799/basedefs/V1_chap12.html#tag_12_02">POSIX
631         * style</a> are single-character and are preceded by the {@code '-'} character, e.g., {@code `-v'}.
632         * <a href="https://www.gnu.org/software/tar/manual/html_node/Long-Options.html">GNU-style</a> long
633         * (or <em>mnemonic</em>) options start with two dashes in a row, e.g., {@code `--file'}.
634         * </p><p>Picocli supports the POSIX convention that short options can be grouped, with the last option
635         * optionally taking a parameter, which may be attached to the option name or separated by a space or
636         * a {@code '='} character. The below examples are all equivalent:
637         * </p><pre>
638         * -xvfFILE
639         * -xvf FILE
640         * -xvf=FILE
641         * -xv --file FILE
642         * -xv --file=FILE
643         * -x -v --file FILE
644         * -x -v --file=FILE
645         * </pre><p>
646         * <b>DOS</b>
647         * </p><p>
648         * DOS options mostly have upper case single-character names and start with a single slash {@code '/'} character.
649         * Option parameters are separated by a {@code ':'} character. Options cannot be grouped together but
650         * must be specified separately. For example:
651         * </p><pre>
652         * DIR /S /A:D /T:C
653         * </pre><p>
654         * <b>PowerShell</b>
655         * </p><p>
656         * Windows PowerShell options generally are a word preceded by a single {@code '-'} character, e.g., {@code `-Help'}.
657         * Option parameters are separated by a space or by a {@code ':'} character.
658         * </p>
659         * @return one or more option names
660         */

728        /**
729         * Specifies the minimum number of required parameters and the maximum number of accepted parameters.

734         * By default, flags (boolean options) have arity zero,

737         * </p><p>

739         * like arrays or Collections. See {@link #type()} for strongly-typed Collection fields.
740         * </p><p>

742         * specify {@code @Option(names = "-example", arity = "2..*")}.

815         */
816        boolean hidden() default false;

1266                    if (existing != null && existing != field) {

1268                    }

1272                        if (existing2 != null && existing2 != field) {
1273                            throw DuplicateOptionAnnotationsException.create(name, field, existing2);

1280                    throw new ParameterException("A field can be either @Option or @Parameters, but '"
1281                            + field.getName() + "' is both.");
1282                }

1448        private void processArguments(List<CommandLine> parsedCommands,
1449                                      Stack<String> args,
1450                                      Collection<Field> required,
1451                                      Set<Field> initialized,
1452                                      String[] originalArgs) throws Exception {
1453            // arg must be one of:
1454            // 1. the "--" double dash separating options from positional arguments
1455            // 1. a stand-alone flag, like "-v" or "--verbose": no value required, must map to boolean or Boolean field
1456            // 2. a short option followed by an argument, like "-f file" or "-ffile": may map to any type of field
1457            // 3. a long option followed by an argument, like "-file out.txt" or "-file=out.txt"
1458            // 3. one or more remaining arguments without any associated options. Must be the last in the list.
1459            // 4. a combination of stand-alone options, like "-vxr". Equivalent to "-v -x -r", "-v true -x true -r true"
1460            // 5. a combination of stand-alone options and one option with an argument, like "-vxrffile"
1461
1462            while (!args.isEmpty()) {
1463                String arg = args.pop();
1464
1465                // Double-dash separates options from positional arguments.
1466                // If found, then interpret the remaining args as positional parameters.
1467                if ("--".equals(arg)) {
1468                    processPositionalParameters(required, args);
1469                    return; // we are done
1470                }
1471
1472                // if we find another command, we are done with the current command
1473                if (commands.containsKey(arg)) {
1474                    if (!isHelpRequested && !required.isEmpty()) { // ensure current command portion is valid
1475                        throw MissingParameterException.create(required);
1476                    }
1477                    commands.get(arg).interpreter.parse(parsedCommands, args, originalArgs);
1478                    return; // remainder done by the command
1479                }
1480
1481                // First try to interpret the argument as a single option (as opposed to a compact group of options).
1482                // A single option may be without option parameters, like "-v" or "--verbose" (a boolean value),
1483                // or an option may have one or more option parameters.
1484                // A parameter may be attached to the option.
1485                boolean paramAttachedToOption = false;
1486                int separatorIndex = arg.indexOf(separator);
1487                if (separatorIndex > 0) {
1488                    String key = arg.substring(0, separatorIndex);
1489                    // be greedy. Consume the whole arg as an option if possible.
1490                    if (optionName2Field.containsKey(key) && !optionName2Field.containsKey(arg)) {
1491                        paramAttachedToOption = true;
1492                        String optionParam = arg.substring(separatorIndex + separator.length());
1493                        args.push(optionParam);
1494                        arg = key;
1495                    }
1496                }
1497                if (optionName2Field.containsKey(arg)) {
1498                    processStandaloneOption(required, initialized, arg, args, paramAttachedToOption);
1499                }
1500                // Compact (single-letter) options can be grouped with other options or with an argument.
1501                // only single-letter options can be combined with other options or with an argument
1502                else if (arg.length() > 2 && arg.startsWith("-")) {
1503                    processClusteredShortOptions(required, initialized, arg, args);
1504                }
1505                // The argument could not be interpreted as an option.
1506                // We take this to mean that the remainder are positional arguments
1507                else {
1508                    args.push(arg);
1509                    processPositionalParameters(required, args);
1510                    return;
1511                }
1512            }

1528        private void processPositionalParameters0(Collection<Field> required, boolean validateOnly, Stack<String> args) throws Exception {
1529            int max = -1;
1530            for (Field positionalParam : positionalParametersFields) {
1531                Range indexRange = Range.parameterIndex(positionalParam);
1532                max = Math.max(max, indexRange.max);
1533                @SuppressWarnings("unchecked")
1534                Stack<String> argsCopy = reverse((Stack<String>) args.clone());
1535                if (!indexRange.isVariable) {
1536                    for (int i = argsCopy.size() - 1; i > indexRange.max; i--) {
1537                        argsCopy.removeElementAt(i);
1538                    }
1539                }
1540                Collections.reverse(argsCopy);
1541                for (int i = 0; i < indexRange.min && !argsCopy.isEmpty(); i++) { argsCopy.pop(); }
1542                Range arity = Range.parameterArity(positionalParam);
1543                assertNoMissingParameters(positionalParam, arity.min, argsCopy);
1544                if (!validateOnly) {
1545                    applyOption(positionalParam, Parameters.class, arity, false, argsCopy, null);
1546                    required.remove(positionalParam);
1547                }
1548            }
1549            // remove processed args from the stack

1570        private void processClusteredShortOptions(Collection<Field> required,
1571                                                  Set<Field> initialized,
1572                                                  String arg,
1573                                                  Stack<String> args)
1574                throws Exception {
1575            String prefix = arg.substring(0, 1);
1576            String cluster = arg.substring(1);
1577            boolean paramAttachedToOption = true;
1578            do {
1579                if (cluster.length() > 0 && singleCharOption2Field.containsKey(cluster.charAt(0))) {
1580                    Field field = singleCharOption2Field.get(cluster.charAt(0));
1581                    required.remove(field);
1582                    cluster = cluster.length() > 0 ? cluster.substring(1) : "";
1583                    paramAttachedToOption = cluster.length() > 0;
1584                    Range arity = Range.optionArity(field);
1585                    if (cluster.startsWith(separator)) {// attached with separator, like -f=FILE or -v=true
1586                        cluster = cluster.substring(separator.length());
1587                        arity = arity.min(Math.max(1, arity.min)); // if key=value, minimum arity is at least 1
1588                    }
1589                    args.push(cluster); // interpret remainder as option parameter (CAUTION: may be empty string!)
1590                    // arity may be >= 1, or
1591                    // arity <= 0 && !cluster.startsWith(separator)
1592                    // e.g., boolean @Option("-v", arity=0, varargs=true); arg "-rvTRUE", remainder cluster="TRUE"
1593                    int consumed = applyOption(field, Option.class, arity, paramAttachedToOption, args, initialized);
1594                    // only return if cluster (and maybe more) was consumed, otherwise continue do-while loop
1595                    if (consumed > 0) {
1596                        return;
1597                    }
1598                    cluster = args.pop();
1599                } else { // cluster is empty || cluster.charAt(0) is not a short option key
1600                    if (cluster.length() == 0) { // we finished parsing a group of short options like -rxv
1601                        return; // return normally and parse the next arg
1602                    }
1603                    // We get here when the remainder of the cluster group is neither an option,
1604                    // nor a parameter that the last option could consume.
1605                    if (arg.endsWith(cluster)) {
1606                        // remainder was part of a clustered group that could not be completely parsed
1607                        args.push(paramAttachedToOption ? prefix + cluster : cluster);
1608                        handleUnmatchedArguments(args);
1609                    }
1610                    args.push(cluster);
1611                    processPositionalParameters(required, args);
1612                    return;
1613                }
1614            } while (true);
1615        }

1617        private int applyOption(Field field,
1618                                Class<?> annotation,
1619                                Range arity,
1620                                boolean valueAttachedToOption,
1621                                Stack<String> args,
1622                                Set<Field> initialized) throws Exception {
1623            updateHelpRequested(field);
1624            if (!args.isEmpty() && args.peek().length() == 0 && !valueAttachedToOption) {
1625                args.pop(); // throw out empty string we get at the end of a group of clustered short options
1626            }
1627            int length = args.size();
1628            assertNoMissingParameters(field, arity.min, args);
1629
1630            Class<?> cls = field.getType();
1631            if (cls.isArray()) {
1632                return applyValuesToArrayField(field, annotation, arity, args, cls);
1633            }
1634            if (Collection.class.isAssignableFrom(cls)) {
1635                return applyValuesToCollectionField(field, annotation, arity, args, cls);
1636            }
1637            return applyValueToSingleValuedField(field, arity, args, cls, initialized);
1638        }
1639        private int applyValueToSingleValuedField(Field field,
1640                                                  Range arity,
1641                                                  Stack<String> args,
1642                                                  Class<?> cls,
1643                                                  Set<Field> initialized) throws Exception {
1644            boolean noMoreValues = args.isEmpty();
1645            String value = args.isEmpty() ? null : trim(args.pop()); // unquote the value
1646            int result = arity.min; // the number or args we need to consume
1647
1648            // special logic for booleans: BooleanConverter accepts only "true" or "false".
1649            if ((cls == Boolean.class || cls == Boolean.TYPE) && arity.min <= 0) {
1650
1651                // boolean option with arity = 0..1 or 0..*: value MAY be a param
1652                if (arity.max > 0 && ("true".equalsIgnoreCase(value) || "false".equalsIgnoreCase(value))) {
1653                    result = 1;            // if it is a varargs we only consume 1 argument if it is a boolean value
1654                } else {
1655                    if (value != null) {
1656                        args.push(value); // we don't consume the value
1657                    }
1658                    Boolean currentValue = (Boolean) field.get(command);
1659                    value = String.valueOf(currentValue == null ? true : !currentValue); // #147 toggle existing boolean value
1660                }
1661            }
1662            if (noMoreValues && value == null) {
1663                return 0;
1664            }
1665            if (initialized != null) {
1666                if (initialized.contains(field) && !isOverwrittenOptionsAllowed()) {
1667                    throw new OverwrittenOptionException(optionDescription("", field, 0) +  " should be specified only once");
1668                }
1669                initialized.add(field);
1670            }
1671            ITypeConverter<?> converter = getTypeConverter(cls);
1672            Object objValue = tryConvert(field, -1, converter, value, cls);
1673            field.set(command, objValue);
1674            return result;
1675        }

1677        private int applyValuesToArrayField(Field field,
1678                                            Class<?> annotation,
1679                                            Range arity,
1680                                            Stack<String> args,
1681                                            Class<?> cls) throws Exception {
1682            Class<?> type = cls.getComponentType();
1683            ITypeConverter<?> converter = getTypeConverter(type);
1684            List<Object> converted = consumeArguments(field, annotation, arity, args, converter, cls);
1685            Object existing = field.get(command);
1686            int length = existing == null ? 0 : Array.getLength(existing);
1687            List<Object> newValues = new ArrayList<Object>();
1688            for (int i = 0; i < length; i++) {
1689                newValues.add(Array.get(existing, i));
1690            }
1691            for (Object obj : converted) {
1692                if (obj instanceof Collection<?>) {
1693                    newValues.addAll((Collection<?>) obj);
1694                } else {
1695                    newValues.add(obj);
1696                }
1697            }
1698            Object array = Array.newInstance(type, newValues.size());
1699            field.set(command, array);
1700            for (int i = 0; i < newValues.size(); i++) {
1701                Array.set(array, i, newValues.get(i));
1702            }
1703            return converted.size(); // return how many args were consumed
1704        }

1708                                                 Class<?> annotation,
1709                                                 Range arity,
1710                                                 Stack<String> args,

1713            Class<?> type = getTypeAttribute(field);

1726            }

1728        }

1730        private List<Object> consumeArguments(Field field,
1731                                              Class<?> annotation,
1732                                              Range arity,
1733                                              Stack<String> args,
1734                                              ITypeConverter<?> converter,
1735                                              Class<?> type) throws Exception {
1736            List<Object> result = new ArrayList<Object>();
1737            int index = 0;
1738
1739            // first do the arity.min mandatory parameters
1740            for (int i = 0; result.size() < arity.min; i++) {
1741                index = consumeOneArgument(field, arity, args, converter, type, result, index);
1742            }
1743            // now process the varargs if any
1744            while (result.size() < arity.max && !args.isEmpty()) {
1745                if (annotation != Parameters.class) {
1746                    if (commands.containsKey(args.peek()) || isOption(args.peek())) {
1747                        return result;
1748                    }
1749                }
1750                index = consumeOneArgument(field, arity, args, converter, type, result, index);
1751            }
1752            return result;
1753        }

1755        private int consumeOneArgument(Field field,
1756                                       Range arity,
1757                                       Stack<String> args,
1758                                       ITypeConverter<?> converter,
1759                                       Class<?> type,
1760                                       List<Object> result, int index) throws Exception {
1761            String[] values = split(trim(args.pop()), field);
1762
1763            // ensure we don't process more than arity.max (as result of splitting args)
1764            int max = Math.min(arity.max - result.size(), values.length);
1765            for (int j = 0; j < max; j++) {
1766                result.add(tryConvert(field, index, converter, values[j], type));
1767            }
1768            // if this option cannot consume values because of its arity.max,
1769            // then push them back on the stack (they are likely processed as positional parameters)
1770            for (int j = values.length - 1; j >= max; j--) {
1771                args.push(values[j]);
1772            }
1773            index++;
1774            return index;
1775        }

1778            if (field.isAnnotationPresent(Option.class)) {
1779                String regex = field.getAnnotation(Option.class).split();
1780                return regex.length() == 0 ? new String[] {value} : value.split(regex);

1783                String regex = field.getAnnotation(Parameters.class).split();
1784                return regex.length() == 0 ? new String[] {value} : value.split(regex);

1786            return new String[] {value};

1790         * Called when parsing varargs parameters for a multi-value option.
1791         * When an option is encountered, the remainder should not be interpreted as vararg elements.
1792         * @param arg the string to determine whether it is an option or not
1793         * @return true if it is an option, false otherwise
1794         */
1795        private boolean isOption(String arg) {
1796            if ("--".equals(arg)) {
1797                return true;
1798            }
1799            // not just arg prefix: we may be in the middle of parsing -xrvfFILE
1800            if (optionName2Field.containsKey(arg)) { // -v or -f or --file (not attached to param or other option)
1801                return true;
1802            }
1803            int separatorIndex = arg.indexOf(separator);
1804            if (separatorIndex > 0) { // -f=FILE or --file==FILE (attached to param via separator)
1805                if (optionName2Field.containsKey(arg.substring(0, separatorIndex))) {
1806                    return true;
1807                }
1808            }
1809            return (arg.length() > 2 && arg.startsWith("-") && singleCharOption2Field.containsKey(arg.charAt(1)));
1810        }

1896        private void assertNoMissingParameters(Field field, int arity, Stack<String> args) {
1897            if (arity > args.size()) {
1898                int actualSize = 0;
1899                @SuppressWarnings("unchecked")
1900                Stack<String> copy = (Stack<String>) args.clone();
1901                while (!copy.isEmpty()) {
1902                    actualSize += split(copy.pop(), field).length;
1903                    if (actualSize >= arity) { return; }
1904                }
1905                if (arity == 1) {
1906                    if (field.isAnnotationPresent(Option.class)) {
1907                        throw new MissingParameterException("Missing required parameter for " +
1908                                optionDescription("", field, 0));
1909                    }
1910                    Range indexRange = Range.parameterIndex(field);
1911                    Help.IParamLabelRenderer labelRenderer = Help.createMinimalParamLabelRenderer();
1912                    String sep = "";
1913                    String names = "";
1914                    int count = 0;
1915                    for (int i = indexRange.min; i < positionalParametersFields.size(); i++) {
1916                        if (Range.parameterArity(positionalParametersFields.get(i)).min > 0) {
1917                            names += sep + labelRenderer.renderParameterLabel(positionalParametersFields.get(i),
1918                                    Help.Ansi.OFF, Collections.<IStyle>emptyList());
1919                            sep = ", ";
1920                            count++;
1921                        }
1922                    }
1923                    String msg = "Missing required parameter";
1924                    Range paramArity = Range.parameterArity(field);
1925                    if (paramArity.isVariable) {
1926                        msg += "s at positions " + indexRange + ": ";
1927                    } else {
1928                        msg += (count > 1 ? "s: " : ": ");
1929                    }
1930                    throw new MissingParameterException(msg + names);
1931                }
1932                throw new MissingParameterException(optionDescription("", field, 0) +
1933                        " requires at least " + arity + " values, but only " + args.size() + " were specified.");
1934            }

2075     * the usage help message to display to end users when help is requested or invalid input values were specified.
2076     * <h3>Layered API</h3>
2077     * <p>The {@link Command} annotation provides the easiest way to customize usage help messages. See
2078     * the <a href="https://remkop.github.io/picocli/index.html#_usage_help">Manual</a> for details.</p>
2079     * <p>This Help class provides high-level functions to create sections of the usage help message and headings
2080     * for these sections. Instead of calling the {@link CommandLine#usage(PrintStream, CommandLine.Help.ColorScheme)}
2081     * method, application authors may want to create a custom usage help message by reorganizing sections in a
2082     * different order and/or adding custom sections.</p>
2083     * <p>Finally, the Help class contains inner classes and interfaces that can be used to create custom help messages.</p>
2084     * <h4>IOptionRenderer and IParameterRenderer</h4>
2085     * <p>Renders a field annotated with {@link Option} or {@link Parameters} to an array of {@link Text} values.
2086     * By default, these values are</p><ul>
2087     * <li>mandatory marker character (if the option/parameter is {@link Option#required() required})</li>
2088     * <li>short option name (empty for parameters)</li>
2089     * <li>comma or empty (empty for parameters)</li>
2090     * <li>long option names (the parameter {@link IParamLabelRenderer label} for parameters)</li>
2091     * <li>description</li>
2092     * </ul>

3603             * Internally keeps both an enriched and a plain text representation to allow layout components to calculate
3604             * text width while remaining unaware of the embedded ANSI escape codes.</p> */
3605            public class Text implements Cloneable {
3606                private final int maxLength;
3607                private int from;
3608                private int length;
3609                private StringBuilder plain = new StringBuilder();
3610                private SortedMap<Integer, String> indexToStyle = new TreeMap<Integer, String>();

3616                /**
3617                 * Constructs a Text with the specified String, which may contain markup like
3618                 * {@code @|bg(red),white,underline some text|@}.
3619                 * @param input the string with markup to parse
3620                 */
3621                public Text(String input) {
3622                    maxLength = -1;
3623                    plain.setLength(0);
3624                    int i = 0;
3625
3626                    while (true) {
3627                        int j = input.indexOf("@|", i);
3628                        if (j == -1) {
3629                            if (i == 0) {
3630                                plain.append(input);
3631                                length = plain.length();
3632                                return;
3633                            }
3634                            plain.append(input.substring(i, input.length()));
3635                            length = plain.length();
3636                            return;
3637                        }
3638                        plain.append(input.substring(i, j));
3639                        int k = input.indexOf("|@", j);
3640                        if (k == -1) {
3641                            plain.append(input);
3642                            length = plain.length();
3643                            return;
3644                        }
3645
3646                        j += 2;
3647                        String spec = input.substring(j, k);
3648                        String[] items = spec.split(" ", 2);
3649                        if (items.length == 1) {
3650                            plain.append(input);
3651                            length = plain.length();
3652                            return;
3653                        }
3654
3655                        IStyle[] styles = Style.parse(items[0]);
3656                        putStyle(plain.length(), Style.on(styles));
3657                        plain.append(items[1]);
3658                        reverse(styles);
3659                        putStyle(plain.length(), Style.off(styles));
3660                        putStyle(plain.length(), Style.reset.off());
3661                        i = k + 2;
3662                    }

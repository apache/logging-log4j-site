019    import java.beans.PropertyChangeEvent;
020    import java.beans.PropertyChangeListener;
021    import java.lang.management.ManagementFactory;
022    import java.util.List;
023    import java.util.Map;
024    import java.util.Set;
025    import java.util.concurrent.Executor;
026    import java.util.concurrent.Executors;
027    
028    import javax.management.InstanceAlreadyExistsException;
029    import javax.management.JMException;
030    import javax.management.MBeanRegistrationException;
031    import javax.management.MBeanServer;
032    import javax.management.NotCompliantMBeanException;
033    import javax.management.ObjectName;
034    
035    import org.apache.logging.log4j.core.Appender;
036    import org.apache.logging.log4j.core.LoggerContext;
037    import org.apache.logging.log4j.core.config.LoggerConfig;
038    import org.apache.logging.log4j.core.selector.ContextSelector;
039    import org.apache.logging.log4j.status.StatusLogger;

041    /**
042     * Creates MBeans to instrument various classes in the log4j class hierarchy.
043     * <p>
044     * All instrumentation for Log4j 2 classes can be disabled by setting system
045     * property {@code -Dlog4j2.disable.jmx=true}.
046     */
047    public final class Server {
048    
049        private static final String PROPERTY_DISABLE_JMX = "log4j2.disable.jmx";
050    
051        private Server() {
052        }
053    
054        /**
055         * Either returns the specified name as is, or returns a quoted value
056         * containing the specified name with the special characters (comma, equals,
057         * colon, quote, asterisk, or question mark) preceded with a backslash.
058         *
059         * @param name
060         *            the name to escape so it can be used as a value in an
061         *            {@link ObjectName}.
062         * @return the escaped name
063         */
064        public static String escape(final String name) {
065            final StringBuilder sb = new StringBuilder(name.length() * 2);
066            boolean needsQuotes = false;
067            for (int i = 0; i < name.length(); i++) {
068                final char c = name.charAt(i);
069                switch (c) {
070                case ',':
071                case '=':
072                case ':':
073                case '\\':
074                case '*':
075                case '?':
076                    sb.append('\\');
077                    needsQuotes = true;
078                }
079                sb.append(c);
080            }
081            if (needsQuotes) {
082                sb.insert(0, '\"');
083                sb.append('\"');
084            }
085            return sb.toString();
086        }
087    
088        /**
089         * Creates MBeans to instrument the specified selector and other classes in
090         * the log4j class hierarchy and registers the MBeans in the platform MBean
091         * server so they can be accessed by remote clients.
092         *
093         * @param selector
094         *            starting point in the log4j class hierarchy
095         * @throws JMException
096         *             if a problem occurs during registration
097         */
098        public static void registerMBeans(final ContextSelector selector)
099                throws JMException {
100    
101            // avoid creating Platform MBean Server if JMX disabled
102            if (Boolean.getBoolean(PROPERTY_DISABLE_JMX)) {
103                StatusLogger.getLogger().debug(
104                        "JMX disabled for log4j2. Not registering MBeans.");
105                return;
106            }
107            final MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
108            registerMBeans(selector, mbs);
109        }
110    
111        /**
112         * Creates MBeans to instrument the specified selector and other classes in
113         * the log4j class hierarchy and registers the MBeans in the specified MBean
114         * server so they can be accessed by remote clients.
115         *
116         * @param selector
117         *            starting point in the log4j class hierarchy
118         * @param mbs
119         *            the MBean Server to register the instrumented objects in
120         * @throws JMException
121         *             if a problem occurs during registration
122         */
123        public static void registerMBeans(final ContextSelector selector,
124                final MBeanServer mbs) throws JMException {
125    
126            if (Boolean.getBoolean(PROPERTY_DISABLE_JMX)) {
127                StatusLogger.getLogger().debug(
128                        "JMX disabled for log4j2. Not registering MBeans.");
129                return;
130            }
131            final Executor executor = Executors.newFixedThreadPool(1);
132            registerStatusLogger(mbs, executor);
133            registerContextSelector(selector, mbs, executor);
134    
135            final List<LoggerContext> contexts = selector.getLoggerContexts();
136            registerContexts(contexts, mbs, executor);
137    
138            for (final LoggerContext context : contexts) {
139                context.addPropertyChangeListener(new PropertyChangeListener() {
140    
141                    @Override
142                    public void propertyChange(final PropertyChangeEvent evt) {
143                        if (!LoggerContext.PROPERTY_CONFIG.equals(evt
144                                .getPropertyName())) {
145                            return;
146                        }
147                        // first unregister the MBeans that instrument the
148                        // previous instrumented LoggerConfigs and Appenders
149                        unregisterLoggerConfigs(context, mbs);
150                        unregisterAppenders(context, mbs);
151    
152                        // now provide instrumentation for the newly configured
153                        // LoggerConfigs and Appenders
154                        try {
155                            registerLoggerConfigs(context, mbs, executor);
156                            registerAppenders(context, mbs, executor);
157                        } catch (final Exception ex) {
158                            StatusLogger.getLogger().error(
159                                    "Could not register mbeans", ex);
160                        }
161                    }
162                });
163            }
164        }
165    
166        private static void registerStatusLogger(final MBeanServer mbs, final Executor executor)
167                throws InstanceAlreadyExistsException, MBeanRegistrationException, NotCompliantMBeanException {
168    
169            final StatusLoggerAdmin mbean = new StatusLoggerAdmin(executor);
170            mbs.registerMBean(mbean, mbean.getObjectName());
171        }
172    
173        private static void registerContextSelector(final ContextSelector selector, final MBeanServer mbs,
174                final Executor executor) throws InstanceAlreadyExistsException, MBeanRegistrationException,
175                NotCompliantMBeanException {
176    
177            final ContextSelectorAdmin mbean = new ContextSelectorAdmin(selector);
178            mbs.registerMBean(mbean, mbean.getObjectName());
179        }
180    
181        private static void registerContexts(final List<LoggerContext> contexts, final MBeanServer mbs,
182                final Executor executor) throws InstanceAlreadyExistsException, MBeanRegistrationException,
183                NotCompliantMBeanException {
184    
185            for (final LoggerContext ctx : contexts) {
186                final LoggerContextAdmin mbean = new LoggerContextAdmin(ctx, executor);
187                mbs.registerMBean(mbean, mbean.getObjectName());
188            }
189        }
190    
191        private static void unregisterLoggerConfigs(final LoggerContext context,
192                final MBeanServer mbs) {
193            final String pattern = LoggerConfigAdminMBean.PATTERN;
194            final String search = String.format(pattern, context.getName(), "*");
195            unregisterAllMatching(search, mbs);
196        }
197    
198        private static void unregisterAppenders(final LoggerContext context,
199                final MBeanServer mbs) {
200            final String pattern = AppenderAdminMBean.PATTERN;
201            final String search = String.format(pattern, context.getName(), "*");
202            unregisterAllMatching(search, mbs);
203        }
204    
205        private static void unregisterAllMatching(final String search, final MBeanServer mbs) {
206            try {
207                final ObjectName pattern = new ObjectName(search);
208                final Set<ObjectName> found = mbs.queryNames(pattern, null);
209                for (final ObjectName objectName : found) {
210                    mbs.unregisterMBean(objectName);
211                }
212            } catch (final Exception ex) {
213                StatusLogger.getLogger()
214                        .error("Could not unregister " + search, ex);
215            }
216        }
217    
218        private static void registerLoggerConfigs(final LoggerContext ctx, final MBeanServer mbs, final Executor executor)
219                throws InstanceAlreadyExistsException, MBeanRegistrationException, NotCompliantMBeanException {
220    
221            final Map<String, LoggerConfig> map = ctx.getConfiguration().getLoggers();
222            for (final String name : map.keySet()) {
223                final LoggerConfig cfg = map.get(name);
224                final LoggerConfigAdmin mbean = new LoggerConfigAdmin(ctx.getName(), cfg);
225                mbs.registerMBean(mbean, mbean.getObjectName());
226            }
227        }
228    
229        private static void registerAppenders(final LoggerContext ctx, final MBeanServer mbs, final Executor executor)
230                throws InstanceAlreadyExistsException, MBeanRegistrationException, NotCompliantMBeanException {
231    
232            final Map<String, Appender> map = ctx.getConfiguration().getAppenders();
233            for (final String name : map.keySet()) {
234                final Appender appender = map.get(name);
235                final AppenderAdmin mbean = new AppenderAdmin(ctx.getName(), appender);
236                mbs.registerMBean(mbean, mbean.getObjectName());
237            }
238        }

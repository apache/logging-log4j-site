041    /**
042     * Creates a Node hierarchy from a JSON file.
043     */
044    public class JSONConfiguration extends BaseConfiguration {
045    
046        private static final String[] VERBOSE_CLASSES = new String[] {ResolverUtil.class.getName()};
047    
048        private static final int BUF_SIZE = 16384;
049    
050        private List<Status> status = new ArrayList<Status>();
051    
052        private JsonNode root;
053    
054        private List<String> messages = new ArrayList<String>();
055    
056        public JSONConfiguration(InputSource source, File configFile) {
057            byte[] buffer;
058    
059            try {
060                buffer = toByteArray(source.getByteStream());
061                InputStream is = new ByteArrayInputStream(buffer);
062                source = new InputSource(is);
063                ObjectMapper mapper = new ObjectMapper().configure(JsonParser.Feature.ALLOW_COMMENTS, true);
064                root = mapper.readTree(is);
065                if (root.size() == 1) {
066                    Iterator<JsonNode> i = root.getElements();
067                    root = i.next();
068                }
069                processAttributes(rootNode, root);
070                Level status = Level.OFF;
071                boolean verbose = false;
072                for (Map.Entry<String, String> entry : rootNode.getAttributes().entrySet()) {
073                    if ("status".equalsIgnoreCase(entry.getKey())) {
074                        status = Level.toLevel(getSubst().replace(entry.getValue()).toUpperCase(), null);
075                        if (status == null) {
076                            status = Level.ERROR;
077                            messages.add("Invalid status specified: " + entry.getValue() + ". Defaulting to ERROR");
078                        }
079                    } else if ("verbose".equalsIgnoreCase(entry.getKey())) {
080                        verbose = Boolean.parseBoolean(getSubst().replace(entry.getValue()));
081                    } else if ("packages".equalsIgnoreCase(entry.getKey())) {
082                        String[] packages = getSubst().replace(entry.getValue()).split(",");
083                        for (String p : packages) {
084                            PluginManager.addPackage(p);
085                        }
086                    } else if ("name".equalsIgnoreCase(entry.getKey())) {
087                        setName(getSubst().replace(entry.getValue()));
088                    } else if ("monitorInterval".equalsIgnoreCase(entry.getKey())) {
089                        int interval = Integer.parseInt(getSubst().replace(entry.getValue()));
090                        if (interval > 0 && configFile != null) {
091                            monitor = new FileConfigurationMonitor(configFile, listeners, interval);
092                        }
093                    }
094                }
095    
096                Iterator<StatusListener> statusIter = ((StatusLogger) LOGGER).getListeners();
097                boolean found = false;
098                while (statusIter.hasNext()) {
099                    StatusListener listener = statusIter.next();
100                    if (listener instanceof StatusConsoleListener) {
101                        found = true;
102                        ((StatusConsoleListener) listener).setLevel(status);
103                        if (!verbose) {
104                            ((StatusConsoleListener) listener).setFilters(VERBOSE_CLASSES);
105                        }
106                    }
107                }
108                if (!found && status != Level.OFF) {
109                    StatusConsoleListener listener = new StatusConsoleListener(status);
110                    if (!verbose) {
111                        listener.setFilters(VERBOSE_CLASSES);
112                    }
113                    ((StatusLogger) LOGGER).registerListener(listener);
114                    for (String msg : messages) {
115                        LOGGER.error(msg);
116                    }
117                }
118                if (getName() == null) {
119                    setName(source.getSystemId());
120                }
121            } catch (Exception ex) {
122                LOGGER.error("Error parsing " + source.getSystemId(), ex);
123                ex.printStackTrace();
124            }
125        }
126    
127         public void setup() {
128            Iterator<Map.Entry<String, JsonNode>> iter = root.getFields();
129            List<Node> children = rootNode.getChildren();
130            while (iter.hasNext()) {
131                Map.Entry<String, JsonNode> entry = iter.next();
132                JsonNode n = entry.getValue();
133                if (n.isObject()) {
134                    LOGGER.debug("Processing node for object " + entry.getKey());
135                    children.add(constructNode(entry.getKey(), rootNode, n));
136                } else if (n.isArray()) {
137                    LOGGER.error("Arrays are not supported at the root configuration.");
138                }
139            }
140            LOGGER.debug("Completed parsing configuration");
141            if (status.size() > 0) {
142                for (Status s : status) {
143                    LOGGER.error("Error processing element " + s.name + ": " + s.errorType);
144                }
145                return;
146            }
147        }
148    
149        private Node constructNode(String name, Node parent, JsonNode jsonNode) {
150            PluginType type = getPluginManager().getPluginType(name);
151            Node node = new Node(parent, name, type);
152            processAttributes(node, jsonNode);
153            Iterator<Map.Entry<String, JsonNode>> iter = jsonNode.getFields();
154            List<Node> children = node.getChildren();
155            while (iter.hasNext()) {
156                Map.Entry<String, JsonNode> entry = iter.next();
157                JsonNode n = entry.getValue();
158                if (n.isArray() || n.isObject()) {
159                    if (type == null) {
160                        status.add(new Status(name, n, ErrorType.CLASS_NOT_FOUND));
161                    }
162                    if (n.isArray()) {
163                        LOGGER.debug("Processing node for array " + entry.getKey());
164                        for (int i = 0; i < n.size(); ++i) {
165                            String pluginType = getType(n.get(i), entry.getKey());
166                            PluginType entryType = getPluginManager().getPluginType(pluginType);
167                            Node item = new Node(node, entry.getKey(), entryType);
168                            processAttributes(item, n.get(i));
169                            if (pluginType.equals(entry.getKey())) {
170                                LOGGER.debug("Processing " + entry.getKey() + "[" + i + "]");
171                            } else {
172                                LOGGER.debug("Processing " + pluginType + " " + entry.getKey() + "[" + i + "]");
173                            }
174                            Iterator<Map.Entry<String, JsonNode>> itemIter = n.get(i).getFields();
175                            List<Node> itemChildren = item.getChildren();
176                            while (itemIter.hasNext()) {
177                                Map.Entry<String, JsonNode> itemEntry = itemIter.next();
178                                if (itemEntry.getValue().isObject()) {
179                                    LOGGER.debug("Processing node for object " + itemEntry.getKey());
180                                    itemChildren.add(constructNode(itemEntry.getKey(), item, itemEntry.getValue()));
181                                }
182                            }
183                            children.add(item);
184                        }
185                    } else {
186                        LOGGER.debug("Processing node for object " + entry.getKey());
187                        children.add(constructNode(entry.getKey(), node, n));
188                    }
189                }
190            }
191    
192            String t;
193            if (type == null) {
194                t = "null";
195            } else {
196                t = type.getElementName() + ":" + type.getPluginClass();
197            }
198    
199            String p = node.getParent() == null ? "null" : node.getParent().getName() == null ?
200                "root" : node.getParent().getName();
201            LOGGER.debug("Returning " + node.getName() + " with parent " + p + " of type " +  t);
202            return node;
203        }
204    
205        private String getType(JsonNode node, String name) {
206            Iterator<Map.Entry<String, JsonNode>> iter = node.getFields();
207            while (iter.hasNext()) {
208                Map.Entry<String, JsonNode> entry = iter.next();
209                if (entry.getKey().equalsIgnoreCase("type")) {
210                    JsonNode n = entry.getValue();
211                    if (n.isValueNode()) {
212                        return n.asText();
213                    }
214                }
215            }
216            return name;
217        }
218    
219        private void processAttributes(Node parent, JsonNode node) {
220            Map<String, String> attrs = parent.getAttributes();
221            Iterator<Map.Entry<String, JsonNode>> iter = node.getFields();
222            while (iter.hasNext()) {
223                Map.Entry<String, JsonNode> entry = iter.next();
224                if (!entry.getKey().equalsIgnoreCase("type")) {
225                    JsonNode n = entry.getValue();
226                    if (n.isValueNode()) {
227                        attrs.put(entry.getKey(), n.asText());
228                    }
229                }
230            }
231        }
232    
233        protected byte[] toByteArray(InputStream is) throws IOException {
234            ByteArrayOutputStream buffer = new ByteArrayOutputStream();
235    
236            int nRead;
237            byte[] data = new byte[BUF_SIZE];
238    
239            while ((nRead = is.read(data, 0, data.length)) != -1) {
240                buffer.write(data, 0, nRead);
241            }
242    
243            return buffer.toByteArray();
244        }
245    
246        /**
247         * The error that occurred.
248         */
249        private enum ErrorType {
250            CLASS_NOT_FOUND
251        }
252    
253        /**
254         * Status for recording errors.
255         */
256        private class Status {
257            private JsonNode node;
258            private String name;
259            private ErrorType errorType;
260    
261            public Status(String name, JsonNode node, ErrorType errorType) {
262                this.name = name;
263                this.node = node;
264                this.errorType = errorType;
265            }
266        }

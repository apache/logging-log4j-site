017    package org.apache.logging.log4j.core.net;
018    
019    /**
020     *  The facility codes used by the Syslog system.
021     *
022     *        Numerical          Facility<br>
023     *           Code<br>
024     *
025     *             0             kernel messages<br>
026     *             1             user-level messages<br>
027     *             2             mail system<br>
028     *             3             system daemons<br>
029     *             4             security/authorization messages<br>
030     *             5             messages generated internally by syslogd<br>
031     *             6             line printer subsystem<br>
032     *             7             network news subsystem<br>
033     *             8             UUCP subsystem<br>
034     *             9             clock daemon<br>
035     *            10             security/authorization messages<br>
036     *            11             FTP daemon<br>
037     *            12             NTP subsystem<br>
038     *            13             log audit<br>
039     *            14             log alert<br>
040     *            15             clock daemon (note 2)<br>
041     *            16             local use 0  (local0)<br>
042     *            17             local use 1  (local1)<br>
043     *            18             local use 2  (local2)<br>
044     *            19             local use 3  (local3)<br>
045     *            20             local use 4  (local4)<br>
046     *            21             local use 5  (local5)<br>
047     *            22             local use 6  (local6)<br>
048     *            23             local use 7  (local7)<br>
049     */
050    public enum Facility {
051        /** Kernel messages. */
052        KERN(0),
053        /** User level messages. */
054        USER(1),
055        /** Mail system. */
056        MAIL(2),
057        /** System daemons. */
058        DAEMON(3),
059        /** Security/Authorization messages. */
060        AUTH(4),
061        /** Messages generated by syslogd. */
062        SYSLOG(5),
063        /** Line printer subsystem. */
064        LPR(6),
065        /** Network news subsystem. */
066        NEWS(7),
067        /** UUCP subsystem. */
068        UUCP(8),
069        /** Clock daemon. */
070        CRON(9),
071        /** Security/Authorization messages. */
072        AUTHPRIV(10),
073        /** FTP daemon. */
074        FTP(11),
075        /** NTP subsystem. */
076        NTP(12),
077        /** Log audit. */
078        LOG_AUDIT(13),
079        /** Log alert. */
080        LOG_ALERT(14),
081        /** Clock daemon. */
082        CLOCK(15),
083        /** Local use 0. */
084        LOCAL0(16),
085        /** Local use 1. */
086        LOCAL1(17),
087        /** Local use 2. */
088        LOCAL2(18),
089        /** Local use 3. */
090        LOCAL3(19),
091        /** Local use 4. */
092        LOCAL4(20),
093        /** Local use 5. */
094        LOCAL5(21),
095        /** Local use 6. */
096        LOCAL6(22),
097        /** Local use 7. */
098        LOCAL7(23);
099    
100        private final int code;
101    
102        private Facility(int code) {
103            this.code = code;
104        }
105    
106        /**
107         * Retrieve the value of the enumeration.
108         * @return The value associated with the enumeration.
109         */
110        public int getCode() {
111            return this.code;
112        }
113    
114        /**
115         * Determine if this enumeration matches the specified name (ignoring case).
116         * @param name The name to check.
117         * @return true if the name matches this enumeration, ignoring case.
118         */
119        public boolean isEqual(String name) {
120            return this.name().equalsIgnoreCase(name);
121        }
122    
123    }

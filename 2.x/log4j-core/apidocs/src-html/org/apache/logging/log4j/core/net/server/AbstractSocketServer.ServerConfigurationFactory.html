046/**
047 * Abstract socket server for TCP and UDP implementations.
048 *
049 * @param <T>
050 *            The kind of input stream read
051 *
052 *            TODO Make a LifeCycle
053 */
054public abstract class AbstractSocketServer<T extends InputStream> extends LogEventListener implements Runnable {
055
056    protected static class CommandLineArguments extends BasicCommandLineArguments {
057
058        @Parameter(names = { "--config", "-c" }, description = "Log4j configuration file location (path or URL).")
059        private String configLocation;
060
061        @Parameter(names = { "--interactive",
062                "-i" }, description = "Accepts commands on standard input (\"exit\" is the only command).")
063        private boolean interactive;
064
065        @Parameter(names = { "--port",
066                "-p" }, validateWith = PositiveInteger.class, description = "Server socket port.")
067        private int port;
068
069        @Parameter(names = { "--localbindaddress",
070                "-a" }, converter = InetAddressConverter.class, description = "Server socket local bind address.")
071        private InetAddress localBindAddress;
072
073        String getConfigLocation() {
074            return configLocation;
075        }
076
077        int getPort() {
078            return port;
079        }
080
081        protected boolean isInteractive() {
082            return interactive;
083        }
084
085        void setConfigLocation(final String configLocation) {
086            this.configLocation = configLocation;
087        }
088
089        void setInteractive(final boolean interactive) {
090            this.interactive = interactive;
091        }
092
093        void setPort(final int port) {
094            this.port = port;
095        }
096
097        InetAddress getLocalBindAddress() {
098            return localBindAddress;
099        }
100
101        void setLocalBindAddress(final InetAddress localBindAddress) {
102            this.localBindAddress = localBindAddress;
103        }
104    }
105
106    /**
107     * Factory that creates a Configuration for the server.
108     */
109    protected static class ServerConfigurationFactory extends XmlConfigurationFactory {
110
111        private final String path;
112
113        public ServerConfigurationFactory(final String path) {
114            this.path = path;
115        }
116
117        @Override
118        public Configuration getConfiguration(final LoggerContext loggerContext, final String name,
119                final URI configLocation) {
120            if (Strings.isNotEmpty(path)) {
121                File file = null;
122                ConfigurationSource source = null;
123                try {
124                    file = new File(path);
125                    final FileInputStream is = new FileInputStream(file);
126                    source = new ConfigurationSource(is, file);
127                } catch (final FileNotFoundException ignored) {
128                    // Ignore this error
129                }
130                if (source == null) {
131                    try {
132                        final URL url = new URL(path);
133                        source = new ConfigurationSource(url.openStream(), url);
134                    } catch (final IOException ignored) {
135                        // Ignore this error
136                    }
137                }
138
139                try {
140                    if (source != null) {
141                        return new XmlConfiguration(loggerContext, source);
142                    }
143                } catch (final Exception ignored) {
144                    // Ignore this error.
145                }
146                System.err.println("Unable to process configuration at " + path + ", using default.");
147            }
148            return super.getConfiguration(loggerContext, name, configLocation);
149        }
150    }
151
152    protected static final int MAX_PORT = 65534;
153
154    private volatile boolean active = true;
155
156    protected final LogEventBridge<T> logEventInput;
157
158    protected final Logger logger;
159
160    /**
161     * Creates a new socket server.
162     *
163     * @param port
164     *            listen to this port
165     * @param logEventInput
166     *            Use this input to read log events.
167     */
168    public AbstractSocketServer(final int port, final LogEventBridge<T> logEventInput) {
169        this.logger = LogManager.getLogger(this.getClass().getName() + '.' + port);
170        this.logEventInput = Objects.requireNonNull(logEventInput, "LogEventInput");
171    }
172
173    protected boolean isActive() {
174        return this.active;
175    }
176
177    protected void setActive(final boolean isActive) {
178        this.active = isActive;
179    }
180
181    /**
182     * Start this server in a new thread.
183     *
184     * @return the new thread that running this server.
185     */
186    public Thread startNewThread() {
187        final Thread thread = new Log4jThread(this);
188        thread.start();
189        return thread;
190    }
191
192    public abstract void shutdown() throws Exception;
193
194    public void awaitTermination(final Thread serverThread) throws Exception {
195        final BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
196        while (true) {
197            final String line = reader.readLine();
198            if (line == null
199                || line.equalsIgnoreCase("quit")
200                || line.equalsIgnoreCase("stop")
201                || line.equalsIgnoreCase("exit")) {
202                this.shutdown();
203                serverThread.join();
204                break;
205            }
206        }
207    }

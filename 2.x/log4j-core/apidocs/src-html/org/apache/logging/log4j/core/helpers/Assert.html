017    package org.apache.logging.log4j.core.helpers;
018    
019    /**
020     * Utility class providing common validation logic.
021     */
022    public final class Assert {
023        private Assert() {
024        }
025    
026        /**
027         * Throws a {@code NullPointerException} if the specified parameter is
028         * {@code null}, otherwise returns the specified parameter.
029         * <p>
030         * Usage:
031         * 
032         * <pre>
033         * // earlier you would write this:
034         * public SomeConstructor(Object param) {
035         *     if (param == null) {
036         *         throw new NullPointerException(name + &quot; is null&quot;);
037         *     }
038         *     this.field = param;
039         * }
040         * 
041         * // now you can do the same in one line:
042         * public SomeConstructor(Object param) {
043         *     this.field = Assert.isNotNull(param);
044         * }
045         * </pre>
046         * 
047         * @param <T> the type of the parameter to check and return
048         * @param checkMe the parameter to check
049         * @param name name of the parameter to use in the error message if
050         *            {@code null}
051         * @return the specified parameter
052         */
053        public static <T> T isNotNull(final T checkMe, final String name) {
054            if (checkMe == null) {
055                throw new NullPointerException(name + " is null");
056            }
057            return checkMe;
058        }
059    }

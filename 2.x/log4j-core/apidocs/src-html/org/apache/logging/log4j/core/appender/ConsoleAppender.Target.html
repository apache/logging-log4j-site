038    /**
039     * ConsoleAppender appends log events to <code>System.out</code> or
040     * <code>System.err</code> using a layout specified by the user. The
041     * default target is <code>System.out</code>.
042     * @doubt accessing System.out or .err as a byte stream instead of a writer
043     *    bypasses the JVM's knowledge of the proper encoding. (RG) Encoding
044     * is handled within the Layout. Typically, a Layout will generate a String
045     * and then call getBytes which may use a configured encoding or the system
046     * default. OTOH, a Writer cannot print byte streams.
047     *
048     * @param <T> The {@link Layout}'s {@link Serializable} type.
049     */
050    @Plugin(name = "Console", category = "Core", elementType = "appender", printObject = true)
051    public final class ConsoleAppender<T extends Serializable> extends AbstractOutputStreamAppender<T> {
052    
053        private static ConsoleManagerFactory factory = new ConsoleManagerFactory();
054    
055        /**
056         * Enumeration of console destinations.
057         */
058        public enum Target {
059            /** Standard output. */
060            SYSTEM_OUT,
061            /** Standard error output. */
062            SYSTEM_ERR
063        }
064    
065        private ConsoleAppender(final String name, final Layout<T> layout, final Filter filter,
066                                final OutputStreamManager manager,
067                                final boolean handleExceptions) {
068            super(name, layout, filter, handleExceptions, true, manager);
069        }
070    
071        /**
072         * Create a Console Appender.
073         * @param layout The layout to use (required).
074         * @param filter The Filter or null.
075         * @param t The target ("SYSTEM_OUT" or "SYSTEM_ERR"). The default is "SYSTEM_OUT".
076         * @param follow If true will follow changes to the underlying output stream.
077         * @param name The name of the Appender (required).
078         * @param suppress "true" if exceptions should be hidden from the application, "false" otherwise.
079         *                 The default is "true".
080         * @param <S> The {@link Layout}'s {@link Serializable} type.
081         * @return The ConsoleAppender.
082         */
083        @PluginFactory
084        public static <S extends Serializable> ConsoleAppender<S> createAppender(@PluginElement("layout") Layout<S> layout,
085                                                     @PluginElement("filters") final Filter filter,
086                                                     @PluginAttr("target") final String t,
087                                                     @PluginAttr("name") final String name,
088                                                     @PluginAttr("follow") final String follow,
089                                                     @PluginAttr("suppressExceptions") final String suppress) {
090            if (name == null) {
091                LOGGER.error("No name provided for ConsoleAppender");
092                return null;
093            }
094            if (layout == null) {
095                @SuppressWarnings({ "unchecked", "UnnecessaryLocalVariable" })
096                final
097                Layout<S> l = (Layout<S>) PatternLayout.createLayout(null, null, null, null, null);
098                layout = l;
099            }
100            final boolean isFollow = Boolean.parseBoolean(follow);
101            final boolean handleExceptions = Booleans.parseBoolean(suppress, true);
102            final Target target = t == null ? Target.SYSTEM_OUT : Target.valueOf(t);
103            return new ConsoleAppender<S>(name, layout, filter, getManager(isFollow, target, layout), handleExceptions);
104        }
105    
106        private static OutputStreamManager getManager(final boolean follow, final Target target, final Layout layout) {
107            final String type = target.name();
108            final OutputStream os = getOutputStream(follow, target);
109            return OutputStreamManager.getManager(target.name() + "." + follow, new FactoryData(os, type, layout), factory);
110        }
111    
112        private static OutputStream getOutputStream(final boolean follow, final Target target) {
113            final String enc = Charset.defaultCharset().name();
114            PrintStream printStream = null;
115            try {
116                printStream = target == Target.SYSTEM_OUT ?
117                follow ? new PrintStream(new SystemOutStream(), true, enc) : System.out :
118                follow ? new PrintStream(new SystemErrStream(), true, enc) : System.err;
119            } catch (final UnsupportedEncodingException ex) { // should never happen
120                throw new IllegalStateException("Unsupported default encoding " + enc, ex);
121            }
122            final PropertiesUtil propsUtil = PropertiesUtil.getProperties();
123            if (!propsUtil.getStringProperty("os.name").startsWith("Windows") ||
124                propsUtil.getBooleanProperty("log4j.skipJansi")) {
125                return printStream;
126            } else {
127                try {
128                    final ClassLoader loader = Loader.getClassLoader();
129                    // We type the parameter as a wildcard to avoid a hard reference to Jansi.
130                    final Class<?> clazz = loader.loadClass("org.fusesource.jansi.WindowsAnsiOutputStream");
131                    final Constructor<?> constructor = clazz.getConstructor(OutputStream.class);
132                    return (OutputStream) constructor.newInstance(printStream);
133                } catch (final ClassNotFoundException cnfe) {
134                    LOGGER.debug("Jansi is not installed");
135                } catch (final NoSuchMethodException nsme) {
136                    LOGGER.warn("WindowsAnsiOutputStream is missing the proper constructor");
137                } catch (final Exception ex) {
138                    LOGGER.warn("Unable to instantiate WindowsAnsiOutputStream");
139                }
140                return printStream;
141            }
142        }
143    
144        /**
145         * An implementation of OutputStream that redirects to the
146         * current System.err.
147         *
148         */
149        private static class SystemErrStream extends OutputStream {
150            public SystemErrStream() {
151            }
152    
153            @Override
154            public void close() {
155            }
156    
157            @Override
158            public void flush() {
159                System.err.flush();
160            }
161    
162            @Override
163            public void write(final byte[] b) throws IOException {
164                System.err.write(b);
165            }
166    
167            @Override
168            public void write(final byte[] b, final int off, final int len)
169                throws IOException {
170                System.err.write(b, off, len);
171            }
172    
173            @Override
174            public void write(final int b) {
175                System.err.write(b);
176            }
177        }
178    
179        /**
180         * An implementation of OutputStream that redirects to the
181         * current System.out.
182         *
183         */
184        private static class SystemOutStream extends OutputStream {
185            public SystemOutStream() {
186            }
187    
188            @Override
189            public void close() {
190            }
191    
192            @Override
193            public void flush() {
194                System.out.flush();
195            }
196    
197            @Override
198            public void write(final byte[] b) throws IOException {
199                System.out.write(b);
200            }
201    
202            @Override
203            public void write(final byte[] b, final int off, final int len)
204                throws IOException {
205                System.out.write(b, off, len);
206            }
207    
208            @Override
209            public void write(final int b) throws IOException {
210                System.out.write(b);
211            }
212        }
213    
214        /**
215         * Data to pass to factory method.
216         */
217        private static class FactoryData {
218            private final OutputStream os;
219            private final String type;
220            private final Layout layout;
221    
222            /**
223             * Constructor.
224             * @param os The OutputStream.
225             * @param type The name of the target.
226             */
227            public FactoryData(final OutputStream os, final String type, final Layout layout) {
228                this.os = os;
229                this.type = type;
230                this.layout = layout;
231            }
232        }
233    
234        /**
235         * Factory to create the Appender.
236         */
237        private static class ConsoleManagerFactory implements ManagerFactory<OutputStreamManager, FactoryData> {
238    
239            /**
240             * Create an OutputStreamManager.
241             * @param name The name of the entity to manage.
242             * @param data The data required to create the entity.
243             * @return The OutputStreamManager
244             */
245            @Override
246            public OutputStreamManager createManager(final String name, final FactoryData data) {
247                return new OutputStreamManager(data.os, data.type, data.layout);
248            }
249        }

017package org.apache.logging.log4j.core.net.ssl;
018
019import org.apache.logging.log4j.Logger;
020import org.apache.logging.log4j.status.StatusLogger;
021import org.apache.logging.log4j.util.PropertiesUtil;
022import org.apache.logging.log4j.util.Strings;
023
024/**
025 * Creates an SSL configuration from Log4j properties.
026 */
027public class SslConfigurationFactory {
028
029    private static final Logger LOGGER = StatusLogger.getLogger();
030    private static SslConfiguration sslConfiguration = createSslConfiguration(PropertiesUtil.getProperties());
031
032    private static final String trustStorelocation = "log4j2.trustStoreLocation";
033    private static final String trustStorePassword = "log4j2.trustStorePassword";
034    private static final String trustStorePasswordFile = "log4j2.trustStorePasswordFile";
035    private static final String trustStorePasswordEnvVar = "log4j2.trustStorePasswordEnvironmentVariable";
036    private static final String trustStoreKeyStoreType = "log4j2.trustStoreKeyStoreType";
037    private static final String trustStoreKeyManagerFactoryAlgorithm = "log4j2.trustStoreKeyManagerFactoryAlgorithm";
038    private static final String keyStoreLocation = "log4j2.keyStoreLocation";
039    private static final String keyStorePassword = "log4j2.keyStorePassword";
040    private static final String keyStorePasswordFile = "log4j2.keyStorePasswordFile";
041    private static final String keyStorePasswordEnvVar = "log4j2.keyStorePasswordEnvironmentVariable";
042    private static final String keyStoreType = "log4j2.keyStoreType";
043    private static final String keyStoreKeyManagerFactoryAlgorithm = "log4j2.keyStoreKeyManagerFactoryAlgorithm";
044    private static final String verifyHostName = "log4j2.sslVerifyHostName";
045
046    static SslConfiguration createSslConfiguration(PropertiesUtil props) {
047        KeyStoreConfiguration keyStoreConfiguration = null;
048        TrustStoreConfiguration trustStoreConfiguration = null;
049        String location = props.getStringProperty(trustStorelocation);
050        String storeType = props.getStringProperty(trustStoreKeyStoreType);
051        if (Strings.isNotEmpty(location) || storeType != null) {
052            String password = props.getStringProperty(trustStorePassword);
053            char[] passwordChars = getPassword(password, storeType);
054            try {
055                trustStoreConfiguration = TrustStoreConfiguration.createKeyStoreConfiguration(Strings.trimToNull(location), passwordChars,
056                    props.getStringProperty(trustStorePasswordEnvVar), props.getStringProperty(trustStorePasswordFile),
057                    storeType, props.getStringProperty(trustStoreKeyManagerFactoryAlgorithm));
058            } catch (Exception ex) {
059                LOGGER.warn("Unable to create trust store configuration due to: {} {}", ex.getClass().getName(),
060                    ex.getMessage());
061            }
062        }
063        location = props.getStringProperty(keyStoreLocation);
064        storeType = props.getStringProperty(keyStoreType);
065        if (Strings.isNotEmpty(location) || storeType != null) {
066            String password = props.getStringProperty(keyStorePassword);
067            char[] passwordChars = getPassword(password, storeType);
068            try {
069                keyStoreConfiguration = KeyStoreConfiguration.createKeyStoreConfiguration(Strings.trimToNull(location), passwordChars,
070                    props.getStringProperty(keyStorePasswordEnvVar), props.getStringProperty(keyStorePasswordFile),
071                    storeType, props.getStringProperty(keyStoreKeyManagerFactoryAlgorithm));
072            } catch (Exception ex) {
073                LOGGER.warn("Unable to create key store configuration due to: {} {}", ex.getClass().getName(),
074                    ex.getMessage());
075            }
076        }
077        if (trustStoreConfiguration != null || keyStoreConfiguration != null) {
078            boolean isVerifyHostName = props.getBooleanProperty(verifyHostName, false);
079            return SslConfiguration.createSSLConfiguration(null, keyStoreConfiguration,
080                trustStoreConfiguration, isVerifyHostName);
081        }
082        return null;
083    }
084
085    private static char[] getPassword(final String password, final String keyStoreType) {
086        // Note from Tomcat's SSLUtiBase#getStore:
087        //
088        // JKS key stores treat null and "" interchangeably.
089        // PKCS12 key stores don't return the cert if null is used.
090        // Key stores that do not use passwords expect null
091        // Therefore:
092        // - generally use null if pass is null or ""
093        // - for JKS or PKCS12 only use null if pass is null
094        //   (because JKS will auto-switch to PKCS12)
095        if (keyStoreType.equals(StoreConfiguration.JKS) || keyStoreType.equals(StoreConfiguration.PKCS12)) {
096            return password != null ? password.toCharArray() : null;
097        }
098        return Strings.isEmpty(password) ? null : password.toCharArray();
099    }
100
101    public static SslConfiguration getSslConfiguration() {
102        return sslConfiguration;
103    }
104}

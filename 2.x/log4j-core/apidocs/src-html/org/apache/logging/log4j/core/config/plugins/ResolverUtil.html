037    /**
038     * <p>ResolverUtil is used to locate classes that are available in the/a class path and meet
039     * arbitrary conditions. The two most common conditions are that a class implements/extends
040     * another class, or that is it annotated with a specific annotation. However, through the use
041     * of the {@link Test} class it is possible to search using arbitrary conditions.</p>
042     *
043     * <p>A ClassLoader is used to locate all locations (directories and jar files) in the class
044     * path that contain classes within certain packages, and then to load those classes and
045     * check them. By default the ClassLoader returned by
046     *  {@code Thread.currentThread().getContextClassLoader()} is used, but this can be overridden
047     * by calling {@link #setClassLoader(ClassLoader)} prior to invoking any of the {@code find()}
048     * methods.</p>
049     *
050     * <p>General searches are initiated by calling the
051     * {@link #find(org.apache.logging.log4j.core..util.ResolverUtil.Test, String...)} ()} method and supplying
052     * a package name and a Test instance. This will cause the named package <b>and all sub-packages</b>
053     * to be scanned for classes that meet the test. There are also utility methods for the common
054     * use cases of scanning multiple packages for extensions of particular classes, or classes
055     * annotated with a specific annotation.</p>
056     *
057     * <p>The standard usage pattern for the ResolverUtil class is as follows:</p>
058     *
059     *<pre>
060     *ResolverUtil&lt;ActionBean&gt; resolver = new ResolverUtil&lt;ActionBean&gt;();
061     *resolver.findImplementation(ActionBean.class, pkg1, pkg2);
062     *resolver.find(new CustomTest(), pkg1);
063     *resolver.find(new CustomTest(), pkg2);
064     *Collection&lt;ActionBean&gt; beans = resolver.getClasses();
065     *</pre>
066     *
067     * <p>This class was copied from Stripes - http://stripes.mc4j.org/confluence/display/stripes/Home
068     * </p>
069     *
070     * @author Tim Fennell
071     * @param <T> The type of the Class that can be returned.
072     */
073    public class ResolverUtil<T> {
074        /** An instance of Log to use for logging in this class. */
075        private static final Logger LOG = StatusLogger.getLogger();
076    
077        private static final String VFSZIP = "vfszip";
078    
079        /** The set of matches being accumulated. */
080        private Set<Class<? extends T>> classMatches = new HashSet<Class<?extends T>>();
081    
082        /** The set of matches being accumulated. */
083        private Set<URI> resourceMatches = new HashSet<URI>();
084    
085        /**
086         * The ClassLoader to use when looking for classes. If null then the ClassLoader returned
087         * by Thread.currentThread().getContextClassLoader() will be used.
088         */
089        private ClassLoader classloader;

091        /**
092         * Provides access to the classes discovered so far. If no calls have been made to
093         * any of the {@code find()} methods, this set will be empty.
094         *
095         * @return the set of classes that have been discovered.
096         */
097        public Set<Class<? extends T>> getClasses() {
098            return classMatches;
099        }

101        /**
102         * Return the matching resources.
103         * @return A Set of URIs that match the criteria.
104         */
105        public Set<URI> getResources() {
106            return resourceMatches;
107        }

114         * @return the ClassLoader that will be used to scan for classes

116        public ClassLoader getClassLoader() {
117            return classloader != null ? classloader : (classloader = Loader.getClassLoader(ResolverUtil.class, null));

128        /**
129         * Attempts to discover classes that are assignable to the type provided. In the case
130         * that an interface is provided this method will collect implementations. In the case
131         * of a non-interface class, subclasses will be collected.  Accumulated classes can be
132         * accessed by calling {@link #getClasses()}.
133         *
134         * @param parent the class of interface to find subclasses or implementations of
135         * @param packageNames one or more package names to scan (including subpackages) for classes
136         */
137        public void findImplementations(Class parent, String... packageNames) {
138            if (packageNames == null) {
139                return;
140            }
141    
142            Test test = new IsA(parent);
143            for (String pkg : packageNames) {
144                findInPackage(test, pkg);
145            }
146        }

157                return;

195        /**
196         * Attempts to discover classes that pass the test. Accumulated
197         * classes can be accessed by calling {@link #getClasses()}.
198         *
199         * @param test the test to determine matching classes
200         * @param packageNames one or more package names to scan (including subpackages) for classes
201         */
202        public void find(Test test, String... packageNames) {
203            if (packageNames == null) {
204                return;
205            }
206    
207            for (String pkg : packageNames) {
208                findInPackage(test, pkg);
209            }
210        }
211    
212        /**
213         * Scans for classes starting at the package provided and descending into subpackages.
214         * Each class is offered up to the Test as it is discovered, and if the Test returns
215         * true the class is retained.  Accumulated classes can be fetched by calling
216         * {@link #getClasses()}.
217         *
218         * @param test an instance of {@link Test} that will be used to filter classes
219         * @param packageName the name of the package from which to start scanning for
220         *        classes, e.g. {@code net.sourceforge.stripes}
221         */
222        public void findInPackage(Test test, String packageName) {
223            packageName = packageName.replace('.', '/');
224            ClassLoader loader = getClassLoader();
225            Enumeration<URL> urls;
226    
227            try {
228                urls = loader.getResources(packageName);
229            } catch (IOException ioe) {
230                LOG.warn("Could not read package: " + packageName, ioe);
231                return;
232            }
233    
234            while (urls.hasMoreElements()) {
235                try {
236                    URL url = urls.nextElement();
237                    String urlPath = url.getFile();
238                    urlPath = URLDecoder.decode(urlPath, "UTF-8");
239    
240                    // If it's a file in a directory, trim the stupid file: spec
241                    if (urlPath.startsWith("file:")) {
242                        urlPath = urlPath.substring(5);
243                    }
244    
245                    // Else it's in a JAR, grab the path to the jar
246                    if (urlPath.indexOf('!') > 0) {
247                        urlPath = urlPath.substring(0, urlPath.indexOf('!'));
248                    }
249    
250                    LOG.info("Scanning for classes in [" + urlPath + "] matching criteria: " + test);
251                    // Check for a jar in a war in JBoss
252                    if (VFSZIP.equals(url.getProtocol())) {
253                        String path = urlPath.substring(0, urlPath.length() - packageName.length() - 2);
254                        URL newURL = new URL(url.getProtocol(), url.getHost(), path);
255                        JarInputStream stream = new JarInputStream(newURL.openStream());
256                        loadImplementationsInJar(test, packageName, path, stream);
257                    } else {
258                        File file = new File(urlPath);
259                        if (file.isDirectory()) {
260                            loadImplementationsInDirectory(test, packageName, file);
261                        } else {
262                            loadImplementationsInJar(test, packageName, file);
263                        }
264                    }
265                } catch (IOException ioe) {
266                    LOG.warn("could not read entries", ioe);
267                }
268            }
269        }

272        /**
273         * Finds matches in a physical directory on a filesystem.  Examines all
274         * files within a directory - if the File object is not a directory, and ends with <i>.class</i>
275         * the file is loaded and tested to see if it is acceptable according to the Test.  Operates
276         * recursively to find classes within a folder structure matching the package structure.
277         *
278         * @param test a Test used to filter the classes that are discovered
279         * @param parent the package name up to this directory in the package hierarchy.  E.g. if
280         *        /classes is in the classpath and we wish to examine files in /classes/org/apache then
281         *        the values of <i>parent</i> would be <i>org/apache</i>
282         * @param location a File object representing a directory
283         */
284        private void loadImplementationsInDirectory(Test test, String parent, File location) {
285            File[] files = location.listFiles();
286            StringBuilder builder;
287    
288            for (File file : files) {
289                builder = new StringBuilder();
290                builder.append(parent).append("/").append(file.getName());
291                String packageOrClass = parent == null ? file.getName() : builder.toString();
292    
293                if (file.isDirectory()) {
294                    loadImplementationsInDirectory(test, packageOrClass, file);
295                } else if (isTestApplicable(test, file.getName())) {
296                    addIfMatching(test, packageOrClass);
297                }
298            }
299        }
300    
301        private boolean isTestApplicable(Test test, String path) {
302            return test.doesMatchResource() || path.endsWith(".class") && test.doesMatchClass();
303        }
304    
305        /**
306         * Finds matching classes within a jar files that contains a folder structure
307         * matching the package structure.  If the File is not a JarFile or does not exist a warning
308         * will be logged, but no error will be raised.
309         *
310         * @param test a Test used to filter the classes that are discovered
311         * @param parent the parent package under which classes must be in order to be considered
312         * @param jarfile the jar file to be examined for classes
313         */
314        private void loadImplementationsInJar(Test test, String parent, File jarfile) {
315            JarInputStream jarStream;
316            try {
317                jarStream = new JarInputStream(new FileInputStream(jarfile));
318                loadImplementationsInJar(test, parent, jarfile.getPath(), jarStream);
319            } catch (FileNotFoundException ex) {
320                LOG.error("Could not search jar file '" + jarfile + "' for classes matching criteria: " +
321                    test + " file not found");
322            } catch (IOException ioe) {
323                LOG.error("Could not search jar file '" + jarfile + "' for classes matching criteria: " +
324                    test + " due to an IOException", ioe);
325            }
326        }
327    
328        /**
329         * Finds matching classes within a jar files that contains a folder structure
330         * matching the package structure.  If the File is not a JarFile or does not exist a warning
331         * will be logged, but no error will be raised.
332         *
333         * @param test a Test used to filter the classes that are discovered
334         * @param parent the parent package under which classes must be in order to be considered
335         * @param stream The jar InputStream
336         */
337        private void loadImplementationsInJar(Test test, String parent, String path, JarInputStream stream) {
338    
339            try {
340                JarEntry entry;
341    
342                while ((entry = stream.getNextJarEntry()) != null) {
343                    String name = entry.getName();
344                    if (!entry.isDirectory() && name.startsWith(parent) && isTestApplicable(test, name)) {
345                        addIfMatching(test, name);
346                    }
347                }
348            } catch (IOException ioe) {
349                LOG.error("Could not search jar file '" + path + "' for classes matching criteria: " +
350                    test + " due to an IOException", ioe);
351            }
352        }

354        /**
355         * Add the class designated by the fully qualified class name provided to the set of
356         * resolved classes if and only if it is approved by the Test supplied.
357         *
358         * @param test the test used to determine if the class matches
359         * @param fqn the fully qualified name of a class
360         */
361        protected void addIfMatching(Test test, String fqn) {
362            try {
363                ClassLoader loader = getClassLoader();
364                if (test.doesMatchClass()) {
365                    String externalName = fqn.substring(0, fqn.indexOf('.')).replace('/', '.');
366                    if (LOG.isDebugEnabled()) {
367                        LOG.debug("Checking to see if class " + externalName + " matches criteria [" + test + "]");
368                    }
369    
370                    Class type = loader.loadClass(externalName);
371                    if (test.matches(type)) {
372                        classMatches.add((Class<T>) type);
373                    }
374                }
375                if (test.doesMatchResource()) {
376                    URL url = loader.getResource(fqn);
377                    if (url == null) {
378                        url = loader.getResource(fqn.substring(1));
379                    }
380                    if (url != null && test.matches(url.toURI())) {
381                        resourceMatches.add(url.toURI());
382                    }
383                }
384            } catch (Throwable t) {
385                LOG.warn("Could not examine class '" + fqn + "' due to a " +
386                         t.getClass().getName() + " with message: " + t.getMessage());
387            }
388        }

390        /**
391         * A simple interface that specifies how to test classes to determine if they
392         * are to be included in the results produced by the ResolverUtil.
393         */
394        public interface Test {
395            /**
396             * Will be called repeatedly with candidate classes. Must return True if a class
397             * is to be included in the results, false otherwise.
398             * @param type The Class to match against.
399             * @return true if the Class matches.
400             */
401            boolean matches(Class type);
402    
403            /**
404             * Test for a resource.
405             * @param resource The URI to the resource.
406             * @return true if the resource matches.
407             */
408            boolean matches(URI resource);
409    
410            boolean doesMatchClass();
411            boolean doesMatchResource();
412        }
413    
414        /**
415         * Test against a Class.
416         */
417        public abstract static class ClassTest implements Test {
418            public boolean matches(URI resource) {
419                throw new UnsupportedOperationException();
420            }
421    
422            public boolean doesMatchClass() {
423                return true;
424            }
425            public boolean doesMatchResource() {
426                return false;
427            }
428        }
429    
430        /**
431         * Test against a resource.
432         */
433        public abstract static class ResourceTest implements Test {
434            public boolean matches(Class cls) {
435                throw new UnsupportedOperationException();
436            }
437    
438            public boolean doesMatchClass() {
439                return false;
440            }
441            public boolean doesMatchResource() {
442                return true;
443            }
444        }

446        /**
447         * A Test that checks to see if each class is assignable to the provided class. Note
448         * that this test will match the parent type itself if it is presented for matching.
449         */
450        public static class IsA extends ClassTest {
451            private Class parent;
452    
453            /**
454             * Constructs an IsA test using the supplied Class as the parent class/interface.
455             * @param parentType The parent class to check for.
456             */
457            public IsA(Class parentType) { this.parent = parentType; }
458    
459            /**
460             * Returns true if type is assignable to the parent type supplied in the constructor.
461             * @param type The Class to check.
462             * @return true if the Class matches.
463             */
464            public boolean matches(Class type) {
465                return type != null && parent.isAssignableFrom(type);
466            }
467    
468            @Override
469            public String toString() {
470                return "is assignable to " + parent.getSimpleName();
471            }
472        }
473    
474        /**
475         * A Test that checks to see if each class name ends with the provided suffix.
476         */
477        public static class NameEndsWith extends ClassTest {
478            private String suffix;
479    
480            /**
481             * Constructs a NameEndsWith test using the supplied suffix.
482             * @param suffix the String suffix to check for.
483             */
484            public NameEndsWith(String suffix) { this.suffix = suffix; }
485    
486            /**
487             * Returns true if type name ends with the suffix supplied in the constructor.
488             * @param type The Class to check.
489             * @return true if the Class matches.
490             */
491            public boolean matches(Class type) {
492                return type != null && type.getName().endsWith(suffix);
493            }
494    
495            @Override
496            public String toString() {
497                return "ends with the suffix " + suffix;
498            }

501        /**
502         * A Test that checks to see if each class is annotated with a specific annotation. If it
503         * is, then the test returns true, otherwise false.
504         */
505        public static class AnnotatedWith extends ClassTest {
506            private Class<? extends Annotation> annotation;
507    
508            /**
509             * Constructs an AnnotatedWith test for the specified annotation type.
510             * @param annotation The annotation to check for.
511             */
512            public AnnotatedWith(Class<? extends Annotation> annotation) {
513                this.annotation = annotation;
514            }
515    
516            /**
517             * Returns true if the type is annotated with the class provided to the constructor.
518             * @param type the Class to match against.
519             * @return true if the Classes match.
520             */
521            public boolean matches(Class type) {
522                return type != null && type.isAnnotationPresent(annotation);
523            }
524    
525            @Override
526            public String toString() {
527                return "annotated with @" + annotation.getSimpleName();
528            }
529        }

531        /**
532         * A Test that checks to see if the class name matches.
533         */
534        public static class NameIs extends ResourceTest {
535            private String name;
536    
537            public NameIs(String name) { this.name = "/" + name; }
538    
539            public boolean matches(URI resource) {
540                return resource.getPath().endsWith(name);
541            }
542    
543            @Override public String toString() {
544                return "named " + name;
545            }

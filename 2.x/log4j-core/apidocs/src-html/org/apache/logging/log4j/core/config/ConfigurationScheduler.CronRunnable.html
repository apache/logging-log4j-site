017package org.apache.logging.log4j.core.config;
018
019import java.util.Date;
020import java.util.List;
021import java.util.Queue;
022import java.util.concurrent.Callable;
023import java.util.concurrent.ScheduledExecutorService;
024import java.util.concurrent.ScheduledFuture;
025import java.util.concurrent.ScheduledThreadPoolExecutor;
026import java.util.concurrent.TimeUnit;
027
028import org.apache.logging.log4j.Logger;
029import org.apache.logging.log4j.core.AbstractLifeCycle;
030import org.apache.logging.log4j.core.util.CronExpression;
031import org.apache.logging.log4j.core.util.Log4jThreadFactory;
032import org.apache.logging.log4j.status.StatusLogger;
033
034/**
035 *
036 */
037public class ConfigurationScheduler extends AbstractLifeCycle {
038
039    private static final Logger LOGGER = StatusLogger.getLogger();
040    private static final String SIMPLE_NAME = "Log4j2 " + ConfigurationScheduler.class.getSimpleName();
041    private static final int MAX_SCHEDULED_ITEMS = 5;
042    private ScheduledExecutorService executorService;
043
044    private int scheduledItems = 0;
045
046    @Override
047    public void start() {
048        super.start();
049    }
050
051    @Override
052    public boolean stop(final long timeout, final TimeUnit timeUnit) {
053        setStopping();
054        if (isExecutorServiceSet()) {
055            LOGGER.debug("{} shutting down threads in {}", SIMPLE_NAME, getExecutorService());
056            executorService.shutdown();
057            try {
058                executorService.awaitTermination(timeout, timeUnit);
059            } catch (final InterruptedException ie) {
060                executorService.shutdownNow();
061                try {
062                    executorService.awaitTermination(timeout, timeUnit);
063                } catch (final InterruptedException inner) {
064                    LOGGER.warn("ConfigurationScheduler stopped but some scheduled services may not have completed.");
065                }
066                // Preserve interrupt status
067                Thread.currentThread().interrupt();
068            }
069        }
070        setStopped();
071        return true;
072    }
073
074    public boolean isExecutorServiceSet() {
075        return executorService != null;
076    }
077
078    /**
079     * Increment the number of threads in the pool.
080     */
081    public void incrementScheduledItems() {
082        if (isExecutorServiceSet()) {
083            LOGGER.error("{} attempted to increment scheduled items after start", SIMPLE_NAME);
084        } else {
085            ++scheduledItems;
086        }
087    }
088
089    /**
090     * Decrement the number of threads in the pool
091     */
092    public void decrementScheduledItems() {
093        if (!isStarted() && scheduledItems > 0) {
094            --scheduledItems;
095        }
096    }
097
098    /**
099     * Creates and executes a ScheduledFuture that becomes enabled after the given delay.
100     * @param <V> The result type returned by this Future
101     * @param callable the function to execute.
102     * @param delay the time from now to delay execution.
103     * @param unit the time unit of the delay parameter.
104     * @return a ScheduledFuture that can be used to extract result or cancel.
105     *
106     */
107    public <V> ScheduledFuture<V> schedule(final Callable<V> callable, final long delay, final TimeUnit unit) {
108        return getExecutorService().schedule(callable, delay, unit);
109    }
110
111    /**
112     * Creates and executes a one-shot action that becomes enabled after the given delay.
113     * @param command the task to execute.
114     * @param delay the time from now to delay execution.
115     * @param unit the time unit of the delay parameter.
116     * @return a ScheduledFuture representing pending completion of the task and whose get() method will return null
117     * upon completion.
118     */
119    public ScheduledFuture<?> schedule(final Runnable command, final long delay, final TimeUnit unit) {
120        return getExecutorService().schedule(command, delay, unit);
121    }
122
123
124    /**
125     * Creates and executes an action that first based on a cron expression.
126     * @param cronExpression the cron expression describing the schedule.
127     * @param command The Runnable to run,
128     * @return a ScheduledFuture representing the next time the command will run.
129     */
130    public CronScheduledFuture<?> scheduleWithCron(final CronExpression cronExpression, final Runnable command) {
131        return scheduleWithCron(cronExpression, new Date(), command);
132    }
133
134    /**
135     * Creates and executes an action that first based on a cron expression.
136     * @param cronExpression the cron expression describing the schedule.
137     * @param startDate The time to use as the time to begin the cron expression. Defaults to the current date and time.
138     * @param command The Runnable to run,
139     * @return a ScheduledFuture representing the next time the command will run.
140     */
141    public CronScheduledFuture<?> scheduleWithCron(final CronExpression cronExpression, final Date startDate, final Runnable command) {
142        final Date fireDate = cronExpression.getNextValidTimeAfter(startDate == null ? new Date() : startDate);
143        final CronRunnable runnable = new CronRunnable(command, cronExpression);
144        final ScheduledFuture<?> future = schedule(runnable, nextFireInterval(fireDate), TimeUnit.MILLISECONDS);
145        final CronScheduledFuture<?> cronScheduledFuture = new CronScheduledFuture<>(future, fireDate);
146        runnable.setScheduledFuture(cronScheduledFuture);
147        LOGGER.debug("Scheduled cron expression {} to fire at {}", cronExpression.getCronExpression(), fireDate);
148        return cronScheduledFuture;
149    }
150
151
152    /**
153     * Creates and executes a periodic action that becomes enabled first after the given initial delay, and subsequently
154     * with the given period; that is executions will commence after initialDelay then initialDelay+period,
155     * then initialDelay + 2 * period, and so on.
156     * @param command the task to execute.
157     * @param initialDelay the time to delay first execution.
158     * @param period the period between successive executions.
159     * @param unit the time unit of the initialDelay and period parameters.
160     * @return a ScheduledFuture representing pending completion of the task, and whose get() method will throw an
161     * exception upon cancellation
162     */
163    public ScheduledFuture<?> scheduleAtFixedRate(final Runnable command, final long initialDelay, final long period, final TimeUnit unit) {
164        return getExecutorService().scheduleAtFixedRate(command, initialDelay, period, unit);
165    }
166
167    /**
168     * Creates and executes a periodic action that becomes enabled first after the given initial delay, and
169     * subsequently with the given delay between the termination of one execution and the commencement of the next.
170     * @param command the task to execute.
171     * @param initialDelay the time to delay first execution.
172     * @param delay the delay between the termination of one execution and the commencement of the next.
173     * @param unit the time unit of the initialDelay and delay parameters
174     * @return a ScheduledFuture representing pending completion of the task, and whose get() method will throw an
175     * exception upon cancellation
176     */
177    public ScheduledFuture<?> scheduleWithFixedDelay(final Runnable command, final long initialDelay, final long delay, final TimeUnit unit) {
178        return getExecutorService().scheduleWithFixedDelay(command, initialDelay, delay, unit);
179    }
180
181    public long nextFireInterval(final Date fireDate) {
182        return fireDate.getTime() - new Date().getTime();
183    }
184
185    private ScheduledExecutorService getExecutorService() {
186        if (executorService == null) {
187            if (scheduledItems > 0) {
188                LOGGER.debug("{} starting {} threads", SIMPLE_NAME, scheduledItems);
189                scheduledItems = Math.min(scheduledItems, MAX_SCHEDULED_ITEMS);
190                ScheduledThreadPoolExecutor executor = new ScheduledThreadPoolExecutor(scheduledItems,
191                        Log4jThreadFactory.createDaemonThreadFactory("Scheduled"));
192                executor.setContinueExistingPeriodicTasksAfterShutdownPolicy(false);
193                executor.setExecuteExistingDelayedTasksAfterShutdownPolicy(false);
194                this.executorService = executor;
195
196            } else {
197                LOGGER.debug("{}: No scheduled items", SIMPLE_NAME);
198            }
199        }
200        return executorService;
201    }
202
203    public class CronRunnable implements Runnable {
204
205        private final CronExpression cronExpression;
206        private final Runnable runnable;
207        private CronScheduledFuture<?> scheduledFuture;
208
209        public CronRunnable(final Runnable runnable, final CronExpression cronExpression) {
210            this.cronExpression = cronExpression;
211            this.runnable = runnable;
212        }
213
214        public void setScheduledFuture(final CronScheduledFuture<?> future) {
215            this.scheduledFuture = future;
216        }
217
218        @Override
219        public void run() {
220            try {
221                long millis = scheduledFuture.getFireTime().getTime() - System.currentTimeMillis();
222                if (millis > 0) {
223                    LOGGER.debug("Cron thread woke up {} millis early. Sleeping", millis);
224                    try {
225                        Thread.sleep(millis);
226                    } catch (InterruptedException ie) {
227                        // Ignore the interruption.
228                    }
229                }
230                runnable.run();
231            } catch(final Throwable ex) {
232                LOGGER.error("{} caught error running command", SIMPLE_NAME, ex);
233            } finally {
234                final Date fireDate = cronExpression.getNextValidTimeAfter(new Date());
235                final ScheduledFuture<?> future = schedule(this, nextFireInterval(fireDate), TimeUnit.MILLISECONDS);
236                LOGGER.debug("Cron expression {} scheduled to fire again at {}", cronExpression.getCronExpression(),
237                        fireDate);
238                scheduledFuture.reset(future, fireDate);
239            }
240        }
241
242        public String toString() {
243            return "CronRunnable{" + cronExpression.getCronExpression() + " - " + scheduledFuture.getFireTime();
244        }
245    }
246
247    public String toString() {
248        StringBuilder sb = new StringBuilder("ConfigurationScheduler {");
249        Queue<Runnable> queue = ((ScheduledThreadPoolExecutor) executorService).getQueue();
250        boolean first = true;
251        for (Runnable runnable : queue) {
252            if (!first) {
253                sb.append(", ");
254            }
255            sb.append(runnable.toString());
256            first = false;
257        }
258        sb.append("}");
259        return sb.toString();
260    }

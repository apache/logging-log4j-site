017    package org.apache.logging.log4j.core.appender.rolling;
018    
019    import org.apache.logging.log4j.core.LogEvent;
020    import org.apache.logging.log4j.core.appender.FileManager;
021    import org.apache.logging.log4j.core.appender.ManagerFactory;
022    import org.apache.logging.log4j.core.appender.rolling.helper.Action;
023    import org.apache.logging.log4j.core.appender.rolling.helper.ActionBase;
024    
025    import java.io.BufferedOutputStream;
026    import java.io.File;
027    import java.io.FileNotFoundException;
028    import java.io.FileOutputStream;
029    import java.io.IOException;
030    import java.io.OutputStream;
031    import java.util.concurrent.Semaphore;
032    
033    /**
034     * The Rolling File Manager.
035     */
036    public class RollingFileManager extends FileManager {
037    
038        private static ManagerFactory factory = new RollingFileManagerFactory();
039    
040        private long size;
041        private long initialTime;
042        private PatternProcessor processor;
043        private final Semaphore semaphore = new Semaphore(1);
044    
045        protected RollingFileManager(String fileName, String pattern, OutputStream os, boolean append, long size,
046                                     long time) {
047            super(fileName, os, append, false);
048            this.size = size;
049            this.initialTime = time;
050            processor = new PatternProcessor(pattern);
051        }
052    
053        /**
054         * Return a RollingFileManager.
055         * @param fileName The file name.
056         * @param pattern The pattern for rolling file.
057         * @param append true if the file should be appended to.
058         * @param bufferedIO true if data should be buffered.
059         * @return A RollingFileManager.
060         */
061        public static RollingFileManager getFileManager(String fileName, String pattern, boolean append,
062                                                        boolean bufferedIO) {
063    
064            return (RollingFileManager) getManager(fileName, new FactoryData(pattern, append,
065                bufferedIO), factory);
066        }
067    
068        protected synchronized void write(byte[] bytes, int offset, int length) {
069            size += length;
070            super.write(bytes, offset, length);
071        }
072    
073        /**
074         * Return the current size of the file.
075         * @return The size of the file in bytes.
076         */
077        public long getFileSize() {
078            return size;
079        }
080    
081        /**
082         * Return the time the file was created.
083         * @return The time the file was created.
084         */
085        public long getFileTime() {
086            return initialTime;
087        }
088    
089        /**
090         * Determine if a rollover should occur.
091         * @param event The LogEvent.
092         * @param policy The TriggeringPolicy.
093         * @param strategy The RolloverStrategy.
094         */
095        public synchronized void checkRollover(LogEvent event, TriggeringPolicy policy, RolloverStrategy strategy) {
096            if (policy.isTriggeringEvent(event) && rollover(strategy)) {
097                try {
098                    size = 0;
099                    initialTime = System.currentTimeMillis();
100                    OutputStream os = new FileOutputStream(getFileName(), isAppend());
101                    setOutputStream(os);
102                } catch (FileNotFoundException ex) {
103                    LOGGER.error("FileManager (" + getFileName() + ") " + ex);
104                }
105            }
106        }
107    
108        /**
109         * Return the pattern processor.
110         * @return The PatternProcessor.
111         */
112        public PatternProcessor getProcessor() {
113            return processor;
114        }
115    
116        private boolean rollover(RolloverStrategy strategy) {
117    
118            try {
119                // Block until the asynchronous operation is completed.
120                semaphore.acquire();
121            } catch (InterruptedException ie) {
122                LOGGER.error("Thread interrupted while attempting to check rollover", ie);
123                return false;
124            }
125    
126            boolean success = false;
127            Thread thread = null;
128    
129            try {
130                RolloverDescription descriptor = strategy.rollover(this);
131    
132                if (descriptor != null) {
133    
134                    close();
135    
136                    if (descriptor.getSynchronous() != null) {
137    
138                        try {
139                            success = descriptor.getSynchronous().execute();
140                        } catch (Exception ex) {
141                            LOGGER.error("Error in synchronous task", ex);
142                        }
143                    }
144    
145                    if (success && descriptor.getAsynchronous() != null) {
146                        thread = new Thread(new AsyncAction(descriptor.getAsynchronous(), this));
147                        thread.start();
148                    }
149                    return true;
150                }
151                return false;
152            } finally {
153                if (thread == null) {
154                    semaphore.release();
155                }
156            }
157    
158        }
159    
160        /**
161         * Performs actions asynchronously.
162         */
163        private static class AsyncAction extends ActionBase {
164    
165            private final Action action;
166            private final RollingFileManager manager;
167    
168            /**
169             * Constructor.
170             * @param act The action to perform.
171             * @param manager The manager.
172             */
173            public AsyncAction(Action act, RollingFileManager manager) {
174                this.action = act;
175                this.manager = manager;
176            }
177    
178            /**
179             * Perform an action.
180             *
181             * @return true if action was successful.  A return value of false will cause
182             *         the rollover to be aborted if possible.
183             * @throws java.io.IOException if IO error, a thrown exception will cause the rollover
184             *                             to be aborted if possible.
185             */
186            public boolean execute() throws IOException {
187                try {
188                    return action.execute();
189                } finally {
190                    manager.semaphore.release();
191                }
192            }
193    
194            /**
195             * Cancels the action if not already initialized or waits till completion.
196             */
197            public void close() {
198                action.close();
199            }
200    
201            /**
202             * Determines if action has been completed.
203             *
204             * @return true if action is complete.
205             */
206            public boolean isComplete() {
207                return action.isComplete();
208            }
209        }
210    
211        /**
212         * Factory data.
213         */
214        private static class FactoryData {
215            private String pattern;
216            private boolean append;
217            private boolean bufferedIO;
218    
219            /**
220             * Create the data for the factory.
221             * @param pattern The pattern.
222             * @param append The append flag.
223             * @param bufferedIO The bufferedIO flag.
224             */
225            public FactoryData(String pattern, boolean append, boolean bufferedIO) {
226                this.pattern = pattern;
227                this.append = append;
228                this.bufferedIO = bufferedIO;
229            }
230        }
231    
232        /**
233         * Factory to create a RollingFileManager.
234         */
235        private static class RollingFileManagerFactory implements ManagerFactory<RollingFileManager, FactoryData> {
236    
237            /**
238             * Create the RollingFileManager.
239             * @param name The name of the entity to manage.
240             * @param data The data required to create the entity.
241             * @return a RollingFileManager.
242             */
243            public RollingFileManager createManager(String name, FactoryData data) {
244                File file = new File(name);
245                final File parent = file.getParentFile();
246                if (null != parent && !parent.exists()) {
247                    parent.mkdirs();
248                }
249                try {
250                    file.createNewFile();
251                } catch (IOException ioe) {
252                    LOGGER.error("Unable to create file " + name, ioe);
253                    return null;
254                }
255                long size = data.append ? file.length() : 0;
256                long time = file.lastModified();
257    
258                OutputStream os;
259                try {
260                    os = new FileOutputStream(name, data.append);
261                    if (data.bufferedIO) {
262                        os = new BufferedOutputStream(os);
263                    }
264                    return new RollingFileManager(name, data.pattern, os, data.append, size, time);
265                } catch (FileNotFoundException ex) {
266                    LOGGER.error("FileManager (" + name + ") " + ex);
267                }
268                return null;
269            }
270        }

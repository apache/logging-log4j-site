017    package org.apache.logging.log4j.core.appender;
018    
019    import org.apache.logging.log4j.core.Filter;
020    import org.apache.logging.log4j.core.Layout;
021    import org.apache.logging.log4j.core.config.plugins.Plugin;
022    import org.apache.logging.log4j.core.config.plugins.PluginAttr;
023    import org.apache.logging.log4j.core.config.plugins.PluginElement;
024    import org.apache.logging.log4j.core.config.plugins.PluginFactory;
025    import org.apache.logging.log4j.core.layout.PatternLayout;
026    
027    import java.io.OutputStream;
028    
029    /**
030     * ConsoleAppender appends log events to <code>System.out</code> or
031     * <code>System.err</code> using a layout specified by the user. The
032     * default target is <code>System.out</code>.
033     * @doubt accessing System.out or .err as a byte stream instead of a writer
034     *    bypasses the JVM's knowledge of the proper encoding. (RG) Encoding
035     * is handled within the Layout. Typically, a Layout will generate a String
036     * and then call getBytes which may use a configured encoding or the system
037     * default. OTOH, a Writer cannot print byte streams.
038     */
039    @Plugin(name = "Console", type = "Core", elementType = "appender", printObject = true)
040    public final class ConsoleAppender extends OutputStreamAppender {
041    
042        private static ManagerFactory factory = new ConsoleManagerFactory();
043    
044        /**
045         * Enumeration of console destinations.
046         */
047        public enum Target {
048            /** Standard output. */
049            SYSTEM_OUT,
050            /** Standard error output. */
051            SYSTEM_ERR
052        }
053    
054        private ConsoleAppender(String name, Layout layout, Filter filter, OutputStreamManager manager,
055                               boolean handleExceptions) {
056            super(name, layout, filter, handleExceptions, true, manager);
057        }
058    
059        /**
060         * Create a Console Appender.
061         * @param layout The layout to use (required).
062         * @param filter The Filter or null.
063         * @param t The target ("SYSTEM_OUT" or "SYSTEM_ERR"). The default is "SYSTEM_OUT".
064         * @param name The name of the Appender (required).
065         * @param suppress "true" if exceptions should be hidden from the application, "false" otherwise.
066         * The default is "true".
067         * @return The ConsoleAppender.
068         */
069        @PluginFactory
070        public static ConsoleAppender createAppender(@PluginElement("layout") Layout layout,
071                                                     @PluginElement("filters") Filter filter,
072                                                     @PluginAttr("target") String t,
073                                                     @PluginAttr("name") String name,
074                                                     @PluginAttr("suppressExceptions") String suppress) {
075            if (name == null) {
076                LOGGER.error("No name provided for ConsoleAppender");
077                return null;
078            }
079            if (layout == null) {
080                layout = PatternLayout.createLayout(null, null, null, null);
081            }
082            boolean handleExceptions = suppress == null ? true : Boolean.valueOf(suppress);
083            Target target = t == null ? Target.SYSTEM_OUT : Target.valueOf(t);
084            return new ConsoleAppender(name, layout, filter, getManager(target), handleExceptions);
085        }
086    
087        private static OutputStreamManager getManager(Target target) {
088            String type = target.name();
089            OutputStream os = target == Target.SYSTEM_OUT ? System.out : System.err;
090            OutputStreamManager manager = OutputStreamManager.getManager(target.name(), factory,
091                new FactoryData(os, type));
092            return manager;
093        }
094    
095        /**
096         * Data to pass to factory method.
097         */
098        private static class FactoryData {
099            private OutputStream os;
100            private String type;
101    
102            /**
103             * Constructor.
104             * @param os The OutputStream.
105             * @param type The name of the target.
106             */
107            public FactoryData(OutputStream os, String type) {
108                this.os = os;
109                this.type = type;
110            }
111        }
112    
113        /**
114         * Factory to create the Appender.
115         */
116        private static class ConsoleManagerFactory implements ManagerFactory<OutputStreamManager, FactoryData> {
117    
118            /**
119             * Create an OutputStreamManager.
120             * @param name The name of the entity to manage.
121             * @param data The data required to create the entity.
122             * @return The OutputStreamManager
123             */
124            public OutputStreamManager createManager(String name, FactoryData data) {
125                return new OutputStreamManager(data.os, data.type);
126            }
127        }
128    
129    }

054         * Not a very good default - it is the Apache Software Foundation's enterprise number.

058         * The default event id.

063        private static final int TWO_DIGITS = 10;
064        private static final int THREE_DIGITS = 100;
065        private static final int MILLIS_PER_MINUTE = 60000;
066        private static final int MINUTES_PER_HOUR = 60;

078        private final List<String> mdcIncludes;

088        private RFC5424Layout(Configuration config, Facility facility, String id, int ein, boolean includeMDC,
089                              boolean includeNL, String mdcId, String appName, String messageId, String excludes,
090                              String includes, String required, Charset charset) {
091            super(charset);
092            this.facility = facility;
093            this.defaultId = id == null ? DEFAULT_ID : id;
094            this.enterpriseNumber = ein;
095            this.includeMDC = includeMDC;
096            this.includeNewLine = includeNL;
097            this.mdcId = mdcId;
098            this.appName = appName;
099            this.messageId = messageId;
100            this.localHostName = getLocalHostname();
101            ListChecker c = null;
102            if (excludes != null) {
103                String[] array = excludes.split(",");
104                if (array.length > 0) {
105                    c = new ExcludeChecker();
106                    mdcExcludes = new ArrayList<String>(array.length);
107                    for (String str : array) {
108                        mdcExcludes.add(str.trim());
109                    }
110                } else {
111                    mdcExcludes = null;
112                }
113            } else {
114                mdcExcludes = null;
115            }
116            if (includes != null) {
117                String[] array = includes.split(",");
118                if (array.length > 0) {
119                    c = new IncludeChecker();
120                    mdcIncludes = new ArrayList<String>(array.length);
121                    for (String str : array) {
122                        mdcIncludes.add(str.trim());
123                    }
124                } else {
125                    mdcIncludes = null;
126                }
127            } else {
128                mdcIncludes = null;
129            }
130            if (required != null) {
131                String[] array = required.split(",");
132                if (array.length > 0) {
133                    mdcRequired = new ArrayList<String>(array.length);
134                    for (String str : array) {
135                        mdcRequired.add(str.trim());
136                    }
137                } else {
138                    mdcRequired = null;
139                }
140    
141            } else {
142                mdcRequired = null;
143            }
144            this.checker = c != null ? c : noopChecker;
145            String name = config == null ? null : config.getName();
146            configName = (name != null && name.length() > 0) ? name : null;
147        }

149        /**
150         * Formats a {@link org.apache.logging.log4j.core.LogEvent} in conformance with the RFC 5424 Syslog specification.
151         * @param event The LogEvent.
152         * @return The RFC 5424 String representation of the LogEvent.
153         */
154        public String formatAs(final LogEvent event) {
155            Message msg = event.getMessage();
156            boolean isStructured = msg instanceof StructuredDataMessage;
157            StringBuilder buf = new StringBuilder();
158    
159            buf.append("<");
160            buf.append(Priority.getPriority(facility, event.getLevel()));
161            buf.append(">1 ");
162            buf.append(computeTimeStampString(event.getMillis()));
163            buf.append(' ');
164            buf.append(localHostName);
165            buf.append(' ');
166            if (appName != null) {
167                buf.append(appName);
168            } else if (configName != null) {
169                buf.append(configName);
170            } else {
171                buf.append("-");
172            }
173            buf.append(" ");
174            buf.append(getProcId());
175            buf.append(" ");
176            String type = isStructured ? ((StructuredDataMessage) msg).getType() : null;
177            if (type != null) {
178                buf.append(type);
179            } else if (messageId != null) {
180                buf.append(messageId);
181            } else {
182                buf.append("-");
183            }
184            buf.append(" ");
185            if (isStructured || includeMDC) {
186                StructuredDataId id = null;
187                String text;
188                if (isStructured) {
189                    StructuredDataMessage data = (StructuredDataMessage) msg;
190                    Map map = data.getData();
191                    id = data.getId();
192                    formatStructuredElement(id, map, buf, noopChecker);
193                    text = data.getFormat();
194                } else {
195                    text = msg.getFormattedMessage();
196                }
197                if (includeMDC) {
198                    if (mdcRequired != null) {
199                        checkRequired(event.getContextMap());
200                    }
201                    int ein = id == null || id.getEnterpriseNumber() < 0 ? enterpriseNumber : id.getEnterpriseNumber();
202                    StructuredDataId mdcSDID = new StructuredDataId(mdcId, ein, null, null);
203                    formatStructuredElement(mdcSDID, event.getContextMap(), buf, checker);
204                }
205                if (text != null && text.length() > 0) {
206                    buf.append(" ").append(text);
207                }
208            } else {
209                buf.append("- ");
210                buf.append(msg.getFormattedMessage());
211            }
212            if (includeNewLine) {
213                buf.append("\n");
214            }
215            return buf.toString();
216        }

218        protected String getProcId() {
219            return "-";

222        /**
223         * This method gets the network name of the machine we are running on.
224         * Returns "UNKNOWN_LOCALHOST" in the unlikely case where the host name
225         * cannot be found.
226         *
227         * @return String the name of the local host
228         */
229        public String getLocalHostname() {
230            try {
231                InetAddress addr = InetAddress.getLocalHost();
232                return addr.getHostName();
233            } catch (UnknownHostException uhe) {
234                try {
235                    Enumeration<NetworkInterface> interfaces = NetworkInterface.getNetworkInterfaces();
236                    while (interfaces.hasMoreElements()) {
237                        NetworkInterface nic = interfaces.nextElement();
238                        Enumeration<InetAddress> addresses = nic.getInetAddresses();
239                        while (addresses.hasMoreElements()) {
240                            InetAddress address = addresses.nextElement();
241                            if (!address.isLoopbackAddress()) {
242                                String hostname = address.getHostName();
243                                if (hostname != null) {
244                                    return hostname;
245                                }
246                            }
247                        }
248                    }
249                } catch (SocketException se) {
250                    LOGGER.error("Could not determine local host name", uhe);
251                    return "UNKNOWN_LOCALHOST";
252                }
253                LOGGER.error("Could not determine local host name", uhe);
254                return "UNKNOWN_LOCALHOST";
255            }
256        }

266        private String computeTimeStampString(long now) {
267            long last;
268            synchronized (this) {
269                last = lastTimestamp;
270                if (now == lastTimestamp) {
271                    return timestamppStr;
272                }
273            }
274    
275            StringBuilder buf = new StringBuilder();
276            Calendar cal = new GregorianCalendar();
277            cal.setTimeInMillis(now);
278            buf.append(Integer.toString(cal.get(Calendar.YEAR)));
279            buf.append("-");
280            pad(cal.get(Calendar.MONTH) + 1, TWO_DIGITS, buf);
281            buf.append("-");
282            pad(cal.get(Calendar.DAY_OF_MONTH), TWO_DIGITS, buf);
283            buf.append("T");
284            pad(cal.get(Calendar.HOUR_OF_DAY), TWO_DIGITS, buf);
285            buf.append(":");
286            pad(cal.get(Calendar.MINUTE), TWO_DIGITS, buf);
287            buf.append(":");
288            pad(cal.get(Calendar.SECOND), TWO_DIGITS, buf);
289    
290            int millis = cal.get(Calendar.MILLISECOND);
291            if (millis != 0) {
292                buf.append(".");
293                pad(millis, THREE_DIGITS, buf);
294            }
295    
296            int tzmin = (cal.get(Calendar.ZONE_OFFSET) + cal.get(Calendar.DST_OFFSET)) / MILLIS_PER_MINUTE;
297            if (tzmin == 0) {
298                buf.append("Z");
299            } else {
300                if (tzmin < 0) {
301                    tzmin = -tzmin;
302                    buf.append("-");
303                } else {
304                    buf.append("+");
305                }
306                int tzhour = tzmin / MINUTES_PER_HOUR;
307                tzmin -= tzhour * MINUTES_PER_HOUR;
308                pad(tzhour, TWO_DIGITS, buf);
309                buf.append(":");
310                pad(tzmin, TWO_DIGITS, buf);
311            }
312            synchronized (this) {
313                if (last == lastTimestamp) {
314                    lastTimestamp = now;
315                    timestamppStr = buf.toString();
316                }
317            }
318            return buf.toString();
319        }

321        private void pad(int val, int max, StringBuilder buf) {
322            while (max > 1) {
323                if (val < max) {
324                    buf.append("0");
325                }
326                max = max / TWO_DIGITS;
327            }
328            buf.append(Integer.toString(val));
329        }

331        private void formatStructuredElement(StructuredDataId id, Map<String, String> data, StringBuilder sb,
332                                             ListChecker checker) {
333            if (id == null && defaultId == null) {
334                return;
335            }
336            sb.append("[");
337            sb.append(getId(id));
338            appendMap(data, sb, checker);
339            sb.append("]");
340        }
341    
342        private String getId(StructuredDataId id) {
343            StringBuilder sb = new StringBuilder();
344            if (id.getName() == null) {
345                sb.append(defaultId);
346            } else {
347                sb.append(id.getName());
348            }
349            int ein = id.getEnterpriseNumber();
350            if (ein < 0) {
351                ein = enterpriseNumber;
352            }
353            if (ein >= 0) {
354                sb.append("@").append(ein);
355            }
356            return sb.toString();
357        }
358    
359        private void checkRequired(Map<String, String> map) {
360            for (String key : mdcRequired) {
361                String value = map.get(key);
362                if (value == null) {
363                    throw new LoggingException("Required key " + key + " is missing from the " + mdcId);
364                }
365            }
366        }
367    
368        private void appendMap(Map<String, String> map, StringBuilder sb, ListChecker checker)
369        {
370            SortedMap<String, String> sorted = new TreeMap<String, String>(map);
371            for (Map.Entry<String, String> entry : sorted.entrySet()) {
372                if (checker.check(entry.getKey())) {
373                    sb.append(" ");
374                    sb.append(entry.getKey()).append("=\"").append(entry.getValue()).append("\"");
375                }
376            }
377        }
378    
379        /**
380         * Interface used to check keys in a Map.
381         */
382        private interface ListChecker {
383            boolean check(String key);
384        }
385    
386        /**
387         * Includes only the listed keys.
388         */
389        private class IncludeChecker implements ListChecker {
390            public boolean check(String key) {
391                return mdcIncludes.contains(key);
392            }
393        }
394    
395        /**
396         * Excludes the listed keys.
397         */
398        private class ExcludeChecker implements ListChecker {
399            public boolean check(String key) {
400                return !mdcExcludes.contains(key);
401            }
402        }
403    
404        /**
405         * Does nothing.
406         */
407        private class NoopChecker implements ListChecker {
408            public boolean check(String key) {
409                return true;
410            }
411        }

413        @Override
414        public String toString() {
415            StringBuilder sb = new StringBuilder();
416            sb.append("facility=").append(facility.name());
417            sb.append(" appName=").append(appName);
418            sb.append(" defaultId=").append(defaultId);
419            sb.append(" enterpriseNumber=").append(enterpriseNumber);
420            sb.append(" newLine=").append(includeNewLine);
421            sb.append(" includeMDC=").append(includeMDC);
422            sb.append(" messageId=").append(messageId);
423            return sb.toString();
424        }

426        /**
427         * Create the RFC 5424 Layout.
428         * @param facility The Facility is used to try to classify the message.
429         * @param id The default structured data id to use when formatting according to RFC 5424.
430         * @param ein The IANA enterprise number.
431         * @param includeMDC Indicates whether data from the ThreadContextMap will be included in the RFC 5424 Syslog
432         * record. Defaults to "true:.
433         * @param mdcId The id to use for the MDC Structured Data Element.
434         * @param includeNL If true, a newline will be appended to the end of the syslog record. The default is false.
435         * @param appName The value to use as the APP-NAME in the RFC 5424 syslog record.
436         * @param msgId The default value to be used in the MSGID field of RFC 5424 syslog records.
437         * @param excludes A comma separated list of mdc keys that should be excluded from the LogEvent.
438         * @param includes A comma separated list of mdc keys that should be included in the FlumeEvent.
439         * @param required A comma separated list of mdc keys that must be present in the MDC.
440         * @param charset The character set.
441         * @param config The Configuration. Some Converters require access to the Interpolator.
442         * @return An RFC5424Layout.
443         */
444        @PluginFactory
445        public static RFC5424Layout createLayout(@PluginAttr("facility") String facility,
446                                                 @PluginAttr("id") String id,
447                                                 @PluginAttr("enterpriseNumber") String ein,
448                                                 @PluginAttr("includeMDC") String includeMDC,
449                                                 @PluginAttr("mdcId") String mdcId,
450                                                 @PluginAttr("newLine") String includeNL,
451                                                 @PluginAttr("appName") String appName,
452                                                 @PluginAttr("messageId") String msgId,
453                                                 @PluginAttr("mdcExcludes") String excludes,
454                                                 @PluginAttr("mdcIncludes") String includes,
455                                                 @PluginAttr("mdcRequired") String required,
456                                                 @PluginAttr("charset") String charset,
457                                                 @PluginConfiguration Configuration config) {
458            Charset c = Charset.isSupported("UTF-8") ? Charset.forName("UTF-8") : Charset.defaultCharset();
459            if (charset != null) {
460                if (Charset.isSupported(charset)) {
461                    c = Charset.forName(charset);
462                } else {
463                    LOGGER.error("Charset " + charset + " is not supported for layout, using " + c.displayName());
464                }
465            }
466            if (includes != null && excludes != null) {
467                LOGGER.error("mdcIncludes and mdcExcludes are mutually exclusive. Includes wil be ignored");
468                includes = null;
469            }
470            Facility f = facility != null ? Facility.valueOf(facility.toUpperCase()) : Facility.LOCAL0;
471            int enterpriseNumber = ein == null ? DEFAULT_ENTERPRISE_NUMBER : Integer.parseInt(ein);
472            boolean isMdc = includeMDC == null ? true : Boolean.valueOf(includeMDC);
473            boolean includeNewLine = includeNL == null ? false : Boolean.valueOf(includeNL);
474            if (mdcId == null) {
475                mdcId = DEFAULT_MDCID;
476            }
477    
478            return new RFC5424Layout(config, f, id, enterpriseNumber, isMdc, includeNewLine, mdcId, appName, msgId,
479                                     excludes, includes, required, c);
480        }

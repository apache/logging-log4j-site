017    package org.apache.log4j;
018    
019    import java.util.Stack;
020    
021    /**
022     *
023     */
024    public final class NDC {
025    
026        private NDC() {
027        }
028    
029        /**
030         * Clear any nested diagnostic information if any. This method is
031         * useful in cases where the same thread can be potentially used
032         * over and over in different unrelated contexts.
033         * <p/>
034         * <p>This method is equivalent to calling the {@link #setMaxDepth}
035         * method with a zero <code>maxDepth</code> argument.
036         */
037        public static void clear() {
038            org.apache.logging.log4j.ThreadContext.clearStack();
039        }
040    
041    
042        /**
043         * Clone the diagnostic context for the current thread.
044         * <p/>
045         * <p>Internally a diagnostic context is represented as a stack.  A
046         * given thread can supply the stack (i.e. diagnostic context) to a
047         * child thread so that the child can inherit the parent thread's
048         * diagnostic context.
049         * <p/>
050         * <p>The child thread uses the {@link #inherit inherit} method to
051         * inherit the parent's diagnostic context.
052         *
053         * @return Stack A clone of the current thread's  diagnostic context.
054         */
055        public static Stack cloneStack() {
056            return org.apache.logging.log4j.ThreadContext.cloneStack();
057        }
058    
059    
060        /**
061         * Inherit the diagnostic context of another thread.
062         * <p/>
063         * <p>The parent thread can obtain a reference to its diagnostic
064         * context using the {@link #cloneStack} method.  It should
065         * communicate this information to its child so that it may inherit
066         * the parent's diagnostic context.
067         * <p/>
068         * <p>The parent's diagnostic context is cloned before being
069         * inherited. In other words, once inherited, the two diagnostic
070         * contexts can be managed independently.
071         * <p/>
072         * <p>In java, a child thread cannot obtain a reference to its
073         * parent, unless it is directly handed the reference. Consequently,
074         * there is no client-transparent way of inheriting diagnostic
075         * contexts. Do you know any solution to this problem?
076         *
077         * @param stack The diagnostic context of the parent thread.
078         */
079        public static void inherit(Stack stack) {
080            org.apache.logging.log4j.ThreadContext.setStack(stack);
081        }
082    
083    
084        /**
085         * <font color="#FF4040"><b>Never use this method directly.</b>
086         * @return The string value of the specified key.
087         */
088        public static String get() {
089            return org.apache.logging.log4j.ThreadContext.peek();
090        }
091    
092        /**
093         * Get the current nesting depth of this diagnostic context.
094         * @return int The number of elements in the call stack.
095         * @see #setMaxDepth
096         */
097        public static int getDepth() {
098            return org.apache.logging.log4j.ThreadContext.getDepth();
099        }
100    
101        /**
102         * Clients should call this method before leaving a diagnostic
103         * context.
104         * <p/>
105         * <p>The returned value is the value that was pushed last. If no
106         * context is available, then the empty string "" is returned.
107         *
108         * @return String The innermost diagnostic context.
109         */
110        public static String pop() {
111            return org.apache.logging.log4j.ThreadContext.pop();
112        }
113    
114        /**
115         * Looks at the last diagnostic context at the top of this NDC
116         * without removing it.
117         * <p/>
118         * <p>The returned value is the value that was pushed last. If no
119         * context is available, then the empty string "" is returned.
120         *
121         * @return String The innermost diagnostic context.
122         */
123        public static String peek() {
124            return org.apache.logging.log4j.ThreadContext.peek();
125        }
126    
127        /**
128         * Push new diagnostic context information for the current thread.
129         * <p/>
130         * <p>The contents of the <code>message</code> parameter is
131         * determined solely by the client.
132         *
133         * @param message The new diagnostic context information.
134         */
135        public static void push(String message) {
136            org.apache.logging.log4j.ThreadContext.push(message);
137        }
138    
139        /**
140         * Remove the diagnostic context for this thread.
141         * <p/>
142         * <p>Each thread that created a diagnostic context by calling
143         * {@link #push} should call this method before exiting. Otherwise,
144         * the memory used by the <b>thread</b> cannot be reclaimed by the
145         * VM.
146         * <p/>
147         * <p>As this is such an important problem in heavy duty systems and
148         * because it is difficult to always guarantee that the remove
149         * method is called before exiting a thread, this method has been
150         * augmented to lazily remove references to dead threads. In
151         * practice, this means that you can be a little sloppy and
152         * occasionally forget to call {@link #remove} before exiting a
153         * thread. However, you must call <code>remove</code> sometime. If
154         * you never call it, then your application is sure to run out of
155         * memory.
156         */
157        public static void remove() {
158            org.apache.logging.log4j.ThreadContext.removeStack();
159        }
160    
161        /**
162         * Set maximum depth of this diagnostic context. If the current
163         * depth is smaller or equal to <code>maxDepth</code>, then no
164         * action is taken.
165         * <p/>
166         * <p>This method is a convenient alternative to multiple {@link
167         * #pop} calls. Moreover, it is often the case that at the end of
168         * complex call sequences, the depth of the NDC is
169         * unpredictable. The <code>setMaxDepth</code> method circumvents
170         * this problem.
171         * <p/>
172         * <p>For example, the combination
173         * <pre>
174         * void foo() {
175         * &nbsp;  int depth = NDC.getDepth();
176         * <p/>
177         * &nbsp;  ... complex sequence of calls
178         * <p/>
179         * &nbsp;  NDC.setMaxDepth(depth);
180         * }
181         * </pre>
182         * <p/>
183         * ensures that between the entry and exit of foo the depth of the
184         * diagnostic stack is conserved.
185         *
186         * @see #getDepth
187         * @param maxDepth The maximum depth of the stack.
188         */
189        public static void setMaxDepth(int maxDepth) {
190            org.apache.logging.log4j.ThreadContext.setMaxDepth(maxDepth);
191        }
192    }

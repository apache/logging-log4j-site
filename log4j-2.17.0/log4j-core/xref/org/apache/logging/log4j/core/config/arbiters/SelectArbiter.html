17  package org.apache.logging.log4j.core.config.arbiters;
18  
19  import java.util.List;
20  import java.util.Optional;
21  
22  import org.apache.logging.log4j.core.config.Node;
23  import org.apache.logging.log4j.core.config.plugins.Plugin;
24  import org.apache.logging.log4j.core.config.plugins.PluginBuilderFactory;
25  
26  /**
27   * Class Description goes here.
28   */
29  @Plugin(name = "Select", category = Node.CATEGORY, elementType = Arbiter.ELEMENT_TYPE, deferChildren = true,
30          printObject = true)
31  public class SelectArbiter {
32  
33      public Arbiter evaluateConditions(List<Arbiter> conditions) {
34          Optional<Arbiter> opt = conditions.stream().filter((c) -> c instanceof DefaultArbiter)
35                  .reduce((a, b) -> {
36                      throw new IllegalStateException("Multiple elements: " + a + ", " + b);
37                  });
38          for (Arbiter condition : conditions) {
39              if (condition instanceof DefaultArbiter) {
40                  continue;
41              }
42              if (condition.isCondition()) {
43                  return condition;
44              }
45          }
46          return opt.orElse(null);
47      }
48  
49      @PluginBuilderFactory
50      public static Builder newBuilder() {
51          return new Builder();
52      }
53  
54      public static class Builder implements org.apache.logging.log4j.core.util.Builder<SelectArbiter> {
55  
56          public Builder asBuilder() {
57              return this;
58          }
59  
60          public SelectArbiter build() {
61              return new SelectArbiter();
62          }
63      }
64  }

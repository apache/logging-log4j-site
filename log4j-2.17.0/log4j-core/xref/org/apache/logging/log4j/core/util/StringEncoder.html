17  package org.apache.logging.log4j.core.util;
18  
19  import java.nio.charset.Charset;
20  
21  /**
22   * Encodes Strings to bytes.
23   *
24   * @since 2.5
25   */
26  public final class StringEncoder {
27  
28      private StringEncoder() {
29      }
30  
31      /**
32       * Converts a String to a byte[].
33       *
34       * @param str if null, return null.
35       * @param charset if null, use the default charset.
36       * @return a byte[]
37       */
38      public static byte[] toBytes(final String str, final Charset charset) {
39          if (str != null) {
40              return str.getBytes(charset != null ? charset : Charset.defaultCharset());
41          }
42          return null;
43      }
44  
45      /**
46       * Prefer standard {@link String#getBytes(Charset)} which performs better in Java 8 and beyond.
47       * Encodes the specified char sequence by casting each character to a byte.
48       *
49       * @param s the char sequence to encode
50       * @return the encoded String
51       * @see <a href="https://issues.apache.org/jira/browse/LOG4J2-1151">LOG4J2-1151</a>
52       * @deprecated No longer necessary given better performance in Java 8
53       */
54      @Deprecated
55      public static byte[] encodeSingleByteChars(final CharSequence s) {
56          final int length = s.length();
57          final byte[] result = new byte[length];
58          encodeString(s, 0, length, result);
59          return result;
60      }
61  
62      // LOG4J2-1151
63      /**
64       * Prefer standard {@link String#getBytes(Charset)} which performs better in Java 8 and beyond.
65       *
66       * Implementation note: this is the fast path. If the char array contains only ISO-8859-1 characters, all the work
67       * will be done here.
68       *
69       * @deprecated No longer necessary given better performance in Java 8
70       */
71      @Deprecated
72      public static int encodeIsoChars(final CharSequence charArray, int charIndex, final byte[] byteArray, int byteIndex, final int length) {
73          int i = 0;
74          for (; i < length; i++) {
75              final char c = charArray.charAt(charIndex++);
76              if (c > 255) {
77                  break;
78              }
79              byteArray[(byteIndex++)] = ((byte) c);
80          }
81          return i;
82      }
83  
84      // LOG4J2-1151
85  
86      /**
87       * Prefer standard {@link String#getBytes(Charset)} which performs better in Java 8 and beyond.
88       * @deprecated No longer necessary given better performance in Java 8
89       */
90      @Deprecated
91      public static int encodeString(final CharSequence charArray, int charOffset, int charLength, final byte[] byteArray) {
92          int byteOffset = 0;
93          int length = Math.min(charLength, byteArray.length);
94          int charDoneIndex = charOffset + length;
95          while (charOffset < charDoneIndex) {
96              final int done = encodeIsoChars(charArray, charOffset, byteArray, byteOffset, length);
97              charOffset += done;
98              byteOffset += done;
99              if (done != length) {
100                 final char c = charArray.charAt(charOffset++);
101                 if ((Character.isHighSurrogate(c)) && (charOffset < charDoneIndex)
102                         && (Character.isLowSurrogate(charArray.charAt(charOffset)))) {
103                     if (charLength > byteArray.length) {
104                         charDoneIndex++;
105                         charLength--;
106                     }
107                     charOffset++;
108                 }
109                 byteArray[(byteOffset++)] = '?';
110                 length = Math.min(charDoneIndex - charOffset, byteArray.length - byteOffset);
111             }
112         }
113         return byteOffset;
114     }
115 }

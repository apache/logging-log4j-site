017package org.apache.log4j.spi;
018
019/**
020
021 The <code>LogManager</code> uses one (and only one)
022 <code>RepositorySelector</code> implementation to select the
023 {@link org.apache.log4j.spi.LoggerRepository} for a particular application context.
024
025 <p>It is the responsibility of the <code>RepositorySelector</code>
026 implementation to track the application context. Log4j makes no
027 assumptions about the application context or on its management.
028
029 <p>See also {@link org.apache.log4j.LogManager LogManager}.
030
031 @since 1.2
032
033 */
034public interface RepositorySelector {
035
036    /**
037     * Returns a {@link org.apache.log4j.spi.LoggerRepository} depending on the
038     * context. Implementers must make sure that a valid (non-null)
039     * LoggerRepository is returned.
040     * @return a LoggerRepository.
041     */
042    LoggerRepository getLoggerRepository();
043}

056    public static String[] concatanateArrays(String[] l, String[] r) {
057        int len = l.length + r.length;
058        String[] a = new String[len];
059
060        System.arraycopy(l, 0, a, 0, l.length);
061        System.arraycopy(r, 0, a, l.length, r.length);
062
063        return a;
064    }

066    public static String convertSpecialChars(String s) {
067        char c;
068        int len = s.length();
069        StringBuilder sbuf = new StringBuilder(len);
070
071        int i = 0;
072        while (i < len) {
073            c = s.charAt(i++);
074            if (c == '\\') {
075                c = s.charAt(i++);
076                for (CharMap entry : charMap) {
077                    if (entry.key == c) {
078                        c = entry.replacement;
079                    }
080                }
081            }
082            sbuf.append(c);
083        }
084        return sbuf.toString();
085    }

089     * Very similar to <code>System.getProperty</code> except
090     * that the {@link SecurityException} is hidden.
091     *
092     * @param key The key to search for.
093     * @param def The default value to return.
094     * @return the string value of the system property, or the default
095     * value if there is no property with that key.
096     * @since 1.1
097     */
098    public static String getSystemProperty(String key, String def) {
099        try {
100            return System.getProperty(key, def);
101        } catch (Throwable e) { // MS-Java throws com.ms.security.SecurityExceptionEx
102            LOGGER.debug("Was not allowed to read system property \"{}\".", key);
103            return def;
104        }
105    }

107    /**
108     * If <code>value</code> is "true", then <code>true</code> is
109     * returned. If <code>value</code> is "false", then
110     * <code>true</code> is returned. Otherwise, <code>default</code> is
111     * returned.
112     *
113     * <p>Case of value is unimportant.
114     * @param value The value to convert.
115     * @param dEfault The default value.
116     * @return the value of the result.
117     */
118    public static boolean toBoolean(String value, boolean dEfault) {
119        if (value == null) {
120            return dEfault;
121        }
122        String trimmedVal = value.trim();
123        if ("true".equalsIgnoreCase(trimmedVal)) {
124            return true;
125        }
126        if ("false".equalsIgnoreCase(trimmedVal)) {
127            return false;
128        }
129        return dEfault;
130    }

132    /**
133     * Converts a standard or custom priority level to a Level
134     * object.  <p> If <code>value</code> is of form
135     * "level#classname", then the specified class' toLevel method
136     * is called to process the specified level string; if no '#'
137     * character is present, then the default {@link org.apache.log4j.Level}
138     * class is used to process the level value.
139     *
140     * <p>As a special case, if the <code>value</code> parameter is
141     * equal to the string "NULL", then the value <code>null</code> will
142     * be returned.
143     *
144     * <p> If any error occurs while converting the value to a level,
145     * the <code>defaultValue</code> parameter, which may be
146     * <code>null</code>, is returned.
147     *
148     * <p> Case of <code>value</code> is insignificant for the level level, but is
149     * significant for the class name part, if present.
150     * @param value The value to convert.
151     * @param defaultValue The default value.
152     * @return the value of the result.
153     *
154     * @since 1.1
155     */
156    public static Level toLevel(String value, Level defaultValue) {
157        if (value == null) {
158            return defaultValue;
159        }
160
161        value = value.trim();
162
163        int hashIndex = value.indexOf('#');
164        if (hashIndex == -1) {
165            if ("NULL".equalsIgnoreCase(value)) {
166                return null;
167            }
168            // no class name specified : use standard Level class
169            return Level.toLevel(value, defaultValue);
170        }
171
172        Level result = defaultValue;
173
174        String clazz = value.substring(hashIndex + 1);
175        String levelName = value.substring(0, hashIndex);
176
177        // This is degenerate case but you never know.
178        if ("NULL".equalsIgnoreCase(levelName)) {
179            return null;
180        }
181
182        LOGGER.debug("toLevel" + ":class=[" + clazz + "]"
183                + ":pri=[" + levelName + "]");
184
185        try {
186            Class<?> customLevel = LoaderUtil.loadClass(clazz);
187
188            // get a ref to the specified class' static method
189            // toLevel(String, org.apache.log4j.Level)
190            Class<?>[] paramTypes = new Class[] { String.class, org.apache.log4j.Level.class };
191            java.lang.reflect.Method toLevelMethod =
192                    customLevel.getMethod("toLevel", paramTypes);
193
194            // now call the toLevel method, passing level string + default
195            Object[] params = new Object[]{levelName, defaultValue};
196            Object o = toLevelMethod.invoke(null, params);
197
198            result = (Level) o;
199        } catch (ClassNotFoundException e) {
200            LOGGER.warn("custom level class [" + clazz + "] not found.");
201        } catch (NoSuchMethodException e) {
202            LOGGER.warn("custom level class [" + clazz + "]"
203                    + " does not have a class function toLevel(String, Level)", e);
204        } catch (java.lang.reflect.InvocationTargetException e) {
205            if (e.getTargetException() instanceof InterruptedException
206                    || e.getTargetException() instanceof InterruptedIOException) {
207                Thread.currentThread().interrupt();
208            }
209            LOGGER.warn("custom level class [" + clazz + "]"
210                    + " could not be instantiated", e);
211        } catch (ClassCastException e) {
212            LOGGER.warn("class [" + clazz
213                    + "] is not a subclass of org.apache.log4j.Level", e);
214        } catch (IllegalAccessException e) {
215            LOGGER.warn("class [" + clazz +
216                    "] cannot be instantiated due to access restrictions", e);
217        } catch (RuntimeException e) {
218            LOGGER.warn("class [" + clazz + "], level [" + levelName +
219                    "] conversion failed.", e);
220        }
221        return result;
222    }

224    /**
225     * Instantiate an object given a class name. Check that the
226     * <code>className</code> is a subclass of
227     * <code>superClass</code>. If that test fails or the object could
228     * not be instantiated, then <code>defaultValue</code> is returned.
229     *
230     * @param className    The fully qualified class name of the object to instantiate.
231     * @param superClass   The class to which the new object should belong.
232     * @param defaultValue The object to return in case of non-fulfillment
233     * @return The created object.
234     */
235    public static Object instantiateByClassName(String className, Class<?> superClass,
236            Object defaultValue) {
237        if (className != null) {
238            try {
239                Object obj = LoaderUtil.newInstanceOf(className);
240                if (!superClass.isAssignableFrom(obj.getClass())) {
241                    LOGGER.error("A \"{}\" object is not assignable to a \"{}\" variable", className,
242                            superClass.getName());
243                    return defaultValue;
244                }
245                return obj;
246            } catch (ClassNotFoundException | NoSuchMethodException | IllegalAccessException
247                    | InstantiationException | InvocationTargetException e) {
248                LOGGER.error("Could not instantiate class [" + className + "].", e);
249            }
250        }
251        return defaultValue;
252    }

255    /**
256     * Perform variable substitution in string <code>val</code> from the
257     * values of keys found in the system propeties.
258     *
259     * <p>The variable substitution delimeters are <b>${</b> and <b>}</b>.
260     *
261     * <p>For example, if the System properties contains "key=value", then
262     * the call
263     * <pre>
264     * String s = OptionConverter.substituteVars("Value of key is ${key}.");
265     * </pre>
266     * <p>
267     * will set the variable <code>s</code> to "Value of key is value.".
268     *
269     * <p>If no value could be found for the specified key, then the
270     * <code>props</code> parameter is searched, if the value could not
271     * be found there, then substitution defaults to the empty string.
272     *
273     * <p>For example, if system propeties contains no value for the key
274     * "inexistentKey", then the call
275     *
276     * <pre>
277     * String s = OptionConverter.subsVars("Value of inexistentKey is [${inexistentKey}]");
278     * </pre>
279     * will set <code>s</code> to "Value of inexistentKey is []"
280     *
281     * <p>An {@link IllegalArgumentException} is thrown if
282     * <code>val</code> contains a start delimeter "${" which is not
283     * balanced by a stop delimeter "}". </p>
284     *
285     * <p><b>Author</b> Avy Sharell</p>
286     *
287     * @param val The string on which variable substitution is performed.
288     * @param props The properties to use for the substitution.
289     * @return The substituted string.
290     * @throws IllegalArgumentException if <code>val</code> is malformed.
291     */
292    public static String substVars(String val, Properties props) throws IllegalArgumentException {
293
294        StringBuilder sbuf = new StringBuilder();
295
296        int i = 0;
297        int j, k;
298
299        while (true) {
300            j = val.indexOf(DELIM_START, i);
301            if (j == -1) {
302                // no more variables
303                if (i == 0) { // this is a simple string
304                    return val;
305                }
306                sbuf.append(val.substring(i, val.length()));
307                return sbuf.toString();
308            }
309            sbuf.append(val.substring(i, j));
310            k = val.indexOf(DELIM_STOP, j);
311            if (k == -1) {
312                throw new IllegalArgumentException('"' + val +
313                        "\" has no closing brace. Opening brace at position " + j
314                        + '.');
315            }
316            j += DELIM_START_LEN;
317            String key = val.substring(j, k);
318            // first try in System properties
319            String replacement = getSystemProperty(key, null);
320            // then try props parameter
321            if (replacement == null && props != null) {
322                replacement = props.getProperty(key);
323            }
324
325            if (replacement != null) {
326                // Do variable substitution on the replacement string
327                // such that we can solve "Hello ${x2}" as "Hello p1"
328                // the where the properties are
329                // x1=p1
330                // x2=${x1}
331                String recursiveReplacement = substVars(replacement, props);
332                sbuf.append(recursiveReplacement);
333            }
334            i = k + DELIM_STOP_LEN;
335        }
336    }

338    public static org.apache.logging.log4j.Level convertLevel(String level,
339            org.apache.logging.log4j.Level defaultLevel) {
340        Level l = toLevel(level, null);
341        return l != null ? convertLevel(l) : defaultLevel;
342    }

344    public static  org.apache.logging.log4j.Level convertLevel(Level level) {
345        if (level == null) {
346            return org.apache.logging.log4j.Level.ERROR;
347        }
348        if (level.isGreaterOrEqual(Level.FATAL)) {
349            return org.apache.logging.log4j.Level.FATAL;
350        } else if (level.isGreaterOrEqual(Level.ERROR)) {
351            return org.apache.logging.log4j.Level.ERROR;
352        } else if (level.isGreaterOrEqual(Level.WARN)) {
353            return org.apache.logging.log4j.Level.WARN;
354        } else if (level.isGreaterOrEqual(Level.INFO)) {
355            return org.apache.logging.log4j.Level.INFO;
356        } else if (level.isGreaterOrEqual(Level.DEBUG)) {
357            return org.apache.logging.log4j.Level.DEBUG;
358        } else if (level.isGreaterOrEqual(Level.TRACE)) {
359            return org.apache.logging.log4j.Level.TRACE;
360        }
361        return org.apache.logging.log4j.Level.ALL;
362    }
363
364    public static Level convertLevel(org.apache.logging.log4j.Level level) {
365        if (level == null) {
366            return Level.ERROR;
367        }
368        switch (level.getStandardLevel()) {
369            case FATAL:
370                return Level.FATAL;
371            case WARN:
372                return Level.WARN;
373            case INFO:
374                return Level.INFO;
375            case DEBUG:
376                return Level.DEBUG;
377            case TRACE:
378                return Level.TRACE;
379            case ALL:
380                return Level.ALL;
381            case OFF:
382                return Level.OFF;
383            default:
384                return Level.ERROR;
385        }
386    }

388    /**
389     * Find the value corresponding to <code>key</code> in
390     * <code>props</code>. Then perform variable substitution on the
391     * found value.
392     * @param key The key used to locate the substitution string.
393     * @param props The properties to use in the substitution.
394     * @return The substituted string.
395     */
396    public static String findAndSubst(String key, Properties props) {
397        String value = props.getProperty(key);
398        if (value == null) {
399            return null;
400        }
401
402        try {
403            return substVars(value, props);
404        } catch (IllegalArgumentException e) {
405            LOGGER.error("Bad option value [{}].", value, e);
406            return value;
407        }
408    }

415            this.key = key;
416            this.replacement = replacement;

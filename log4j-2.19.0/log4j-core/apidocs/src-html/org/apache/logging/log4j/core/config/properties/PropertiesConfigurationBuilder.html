049/**
050 * Helper builder for parsing properties files into a PropertiesConfiguration.
051 *
052 * @since 2.6
053 */
054public class PropertiesConfigurationBuilder extends ConfigurationBuilderFactory
055    implements Builder<PropertiesConfiguration> {
056
057    private static final String ADVERTISER_KEY = "advertiser";
058    private static final String STATUS_KEY = "status";
059    private static final String SHUTDOWN_HOOK = "shutdownHook";
060    private static final String SHUTDOWN_TIMEOUT = "shutdownTimeout";
061    private static final String VERBOSE = "verbose";
062    private static final String DEST = "dest";
063    private static final String PACKAGES = "packages";
064    private static final String CONFIG_NAME = "name";
065    private static final String MONITOR_INTERVAL = "monitorInterval";
066    private static final String CONFIG_TYPE = "type";
067
068    private final ConfigurationBuilder<PropertiesConfiguration> builder;
069    private LoggerContext loggerContext;
070    private Properties rootProperties;
071
072    public PropertiesConfigurationBuilder() {
073        this.builder = newConfigurationBuilder(PropertiesConfiguration.class);
074    }

076    public PropertiesConfigurationBuilder setRootProperties(final Properties rootProperties) {
077        this.rootProperties = rootProperties;
078        return this;
079    }

081    public PropertiesConfigurationBuilder setConfigurationSource(final ConfigurationSource source) {
082        builder.setConfigurationSource(source);
083        return this;
084    }

086    @Override
087    public PropertiesConfiguration build() {
088        for (final String key : rootProperties.stringPropertyNames()) {
089            if (!key.contains(".")) {
090                builder.addRootProperty(key, rootProperties.getProperty(key));
091            }
092        }
093        builder
094            .setStatusLevel(Level.toLevel(rootProperties.getProperty(STATUS_KEY), Level.ERROR))
095            .setShutdownHook(rootProperties.getProperty(SHUTDOWN_HOOK))
096            .setShutdownTimeout(Long.parseLong(rootProperties.getProperty(SHUTDOWN_TIMEOUT, "0")), TimeUnit.MILLISECONDS)
097            .setVerbosity(rootProperties.getProperty(VERBOSE))
098            .setDestination(rootProperties.getProperty(DEST))
099            .setPackages(rootProperties.getProperty(PACKAGES))
100            .setConfigurationName(rootProperties.getProperty(CONFIG_NAME))
101            .setMonitorInterval(rootProperties.getProperty(MONITOR_INTERVAL, "0"))
102            .setAdvertiser(rootProperties.getProperty(ADVERTISER_KEY));
103
104        final Properties propertyPlaceholders = PropertiesUtil.extractSubset(rootProperties, "property");
105        for (final String key : propertyPlaceholders.stringPropertyNames()) {
106            builder.addProperty(key, propertyPlaceholders.getProperty(key));
107        }
108
109        final Map<String, Properties> scripts = PropertiesUtil.partitionOnCommonPrefixes(
110            PropertiesUtil.extractSubset(rootProperties, "script"));
111        for (final Map.Entry<String, Properties> entry : scripts.entrySet()) {
112            final Properties scriptProps = entry.getValue();
113            final String type = (String) scriptProps.remove("type");
114            if (type == null) {
115                throw new ConfigurationException("No type provided for script - must be Script or ScriptFile");
116            }
117            if (type.equalsIgnoreCase("script")) {
118                builder.add(createScript(scriptProps));
119            } else {
120                builder.add(createScriptFile(scriptProps));
121            }
122        }
123
124        final Properties levelProps = PropertiesUtil.extractSubset(rootProperties, "customLevel");
125        if (levelProps.size() > 0) {
126            for (final String key : levelProps.stringPropertyNames()) {
127                builder.add(builder.newCustomLevel(key, Integers.parseInt(levelProps.getProperty(key))));
128            }
129        }
130
131        final String filterProp = rootProperties.getProperty("filters");
132        if (filterProp != null) {
133            final String[] filterNames = filterProp.split(",");
134            for (final String filterName : filterNames) {
135                final String name = filterName.trim();
136                builder.add(createFilter(name, PropertiesUtil.extractSubset(rootProperties, "filter." + name)));
137            }
138        } else {
139
140            final Map<String, Properties> filters = PropertiesUtil
141                    .partitionOnCommonPrefixes(PropertiesUtil.extractSubset(rootProperties, "filter"));
142            for (final Map.Entry<String, Properties> entry : filters.entrySet()) {
143                builder.add(createFilter(entry.getKey().trim(), entry.getValue()));
144            }
145        }
146
147        final String appenderProp = rootProperties.getProperty("appenders");
148        if (appenderProp != null) {
149            final String[] appenderNames = appenderProp.split(",");
150            for (final String appenderName : appenderNames) {
151                final String name = appenderName.trim();
152                builder.add(createAppender(appenderName.trim(),
153                        PropertiesUtil.extractSubset(rootProperties, "appender." + name)));
154            }
155        } else {
156            final Map<String, Properties> appenders = PropertiesUtil
157                    .partitionOnCommonPrefixes(PropertiesUtil.extractSubset(rootProperties, Appender.ELEMENT_TYPE));
158            for (final Map.Entry<String, Properties> entry : appenders.entrySet()) {
159                builder.add(createAppender(entry.getKey().trim(), entry.getValue()));
160            }
161        }
162
163        final String loggerProp = rootProperties.getProperty("loggers");
164        if (loggerProp != null) {
165            final String[] loggerNames = loggerProp.split(",");
166            for (final String loggerName : loggerNames) {
167                final String name = loggerName.trim();
168                if (!name.equals(LoggerConfig.ROOT)) {
169                    builder.add(createLogger(name, PropertiesUtil.extractSubset(rootProperties, "logger." +
170                            name)));
171                }
172            }
173        } else {
174
175            final Map<String, Properties> loggers = PropertiesUtil
176                    .partitionOnCommonPrefixes(PropertiesUtil.extractSubset(rootProperties, "logger"), true);
177            for (final Map.Entry<String, Properties> entry : loggers.entrySet()) {
178                final String name = entry.getKey().trim();
179                if (!name.equals(LoggerConfig.ROOT)) {
180                    builder.add(createLogger(name, entry.getValue()));
181                }
182            }
183        }
184
185        String rootProp = rootProperties.getProperty("rootLogger");
186        final Properties props = PropertiesUtil.extractSubset(rootProperties, "rootLogger");
187        if (rootProp != null) {
188            props.setProperty("", rootProp);
189            rootProperties.remove("rootLogger");
190        }
191        if (props.size() > 0) {
192            builder.add(createRootLogger(props));
193        }
194
195        builder.setLoggerContext(loggerContext);
196
197        return builder.build(false);
198    }

200    private ScriptComponentBuilder createScript(final Properties properties) {
201        final String name = (String) properties.remove("name");
202        final String language = (String) properties.remove("language");
203        final String text = (String) properties.remove("text");
204        final ScriptComponentBuilder scriptBuilder = builder.newScript(name, language, text);
205        return processRemainingProperties(scriptBuilder, properties);
206    }

209    private ScriptFileComponentBuilder createScriptFile(final Properties properties) {
210        final String name = (String) properties.remove("name");
211        final String path = (String) properties.remove("path");
212        final ScriptFileComponentBuilder scriptFileBuilder = builder.newScriptFile(name, path);
213        return processRemainingProperties(scriptFileBuilder, properties);
214    }
215
216    private AppenderComponentBuilder createAppender(final String key, final Properties properties) {
217        final String name = (String) properties.remove(CONFIG_NAME);
218        if (Strings.isEmpty(name)) {
219            throw new ConfigurationException("No name attribute provided for Appender " + key);
220        }
221        final String type = (String) properties.remove(CONFIG_TYPE);
222        if (Strings.isEmpty(type)) {
223            throw new ConfigurationException("No type attribute provided for Appender " + key);
224        }
225        final AppenderComponentBuilder appenderBuilder = builder.newAppender(name, type);
226        addFiltersToComponent(appenderBuilder, properties);
227        final Properties layoutProps = PropertiesUtil.extractSubset(properties, "layout");
228        if (layoutProps.size() > 0) {
229            appenderBuilder.add(createLayout(name, layoutProps));
230        }
231
232        return processRemainingProperties(appenderBuilder, properties);
233    }
234
235    private FilterComponentBuilder createFilter(final String key, final Properties properties) {
236        final String type = (String) properties.remove(CONFIG_TYPE);
237        if (Strings.isEmpty(type)) {
238            throw new ConfigurationException("No type attribute provided for Filter " + key);
239        }
240        final String onMatch = (String) properties.remove(AbstractFilterBuilder.ATTR_ON_MATCH);
241        final String onMismatch = (String) properties.remove(AbstractFilterBuilder.ATTR_ON_MISMATCH);
242        final FilterComponentBuilder filterBuilder = builder.newFilter(type, onMatch, onMismatch);
243        return processRemainingProperties(filterBuilder, properties);
244    }
245
246    private AppenderRefComponentBuilder createAppenderRef(final String key, final Properties properties) {
247        final String ref = (String) properties.remove("ref");
248        if (Strings.isEmpty(ref)) {
249            throw new ConfigurationException("No ref attribute provided for AppenderRef " + key);
250        }
251        final AppenderRefComponentBuilder appenderRefBuilder = builder.newAppenderRef(ref);
252        final String level = Strings.trimToNull((String) properties.remove("level"));
253        if (!Strings.isEmpty(level)) {
254            appenderRefBuilder.addAttribute("level", level);
255        }
256        return addFiltersToComponent(appenderRefBuilder, properties);
257    }
258
259    private LoggerComponentBuilder createLogger(final String key, final Properties properties) {
260        final String levelAndRefs = properties.getProperty("");
261        final String name = (String) properties.remove(CONFIG_NAME);
262        final String location = (String) properties.remove("includeLocation");
263        if (Strings.isEmpty(name)) {
264            throw new ConfigurationException("No name attribute provided for Logger " + key);
265        }
266        final String level = Strings.trimToNull((String) properties.remove("level"));
267        final String type = (String) properties.remove(CONFIG_TYPE);
268        final LoggerComponentBuilder loggerBuilder;
269        boolean includeLocation;
270        if (type != null) {
271            if (type.equalsIgnoreCase("asyncLogger")) {
272                if (location != null) {
273                    includeLocation = Boolean.parseBoolean(location);
274                    loggerBuilder = builder.newAsyncLogger(name, level, includeLocation);
275                } else {
276                    loggerBuilder = builder.newAsyncLogger(name, level);
277                }
278            } else {
279                throw new ConfigurationException("Unknown Logger type " + type + " for Logger " + name);
280            }
281        } else if (location != null) {
282            includeLocation = Boolean.parseBoolean(location);
283            loggerBuilder = builder.newLogger(name, level, includeLocation);
284        } else {
285            loggerBuilder = builder.newLogger(name, level);
286        }
287        addLoggersToComponent(loggerBuilder, properties);
288        addFiltersToComponent(loggerBuilder, properties);
289        final String additivity = (String) properties.remove("additivity");
290        if (!Strings.isEmpty(additivity)) {
291            loggerBuilder.addAttribute("additivity", additivity);
292        }
293        if (levelAndRefs != null) {
294            loggerBuilder.addAttribute("levelAndRefs", levelAndRefs);
295        }
296        return loggerBuilder;
297    }
298
299    private RootLoggerComponentBuilder createRootLogger(final Properties properties) {
300        final String levelAndRefs = properties.getProperty("");
301        final String level = Strings.trimToNull((String) properties.remove("level"));
302        final String type = (String) properties.remove(CONFIG_TYPE);
303        final String location = (String) properties.remove("includeLocation");
304        final boolean includeLocation;
305        final RootLoggerComponentBuilder loggerBuilder;
306        if (type != null) {
307            if (type.equalsIgnoreCase("asyncRoot")) {
308                if (location != null) {
309                    includeLocation = Boolean.parseBoolean(location);
310                    loggerBuilder = builder.newAsyncRootLogger(level, includeLocation);
311                } else {
312                    loggerBuilder = builder.newAsyncRootLogger(level);
313                }
314            } else {
315                throw new ConfigurationException("Unknown Logger type for root logger" + type);
316            }
317        } else if (location != null) {
318            includeLocation = Boolean.parseBoolean(location);
319            loggerBuilder = builder.newRootLogger(level, includeLocation);
320        } else {
321            loggerBuilder = builder.newRootLogger(level);
322        }
323        addLoggersToComponent(loggerBuilder, properties);
324        if (levelAndRefs != null) {
325            loggerBuilder.addAttribute("levelAndRefs", levelAndRefs);
326        }
327        return addFiltersToComponent(loggerBuilder, properties);
328    }

330    private LayoutComponentBuilder createLayout(final String appenderName, final Properties properties) {
331        final String type = (String) properties.remove(CONFIG_TYPE);
332        if (Strings.isEmpty(type)) {
333            throw new ConfigurationException("No type attribute provided for Layout on Appender " + appenderName);
334        }
335        final LayoutComponentBuilder layoutBuilder = builder.newLayout(type);
336        return processRemainingProperties(layoutBuilder, properties);
337    }
338
339    private static <B extends ComponentBuilder<B>> ComponentBuilder<B> createComponent(final ComponentBuilder<?> parent,
340                                                                                       final String key,
341                                                                                       final Properties properties) {
342        final String name = (String) properties.remove(CONFIG_NAME);
343        final String type = (String) properties.remove(CONFIG_TYPE);
344        if (Strings.isEmpty(type)) {
345            throw new ConfigurationException("No type attribute provided for component " + key);
346        }
347        final ComponentBuilder<B> componentBuilder = parent.getBuilder().newComponent(name, type);
348        return processRemainingProperties(componentBuilder, properties);
349    }
350
351    private static <B extends ComponentBuilder<?>> B processRemainingProperties(final B builder,
352                                                                                final Properties properties) {
353        while (properties.size() > 0) {
354            final String propertyName = properties.stringPropertyNames().iterator().next();
355            final int index = propertyName.indexOf('.');
356            if (index > 0) {
357                final String prefix = propertyName.substring(0, index);
358                final Properties componentProperties = PropertiesUtil.extractSubset(properties, prefix);
359                builder.addComponent(createComponent(builder, prefix, componentProperties));
360            } else {
361                builder.addAttribute(propertyName, properties.getProperty(propertyName));
362                properties.remove(propertyName);
363            }
364        }
365        return builder;
366    }
367
368    private <B extends FilterableComponentBuilder<? extends ComponentBuilder<?>>> B addFiltersToComponent(
369        final B componentBuilder, final Properties properties) {
370        final Map<String, Properties> filters = PropertiesUtil.partitionOnCommonPrefixes(
371            PropertiesUtil.extractSubset(properties, "filter"));
372        for (final Map.Entry<String, Properties> entry : filters.entrySet()) {
373            componentBuilder.add(createFilter(entry.getKey().trim(), entry.getValue()));
374        }
375        return componentBuilder;
376    }
377
378    private <B extends LoggableComponentBuilder<? extends ComponentBuilder<?>>> B addLoggersToComponent(
379        final B loggerBuilder, final Properties properties) {
380        final Map<String, Properties> appenderRefs = PropertiesUtil.partitionOnCommonPrefixes(
381            PropertiesUtil.extractSubset(properties, "appenderRef"));
382        for (final Map.Entry<String, Properties> entry : appenderRefs.entrySet()) {
383            loggerBuilder.add(createAppenderRef(entry.getKey().trim(), entry.getValue()));
384        }
385        return loggerBuilder;
386    }

388    public PropertiesConfigurationBuilder setLoggerContext(final LoggerContext loggerContext) {
389        this.loggerContext = loggerContext;
390        return this;
391    }
392
393    public LoggerContext getLoggerContext() {
394        return loggerContext;
395    }

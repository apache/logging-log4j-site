061/**
062 * Override Spring's implementation of the Log4j 2 Logging System to properly support Spring Cloud Config.
063 */
064public class Log4j2SpringBootLoggingSystem extends Log4J2LoggingSystem {
065
066    /**
067     * Property that disables the usage of this {@link LoggingSystem}.
068     */
069    public static final String LOG4J2_DISABLE_CLOUD_CONFIG_LOGGING_SYSTEM = "log4j2.disableCloudConfigLoggingSystem";
070
071    public static final String ENVIRONMENT_KEY = "SpringEnvironment";
072    private static final String HTTPS = "https";
073    private static final String OVERRIDE_PARAM = "override";
074    private static Logger LOGGER = StatusLogger.getLogger();
075    private static final int PRECEDENCE = 0;
076
077    public Log4j2SpringBootLoggingSystem(ClassLoader loader) {
078        super(loader);
079    }
080
081    /**
082     * Set the environment into the ExternalContext field so that it can be obtained by SpringLookup when it
083     * is constructed. Spring will replace the ExternalContext field with a String once initialization is
084     * complete.
085     * @param initializationContext The initialization context.
086     * @param configLocation The configuration location.
087     * @param logFile the log file.
088     */
089    @Override
090    public void initialize(LoggingInitializationContext initializationContext, String configLocation, LogFile logFile) {
091        Environment environment = initializationContext.getEnvironment();
092        PropertiesUtil.getProperties().addPropertySource(new SpringPropertySource(environment));
093        getLoggerContext().putObjectIfAbsent(ENVIRONMENT_KEY, environment);
094        super.initialize(initializationContext, configLocation, logFile);
095    }
096
097    @Override
098    protected String[] getStandardConfigLocations() {
099        String[] locations = super.getStandardConfigLocations();
100        PropertiesUtil props = new PropertiesUtil(new Properties());
101        String location = props.getStringProperty(ConfigurationFactory.CONFIGURATION_FILE_PROPERTY);
102        if (location != null) {
103            List<String> list = new ArrayList<>(Arrays.asList(super.getStandardConfigLocations()));
104            list.add(location);
105            locations = list.toArray(Strings.EMPTY_ARRAY);
106        }
107        return locations;
108    }
109
110    @Override
111    protected void loadDefaults(LoggingInitializationContext initializationContext, LogFile logFile) {
112        if (logFile != null) {
113            this.loadConfiguration(this.getBootPackagedConfigFile("log4j2-file.xml"), logFile);
114        } else {
115            this.loadConfiguration(this.getBootPackagedConfigFile("log4j2.xml"), logFile);
116        }
117    }
118
119    private String getBootPackagedConfigFile(String fileName) {
120        String defaultPath = ClassUtils.getPackageName(Log4J2LoggingSystem.class);
121        defaultPath = defaultPath.replace('.', '/');
122        defaultPath = defaultPath + "/" + fileName;
123        defaultPath = "classpath:" + defaultPath;
124        return defaultPath;
125    }
126
127    /**
128     * This method is removed from Spring in 3.x and is scheduled to be removed in 2.8.x. It must be left in
129     * to support older Spring releases.
130     * @param location the location
131     * @param logFile log file configuration
132     */
133    @Override
134    protected void loadConfiguration(String location, LogFile logFile) {
135        loadConfiguration(location, logFile, Collections.emptyList());
136    }
137
138    /**
139     * Added in Spring 2.6.0, the overrides parameter allows override files to be specified in the
140     * "logging.log4j2.config.override" property. However, spring does not support passing credentials
141     * when accessing the location. We do.
142     * @param location The location of the primary configuration.
143     * @param logFile log file configuration.
144     * @param overrides Any override files.
145     */
146    @Override
147    protected void loadConfiguration(String location, LogFile logFile, List<String> overrides) {
148        Assert.notNull(location, "Location must not be null");
149        try {
150            LoggerContext ctx = getLoggerContext();
151            List<String> locations = parseConfigLocations(location);
152            if (overrides != null) {
153                locations.addAll(overrides);
154            }
155            if (locations.size() == 1) {
156                final URL url = ResourceUtils.getURL(location);
157                final ConfigurationSource source = getConfigurationSource(url);
158                if (source != null) {
159                    ctx.start(ConfigurationFactory.getInstance().getConfiguration(ctx, source));
160                }
161            } else {
162                final List<AbstractConfiguration> configs = new ArrayList<>();
163                boolean first = true;
164                for (final String sourceLocation : locations) {
165                    final ConfigurationSource source = getConfigurationSource(ResourceUtils.getURL(sourceLocation));
166                    if (source != null) {
167                        try {
168                            final Configuration config = ConfigurationFactory.getInstance().getConfiguration(ctx, source);
169                            if (config instanceof AbstractConfiguration) {
170                                configs.add((AbstractConfiguration) config);
171                            } else {
172                                LOGGER.warn("Configuration at {} cannot be combined in a CompositeConfiguration", sourceLocation);
173                                return;
174                            }
175                        } catch (Exception ex) {
176                            if (!first) {
177                                LOGGER.warn("Error accessing {}: {}. Ignoring override", sourceLocation, ex.getMessage());
178                            } else {
179                                throw ex;
180                            }
181                        }
182                    }
183                    first = false;
184                }
185                if (configs.size() > 1) {
186                    ctx.start(new CompositeConfiguration(configs));
187                } else {
188                    ctx.start(configs.get(0));
189                }
190            }
191        }
192        catch (Exception ex) {
193            throw new IllegalStateException(
194                    "Could not initialize Log4J2 logging from " + location, ex);
195        }
196    }
197
198    @Override
199    public void cleanUp() {
200        getLoggerContext().removeObject(ENVIRONMENT_KEY);
201        super.cleanUp();
202    }
203
204    private List<String> parseConfigLocations(String configLocations) {
205        final String[] uris = configLocations.split("\\?");
206        final List<String> locations = new ArrayList<>();
207        locations.add(uris[0]);
208        if (uris.length > 1) {
209            try {
210                final URL url = new URL(configLocations);
211                final String[] pairs = url.getQuery().split("&");
212                for (String pair : pairs) {
213                    final int idx = pair.indexOf("=");
214                    try {
215                        final String key = idx > 0 ? URLDecoder.decode(pair.substring(0, idx), "UTF-8") : pair;
216                        if (key.equalsIgnoreCase(OVERRIDE_PARAM)) {
217                            locations.add(URLDecoder.decode(pair.substring(idx + 1), "UTF-8"));
218                        }
219                    } catch (UnsupportedEncodingException ex) {
220                        LOGGER.warn("Bad data in configuration string: {}", pair);
221                    }
222                }
223                return locations;
224            } catch (MalformedURLException ex) {
225                LOGGER.warn("Unable to parse configuration URL {}", configLocations);
226            }
227        }
228        return locations;
229    }
230
231    private ConfigurationSource getConfigurationSource(URL url) throws IOException, URISyntaxException {
232        AuthorizationProvider provider = ConfigurationFactory.authorizationProvider(PropertiesUtil.getProperties());
233        SslConfiguration sslConfiguration = url.getProtocol().equals(HTTPS)
234                ? SslConfigurationFactory.getSslConfiguration() : null;
235        URLConnection urlConnection = UrlConnectionFactory.createConnection(url, 0, sslConfiguration,
236                provider);
237
238        File file = FileUtils.fileFromUri(url.toURI());
239        try {
240            if (file != null) {
241                return new ConfigurationSource(urlConnection.getInputStream(), FileUtils.fileFromUri(url.toURI()));
242            }
243            return new ConfigurationSource(urlConnection.getInputStream(), url, urlConnection.getLastModified());
244        } catch (FileNotFoundException ex) {
245            LOGGER.info("Unable to locate file {}, ignoring.", url.toString());
246            return null;
247        }
248    }
249
250    private LoggerContext getLoggerContext() {
251        return (LoggerContext) LogManager.getContext(false);
252    }
253
254    @Order(PRECEDENCE)
255    public static class Factory implements LoggingSystemFactory {
256
257        @Override
258        public LoggingSystem getLoggingSystem(ClassLoader classLoader) {
259            if (PropertiesUtil.getProperties().getBooleanProperty(LOG4J2_DISABLE_CLOUD_CONFIG_LOGGING_SYSTEM)) {
260                return null;
261            }
262            return new Log4j2SpringBootLoggingSystem(classLoader);
263        }
264
265    }

093    public static <V, C extends TemplateResolverContext<V, C>> TemplateResolver<V> ofTemplate(
094            final C context,
095            final String template) {
096
097        // Check arguments.
098        Objects.requireNonNull(context, "context");
099        Objects.requireNonNull(template, "template");
100
101        // Read the template.
102        Object node;
103        try {
104            node = JsonReader.read(template);
105        } catch (final Exception error) {
106            final String message = String.format("failed parsing template (template=%s)", template);
107            throw new RuntimeException(message, error);
108        }
109
110        // Perform contextual interception.
111        final List<? extends TemplateResolverInterceptor<V, C>> interceptors =
112                context.getResolverInterceptors();
113        // noinspection ForLoopReplaceableByForEach
114        for (int interceptorIndex = 0;
115             interceptorIndex < interceptors.size();
116             interceptorIndex++) {
117            final TemplateResolverInterceptor<V, C> interceptor =
118                    interceptors.get(interceptorIndex);
119            node = interceptor.processTemplateBeforeResolverInjection(context, node);
120        }
121
122        // Resolve the template.
123        return ofObject(context, node);
124
125    }

127    static <V, C extends TemplateResolverContext<V, C>> TemplateResolver<V> ofObject(
128            final C context,
129            final Object object) {
130        if (object == null) {
131            @SuppressWarnings("unchecked")
132            final TemplateResolver<V> nullResolver = (TemplateResolver<V>) NULL_RESOLVER;
133            return nullResolver;
134        } else if (object instanceof List) {
135            @SuppressWarnings("unchecked")
136            final List<Object> list = (List<Object>) object;
137            return ofList(context, list);
138        } else if (object instanceof Map) {
139            @SuppressWarnings("unchecked")
140            final Map<String, Object> map = (Map<String, Object>) object;
141            return ofMap(context, map);
142        } else if (object instanceof String) {
143            final String string = (String) object;
144            return ofString(context, string);
145        } else if (object instanceof Number) {
146            final Number number = (Number) object;
147            return ofNumber(number);
148        } else if (object instanceof Boolean) {
149            final boolean value = (boolean) object;
150            return ofBoolean(value);
151        } else {
152            final String message = String.format(
153                    "invalid JSON node type (class=%s)",
154                    object.getClass().getName());
155            throw new IllegalArgumentException(message);
156        }
157    }

159    private static <V, C extends TemplateResolverContext<V, C>> TemplateResolver<V> ofList(
160            final C context,
161            final List<Object> list) {
162
163        // Create resolver for each child.
164        final List<TemplateResolver<V>> itemResolvers = list
165                .stream()
166                .map(item -> {
167                    final TemplateResolver<V> itemResolver = ofObject(context, item);
168                    if (itemResolver.isFlattening()) {
169                        throw new IllegalArgumentException(
170                                "flattening resolvers are not allowed in lists");
171                    }
172                    return itemResolver;
173                })
174                .collect(Collectors.toList());
175
176        // Short-circuit if the array is empty.
177        if (itemResolvers.isEmpty()) {
178            @SuppressWarnings("unchecked")
179            final TemplateResolver<V> emptyArrayResolver =
180                    (TemplateResolver<V>) EMPTY_ARRAY_RESOLVER;
181            return emptyArrayResolver;
182        }
183
184        // Create a parent resolver collecting each child resolver execution.
185        return new ArrayResolver<>(itemResolvers);
186
187    }

189    private static final class ArrayResolver<V> implements TemplateResolver<V> {
190
191        private final List<TemplateResolver<V>> itemResolvers;
192
193        private ArrayResolver(final List<TemplateResolver<V>> itemResolvers) {
194            this.itemResolvers = itemResolvers;
195        }
196
197        @Override
198        public void resolve(final V value, final JsonWriter jsonWriter) {
199            jsonWriter.writeArrayStart();
200            for (int itemResolverIndex = 0;
201                 itemResolverIndex < itemResolvers.size();
202                 itemResolverIndex++) {
203                if (itemResolverIndex > 0) {
204                    jsonWriter.writeSeparator();
205                }
206                final TemplateResolver<V> itemResolver =
207                        itemResolvers.get(itemResolverIndex);
208                itemResolver.resolve(value, jsonWriter);
209            }
210            jsonWriter.writeArrayEnd();
211        }
212
213    }

215    private static <V, C extends TemplateResolverContext<V, C>> TemplateResolver<V> ofMap(
216            final C context,
217            final Map<String, Object> map) {
218
219        // Check if this is a resolver request.
220        if (map.containsKey(RESOLVER_FIELD_NAME)) {
221            return ofResolver(context, map);
222        }
223
224        // Collect field resolver contexts.
225        List<FieldResolverContext<V>> fieldResolverContexts =
226                populateFieldResolverMethods(context, map);
227
228        // Short-circuit if the object is empty.
229        final int fieldCount = fieldResolverContexts.size();
230        if (fieldCount == 0) {
231            @SuppressWarnings("unchecked")
232            final TemplateResolver<V> emptyObjectResolver =
233                    (TemplateResolver<V>) EMPTY_OBJECT_RESOLVER;
234            return emptyObjectResolver;
235        }
236
237        // Create the resolver.
238        return new MapResolver<>(fieldResolverContexts);
239
240    }

242    private static <V, C extends TemplateResolverContext<V, C>> List<FieldResolverContext<V>> populateFieldResolverMethods(
243            final C context,
244            final Map<String, Object> map) {
245
246        // Create resolver for each object field.
247        final List<String> fieldNames = new ArrayList<>();
248        final List<TemplateResolver<V>> fieldResolvers = new ArrayList<>();
249        map.forEach((fieldName, fieldValue) -> {
250            final TemplateResolver<V> fieldResolver = ofObject(context, fieldValue);
251            final boolean resolvable = fieldResolver.isResolvable();
252            if (resolvable) {
253                fieldNames.add(fieldName);
254                fieldResolvers.add(fieldResolver);
255            }
256        });
257
258        // Prepare field names to avoid escape and truncation costs at runtime.
259        final List<String> fieldPrefixes = fieldNames
260                .stream()
261                .map(fieldName -> {
262                    try (JsonWriter jsonWriter = context.getJsonWriter()) {
263                        jsonWriter.writeString(fieldName);
264                        jsonWriter.getStringBuilder().append(':');
265                        return jsonWriter.getStringBuilder().toString();
266                    }
267                })
268                .collect(Collectors.toList());
269
270        // Collect field resolver contexts.
271        final int fieldCount = fieldNames.size();
272        return IntStream
273                .range(0, fieldCount)
274                .mapToObj(fieldIndex -> {
275                    final TemplateResolver<V> fieldResolver = fieldResolvers.get(fieldIndex);
276                    final FieldResolverMethod<V> fieldResolverMethod;
277                    final boolean flattening = fieldResolver.isFlattening();
278                    if (flattening) {
279                        fieldResolverMethod = new FlatteningFieldResolverMethod<>(fieldResolver);
280                    } else {
281                        final String fieldPrefix = fieldPrefixes.get(fieldIndex);
282                        fieldResolverMethod = new PrefixedFieldResolverMethod<>(fieldPrefix, fieldResolver);
283                    }
284                    return new FieldResolverContext<>(fieldResolver, fieldResolverMethod);
285                })
286                .collect(Collectors.toList());
287
288    }

290    private static final class FieldResolverContext<V> {
291
292        private final TemplateResolver<V> resolver;
293
294        private final FieldResolverMethod<V> resolverMethod;
295
296        private FieldResolverContext(final TemplateResolver<V> resolver, final FieldResolverMethod<V> resolverMethod) {
297            this.resolver = resolver;
298            this.resolverMethod = resolverMethod;
299        }
300
301    }
302
303    @FunctionalInterface
304    private interface FieldResolverMethod<V> {
305
306        boolean resolve(V value, JsonWriter jsonWriter, boolean succeedingEntry);
307
308    }
309
310    private static final class FlatteningFieldResolverMethod<V> implements FieldResolverMethod<V> {
311
312        private final TemplateResolver<V> fieldResolver;
313
314        private FlatteningFieldResolverMethod(final TemplateResolver<V> fieldResolver) {
315            this.fieldResolver = fieldResolver;
316        }
317
318        @Override
319        public boolean resolve(final V value, final JsonWriter jsonWriter, final boolean succeedingEntry) {
320            final boolean resolvable = fieldResolver.isResolvable(value);
321            if (!resolvable) {
322                return false;
323            }
324            final StringBuilder jsonWriterStringBuilder = jsonWriter.getStringBuilder();
325            final int initLength = jsonWriterStringBuilder.length();
326            fieldResolver.resolve(value, jsonWriter, succeedingEntry);
327            return jsonWriterStringBuilder.length() > initLength;
328        }
329
330    }
331
332    private static final class PrefixedFieldResolverMethod<V> implements FieldResolverMethod<V> {
333
334        private final String fieldPrefix;
335
336        private final TemplateResolver<V> fieldResolver;
337
338        private PrefixedFieldResolverMethod(final String fieldPrefix, final TemplateResolver<V> fieldResolver) {
339            this.fieldPrefix = fieldPrefix;
340            this.fieldResolver = fieldResolver;
341        }
342
343        @Override
344        public boolean resolve(final V value, final JsonWriter jsonWriter, final boolean succeedingEntry) {
345            final boolean resolvable = fieldResolver.isResolvable(value);
346            if (!resolvable) {
347                return false;
348            }
349            if (succeedingEntry) {
350                jsonWriter.writeSeparator();
351            }
352            jsonWriter.writeRawString(fieldPrefix);
353            fieldResolver.resolve(value, jsonWriter, succeedingEntry);
354            return true;
355        }
356
357    }

359    private static final class MapResolver<V> implements TemplateResolver<V> {
360
361        private final List<FieldResolverContext<V>> fieldResolverContexts;
362
363        private MapResolver(final List<FieldResolverContext<V>> fieldResolverContexts) {
364            this.fieldResolverContexts = fieldResolverContexts;
365        }
366
367        @Override
368        public boolean isResolvable() {
369            // We have already excluded unresolvable ones while collecting
370            // the resolvers; it is safe to return true here.
371            return true;
372        }
373
374        /**
375         * The parent resolver checking if each child is resolvable given
376         * the passed {@code value}.
377         *
378         * This is an optimization to skip the rendering of a parent if all
379         * its children are not resolvable for the given {@code value}.
380         */
381        @Override
382        public boolean isResolvable(final V value) {
383            int fieldCount = fieldResolverContexts.size();
384            // noinspection ForLoopReplaceableByForEach (avoid iterator instantiation)
385            for (int fieldIndex = 0; fieldIndex < fieldCount; fieldIndex++) {
386                final TemplateResolver<V> fieldResolver = fieldResolverContexts.get(fieldIndex).resolver;
387                final boolean resolvable = fieldResolver.isResolvable(value);
388                if (resolvable) {
389                    return true;
390                }
391            }
392            return false;
393        }
394
395        /**
396         * The parent resolver combining all child resolver executions.
397         */
398        @Override
399        public void resolve(final V value, final JsonWriter jsonWriter) {
400            jsonWriter.writeObjectStart();
401            int fieldCount = fieldResolverContexts.size();
402            for (int resolvedFieldCount = 0, fieldIndex = 0; fieldIndex < fieldCount; fieldIndex++) {
403                FieldResolverContext<V> fieldResolverContext = fieldResolverContexts.get(fieldIndex);
404                final boolean resolvable = fieldResolverContext.resolver.isResolvable(value);
405                if (!resolvable) {
406                    continue;
407                }
408                final boolean succeedingEntry = resolvedFieldCount > 0;
409                final boolean resolved = fieldResolverContext.resolverMethod.resolve(value, jsonWriter, succeedingEntry);
410                if (resolved) {
411                    resolvedFieldCount++;
412                }
413            }
414            jsonWriter.writeObjectEnd();
415        }
416
417    }

419    private static <V, C extends TemplateResolverContext<V, C>> TemplateResolver<V> ofResolver(
420            final C context,
421            final Map<String, Object> configMap) {
422
423        // Check arguments.
424        Objects.requireNonNull(context, "context");
425        Objects.requireNonNull(configMap, "configMap");
426
427        // Extract the resolver name.
428        final Object resolverNameObject = configMap.get(RESOLVER_FIELD_NAME);
429        if (!(resolverNameObject instanceof String)) {
430            throw new IllegalArgumentException(
431                    "invalid resolver name: " + resolverNameObject);
432        }
433        final String resolverName = (String) resolverNameObject;
434
435        // Retrieve the resolver.
436        final TemplateResolverFactory<V, C> resolverFactory =
437                context.getResolverFactoryByName().get(resolverName);
438        if (resolverFactory == null) {
439            throw new IllegalArgumentException("unknown resolver: " + resolverName);
440        }
441        final TemplateResolverConfig resolverConfig = new TemplateResolverConfig(configMap);
442        return resolverFactory.create(context, resolverConfig);
443
444    }

446    private static <V, C extends TemplateResolverContext<V, C>> TemplateResolver<V> ofString(
447            final C context,
448            final String fieldValue) {
449
450        // Check if substitution is needed.
451        final boolean substitutionNeeded = fieldValue.contains("${");
452        final JsonWriter contextJsonWriter = context.getJsonWriter();
453        if (substitutionNeeded) {
454            final TemplateResolverStringSubstitutor<V> substitutor = context.getSubstitutor();
455
456            // If the substitutor is stable, we can get the replacement right
457            // away and avoid runtime substitution.
458            if (substitutor.isStable()) {
459                final String replacedText = substitutor.replace(null, fieldValue);
460                if (replacedText == null) {
461                    @SuppressWarnings("unchecked")
462                    final TemplateResolver<V> resolver =
463                            (TemplateResolver<V>) NULL_RESOLVER;
464                    return resolver;
465                } else {
466                    // Prepare the escaped replacement first.
467                    final String escapedReplacedText =
468                            contextJsonWriter.use(() ->
469                                    contextJsonWriter.writeString(replacedText));
470                    // Create a resolver dedicated to the escaped replacement.
471                    return new RawStringResolver<>(escapedReplacedText);
472                }
473            }
474
475            // Otherwise, the unstable substitutor needs to be invoked always at
476            // runtime.
477            else {
478                return new SubstitutingStringResolver<>(substitutor, fieldValue);
479            }
480
481        }
482
483        // Write the field value as is.
484        else {
485            final String escapedFieldValue =
486                    contextJsonWriter.use(() ->
487                            contextJsonWriter.writeString(fieldValue));
488            return new RawStringResolver<>(escapedFieldValue);
489        }
490
491    }

493    private static final class SubstitutingStringResolver<V>
494            implements TemplateResolver<V> {
495
496        private final TemplateResolverStringSubstitutor<V> substitutor;
497
498        private final String string;
499
500        private SubstitutingStringResolver(
501                final TemplateResolverStringSubstitutor<V> substitutor,
502                final String string) {
503            this.substitutor = substitutor;
504            this.string = string;
505        }
506
507        @Override
508        public void resolve(final V value, final JsonWriter jsonWriter) {
509            final String replacedString = substitutor.replace(value, string);
510            jsonWriter.writeString(replacedString);
511        }
512
513    }

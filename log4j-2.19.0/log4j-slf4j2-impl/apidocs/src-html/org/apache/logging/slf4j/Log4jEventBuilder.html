017package org.apache.logging.slf4j;
018
019import java.util.ArrayList;
020import java.util.HashMap;
021import java.util.List;
022import java.util.Map;
023import java.util.function.Supplier;
024
025import org.apache.logging.log4j.BridgeAware;
026import org.apache.logging.log4j.CloseableThreadContext;
027import org.apache.logging.log4j.CloseableThreadContext.Instance;
028import org.apache.logging.log4j.LogBuilder;
029import org.slf4j.Marker;
030import org.slf4j.spi.LoggingEventBuilder;
031
032public class Log4jEventBuilder implements LoggingEventBuilder {
033
034    private static final String FQCN = Log4jEventBuilder.class.getName();
035
036    private final Log4jMarkerFactory markerFactory;
037    private final LogBuilder logBuilder;
038    private final List<Object> arguments = new ArrayList<>();
039    private String message = null;
040    private Map<String, String> keyValuePairs = null;
041
042    public Log4jEventBuilder(final Log4jMarkerFactory markerFactory, final LogBuilder logBuilder) {
043        this.markerFactory = markerFactory;
044        this.logBuilder = logBuilder;
045        if (logBuilder instanceof BridgeAware) {
046            ((BridgeAware) logBuilder).setEntryPoint(FQCN);
047        }
048    }
049
050    @Override
051    public LoggingEventBuilder setCause(Throwable cause) {
052        logBuilder.withThrowable(cause);
053        return this;
054    }
055
056    @Override
057    public LoggingEventBuilder addMarker(Marker marker) {
058        logBuilder.withMarker(markerFactory.getLog4jMarker(marker));
059        return this;
060    }
061
062    @Override
063    public LoggingEventBuilder addArgument(Object p) {
064        arguments.add(p);
065        return this;
066    }
067
068    @Override
069    public LoggingEventBuilder addArgument(Supplier<?> objectSupplier) {
070        arguments.add(objectSupplier.get());
071        return this;
072    }
073
074    @Override
075    public LoggingEventBuilder addKeyValue(String key, Object value) {
076        if (keyValuePairs == null) {
077            keyValuePairs = new HashMap<>();
078        }
079        keyValuePairs.put(key, String.valueOf(value));
080        return this;
081    }
082
083    @Override
084    public LoggingEventBuilder addKeyValue(String key, Supplier<Object> valueSupplier) {
085        if (keyValuePairs == null) {
086            keyValuePairs = new HashMap<>();
087        }
088        keyValuePairs.put(key, String.valueOf(valueSupplier.get()));
089        return this;
090    }
091
092    @Override
093    public LoggingEventBuilder setMessage(String message) {
094        this.message = message;
095        return this;
096    }
097
098    @Override
099    public LoggingEventBuilder setMessage(Supplier<String> messageSupplier) {
100        this.message = messageSupplier.get();
101        return this;
102    }
103
104    @Override
105    public void log() {
106        if (keyValuePairs == null || keyValuePairs.isEmpty()) {
107            logBuilder.log(message, arguments.toArray());
108        } else {
109            try (Instance c = CloseableThreadContext.putAll(keyValuePairs)) {
110                logBuilder.log(message, arguments.toArray());
111            }
112        }
113    }
114
115    @Override
116    public void log(String message) {
117        setMessage(message);
118        log();
119    }
120
121    @Override
122    public void log(String message, Object arg) {
123        setMessage(message);
124        addArgument(arg);
125        log();
126    }
127
128    @Override
129    public void log(String message, Object arg0, Object arg1) {
130        setMessage(message);
131        addArgument(arg0);
132        addArgument(arg1);
133        log();
134    }
135
136    @Override
137    public void log(String message, Object... args) {
138        setMessage(message);
139        for (final Object arg : args) {
140            addArgument(arg);
141        }
142        log();
143    }
144
145    @Override
146    public void log(Supplier<String> messageSupplier) {
147        setMessage(messageSupplier);
148        log();
149    }
150
151}

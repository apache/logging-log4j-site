051/**
052 * Manager for sending SMTP events.
053 */
054public class SmtpManager extends MailManager {
055
056    private final Session session;
057
058    private final CyclicBuffer<LogEvent> buffer;
059
060    private volatile MimeMessage message;
061
062    private final FactoryData data;
063
064    private static MimeMessage createMimeMessage(final FactoryData data, final Session session, final LogEvent appendEvent)
065            throws MessagingException {
066        return new MimeMessageBuilder(session).setFrom(data.getFrom())
067                .setReplyTo(data.getReplyTo())
068                .setRecipients(Message.RecipientType.TO, data.getTo())
069                .setRecipients(Message.RecipientType.CC, data.getCc())
070                .setRecipients(Message.RecipientType.BCC, data.getBcc())
071                .setSubject(data.getSubjectSerializer().toSerializable(appendEvent))
072                .build();
073    }
074
075    protected SmtpManager(final String name, final Session session, final MimeMessage message,
076                          final FactoryData data) {
077        super(null, name);
078        this.session = session;
079        this.message = message;
080        this.data = data;
081        this.buffer = new CyclicBuffer<>(LogEvent.class, data.getBufferSize());
082    }
083
084    @Override
085    public void add(LogEvent event) {
086        buffer.add(event.toImmutable());
087    }
088
089    @Override
090    public void sendEvents(final Layout<?> layout, final LogEvent appendEvent) {
091        if (message == null) {
092            connect(appendEvent);
093        }
094        try {
095            final LogEvent[] priorEvents = removeAllBufferedEvents();
096            // LOG4J-310: log appendEvent even if priorEvents is empty
097
098            final byte[] rawBytes = formatContentToBytes(priorEvents, appendEvent, layout);
099
100            final String contentType = layout.getContentType();
101            final String encoding = getEncoding(rawBytes, contentType);
102            final byte[] encodedBytes = encodeContentToBytes(rawBytes, encoding);
103
104            final InternetHeaders headers = getHeaders(contentType, encoding);
105            final MimeMultipart mp = getMimeMultipart(encodedBytes, headers);
106
107            final String subject = data.getSubjectSerializer().toSerializable(appendEvent);
108
109            sendMultipartMessage(message, mp, subject);
110        } catch (final MessagingException | IOException | RuntimeException e) {
111            logError("Caught exception while sending e-mail notification.", e);
112            throw new LoggingException("Error occurred while sending email", e);
113        }
114    }
115
116    LogEvent[] removeAllBufferedEvents() {
117        return buffer.removeAll();
118    }
119
120    protected byte[] formatContentToBytes(final LogEvent[] priorEvents, final LogEvent appendEvent,
121                                          final Layout<?> layout) throws IOException {
122        final ByteArrayOutputStream raw = new ByteArrayOutputStream();
123        writeContent(priorEvents, appendEvent, layout, raw);
124        return raw.toByteArray();
125    }
126
127    private void writeContent(final LogEvent[] priorEvents, final LogEvent appendEvent, final Layout<?> layout,
128                              final ByteArrayOutputStream out)
129        throws IOException {
130        writeHeader(layout, out);
131        writeBuffer(priorEvents, appendEvent, layout, out);
132        writeFooter(layout, out);
133    }
134
135    protected void writeHeader(final Layout<?> layout, final OutputStream out) throws IOException {
136        final byte[] header = layout.getHeader();
137        if (header != null) {
138            out.write(header);
139        }
140    }
141
142    protected void writeBuffer(final LogEvent[] priorEvents, final LogEvent appendEvent, final Layout<?> layout,
143                               final OutputStream out) throws IOException {
144        for (final LogEvent priorEvent : priorEvents) {
145            final byte[] bytes = layout.toByteArray(priorEvent);
146            out.write(bytes);
147        }
148
149        final byte[] bytes = layout.toByteArray(appendEvent);
150        out.write(bytes);
151    }
152
153    protected void writeFooter(final Layout<?> layout, final OutputStream out) throws IOException {
154        final byte[] footer = layout.getFooter();
155        if (footer != null) {
156            out.write(footer);
157        }
158    }
159
160    protected String getEncoding(final byte[] rawBytes, final String contentType) {
161        final DataSource dataSource = new ByteArrayDataSource(rawBytes, contentType);
162        return MimeUtility.getEncoding(dataSource);
163    }
164
165    protected byte[] encodeContentToBytes(final byte[] rawBytes, final String encoding)
166        throws MessagingException, IOException {
167        final ByteArrayOutputStream encoded = new ByteArrayOutputStream();
168        encodeContent(rawBytes, encoding, encoded);
169        return encoded.toByteArray();
170    }
171
172    protected void encodeContent(final byte[] bytes, final String encoding, final ByteArrayOutputStream out)
173            throws MessagingException, IOException {
174        try (final OutputStream encoder = MimeUtility.encode(out, encoding)) {
175            encoder.write(bytes);
176        }
177    }
178
179    protected InternetHeaders getHeaders(final String contentType, final String encoding) {
180        final InternetHeaders headers = new InternetHeaders();
181        headers.setHeader("Content-Type", contentType + "; charset=UTF-8");
182        headers.setHeader("Content-Transfer-Encoding", encoding);
183        return headers;
184    }
185
186    protected MimeMultipart getMimeMultipart(final byte[] encodedBytes, final InternetHeaders headers)
187        throws MessagingException {
188        final MimeMultipart mp = new MimeMultipart();
189        final MimeBodyPart part = new MimeBodyPart(headers, encodedBytes);
190        mp.addBodyPart(part);
191        return mp;
192    }
193
194    /**
195     * @deprecated Please use the {@link #sendMultipartMessage(MimeMessage, MimeMultipart, String)} method instead.
196     */
197    @Deprecated
198    protected void sendMultipartMessage(final MimeMessage msg, final MimeMultipart mp) throws MessagingException {
199        synchronized (msg) {
200            msg.setContent(mp);
201            msg.setSentDate(new Date());
202            Transport.send(msg);
203        }
204    }
205
206    protected void sendMultipartMessage(final MimeMessage msg, final MimeMultipart mp, final String subject) throws MessagingException {
207        synchronized (msg) {
208            msg.setContent(mp);
209            msg.setSentDate(new Date());
210            msg.setSubject(subject);
211            Transport.send(msg);
212        }
213    }
214
215    private synchronized void connect(final LogEvent appendEvent) {
216        if (message != null) {
217            return;
218        }
219        try {
220            message = createMimeMessage(data, session, appendEvent);
221        } catch (final MessagingException e) {
222            logError("Could not set SmtpAppender message options", e);
223            message = null;
224        }
225    }
226
227    /**
228     * Factory to create the SMTP Manager.
229     */
230    public static class SMTPManagerFactory implements MailManagerFactory {
231
232        @Override
233        public SmtpManager createManager(final String name, final FactoryData data) {
234            final String smtpProtocol = data.getSmtpProtocol();
235            final String prefix = "mail." + smtpProtocol;
236
237            final Properties properties = PropertiesUtil.getSystemProperties();
238            properties.setProperty("mail.transport.protocol", smtpProtocol);
239            if (properties.getProperty("mail.host") == null) {
240                // Prevent an UnknownHostException in Java 7
241                properties.setProperty("mail.host", NetUtils.getLocalHostname());
242            }
243
244            final String smtpHost = data.getSmtpHost();
245            if (null != smtpHost) {
246                properties.setProperty(prefix + ".host", smtpHost);
247            }
248            if (data.getSmtpPort() > 0) {
249                properties.setProperty(prefix + ".port", String.valueOf(data.getSmtpPort()));
250            }
251
252            final Authenticator authenticator = buildAuthenticator(data.getSmtpUsername(), data.getSmtpPassword());
253            if (null != authenticator) {
254                properties.setProperty(prefix + ".auth", "true");
255            }
256
257            if (smtpProtocol.equals("smtps")) {
258                final SslConfiguration sslConfiguration = data.getSslConfiguration();
259                if (sslConfiguration != null) {
260                    final SSLSocketFactory sslSocketFactory = sslConfiguration.getSslSocketFactory();
261                    properties.put(prefix + ".ssl.socketFactory", sslSocketFactory);
262                    properties.setProperty(prefix + ".ssl.checkserveridentity", Boolean.toString(sslConfiguration.isVerifyHostName()));
263                }
264            }
265
266            final Session session = Session.getInstance(properties, authenticator);
267            session.setProtocolForAddress("rfc822", smtpProtocol);
268            session.setDebug(data.isSmtpDebug());
269            return new SmtpManager(name, session, null, data);
270        }
271
272        private Authenticator buildAuthenticator(final String username, final String password) {
273            if (null != password && null != username) {
274                return new Authenticator() {
275                    private final PasswordAuthentication passwordAuthentication =
276                        new PasswordAuthentication(username, password);
277
278                    @Override
279                    protected PasswordAuthentication getPasswordAuthentication() {
280                        return passwordAuthentication;
281                    }
282                };
283            }
284            return null;
285        }
286    }

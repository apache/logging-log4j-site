018package org.apache.log4j.varia;
019
020import org.apache.log4j.Level;
021import org.apache.log4j.spi.Filter;
022import org.apache.log4j.spi.LoggingEvent;
023
024/**
025 * This is a very simple filter based on level matching, which can be used to reject messages with priorities outside a
026 * certain range.
027 * <p>
028 * The filter admits three options <b>LevelMin</b>, <b>LevelMax</b> and <b>AcceptOnMatch</b>.
029 * </p>
030 * <p>
031 * If the level of the {@link LoggingEvent} is not between Min and Max (inclusive), then {@link Filter#DENY} is
032 * returned.
033 * </p>
034 * <p>
035 * If the Logging event level is within the specified range, then if <b>AcceptOnMatch</b> is true, {@link Filter#ACCEPT}
036 * is returned, and if <b>AcceptOnMatch</b> is false, {@link Filter#NEUTRAL} is returned.
037 * </p>
038 * <p>
039 * If <code>LevelMin</code>w is not defined, then there is no minimum acceptable level (ie a level is never rejected for
040 * being too "low"/unimportant). If <code>LevelMax</code> is not defined, then there is no maximum acceptable level (ie
041 * a level is never rejected for beeing too "high"/important).
042 * </p>
043 * <p>
044 * Refer to the {@link org.apache.log4j.AppenderSkeleton#setThreshold setThreshold} method available to <code>all</code>
045 * appenders extending {@link org.apache.log4j.AppenderSkeleton} for a more convenient way to filter out events by
046 * level.
047 * </p>
048 */
049public class LevelRangeFilter extends Filter {
050
051    /**
052     * Do we return ACCEPT when a match occurs. Default is <code>false</code>, so that later filters get run by default
053     */
054    boolean acceptOnMatch;
055
056    Level levelMin;
057    Level levelMax;
058
059    /**
060     * Return the decision of this filter.
061     */
062    @Override
063    public int decide(final LoggingEvent event) {
064        if (this.levelMin != null) {
065            if (!event.getLevel().isGreaterOrEqual(levelMin)) {
066                // level of event is less than minimum
067                return Filter.DENY;
068            }
069        }
070
071        if (this.levelMax != null) {
072            if (event.getLevel().toInt() > levelMax.toInt()) {
073                // level of event is greater than maximum
074                // Alas, there is no Level.isGreater method. and using
075                // a combo of isGreaterOrEqual && !Equal seems worse than
076                // checking the int values of the level objects..
077                return Filter.DENY;
078            }
079        }
080
081        if (acceptOnMatch) {
082            // this filter set up to bypass later filters and always return
083            // accept if level in range
084            return Filter.ACCEPT;
085        }
086        // event is ok for this filter; allow later filters to have a look..
087        return Filter.NEUTRAL;
088    }
089
090    /**
091     * Get the value of the <code>AcceptOnMatch</code> option.
092     */
093    public boolean getAcceptOnMatch() {
094        return acceptOnMatch;
095    }
096
097    /**
098     * Get the value of the <code>LevelMax</code> option.
099     */
100    public Level getLevelMax() {
101        return levelMax;
102    }
103
104    /**
105     * Get the value of the <code>LevelMin</code> option.
106     */
107    public Level getLevelMin() {
108        return levelMin;
109    }
110
111    /**
112     * Set the <code>AcceptOnMatch</code> option.
113     */
114    public void setAcceptOnMatch(final boolean acceptOnMatch) {
115        this.acceptOnMatch = acceptOnMatch;
116    }
117
118    /**
119     * Set the <code>LevelMax</code> option.
120     */
121    public void setLevelMax(final Level levelMax) {
122        this.levelMax = levelMax;
123    }
124
125    /**
126     * Set the <code>LevelMin</code> option.
127     */
128    public void setLevelMin(final Level levelMin) {
129        this.levelMin = levelMin;
130    }
131}

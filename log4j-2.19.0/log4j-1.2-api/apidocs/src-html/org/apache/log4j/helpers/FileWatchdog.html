020package org.apache.log4j.helpers;
021
022import java.io.File;
023
024/**
025 * Checks every now and then that a certain file has not changed. If it has, then call the {@link #doOnChange} method.
026 *
027 * @since version 0.9.1
028 */
029public abstract class FileWatchdog extends Thread {
030
031    /**
032     * The default delay between every file modification check, set to 60 seconds.
033     */
034    static final public long DEFAULT_DELAY = 60_000;
035
036    /**
037     * The name of the file to observe for changes.
038     */
039    protected String filename;
040
041    /**
042     * The delay to observe between every check. By default set {@link #DEFAULT_DELAY}.
043     */
044    protected long delay = DEFAULT_DELAY;
045
046    File file;
047    long lastModified;
048    boolean warnedAlready;
049    boolean interrupted;
050
051    protected FileWatchdog(final String fileName) {
052        super("FileWatchdog");
053        this.filename = fileName;
054        this.file = new File(fileName);
055        setDaemon(true);
056        checkAndConfigure();
057    }
058
059    protected void checkAndConfigure() {
060        boolean fileExists;
061        try {
062            fileExists = file.exists();
063        } catch (final SecurityException e) {
064            LogLog.warn("Was not allowed to read check file existance, file:[" + filename + "].");
065            interrupted = true; // there is no point in continuing
066            return;
067        }
068
069        if (fileExists) {
070            final long fileLastMod = file.lastModified(); // this can also throw a SecurityException
071            if (fileLastMod > lastModified) { // however, if we reached this point this
072                lastModified = fileLastMod; // is very unlikely.
073                doOnChange();
074                warnedAlready = false;
075            }
076        } else {
077            if (!warnedAlready) {
078                LogLog.debug("[" + filename + "] does not exist.");
079                warnedAlready = true;
080            }
081        }
082    }
083
084    abstract protected void doOnChange();
085
086    @Override
087    public void run() {
088        while (!interrupted) {
089            try {
090                Thread.sleep(delay);
091            } catch (final InterruptedException e) {
092                // no interruption expected
093            }
094            checkAndConfigure();
095        }
096    }
097
098    /**
099     * Sets the delay in milliseconds to observe between each check of the file changes.
100     * 
101     * @param delayMillis the delay in milliseconds
102     */
103    public void setDelay(final long delayMillis) {
104        this.delay = delayMillis;
105    }
106}

018package org.apache.log4j.helpers;
019
020import java.text.DateFormat;
021import java.text.FieldPosition;
022import java.text.ParsePosition;
023import java.util.Calendar;
024import java.util.Date;
025import java.util.TimeZone;
026
027/**
028 * Formats a {@link Date} in the format "HH:mm:ss,SSS" for example, "15:49:37,459".
029 *
030 * @since 0.7.5
031 */
032public class AbsoluteTimeDateFormat extends DateFormat {
033
034    private static final long serialVersionUID = -388856345976723342L;
035
036    /**
037     * String constant used to specify {@link org.apache.log4j.helpers.AbsoluteTimeDateFormat} in layouts. Current value is
038     * <b>ABSOLUTE</b>.
039     */
040    public final static String ABS_TIME_DATE_FORMAT = "ABSOLUTE";
041
042    /**
043     * String constant used to specify {@link org.apache.log4j.helpers.DateTimeDateFormat} in layouts. Current value is
044     * <b>DATE</b>.
045     */
046    public final static String DATE_AND_TIME_DATE_FORMAT = "DATE";
047
048    /**
049     * String constant used to specify {@link org.apache.log4j.helpers.ISO8601DateFormat} in layouts. Current value is
050     * <b>ISO8601</b>.
051     */
052    public final static String ISO8601_DATE_FORMAT = "ISO8601";
053
054    private static long previousTime;
055
056    private static char[] previousTimeWithoutMillis = new char[9]; // "HH:mm:ss."
057
058    public AbsoluteTimeDateFormat() {
059        setCalendar(Calendar.getInstance());
060    }
061
062    public AbsoluteTimeDateFormat(final TimeZone timeZone) {
063        setCalendar(Calendar.getInstance(timeZone));
064    }
065
066    /**
067     * Appends to <code>sbuf</code> the time in the format "HH:mm:ss,SSS" for example, "15:49:37,459"
068     *
069     * @param date the date to format
070     * @param sbuf the string buffer to write to
071     * @param fieldPosition remains untouched
072     */
073    @Override
074    public StringBuffer format(final Date date, final StringBuffer sbuf, final FieldPosition fieldPosition) {
075
076        final long now = date.getTime();
077        final int millis = (int) (now % 1000);
078
079        if ((now - millis) != previousTime || previousTimeWithoutMillis[0] == 0) {
080            // We reach this point at most once per second
081            // across all threads instead of each time format()
082            // is called. This saves considerable CPU time.
083
084            calendar.setTime(date);
085
086            final int start = sbuf.length();
087
088            final int hour = calendar.get(Calendar.HOUR_OF_DAY);
089            if (hour < 10) {
090                sbuf.append('0');
091            }
092            sbuf.append(hour);
093            sbuf.append(':');
094
095            final int mins = calendar.get(Calendar.MINUTE);
096            if (mins < 10) {
097                sbuf.append('0');
098            }
099            sbuf.append(mins);
100            sbuf.append(':');
101
102            final int secs = calendar.get(Calendar.SECOND);
103            if (secs < 10) {
104                sbuf.append('0');
105            }
106            sbuf.append(secs);
107            sbuf.append(',');
108
109            // store the time string for next time to avoid recomputation
110            sbuf.getChars(start, sbuf.length(), previousTimeWithoutMillis, 0);
111
112            previousTime = now - millis;
113        } else {
114            sbuf.append(previousTimeWithoutMillis);
115        }
116
117        if (millis < 100) {
118            sbuf.append('0');
119        }
120        if (millis < 10) {
121            sbuf.append('0');
122        }
123
124        sbuf.append(millis);
125        return sbuf;
126    }
127
128    /**
129     * Always returns null.
130     */
131    @Override
132    public Date parse(final String s, final ParsePosition pos) {
133        return null;
134    }
135}

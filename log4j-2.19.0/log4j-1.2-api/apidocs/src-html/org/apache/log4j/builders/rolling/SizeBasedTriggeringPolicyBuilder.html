017package org.apache.log4j.builders.rolling;
018
019import static org.apache.log4j.builders.BuilderManager.CATEGORY;
020import static org.apache.log4j.xml.XmlConfiguration.PARAM_TAG;
021import static org.apache.log4j.xml.XmlConfiguration.forEachElement;
022
023import java.util.Properties;
024import java.util.concurrent.atomic.AtomicLong;
025
026import org.apache.log4j.builders.AbstractBuilder;
027import org.apache.log4j.config.PropertiesConfiguration;
028import org.apache.log4j.xml.XmlConfiguration;
029import org.apache.logging.log4j.core.appender.rolling.SizeBasedTriggeringPolicy;
030import org.apache.logging.log4j.core.appender.rolling.TriggeringPolicy;
031import org.apache.logging.log4j.core.config.plugins.Plugin;
032import org.w3c.dom.Element;
033
034@Plugin(name = "org.apache.log4j.rolling.SizeBasedTriggeringPolicy", category = CATEGORY)
035public class SizeBasedTriggeringPolicyBuilder extends AbstractBuilder<TriggeringPolicy>
036        implements TriggeringPolicyBuilder {
037
038    private static final String MAX_SIZE_PARAM = "MaxFileSize";
039    private static final long DEFAULT_MAX_SIZE = 10 * 1024 * 1024;
040
041    public SizeBasedTriggeringPolicyBuilder() {
042        super();
043    }
044
045    public SizeBasedTriggeringPolicyBuilder(String prefix, Properties props) {
046        super(prefix, props);
047    }
048
049    @Override
050    public SizeBasedTriggeringPolicy parse(Element element, XmlConfiguration configuration) {
051        final AtomicLong maxSize = new AtomicLong(DEFAULT_MAX_SIZE);
052        forEachElement(element.getChildNodes(), currentElement -> {
053            switch (currentElement.getTagName()) {
054                case PARAM_TAG:
055                    switch (getNameAttributeKey(currentElement)) {
056                        case MAX_SIZE_PARAM:
057                            set(MAX_SIZE_PARAM, currentElement, maxSize);
058                            break;
059                    }
060                    break;
061            }
062        });
063        return createTriggeringPolicy(maxSize.get());
064    }
065
066    @Override
067    public SizeBasedTriggeringPolicy parse(PropertiesConfiguration configuration) {
068        final long maxSize = getLongProperty(MAX_SIZE_PARAM, DEFAULT_MAX_SIZE);
069        return createTriggeringPolicy(maxSize);
070    }
071
072    private SizeBasedTriggeringPolicy createTriggeringPolicy(long maxSize) {
073        return SizeBasedTriggeringPolicy.createPolicy(Long.toString(maxSize));
074    }
075}

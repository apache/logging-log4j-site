049/**
050 * Build a File Appender
051 */
052@Plugin(name = "org.apache.log4j.rolling.RollingFileAppender", category = CATEGORY)
053
054public class EnhancedRollingFileAppenderBuilder extends AbstractBuilder<Appender> implements AppenderBuilder<Appender> {
055
056    private static final String TIME_BASED_ROLLING_POLICY = "org.apache.log4j.rolling.TimeBasedRollingPolicy";
057    private static final String FIXED_WINDOW_ROLLING_POLICY = "org.apache.log4j.rolling.FixedWindowRollingPolicy";
058    private static final Logger LOGGER = StatusLogger.getLogger();
059    private static final String TRIGGERING_TAG = "triggeringPolicy";
060    private static final String ROLLING_TAG = "rollingPolicy";
061    private static final int DEFAULT_MIN_INDEX = 1;
062    private static final int DEFAULT_MAX_INDEX = 7;
063    private static final String ACTIVE_FILE_PARAM = "ActiveFileName";
064    private static final String FILE_PATTERN_PARAM = "FileNamePattern";
065    private static final String MIN_INDEX_PARAM = "MinIndex";
066    private static final String MAX_INDEX_PARAM = "MaxIndex";
067
068    public EnhancedRollingFileAppenderBuilder() {
069    }
070
071    public EnhancedRollingFileAppenderBuilder(final String prefix, final Properties properties) {
072        super(prefix, properties);
073    }
074
075    private void parseRollingPolicy(Element element, XmlConfiguration configuration,
076            AtomicReference<String> rollingPolicyClassName, AtomicReference<String> activeFileName,
077            AtomicReference<String> fileNamePattern, AtomicInteger minIndex,
078            AtomicInteger maxIndex) {
079        rollingPolicyClassName.set(configuration.subst(element.getAttribute("class"), getProperties()));
080        forEachElement(element.getChildNodes(), currentElement -> {
081            switch (currentElement.getTagName()) {
082                case PARAM_TAG:
083                    switch (getNameAttributeKey(currentElement)) {
084                        case ACTIVE_FILE_PARAM:
085                            set(ACTIVE_FILE_PARAM, currentElement, activeFileName);
086                            break;
087                        case FILE_PATTERN_PARAM:
088                            set(FILE_PATTERN_PARAM, currentElement, fileNamePattern);
089                            break;
090                        case MIN_INDEX_PARAM:
091                            set(MIN_INDEX_PARAM, currentElement, minIndex);
092                            break;
093                        case MAX_INDEX_PARAM:
094                            set(MAX_INDEX_PARAM, currentElement, maxIndex);
095                    }
096            }
097        });
098    }
099
100    @Override
101    public Appender parseAppender(Element element, XmlConfiguration configuration) {
102        // FileAppender
103        final String name = getNameAttribute(element);
104        final AtomicReference<Layout> layout = new AtomicReference<>();
105        final AtomicReference<Filter> filter = new AtomicReference<>();
106        final AtomicReference<String> fileName = new AtomicReference<>();
107        final AtomicReference<String> level = new AtomicReference<>();
108        final AtomicBoolean immediateFlush = new AtomicBoolean(true);
109        final AtomicBoolean append = new AtomicBoolean(true);
110        final AtomicBoolean bufferedIo = new AtomicBoolean();
111        final AtomicInteger bufferSize = new AtomicInteger(8192);
112        // specific to RollingFileAppender
113        final AtomicReference<String> rollingPolicyClassName = new AtomicReference<>();
114        final AtomicReference<String> activeFileName = new AtomicReference<>();
115        final AtomicReference<String> fileNamePattern = new AtomicReference<>();
116        final AtomicInteger minIndex = new AtomicInteger(DEFAULT_MIN_INDEX);
117        final AtomicInteger maxIndex = new AtomicInteger(DEFAULT_MAX_INDEX);
118        final AtomicReference<TriggeringPolicy> triggeringPolicy = new AtomicReference<>();
119        forEachElement(element.getChildNodes(), currentElement -> {
120            switch (currentElement.getTagName()) {
121                case ROLLING_TAG:
122                    parseRollingPolicy(currentElement, configuration, rollingPolicyClassName, activeFileName,
123                            fileNamePattern, minIndex, maxIndex);
124                    break;
125                case TRIGGERING_TAG:
126                    triggeringPolicy.set(configuration.parseTriggeringPolicy(currentElement));
127                    break;
128                case LAYOUT_TAG:
129                    layout.set(configuration.parseLayout(currentElement));
130                    break;
131                case FILTER_TAG:
132                    configuration.addFilter(filter, currentElement);
133                    break;
134                case PARAM_TAG:
135                    switch (getNameAttributeKey(currentElement)) {
136                        case FILE_PARAM:
137                            set(FILE_PARAM, currentElement, fileName);
138                            break;
139                        case APPEND_PARAM:
140                            set(APPEND_PARAM, currentElement, append);
141                            break;
142                        case BUFFERED_IO_PARAM:
143                            set(BUFFERED_IO_PARAM, currentElement, bufferedIo);
144                            break;
145                        case BUFFER_SIZE_PARAM:
146                            set(BUFFER_SIZE_PARAM, currentElement, bufferSize);
147                            break;
148                        case THRESHOLD_PARAM:
149                            set(THRESHOLD_PARAM, currentElement, level);
150                            break;
151                        case IMMEDIATE_FLUSH_PARAM:
152                            set(IMMEDIATE_FLUSH_PARAM, currentElement, immediateFlush);
153                            break;
154                    }
155                    break;
156            }
157        });
158        return createAppender(name, layout.get(), filter.get(), fileName.get(), level.get(), immediateFlush.get(),
159                append.get(), bufferedIo.get(), bufferSize.get(), rollingPolicyClassName.get(), activeFileName.get(),
160                fileNamePattern.get(), minIndex.get(), maxIndex.get(), triggeringPolicy.get(), configuration);
161    }
162
163    @Override
164    public Appender parseAppender(String name, String appenderPrefix, String layoutPrefix, String filterPrefix,
165            Properties props, PropertiesConfiguration configuration) {
166        final Layout layout = configuration.parseLayout(layoutPrefix, name, props);
167        final Filter filter = configuration.parseAppenderFilters(props, filterPrefix, name);
168        final String level = getProperty(THRESHOLD_PARAM);
169        final String fileName = getProperty(FILE_PARAM);
170        final boolean append = getBooleanProperty(APPEND_PARAM, true);
171        final boolean immediateFlush = getBooleanProperty(IMMEDIATE_FLUSH_PARAM, true);
172        final boolean bufferedIo = getBooleanProperty(BUFFERED_IO_PARAM, false);
173        final int bufferSize = Integer.parseInt(getProperty(BUFFER_SIZE_PARAM, "8192"));
174        final String rollingPolicyClassName = getProperty(ROLLING_TAG);
175        final int minIndex = getIntegerProperty(ROLLING_TAG + "." + MIN_INDEX_PARAM, DEFAULT_MIN_INDEX);
176        final int maxIndex = getIntegerProperty(ROLLING_TAG + "." + MAX_INDEX_PARAM, DEFAULT_MAX_INDEX);
177        final String activeFileName = getProperty(ROLLING_TAG + "." + ACTIVE_FILE_PARAM);
178        final String fileNamePattern = getProperty(ROLLING_TAG + "." + FILE_PATTERN_PARAM);
179        final TriggeringPolicy triggeringPolicy = configuration.parseTriggeringPolicy(props,
180                appenderPrefix + "." + TRIGGERING_TAG);
181        return createAppender(name, layout, filter, fileName, level, immediateFlush, append, bufferedIo, bufferSize,
182                rollingPolicyClassName, activeFileName, fileNamePattern, minIndex, maxIndex, triggeringPolicy,
183                configuration);
184    }
185
186    private Appender createAppender(final String name, final Layout layout, final Filter filter, final String fileName,
187            final String level, final boolean immediateFlush, final boolean append, final boolean bufferedIo,
188            final int bufferSize, final String rollingPolicyClassName, final String activeFileName,
189            final String fileNamePattern, final int minIndex, final int maxIndex,
190            final TriggeringPolicy triggeringPolicy, final Configuration configuration) {
191        final org.apache.logging.log4j.core.Layout<?> fileLayout = LayoutAdapter.adapt(layout);
192        final boolean actualImmediateFlush = bufferedIo ? false : immediateFlush;
193        final org.apache.logging.log4j.core.Filter fileFilter = buildFilters(level, filter);
194        if (rollingPolicyClassName == null) {
195            LOGGER.error("Unable to create RollingFileAppender, no rolling policy provided.");
196            return null;
197        }
198        final String actualFileName = activeFileName != null ? activeFileName : fileName;
199        if (actualFileName == null) {
200            LOGGER.error("Unable to create RollingFileAppender, no file name provided.");
201            return null;
202        }
203        if (fileNamePattern == null) {
204            LOGGER.error("Unable to create RollingFileAppender, no file name pattern provided.");
205            return null;
206        }
207        final DefaultRolloverStrategy.Builder rolloverStrategyBuilder = DefaultRolloverStrategy.newBuilder();
208        switch (rollingPolicyClassName) {
209            case FIXED_WINDOW_ROLLING_POLICY:
210                rolloverStrategyBuilder.withMin(Integer.toString(minIndex)).withMax(Integer.toString(maxIndex));
211                break;
212            case TIME_BASED_ROLLING_POLICY:
213                break;
214            default:
215                LOGGER.warn("Unsupported rolling policy: {}", rollingPolicyClassName);
216        }
217        final TriggeringPolicy actualTriggeringPolicy;
218        if (triggeringPolicy != null) {
219            actualTriggeringPolicy = triggeringPolicy;
220        } else if (rollingPolicyClassName.equals(TIME_BASED_ROLLING_POLICY)) {
221            actualTriggeringPolicy = TimeBasedTriggeringPolicy.newBuilder().build();
222        } else {
223            LOGGER.error("Unable to create RollingFileAppender, no triggering policy provided.");
224            return null;
225        }
226        return AppenderWrapper.adapt(RollingFileAppender.newBuilder()
227                .withAppend(append)
228                .setBufferedIo(bufferedIo)
229                .setBufferSize(bufferedIo ? bufferSize : 0)
230                .setConfiguration(configuration)
231                .withFileName(actualFileName)
232                .withFilePattern(fileNamePattern)
233                .setFilter(fileFilter)
234                .setImmediateFlush(actualImmediateFlush)
235                .setLayout(fileLayout)
236                .setName(name)
237                .withPolicy(actualTriggeringPolicy)
238                .withStrategy(rolloverStrategyBuilder.build())
239                .build());
240    }

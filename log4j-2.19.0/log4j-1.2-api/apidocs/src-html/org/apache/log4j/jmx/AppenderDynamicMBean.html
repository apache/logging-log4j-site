056public class AppenderDynamicMBean extends AbstractDynamicMBean {
057
058    // This category instance is for logging.
059    private static Logger cat = Logger.getLogger(AppenderDynamicMBean.class);
060    private final MBeanConstructorInfo[] dConstructors = new MBeanConstructorInfo[1];
061    private final Vector dAttributes = new Vector();
062
063    private final String dClassName = this.getClass().getName();
064    private final Hashtable dynamicProps = new Hashtable(5);
065    private final MBeanOperationInfo[] dOperations = new MBeanOperationInfo[2];
066
067    private final String dDescription = "This MBean acts as a management facade for log4j appenders.";
068
069    // We wrap this appender instance.
070    private final Appender appender;
071
072    public AppenderDynamicMBean(final Appender appender) throws IntrospectionException {
073        this.appender = appender;
074        buildDynamicMBeanInfo();
075    }
076
077    private void buildDynamicMBeanInfo() throws IntrospectionException {
078        final Constructor[] constructors = this.getClass().getConstructors();
079        dConstructors[0] = new MBeanConstructorInfo("AppenderDynamicMBean(): Constructs a AppenderDynamicMBean instance", constructors[0]);
080
081        final BeanInfo bi = Introspector.getBeanInfo(appender.getClass());
082        final PropertyDescriptor[] pd = bi.getPropertyDescriptors();
083
084        final int size = pd.length;
085
086        for (int i = 0; i < size; i++) {
087            final String name = pd[i].getName();
088            final Method readMethod = pd[i].getReadMethod();
089            final Method writeMethod = pd[i].getWriteMethod();
090            if (readMethod != null) {
091                final Class returnClass = readMethod.getReturnType();
092                if (isSupportedType(returnClass)) {
093                    String returnClassName;
094                    if (returnClass.isAssignableFrom(Priority.class)) {
095                        returnClassName = "java.lang.String";
096                    } else {
097                        returnClassName = returnClass.getName();
098                    }
099
100                    dAttributes.add(new MBeanAttributeInfo(name, returnClassName, "Dynamic", true, writeMethod != null, false));
101                    dynamicProps.put(name, new MethodUnion(readMethod, writeMethod));
102                }
103            }
104        }
105
106        MBeanParameterInfo[] params = new MBeanParameterInfo[0];
107
108        dOperations[0] = new MBeanOperationInfo("activateOptions", "activateOptions(): add an appender", params, "void", MBeanOperationInfo.ACTION);
109
110        params = new MBeanParameterInfo[1];
111        params[0] = new MBeanParameterInfo("layout class", "java.lang.String", "layout class");
112
113        dOperations[1] = new MBeanOperationInfo("setLayout", "setLayout(): add a layout", params, "void", MBeanOperationInfo.ACTION);
114    }
115
116    @Override
117    public Object getAttribute(final String attributeName) throws AttributeNotFoundException, MBeanException, ReflectionException {
118
119        // Check attributeName is not null to avoid NullPointerException later on
120        if (attributeName == null) {
121            throw new RuntimeOperationsException(new IllegalArgumentException("Attribute name cannot be null"),
122                "Cannot invoke a getter of " + dClassName + " with null attribute name");
123        }
124
125        cat.debug("getAttribute called with [" + attributeName + "].");
126        if (attributeName.startsWith("appender=" + appender.getName() + ",layout")) {
127            try {
128                return new ObjectName("log4j:" + attributeName);
129            } catch (final MalformedObjectNameException e) {
130                cat.error("attributeName", e);
131            } catch (final RuntimeException e) {
132                cat.error("attributeName", e);
133            }
134        }
135
136        final MethodUnion mu = (MethodUnion) dynamicProps.get(attributeName);
137
138        // cat.debug("----name="+attributeName+", b="+b);
139
140        if (mu != null && mu.readMethod != null) {
141            try {
142                return mu.readMethod.invoke(appender, null);
143            } catch (final IllegalAccessException e) {
144                return null;
145            } catch (final InvocationTargetException e) {
146                if (e.getTargetException() instanceof InterruptedException || e.getTargetException() instanceof InterruptedIOException) {
147                    Thread.currentThread().interrupt();
148                }
149                return null;
150            } catch (final RuntimeException e) {
151                return null;
152            }
153        }
154
155        // If attributeName has not been recognized throw an AttributeNotFoundException
156        throw (new AttributeNotFoundException("Cannot find " + attributeName + " attribute in " + dClassName));
157
158    }
159
160    @Override
161    protected Logger getLogger() {
162        return cat;
163    }
164
165    @Override
166    public MBeanInfo getMBeanInfo() {
167        cat.debug("getMBeanInfo called.");
168
169        final MBeanAttributeInfo[] attribs = new MBeanAttributeInfo[dAttributes.size()];
170        dAttributes.toArray(attribs);
171
172        return new MBeanInfo(dClassName, dDescription, attribs, dConstructors, dOperations, new MBeanNotificationInfo[0]);
173    }
174
175    @Override
176    public Object invoke(final String operationName, final Object params[], final String signature[]) throws MBeanException, ReflectionException {
177
178        if (operationName.equals("activateOptions") && appender instanceof OptionHandler) {
179            final OptionHandler oh = (OptionHandler) appender;
180            oh.activateOptions();
181            return "Options activated.";
182        } else if (operationName.equals("setLayout")) {
183            final Layout layout = (Layout) OptionConverter.instantiateByClassName((String) params[0], Layout.class, null);
184            appender.setLayout(layout);
185            registerLayoutMBean(layout);
186        }
187        return null;
188    }
189
190    private boolean isSupportedType(final Class clazz) {
191        if (clazz.isPrimitive() || (clazz == String.class) || clazz.isAssignableFrom(Priority.class)) {
192            return true;
193        }
194
195        return false;
196
197    }
198
199    @Override
200    public ObjectName preRegister(final MBeanServer server, final ObjectName name) {
201        cat.debug("preRegister called. Server=" + server + ", name=" + name);
202        this.server = server;
203        registerLayoutMBean(appender.getLayout());
204
205        return name;
206    }
207
208    void registerLayoutMBean(final Layout layout) {
209        if (layout == null) {
210            return;
211        }
212
213        final String name = getAppenderName(appender) + ",layout=" + layout.getClass().getName();
214        cat.debug("Adding LayoutMBean:" + name);
215        ObjectName objectName = null;
216        try {
217            final LayoutDynamicMBean appenderMBean = new LayoutDynamicMBean(layout);
218            objectName = new ObjectName("log4j:appender=" + name);
219            if (!server.isRegistered(objectName)) {
220                registerMBean(appenderMBean, objectName);
221                dAttributes.add(new MBeanAttributeInfo("appender=" + name, "javax.management.ObjectName", "The " + name + " layout.", true, true, false));
222            }
223
224        } catch (final JMException e) {
225            cat.error("Could not add DynamicLayoutMBean for [" + name + "].", e);
226        } catch (final java.beans.IntrospectionException e) {
227            cat.error("Could not add DynamicLayoutMBean for [" + name + "].", e);
228        } catch (final RuntimeException e) {
229            cat.error("Could not add DynamicLayoutMBean for [" + name + "].", e);
230        }
231    }
232
233    @Override
234    public void setAttribute(final Attribute attribute) throws AttributeNotFoundException, InvalidAttributeValueException, MBeanException, ReflectionException {
235
236        // Check attribute is not null to avoid NullPointerException later on
237        if (attribute == null) {
238            throw new RuntimeOperationsException(new IllegalArgumentException("Attribute cannot be null"),
239                "Cannot invoke a setter of " + dClassName + " with null attribute");
240        }
241        final String name = attribute.getName();
242        Object value = attribute.getValue();
243
244        if (name == null) {
245            throw new RuntimeOperationsException(new IllegalArgumentException("Attribute name cannot be null"),
246                "Cannot invoke the setter of " + dClassName + " with null attribute name");
247        }
248
249        final MethodUnion mu = (MethodUnion) dynamicProps.get(name);
250
251        if (mu != null && mu.writeMethod != null) {
252            final Object[] o = new Object[1];
253
254            final Class[] params = mu.writeMethod.getParameterTypes();
255            if (params[0] == org.apache.log4j.Priority.class) {
256                value = OptionConverter.toLevel((String) value, (Level) getAttribute(name));
257            }
258            o[0] = value;
259
260            try {
261                mu.writeMethod.invoke(appender, o);
262
263            } catch (final InvocationTargetException e) {
264                if (e.getTargetException() instanceof InterruptedException || e.getTargetException() instanceof InterruptedIOException) {
265                    Thread.currentThread().interrupt();
266                }
267                cat.error("FIXME", e);
268            } catch (final IllegalAccessException e) {
269                cat.error("FIXME", e);
270            } catch (final RuntimeException e) {
271                cat.error("FIXME", e);
272            }
273        } else if (name.endsWith(".layout")) {
274
275        } else {
276            throw (new AttributeNotFoundException("Attribute " + name + " not found in " + this.getClass().getName()));
277        }
278    }

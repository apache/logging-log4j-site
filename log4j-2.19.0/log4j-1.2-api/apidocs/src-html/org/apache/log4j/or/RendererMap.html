018package org.apache.log4j.or;
019
020import java.util.Hashtable;
021
022import org.apache.log4j.helpers.OptionConverter;
023import org.apache.log4j.spi.RendererSupport;
024import org.apache.logging.log4j.core.util.Loader;
025import org.apache.logging.log4j.status.StatusLogger;
026
027/**
028 * Map class objects to an {@link ObjectRenderer}.
029 * 
030 * @since version 1.0
031 */
032public class RendererMap {
033
034    Hashtable map;
035
036    static ObjectRenderer defaultRenderer = new DefaultRenderer();
037
038    public RendererMap() {
039        map = new Hashtable();
040    }
041
042    /**
043     * Add a renderer to a hierarchy passed as parameter.
044     */
045    static public void addRenderer(RendererSupport repository, String renderedClassName, String renderingClassName) {
046        StatusLogger.getLogger().debug("Rendering class: [" + renderingClassName + "], Rendered class: [" + renderedClassName + "].");
047        ObjectRenderer renderer = (ObjectRenderer) OptionConverter.instantiateByClassName(renderingClassName, ObjectRenderer.class, null);
048        if (renderer == null) {
049            StatusLogger.getLogger().error("Could not instantiate renderer [" + renderingClassName + "].");
050            return;
051        }
052        try {
053            Class renderedClass = Loader.loadClass(renderedClassName);
054            repository.setRenderer(renderedClass, renderer);
055        } catch (ClassNotFoundException e) {
056            StatusLogger.getLogger().error("Could not find class [" + renderedClassName + "].", e);
057        }
058    }
059
060    /**
061     * Find the appropriate renderer for the class type of the <code>o</code> parameter. This is accomplished by calling the
062     * {@link #get(Class)} method. Once a renderer is found, it is applied on the object <code>o</code> and the result is
063     * returned as a {@link String}.
064     */
065    public String findAndRender(Object o) {
066        if (o == null)
067            return null;
068        else
069            return get(o.getClass()).doRender(o);
070    }
071
072    /**
073     * Syntactic sugar method that calls {@link #get(Class)} with the class of the object parameter.
074     */
075    public ObjectRenderer get(Object o) {
076        if (o == null)
077            return null;
078        else
079            return get(o.getClass());
080    }
081
082    /**
083     * Search the parents of <code>clazz</code> for a renderer. The renderer closest in the hierarchy will be returned. If
084     * no renderers could be found, then the default renderer is returned.
085     * 
086     * <p>
087     * The search first looks for a renderer configured for <code>clazz</code>. If a renderer could not be found, then the
088     * search continues by looking at all the interfaces implemented by <code>clazz</code> including the super-interfaces of
089     * each interface. If a renderer cannot be found, then the search looks for a renderer defined for the parent
090     * (superclass) of <code>clazz</code>. If that fails, then all the interfaces implemented by the parent of
091     * <code>clazz</code> are searched and so on.
092     * 
093     * <p>
094     * For example, if A0, A1, A2 are classes and X0, X1, X2, Y0, Y1 are interfaces where A2 extends A1 which in turn
095     * extends A0 and similarly X2 extends X1 which extends X0 and Y1 extends Y0. Let us also assume that A1 implements the
096     * Y0 interface and that A2 implements the X2 interface.
097     * 
098     * <p>
099     * The table below shows the results returned by the <code>get(A2.class)</code> method depending on the renderers added
100     * to the map.
101     * 
102     * <p>
103     * <table border="1">
104     * <tr>
105     * <th>Added renderers</th>
106     * <th>Value returned by <code>get(A2.class)</code></th>
107     * 
108     * <tr>
109     * <td><code>A0Renderer</code>
110     * <td align="center"><code>A0Renderer</code>
111     * 
112     * <tr>
113     * <td><code>A0Renderer, A1Renderer</code>
114     * <td align="center"><code>A1Renderer</code>
115     * 
116     * <tr>
117     * <td><code>X0Renderer</code>
118     * <td align="center"><code>X0Renderer</code>
119     * 
120     * <tr>
121     * <td><code>A1Renderer, X0Renderer</code>
122     * <td align="center"><code>X0Renderer</code>
123     * 
124     * </table>
125     * 
126     * <p>
127     * This search algorithm is not the most natural, although it is particularly easy to implement. Future log4j versions
128     * <em>may</em> implement a more intuitive search algorithm. However, the present algorithm should be acceptable in the
129     * vast majority of circumstances.
130     * 
131     */
132    public ObjectRenderer get(Class clazz) {
133        // System.out.println("\nget: "+clazz);
134        ObjectRenderer r = null;
135        for (Class c = clazz; c != null; c = c.getSuperclass()) {
136            // System.out.println("Searching for class: "+c);
137            r = (ObjectRenderer) map.get(c);
138            if (r != null) {
139                return r;
140            }
141            r = searchInterfaces(c);
142            if (r != null)
143                return r;
144        }
145        return defaultRenderer;
146    }
147
148    ObjectRenderer searchInterfaces(Class c) {
149        // System.out.println("Searching interfaces of class: "+c);
150
151        ObjectRenderer r = (ObjectRenderer) map.get(c);
152        if (r != null) {
153            return r;
154        }
155        Class[] ia = c.getInterfaces();
156        for (int i = 0; i < ia.length; i++) {
157            r = searchInterfaces(ia[i]);
158            if (r != null)
159                return r;
160        }
161        return null;
162    }
163
164    public ObjectRenderer getDefaultRenderer() {
165        return defaultRenderer;
166    }
167
168    public void clear() {
169        map.clear();
170    }
171
172    /**
173     * Register an {@link ObjectRenderer} for <code>clazz</code>.
174     */
175    public void put(Class clazz, ObjectRenderer or) {
176        map.put(clazz, or);
177    }
178}

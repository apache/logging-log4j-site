017package org.apache.log4j;
018
019import java.util.Collections;
020import java.util.Enumeration;
021import java.util.stream.Collectors;
022
023import org.apache.log4j.legacy.core.ContextUtil;
024import org.apache.log4j.spi.DefaultRepositorySelector;
025import org.apache.log4j.spi.LoggerFactory;
026import org.apache.log4j.spi.LoggerRepository;
027import org.apache.log4j.spi.NOPLoggerRepository;
028import org.apache.log4j.spi.RepositorySelector;
029import org.apache.log4j.spi.RootLogger;
030import org.apache.logging.log4j.spi.LoggerContext;
031import org.apache.logging.log4j.util.StackLocatorUtil;
032
033/**
034 * The main entry point to Log4j 1.
035 */
036public final class LogManager {
037
038    /**
039     * @deprecated This variable is for internal use only. It will become package protected in future versions.
040     */
041    @Deprecated
042    public static final String DEFAULT_CONFIGURATION_FILE = "log4j.properties";
043
044    /**
045     * @deprecated This variable is for internal use only. It will become private in future versions.
046     */
047    @Deprecated
048    public static final String DEFAULT_CONFIGURATION_KEY = "log4j.configuration";
049
050    /**
051     * @deprecated This variable is for internal use only. It will become private in future versions.
052     */
053    @Deprecated
054    public static final String CONFIGURATOR_CLASS_KEY = "log4j.configuratorClass";
055
056    /**
057     * @deprecated This variable is for internal use only. It will become private in future versions.
058     */
059    @Deprecated
060    public static final String DEFAULT_INIT_OVERRIDE_KEY = "log4j.defaultInitOverride";
061
062    static final String DEFAULT_XML_CONFIGURATION_FILE = "log4j.xml";
063
064    static private RepositorySelector repositorySelector;
065
066    private static final boolean LOG4J_CORE_PRESENT;
067
068    static {
069        LOG4J_CORE_PRESENT = checkLog4jCore();
070        //
071        // By default we use a DefaultRepositorySelector which always returns 'hierarchy'.
072        final Hierarchy hierarchy = new Hierarchy(new RootLogger(Level.DEBUG));
073        repositorySelector = new DefaultRepositorySelector(hierarchy);
074    }
075
076    private static boolean checkLog4jCore() {
077        try {
078            return Class.forName("org.apache.logging.log4j.core.LoggerContext") != null;
079        } catch (final Throwable ex) {
080            return false;
081        }
082    }
083
084    /**
085     * Tests if a logger for the given name exists.
086     *
087     * @param name logger name to test.
088     * @return whether a logger for the given name exists.
089     */
090    public static Logger exists(final String name) {
091        return exists(name, StackLocatorUtil.getCallerClassLoader(2));
092    }
093
094    static Logger exists(final String name, final ClassLoader classLoader) {
095        return getHierarchy().exists(name, classLoader);
096    }
097
098    /**
099     * Gets a LoggerContext.
100     *
101     * @param loader The ClassLoader for the context. If null the context will attempt to determine the appropriate
102     *        ClassLoader.
103     * @return a LoggerContext.
104     */
105    static LoggerContext getContext(final ClassLoader classLoader) {
106        return org.apache.logging.log4j.LogManager.getContext(classLoader, false);
107    }
108
109    /**
110     * Gets an enumeration of the current loggers.
111     *
112     * @return an enumeration of the current loggers.
113     */
114    @SuppressWarnings("rawtypes")
115    public static Enumeration getCurrentLoggers() {
116        return getCurrentLoggers(StackLocatorUtil.getCallerClassLoader(2));
117    }
118
119    @SuppressWarnings("rawtypes")
120    static Enumeration getCurrentLoggers(final ClassLoader classLoader) {
121        // @formatter:off
122        return Collections.enumeration(
123            LogManager.getContext(classLoader).getLoggerRegistry()
124                .getLoggers().stream().map(e -> LogManager.getLogger(e.getName(), classLoader))
125                .collect(Collectors.toList()));
126        // @formatter:on
127    }
128
129    static Hierarchy getHierarchy() {
130        final LoggerRepository loggerRepository = getLoggerRepository();
131        return loggerRepository instanceof Hierarchy ? (Hierarchy) loggerRepository : null;
132    }
133
134    /**
135     * Gets the logger for the given class.
136     */
137    public static Logger getLogger(final Class<?> clazz) {
138        final Hierarchy hierarchy = getHierarchy();
139        return hierarchy != null ? hierarchy.getLogger(clazz.getName(), StackLocatorUtil.getCallerClassLoader(2))
140            : getLoggerRepository().getLogger(clazz.getName());
141    }
142
143    /**
144     * Gets the logger for the given name.
145     */
146    public static Logger getLogger(final String name) {
147        final Hierarchy hierarchy = getHierarchy();
148        return hierarchy != null ? hierarchy.getLogger(name, StackLocatorUtil.getCallerClassLoader(2)) : getLoggerRepository().getLogger(name);
149    }
150
151    static Logger getLogger(final String name, final ClassLoader classLoader) {
152        final Hierarchy hierarchy = getHierarchy();
153        return hierarchy != null ? hierarchy.getLogger(name, classLoader) : getLoggerRepository().getLogger(name);
154    }
155
156    public static Logger getLogger(final String name, final LoggerFactory factory) {
157        final Hierarchy hierarchy = getHierarchy();
158        return hierarchy != null ? hierarchy.getLogger(name, factory, StackLocatorUtil.getCallerClassLoader(2))
159            : getLoggerRepository().getLogger(name, factory);
160    }
161
162    static Logger getLogger(final String name, final LoggerFactory factory, final ClassLoader classLoader) {
163        final Hierarchy hierarchy = getHierarchy();
164        return hierarchy != null ? hierarchy.getLogger(name, factory, classLoader) : getLoggerRepository().getLogger(name, factory);
165    }
166
167    public static LoggerRepository getLoggerRepository() {
168        if (repositorySelector == null) {
169            repositorySelector = new DefaultRepositorySelector(new NOPLoggerRepository());
170        }
171        return repositorySelector.getLoggerRepository();
172    }
173
174    /**
175     * Gets the root logger.
176     */
177    public static Logger getRootLogger() {
178        return getRootLogger(StackLocatorUtil.getCallerClassLoader(2));
179    }
180
181    static Logger getRootLogger(final ClassLoader classLoader) {
182        final Hierarchy hierarchy = getHierarchy();
183        return hierarchy != null ? hierarchy.getRootLogger(classLoader) : getLoggerRepository().getRootLogger();
184    }
185
186    static boolean isLog4jCorePresent() {
187        return LOG4J_CORE_PRESENT;
188    }
189
190    static void reconfigure(final ClassLoader classLoader) {
191        if (isLog4jCorePresent()) {
192            ContextUtil.reconfigure(LogManager.getContext(classLoader));
193        }
194    }
195
196    public static void resetConfiguration() {
197        resetConfiguration(StackLocatorUtil.getCallerClassLoader(2));
198    }
199
200    static void resetConfiguration(final ClassLoader classLoader) {
201        final Hierarchy hierarchy = getHierarchy();
202        if (hierarchy != null) {
203            hierarchy.resetConfiguration(classLoader);
204        } else {
205            getLoggerRepository().resetConfiguration();
206        }
207    }
208
209    public static void setRepositorySelector(final RepositorySelector selector, final Object guard) throws IllegalArgumentException {
210        if (selector == null) {
211            throw new IllegalArgumentException("RepositorySelector must be non-null.");
212        }
213        LogManager.repositorySelector = selector;
214    }
215
216    /**
217     * Shuts down the current configuration.
218     */
219    public static void shutdown() {
220        shutdown(StackLocatorUtil.getCallerClassLoader(2));
221    }
222
223    static void shutdown(final ClassLoader classLoader) {
224        final Hierarchy hierarchy = getHierarchy();
225        if (hierarchy != null) {
226            hierarchy.shutdown(classLoader);
227        } else {
228            getLoggerRepository().shutdown();
229        }
230    }
231
232}

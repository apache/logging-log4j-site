32  /**
33   * FileAppender appends log events to a file.
34   * <p>
35   * Support for <code>java.io.Writer</code> and console appending has been deprecated and then removed. See the
36   * replacement solutions: {@link WriterAppender} and {@link ConsoleAppender}.
37   * </p>
38   */
39  public class FileAppender extends WriterAppender {
40  
41      /**
42       * Controls file truncatation. The default value for this variable is <code>true</code>, meaning that by default a
43       * <code>FileAppender</code> will append to an existing file and not truncate it.
44       * <p>
45       * This option is meaningful only if the FileAppender opens the file.
46       * </p>
47       */
48      protected boolean fileAppend = true;
49  
50      /**
51       * The name of the log file.
52       */
53      protected String fileName = null;
54  
55      /**
56       * Do we do bufferedIO?
57       */
58      protected boolean bufferedIO = false;
59  
60      /**
61       * Determines the size of IO buffer be. Default is 8K.
62       */
63      protected int bufferSize = 8 * 1024;
64  
65      /**
66       * The default constructor does not do anything.
67       */
68      public FileAppender() {
69      }
70  
71      /**
72       * Constructs a FileAppender and open the file designated by <code>filename</code>. The opened filename will become the
73       * output destination for this appender.
74       * <p>
75       * The file will be appended to.
76       * </p>
77       */
78      public FileAppender(Layout layout, String filename) throws IOException {
79          this(layout, filename, true);
80      }
81  
82      /**
83       * Constructs a FileAppender and open the file designated by <code>filename</code>. The opened filename will become the
84       * output destination for this appender.
85       * <p>
86       * If the <code>append</code> parameter is true, the file will be appended to. Otherwise, the file designated by
87       * <code>filename</code> will be truncated before being opened.
88       * </p>
89       */
90      public FileAppender(Layout layout, String filename, boolean append) throws IOException {
91          this.layout = layout;
92          this.setFile(filename, append, false, bufferSize);
93      }
94  
95      /**
96       * Constructs a <code>FileAppender</code> and open the file designated by <code>filename</code>. The opened filename
97       * will become the output destination for this appender.
98       * <p>
99       * If the <code>append</code> parameter is true, the file will be appended to. Otherwise, the file designated by
100      * <code>filename</code> will be truncated before being opened.
101      * </p>
102      * <p>
103      * If the <code>bufferedIO</code> parameter is <code>true</code>, then buffered IO will be used to write to the output
104      * file.
105      * </p>
106      */
107     public FileAppender(Layout layout, String filename, boolean append, boolean bufferedIO, int bufferSize) throws IOException {
108         this.layout = layout;
109         this.setFile(filename, append, bufferedIO, bufferSize);
110     }
111 
112     /**
113      * If the value of <b>File</b> is not <code>null</code>, then {@link #setFile} is called with the values of <b>File</b>
114      * and <b>Append</b> properties.
115      * 
116      * @since 0.8.1
117      */
118     public void activateOptions() {
119         if (fileName != null) {
120             try {
121                 setFile(fileName, fileAppend, bufferedIO, bufferSize);
122             } catch (java.io.IOException e) {
123                 errorHandler.error("setFile(" + fileName + "," + fileAppend + ") call failed.", e, ErrorCode.FILE_OPEN_FAILURE);
124             }
125         } else {
126             // LogLog.error("File option not set for appender ["+name+"].");
127             LogLog.warn("File option not set for appender [" + name + "].");
128             LogLog.warn("Are you using FileAppender instead of ConsoleAppender?");
129         }
130     }
131 
132     /**
133      * Closes the previously opened file.
134      */
135     protected void closeFile() {
136         if (this.qw != null) {
137             try {
138                 this.qw.close();
139             } catch (java.io.IOException e) {
140                 if (e instanceof InterruptedIOException) {
141                     Thread.currentThread().interrupt();
142                 }
143                 // Exceptionally, it does not make sense to delegate to an
144                 // ErrorHandler. Since a closed appender is basically dead.
145                 LogLog.error("Could not close " + qw, e);
146             }
147         }
148     }

150     /**
151      * Returns the value of the <b>Append</b> option.
152      */
153     public boolean getAppend() {
154         return fileAppend;
155     }
156 
157     /**
158      * Get the value of the <b>BufferedIO</b> option.
159      * 
160      * <p>
161      * BufferedIO will significatnly increase performance on heavily loaded systems.
162      * </p>
163      */
164     public boolean getBufferedIO() {
165         return this.bufferedIO;
166     }
167 
168     /**
169      * Get the size of the IO buffer.
170      */
171     public int getBufferSize() {
172         return this.bufferSize;
173     }
174 
175     /** Returns the value of the <b>File</b> option. */
176     public String getFile() {
177         return fileName;
178     }
179 
180     /**
181      * Close any previously opened file and call the parent's <code>reset</code>.
182      */
183     protected void reset() {
184         closeFile();
185         this.fileName = null;
186         super.reset();
187     }
188 
189     /**
190      * The <b>Append</b> option takes a boolean value. It is set to <code>true</code> by default. If true, then
191      * <code>File</code> will be opened in append mode by {@link #setFile setFile} (see above). Otherwise, {@link #setFile
192      * setFile} will open <code>File</code> in truncate mode.
193      * 
194      * <p>
195      * Note: Actual opening of the file is made when {@link #activateOptions} is called, not when the options are set.
196      * </p>
197      */
198     public void setAppend(boolean flag) {
199         fileAppend = flag;
200     }
201 
202     /**
203      * The <b>BufferedIO</b> option takes a boolean value. It is set to <code>false</code> by default. If true, then
204      * <code>File</code> will be opened and the resulting {@link java.io.Writer} wrapped around a {@link BufferedWriter}.
205      * 
206      * BufferedIO will significatnly increase performance on heavily loaded systems.
207      * 
208      */
209     public void setBufferedIO(boolean bufferedIO) {
210         this.bufferedIO = bufferedIO;
211         if (bufferedIO) {
212             immediateFlush = false;
213         }
214     }
215 
216     /**
217      * Set the size of the IO buffer.
218      */
219     public void setBufferSize(int bufferSize) {
220         this.bufferSize = bufferSize;
221     }
222 
223     /**
224      * The <b>File</b> property takes a string value which should be the name of the file to append to.
225      * <p>
226      * <font color="#DD0044"><b>Note that the special values "System.out" or "System.err" are no longer honored.</b></font>
227      * </p>
228      * <p>
229      * Note: Actual opening of the file is made when {@link #activateOptions} is called, not when the options are set.
230      * </p>
231      */
232     public void setFile(String file) {
233         // Trim spaces from both ends. The users probably does not want
234         // trailing spaces in file names.
235         String val = file.trim();
236         fileName = val;
237     }
238 
239     /**
240      * Sets and <i>opens</i> the file where the log output will go. The specified file must be writable.
241      * <p>
242      * If there was already an opened file, then the previous file is closed first.
243      * </p>
244      * <p>
245      * <b>Do not use this method directly. To configure a FileAppender or one of its subclasses, set its properties one by
246      * one and then call activateOptions.</b>
247      * </p>
248      * 
249      * @param fileName The path to the log file.
250      * @param append If true will append to fileName. Otherwise will truncate fileName.
251      */
252     public synchronized void setFile(String fileName, boolean append, boolean bufferedIO, int bufferSize) throws IOException {
253         LogLog.debug("setFile called: " + fileName + ", " + append);
254 
255         // It does not make sense to have immediate flush and bufferedIO.
256         if (bufferedIO) {
257             setImmediateFlush(false);
258         }
259 
260         reset();
261         FileOutputStream ostream = null;
262         try {
263             //
264             // attempt to create file
265             //
266             ostream = new FileOutputStream(fileName, append);
267         } catch (FileNotFoundException ex) {
268             //
269             // if parent directory does not exist then
270             // attempt to create it and try to create file
271             // see bug 9150
272             //
273             String parentName = new File(fileName).getParent();
274             if (parentName != null) {
275                 File parentDir = new File(parentName);
276                 if (!parentDir.exists() && parentDir.mkdirs()) {
277                     ostream = new FileOutputStream(fileName, append);
278                 } else {
279                     throw ex;
280                 }
281             } else {
282                 throw ex;
283             }
284         }
285         Writer fw = createWriter(ostream);
286         if (bufferedIO) {
287             fw = new BufferedWriter(fw, bufferSize);
288         }
289         this.setQWForFiles(fw);
290         this.fileName = fileName;
291         this.fileAppend = append;
292         this.bufferedIO = bufferedIO;
293         this.bufferSize = bufferSize;
294         writeHeader();
295         LogLog.debug("setFile ended");
296     }
297 
298     /**
299      * Sets the quiet writer being used.
300      * 
301      * This method is overriden by {@link RollingFileAppender}.
302      */
303     protected void setQWForFiles(Writer writer) {
304         this.qw = new QuietWriter(writer, errorHandler);
305     }

54  public class HierarchyDynamicMBean extends AbstractDynamicMBean implements HierarchyEventListener, NotificationBroadcaster {
55  
56      static final String ADD_APPENDER = "addAppender.";
57      static final String THRESHOLD = "threshold";
58  
59      private static Logger log = Logger.getLogger(HierarchyDynamicMBean.class);
60      private final MBeanConstructorInfo[] dConstructors = new MBeanConstructorInfo[1];
61  
62      private final MBeanOperationInfo[] dOperations = new MBeanOperationInfo[1];
63      private final Vector vAttributes = new Vector();
64      private final String dClassName = this.getClass().getName();
65  
66      private final String dDescription = "This MBean acts as a management facade for org.apache.log4j.Hierarchy.";
67  
68      private final NotificationBroadcasterSupport nbs = new NotificationBroadcasterSupport();
69  
70      private final LoggerRepository hierarchy;
71  
72      public HierarchyDynamicMBean() {
73          hierarchy = LogManager.getLoggerRepository();
74          buildDynamicMBeanInfo();
75      }
76  
77      @Override
78      public void addAppenderEvent(final Category logger, final Appender appender) {
79          log.debug("addAppenderEvent called: logger=" + logger.getName() + ", appender=" + appender.getName());
80          final Notification n = new Notification(ADD_APPENDER + logger.getName(), this, 0);
81          n.setUserData(appender);
82          log.debug("sending notification.");
83          nbs.sendNotification(n);
84      }
85  
86      ObjectName addLoggerMBean(final Logger logger) {
87          final String name = logger.getName();
88          ObjectName objectName = null;
89          try {
90              final LoggerDynamicMBean loggerMBean = new LoggerDynamicMBean(logger);
91              objectName = new ObjectName("log4j", "logger", name);
92  
93              if (!server.isRegistered(objectName)) {
94                  registerMBean(loggerMBean, objectName);
95                  final NotificationFilterSupport nfs = new NotificationFilterSupport();
96                  nfs.enableType(ADD_APPENDER + logger.getName());
97                  log.debug("---Adding logger [" + name + "] as listener.");
98                  nbs.addNotificationListener(loggerMBean, nfs, null);
99                  vAttributes.add(new MBeanAttributeInfo("logger=" + name, "javax.management.ObjectName", "The " + name + " logger.", true, true, // this makes
100                                                                                                                                                 // the object
101                     // clickable
102                     false));
103 
104             }
105 
106         } catch (final JMException e) {
107             log.error("Could not add loggerMBean for [" + name + "].", e);
108         } catch (final RuntimeException e) {
109             log.error("Could not add loggerMBean for [" + name + "].", e);
110         }
111         return objectName;
112     }
113 
114     public ObjectName addLoggerMBean(final String name) {
115         final Logger cat = LogManager.exists(name);
116 
117         if (cat != null) {
118             return addLoggerMBean(cat);
119         } else {
120             return null;
121         }
122     }
123 
124     @Override
125     public void addNotificationListener(final NotificationListener listener, final NotificationFilter filter, final java.lang.Object handback) {
126         nbs.addNotificationListener(listener, filter, handback);
127     }
128 
129     private void buildDynamicMBeanInfo() {
130         final Constructor[] constructors = this.getClass().getConstructors();
131         dConstructors[0] = new MBeanConstructorInfo("HierarchyDynamicMBean(): Constructs a HierarchyDynamicMBean instance", constructors[0]);
132 
133         vAttributes.add(new MBeanAttributeInfo(THRESHOLD, "java.lang.String", "The \"threshold\" state of the hiearchy.", true, true, false));
134 
135         final MBeanParameterInfo[] params = new MBeanParameterInfo[1];
136         params[0] = new MBeanParameterInfo("name", "java.lang.String", "Create a logger MBean");
137         dOperations[0] = new MBeanOperationInfo("addLoggerMBean", "addLoggerMBean(): add a loggerMBean", params, "javax.management.ObjectName",
138             MBeanOperationInfo.ACTION);
139     }
140 
141     @Override
142     public Object getAttribute(final String attributeName) throws AttributeNotFoundException, MBeanException, ReflectionException {
143 
144         // Check attributeName is not null to avoid NullPointerException later on
145         if (attributeName == null) {
146             throw new RuntimeOperationsException(new IllegalArgumentException("Attribute name cannot be null"),
147                 "Cannot invoke a getter of " + dClassName + " with null attribute name");
148         }
149 
150         log.debug("Called getAttribute with [" + attributeName + "].");
151 
152         // Check for a recognized attributeName and call the corresponding getter
153         if (attributeName.equals(THRESHOLD)) {
154             return hierarchy.getThreshold();
155         } else if (attributeName.startsWith("logger")) {
156             final int k = attributeName.indexOf("%3D");
157             String val = attributeName;
158             if (k > 0) {
159                 val = attributeName.substring(0, k) + '=' + attributeName.substring(k + 3);
160             }
161             try {
162                 return new ObjectName("log4j:" + val);
163             } catch (final JMException e) {
164                 log.error("Could not create ObjectName" + val);
165             } catch (final RuntimeException e) {
166                 log.error("Could not create ObjectName" + val);
167             }
168         }
169 
170         // If attributeName has not been recognized throw an AttributeNotFoundException
171         throw (new AttributeNotFoundException("Cannot find " + attributeName + " attribute in " + dClassName));
172 
173     }
174 
175     @Override
176     protected Logger getLogger() {
177         return log;
178     }

180     @Override
181     public MBeanInfo getMBeanInfo() {
182         // cat.debug("getMBeanInfo called.");
183 
184         final MBeanAttributeInfo[] attribs = new MBeanAttributeInfo[vAttributes.size()];
185         vAttributes.toArray(attribs);
186 
187         return new MBeanInfo(dClassName, dDescription, attribs, dConstructors, dOperations, new MBeanNotificationInfo[0]);
188     }
189 
190     @Override
191     public MBeanNotificationInfo[] getNotificationInfo() {
192         return nbs.getNotificationInfo();
193     }
194 
195     @Override
196     public Object invoke(final String operationName, final Object params[], final String signature[]) throws MBeanException, ReflectionException {
197 
198         if (operationName == null) {
199             throw new RuntimeOperationsException(new IllegalArgumentException("Operation name cannot be null"),
200                 "Cannot invoke a null operation in " + dClassName);
201         }
202         // Check for a recognized operation name and call the corresponding operation
203 
204         if (operationName.equals("addLoggerMBean")) {
205             return addLoggerMBean((String) params[0]);
206         } else {
207             throw new ReflectionException(new NoSuchMethodException(operationName), "Cannot find the operation " + operationName + " in " + dClassName);
208         }
209 
210     }
211 
212     @Override
213     public void postRegister(final java.lang.Boolean registrationDone) {
214         log.debug("postRegister is called.");
215         hierarchy.addHierarchyEventListener(this);
216         final Logger root = hierarchy.getRootLogger();
217         addLoggerMBean(root);
218     }
219 
220     @Override
221     public void removeAppenderEvent(final Category cat, final Appender appender) {
222         log.debug("removeAppenderCalled: logger=" + cat.getName() + ", appender=" + appender.getName());
223     }
224 
225     @Override
226     public void removeNotificationListener(final NotificationListener listener) throws ListenerNotFoundException {
227         nbs.removeNotificationListener(listener);
228     }
229 
230     @Override
231     public void setAttribute(final Attribute attribute) throws AttributeNotFoundException, InvalidAttributeValueException, MBeanException, ReflectionException {
232 
233         // Check attribute is not null to avoid NullPointerException later on
234         if (attribute == null) {
235             throw new RuntimeOperationsException(new IllegalArgumentException("Attribute cannot be null"),
236                 "Cannot invoke a setter of " + dClassName + " with null attribute");
237         }
238         final String name = attribute.getName();
239         final Object value = attribute.getValue();
240 
241         if (name == null) {
242             throw new RuntimeOperationsException(new IllegalArgumentException("Attribute name cannot be null"),
243                 "Cannot invoke the setter of " + dClassName + " with null attribute name");
244         }
245 
246         if (name.equals(THRESHOLD)) {
247             final Level l = OptionConverter.toLevel((String) value, hierarchy.getThreshold());
248             hierarchy.setThreshold(l);
249         }
250 
251     }

56      @Override
57      public void logMessage(final String fqcn, final Level level, final Marker marker, final Message message, final Throwable t) {
58          java.util.logging.Level julLevel = convertLevel(level);
59          if (!logger.isLoggable(julLevel)) {
60              return;
61          }
62          LazyLog4jLogRecord record = new LazyLog4jLogRecord(fqcn, julLevel, message.getFormattedMessage()); // NOT getFormat()
63          // NOT record.setParameters(message.getParameters()); BECAUSE getFormattedMessage() NOT getFormat()
64          record.setLoggerName(getName());
65          record.setThrown(t == null ? message.getThrowable() : t);
66          logger.log(record);
67      }

69      // Convert Level in Log4j scale to JUL scale.
70      // See getLevel() for the mapping. Note that JUL's FINEST & CONFIG are never returned because Log4j has no such levels, and
71      // that Log4j's FATAL is simply mapped to JUL's SEVERE as is Log4j's ERROR because JUL does not distinguish between ERROR and FATAL.
72      private java.util.logging.Level convertLevel(final Level level) {
73          switch (level.getStandardLevel()) {
74              // Test in logical order of likely frequency of use
75              // Must be kept in sync with #getLevel()
76              case ALL:
77                  return java.util.logging.Level.ALL;
78              case TRACE:
79                  return java.util.logging.Level.FINER;
80              case DEBUG:
81                  return java.util.logging.Level.FINE;
82              case INFO:
83                  return java.util.logging.Level.INFO;
84              case WARN:
85                  return java.util.logging.Level.WARNING;
86              case ERROR:
87                  return java.util.logging.Level.SEVERE;
88              case FATAL:
89                  return java.util.logging.Level.SEVERE;
90              case OFF:
91                  return java.util.logging.Level.OFF;
92              default:
93                  // This is tempting: throw new IllegalStateException("Impossible Log4j Level encountered: " + level.intLevel());
94                  // But it's not a great idea, security wise. If an attacker *SOMEHOW* managed to create a Log4j Level instance
95                  // with an unexpected level (through JVM de-serialization, despite readResolve() { return Level.valueOf(this.name); },
96                  // or whatever other means), then we would blow up in a very unexpected place and way. Let us therefore instead just
97                  // return SEVERE for unexpected values, because that's more likely to be noticed than a FINER.
98                  // Greetings, Michael Vorburger.ch <http://www.vorburger.ch>, for Google, on 2021.12.24.
99                  return java.util.logging.Level.SEVERE;
100         }
101     }

103     /**
104      * Level in Log4j scale.
105      * JUL Levels are mapped as follows:
106      * <ul>
107      * <li>OFF => OFF
108      * <li>SEVERE => ERROR
109      * <li>WARNING => WARN
110      * <li>INFO => INFO
111      * <li>CONFIG => INFO
112      * <li>FINE => DEBUG
113      * <li>FINER => TRACE (as in https://github.com/apache/logging-log4j2/blob/a58a06bf2365165ac5abdde931bb4ecd1adf0b3c/log4j-jul/src/main/java/org/apache/logging/log4j/jul/DefaultLevelConverter.java#L55-L75)
114      * <li>FINEST => TRACE
115      * <li>ALL => ALL
116      * </ul>
117      *
118      * Numeric JUL Levels that don't match the known levels are matched to the closest one.
119      * For example, anything between OFF (Integer.MAX_VALUE) and SEVERE (1000) is returned as a Log4j FATAL.
120      */
121     @Override
122     public Level getLevel() {
123         int julLevel = getEffectiveJULLevel().intValue();
124         // Test in logical order of likely frequency of use
125         // Must be kept in sync with #convertLevel()
126         if (julLevel == java.util.logging.Level.ALL.intValue()) {
127             return Level.ALL;
128         }
129         if (julLevel <= java.util.logging.Level.FINER.intValue()) {
130             return Level.TRACE;
131         }
132         if (julLevel <= java.util.logging.Level.FINE.intValue()) {  // includes FINER
133             return Level.DEBUG;
134         }
135         if (julLevel <= java.util.logging.Level.INFO.intValue()) { // includes CONFIG
136             return Level.INFO;
137         }
138         if (julLevel <= java.util.logging.Level.WARNING.intValue()) {
139             return Level.WARN;
140         }
141         if (julLevel <= java.util.logging.Level.SEVERE.intValue()) {
142             return Level.ERROR;
143         }
144         return Level.OFF;
145     }
146 
147     private java.util.logging.Level getEffectiveJULLevel() {
148         Logger current = logger;
149         while (current.getLevel() == null && current.getParent() != null) {
150             current = current.getParent();
151         }
152         if (current.getLevel() != null) {
153             return current.getLevel();
154         }
155         // This is a safety fallback that is typically never reached, because usually the root Logger.getLogger("") has a Level.
156         // Since JDK 8 the LogManager$RootLogger does not have a default level, just a default effective level of INFO.
157         return java.util.logging.Level.INFO;
158     }

160     private boolean isEnabledFor(final Level level, final Marker marker) {
161         // E.g. we're logging WARN and more, so getLevel() is 300, if we're asked if we're
162         // enabled for level ERROR which is 200, isLessSpecificThan() tests for >= so return true.
163         return getLevel().isLessSpecificThan(level);
164     }

095    private JsonTemplateLayout(final Builder builder) {
096        this.charset = builder.charset;
097        this.contentType = "application/json; charset=" + charset;
098        final String eventDelimiterSuffix = builder.isNullEventDelimiterEnabled() ? "\0" : "";
099        this.eventDelimiter = builder.eventDelimiter + eventDelimiterSuffix;
100        final Configuration configuration = builder.configuration;
101        final JsonWriter jsonWriter = JsonWriter
102                .newBuilder()
103                .setMaxStringLength(builder.maxStringLength)
104                .setTruncatedStringSuffix(builder.truncatedStringSuffix)
105                .build();
106        this.eventResolver = createEventResolver(
107                builder,
108                configuration,
109                charset,
110                jsonWriter);
111        this.contextRecycler = createContextRecycler(builder, jsonWriter);
112    }
113
114    private TemplateResolver<LogEvent> createEventResolver(
115            final Builder builder,
116            final Configuration configuration,
117            final Charset charset,
118            final JsonWriter jsonWriter) {
119
120        // Inject resolver factory and interceptor plugins.
121        final List<String> pluginPackages = configuration.getPluginPackages();
122        final Map<String, EventResolverFactory> resolverFactoryByName =
123                EventResolverFactories.populateResolverFactoryByName(pluginPackages);
124        final List<EventResolverInterceptor> resolverInterceptors =
125                EventResolverInterceptors.populateInterceptors(pluginPackages);
126        final EventResolverStringSubstitutor substitutor =
127                new EventResolverStringSubstitutor(configuration.getStrSubstitutor());
128
129        // Read event and stack trace element templates.
130        final String eventTemplate = readEventTemplate(builder);
131        final String stackTraceElementTemplate = readStackTraceElementTemplate(builder);
132
133        // Determine the max. string byte count.
134        final float maxByteCountPerChar = builder.charset.newEncoder().maxBytesPerChar();
135        final int maxStringByteCount =
136                Math.toIntExact(Math.round(Math.ceil(
137                        maxByteCountPerChar * builder.maxStringLength)));
138
139        // Replace null event template additional fields with an empty array.
140        final EventTemplateAdditionalField[] eventTemplateAdditionalFields =
141                builder.eventTemplateAdditionalFields != null
142                        ? builder.eventTemplateAdditionalFields
143                        : EventTemplateAdditionalField.EMPTY_ARRAY;
144
145        // Create the resolver context.
146        final EventResolverContext resolverContext = EventResolverContext
147                .newBuilder()
148                .setConfiguration(configuration)
149                .setResolverFactoryByName(resolverFactoryByName)
150                .setResolverInterceptors(resolverInterceptors)
151                .setSubstitutor(substitutor)
152                .setCharset(charset)
153                .setJsonWriter(jsonWriter)
154                .setRecyclerFactory(builder.recyclerFactory)
155                .setMaxStringByteCount(maxStringByteCount)
156                .setTruncatedStringSuffix(builder.truncatedStringSuffix)
157                .setLocationInfoEnabled(builder.locationInfoEnabled)
158                .setStackTraceEnabled(builder.stackTraceEnabled)
159                .setStackTraceElementTemplate(stackTraceElementTemplate)
160                .setEventTemplateRootObjectKey(builder.eventTemplateRootObjectKey)
161                .setEventTemplateAdditionalFields(eventTemplateAdditionalFields)
162                .build();
163
164        // Compile the resolver template.
165        return TemplateResolvers.ofTemplate(resolverContext, eventTemplate);
166
167    }

169    private static String readEventTemplate(final Builder builder) {
170        return readTemplate(
171                builder.eventTemplate,
172                builder.eventTemplateUri,
173                builder.charset);
174    }
175
176    private static String readStackTraceElementTemplate(final Builder builder) {
177        return readTemplate(
178                builder.stackTraceElementTemplate,
179                builder.stackTraceElementTemplateUri,
180                builder.charset);
181    }
182
183    private static String readTemplate(
184            final String template,
185            final String templateUri,
186            final Charset charset) {
187        return Strings.isBlank(template)
188                ? Uris.readUri(templateUri, charset)
189                : template;
190    }
191
192    private static Recycler<Context> createContextRecycler(
193            final Builder builder,
194            final JsonWriter jsonWriter) {
195        final Supplier<Context> supplier =
196                createContextSupplier(builder.charset, jsonWriter);
197        return builder
198                .recyclerFactory
199                .create(supplier, Context::close);
200    }
201
202    private static Supplier<Context> createContextSupplier(
203            final Charset charset,
204            final JsonWriter jsonWriter) {
205        return () -> {
206            final JsonWriter clonedJsonWriter = jsonWriter.clone();
207            final Encoder<StringBuilder> encoder = createStringBuilderEncoder(charset);
208            return new Context(clonedJsonWriter, encoder);
209        };
210    }
211
212    private static Encoder<StringBuilder> createStringBuilderEncoder(
213            final Charset charset) {
214        if (Constants.ENABLE_DIRECT_ENCODERS) {
215            return Constants.ENABLE_THREADLOCALS
216                    ? new StringBuilderEncoder(charset)
217                    : new LockingStringBuilderEncoder(charset);
218        }
219        return null;
220    }

222    @Override
223    public byte[] toByteArray(final LogEvent event) {
224        final String eventJson = toSerializable(event);
225        return StringEncoder.toBytes(eventJson, charset);
226    }
227
228    @Override
229    public String toSerializable(final LogEvent event) {
230        final Context context = acquireContext();
231        final JsonWriter jsonWriter = context.jsonWriter;
232        final StringBuilder stringBuilder = jsonWriter.getStringBuilder();
233        try {
234            eventResolver.resolve(event, jsonWriter);
235            stringBuilder.append(eventDelimiter);
236            return stringBuilder.toString();
237        } finally {
238            contextRecycler.release(context);
239        }
240    }
241
242    @Override
243    public void encode(final LogEvent event, final ByteBufferDestination destination) {
244
245        // Acquire a context.
246        final Context context = acquireContext();
247        final JsonWriter jsonWriter = context.jsonWriter;
248        final StringBuilder stringBuilder = jsonWriter.getStringBuilder();
249        final Encoder<StringBuilder> encoder = context.encoder;
250
251        try {
252
253            // Render the JSON.
254            eventResolver.resolve(event, jsonWriter);
255            stringBuilder.append(eventDelimiter);
256
257            // Write to the destination.
258            if (encoder == null) {
259                final String eventJson = stringBuilder.toString();
260                final byte[] eventJsonBytes = StringEncoder.toBytes(eventJson, charset);
261                destination.writeBytes(eventJsonBytes, 0, eventJsonBytes.length);
262            } else {
263                encoder.encode(stringBuilder, destination);
264            }
265
266        }
267
268        // Release the context.
269        finally {
270            contextRecycler.release(context);
271        }
272
273    }

518        private void validate() {
519            Objects.requireNonNull(configuration, "config");
520            if (Strings.isBlank(eventTemplate) && Strings.isBlank(eventTemplateUri)) {
521                    throw new IllegalArgumentException(
522                            "both eventTemplate and eventTemplateUri are blank");
523            }
524            if (stackTraceEnabled &&
525                    Strings.isBlank(stackTraceElementTemplate)
526                    && Strings.isBlank(stackTraceElementTemplateUri)) {
527                throw new IllegalArgumentException(
528                        "both stackTraceElementTemplate and stackTraceElementTemplateUri are blank");
529            }
530            if (maxStringLength <= 0) {
531                throw new IllegalArgumentException(
532                        "was expecting a non-zero positive maxStringLength: " +
533                                maxStringLength);
534            }
535            Objects.requireNonNull(truncatedStringSuffix, "truncatedStringSuffix");
536            Objects.requireNonNull(recyclerFactory, "recyclerFactory");
537        }

541    @Plugin(name = "EventTemplateAdditionalField",
542            category = Node.CATEGORY,
543            printObject = true)
544    public static final class EventTemplateAdditionalField {
545
546        static final EventTemplateAdditionalField[] EMPTY_ARRAY = {};
547
548        public enum Format { STRING, JSON }
549
550        private final String key;
551
552        private final String value;
553
554        private final Format format;
555
556        private EventTemplateAdditionalField(final Builder builder) {
557            this.key = builder.key;
558            this.value = builder.value;
559            this.format = builder.format;
560        }
561
562        public String getKey() {
563            return key;
564        }
565
566        public String getValue() {
567            return value;
568        }
569
570        public Format getFormat() {
571            return format;
572        }
573
574        @Override
575        public boolean equals(Object object) {
576            if (this == object) {
577                return true;
578            }
579            if (object == null || getClass() != object.getClass()) {
580                return false;
581            }
582            EventTemplateAdditionalField that = (EventTemplateAdditionalField) object;
583            return key.equals(that.key) &&
584                    value.equals(that.value) &&
585                    format == that.format;
586        }
587
588        @Override
589        public int hashCode() {
590            return Objects.hash(key, value, format);
591        }
592
593        @Override
594        public String toString() {
595            final String formattedValue = Format.STRING.equals(format)
596                    ? String.format("\"%s\"", value)
597                    : value;
598            return String.format("%s=%s", key, formattedValue);
599        }
600
601        @PluginBuilderFactory
602        public static EventTemplateAdditionalField.Builder newBuilder() {
603            return new EventTemplateAdditionalField.Builder();
604        }

017package org.apache.logging.log4j.core.config;
018
019import java.util.Objects;
020
021import org.apache.logging.log4j.Logger;
022import org.apache.logging.log4j.core.config.plugins.Plugin;
023import org.apache.logging.log4j.core.config.plugins.PluginAttribute;
024import org.apache.logging.log4j.core.config.plugins.PluginFactory;
025import org.apache.logging.log4j.core.config.plugins.PluginValue;
026import org.apache.logging.log4j.core.lookup.StrSubstitutor;
027import org.apache.logging.log4j.status.StatusLogger;
028import org.apache.logging.log4j.util.Strings;
029
030/**
031 * Represents a key/value pair in the configuration.
032 */
033@Plugin(name = "property", category = Node.CATEGORY, printObject = true)
034public final class Property {
035
036    /**
037     * @since 2.11.2
038     */
039    public static final Property[] EMPTY_ARRAY = {};
040
041    private static final Logger LOGGER = StatusLogger.getLogger();
042
043    private final String name;
044    private final String rawValue;
045    private final String value;
046    private final boolean valueNeedsLookup;
047
048    private Property(final String name, final String rawValue, final String value) {
049        this.name = name;
050        this.rawValue = rawValue;
051        this.value = value;
052        this.valueNeedsLookup = value != null && value.contains("${");
053    }
054
055    /**
056     * Returns the property name.
057     * @return the property name.
058     */
059    public String getName() {
060        return name;
061    }
062
063    /**
064     * Returns the original raw property value without substitution.
065     * @return the raw value of the property, or empty string if it is not set.
066     */
067    public String getRawValue() {
068        return Objects.toString(rawValue, Strings.EMPTY);
069    }
070
071    /**
072     * Returns the property value.
073     * @return the value of the property.
074     */
075    public String getValue() {
076        return Objects.toString(value, Strings.EMPTY); // LOG4J2-1313 null would be same as Property not existing
077    }
078
079    /**
080     * Returns {@code true} if the value contains a substitutable property that requires a lookup to be resolved.
081     * @return {@code true} if the value contains {@code "${"}, {@code false} otherwise
082     */
083    public boolean isValueNeedsLookup() {
084        return valueNeedsLookup;
085    }
086
087    /**
088     * Evaluate this property with the provided substitutor. If {@link #isValueNeedsLookup()} is {@code false},
089     * the {@link #getValue() value} is returned, otherwise the {@link #getRawValue() raw value} is evaluated
090     * with the given substitutor.
091     */
092    public String evaluate(StrSubstitutor substitutor) {
093        return valueNeedsLookup
094                // Unescape the raw value first, handling '$${ctx:foo}' -> '${ctx:foo}'
095                ? substitutor.replace(PropertiesPlugin.unescape(getRawValue()))
096                : getValue();
097    }
098
099    /**
100     * Creates a Property.
101     *
102     * @param name The key.
103     * @param value The value.
104     * @return A Property.
105     */
106    public static Property createProperty(final String name, final String value) {
107        if (name == null) {
108            LOGGER.error("Property name cannot be null");
109        }
110        return new Property(name, value, value);
111    }
112
113    /**
114     * Creates a Property.
115     *
116     * @param name The key.
117     * @param rawValue The value without any substitution applied.
118     * @param value The value.
119     * @return A Property.
120     */
121    @PluginFactory
122    public static Property createProperty(
123            @PluginAttribute("name") final String name,
124            @PluginValue(value = "value", substitute = false) final String rawValue,
125            @PluginValue("value") final String value) {
126        if (name == null) {
127            LOGGER.error("Property name cannot be null");
128        }
129        return new Property(name, rawValue, value);
130    }
131
132    @Override
133    public String toString() {
134        return name + '=' + getValue();
135    }
136}

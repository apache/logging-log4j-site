112    @Override
113    public void doConfigure() {
114        if (properties == null) {
115            properties = new Properties();
116            final InputStream inputStream = getConfigurationSource().getInputStream();
117            if (inputStream != null) {
118                try {
119                    properties.load(inputStream);
120                } catch (final Exception e) {
121                    LOGGER.error("Could not read configuration file [{}].", getConfigurationSource().toString(), e);
122                    return;
123                }
124            }
125        }
126        // If we reach here, then the config file is alright.
127        doConfigure(properties);
128    }

130    @Override
131    public Configuration reconfigure() {
132        try {
133            final ConfigurationSource source = getConfigurationSource().resetInputStream();
134            if (source == null) {
135                return null;
136            }
137            final Configuration config = new PropertiesConfigurationFactory().getConfiguration(getLoggerContext(), source);
138            return config == null || config.getState() != State.INITIALIZING ? null : config;
139        } catch (final IOException ex) {
140            LOGGER.error("Cannot locate file {}: {}", getConfigurationSource(), ex);
141        }
142        return null;
143    }

145    /**
146     * Reads a configuration from a file. <b>The existing configuration is not cleared nor reset.</b> If you require a
147     * different behavior, then call {@link LogManager#resetConfiguration resetConfiguration} method before calling
148     * <code>doConfigure</code>.
149     * <p>
150     * The configuration file consists of statements in the format <code>key=value</code>. The syntax of different
151     * configuration elements are discussed below.
152     * </p>
153     * <p>
154     * The level value can consist of the string values OFF, FATAL, ERROR, WARN, INFO, DEBUG, ALL or a <em>custom level</em>
155     * value. A custom level value can be specified in the form level#classname. By default the repository-wide threshold is
156     * set to the lowest possible value, namely the level <code>ALL</code>.
157     * </p>
158     *
159     * <h3>Appender configuration</h3>
160     * <p>
161     * Appender configuration syntax is:
162     * </p>
163     * <pre>
164     * # For appender named <i>appenderName</i>, set its class.
165     * # Note: The appender name can contain dots.
166     * log4j.appender.appenderName=fully.qualified.name.of.appender.class
167     *
168     * # Set appender specific options.
169     * log4j.appender.appenderName.option1=value1
170     * ...
171     * log4j.appender.appenderName.optionN=valueN
172     * </pre>
173     * <p>
174     * For each named appender you can configure its {@link Layout}. The syntax for configuring an appender's layout is:
175     * </p>
176     * <pre>
177     * log4j.appender.appenderName.layout=fully.qualified.name.of.layout.class
178     * log4j.appender.appenderName.layout.option1=value1
179     * ....
180     * log4j.appender.appenderName.layout.optionN=valueN
181     * </pre>
182     * <p>
183     * The syntax for adding {@link Filter}s to an appender is:
184     * </p>
185     * <pre>
186     * log4j.appender.appenderName.filter.ID=fully.qualified.name.of.filter.class
187     * log4j.appender.appenderName.filter.ID.option1=value1
188     * ...
189     * log4j.appender.appenderName.filter.ID.optionN=valueN
190     * </pre>
191     * <p>
192     * The first line defines the class name of the filter identified by ID; subsequent lines with the same ID specify
193     * filter option - value pairs. Multiple filters are added to the appender in the lexicographic order of IDs.
194     * </p>
195     * <p>
196     * The syntax for adding an {@link ErrorHandler} to an appender is:
197     * </p>
198     * <pre>
199     * log4j.appender.appenderName.errorhandler=fully.qualified.name.of.errorhandler.class
200     * log4j.appender.appenderName.errorhandler.appender-ref=appenderName
201     * log4j.appender.appenderName.errorhandler.option1=value1
202     * ...
203     * log4j.appender.appenderName.errorhandler.optionN=valueN
204     * </pre>
205     *
206     * <h3>Configuring loggers</h3>
207     * <p>
208     * The syntax for configuring the root logger is:
209     * </p>
210     * <pre>
211     * log4j.rootLogger=[level], appenderName, appenderName, ...
212     * </pre>
213     * <p>
214     * This syntax means that an optional <em>level</em> can be supplied followed by appender names separated by commas.
215     * </p>
216     * <p>
217     * The level value can consist of the string values OFF, FATAL, ERROR, WARN, INFO, DEBUG, ALL or a <em>custom level</em>
218     * value. A custom level value can be specified in the form <code>level#classname</code>.
219     * </p>
220     * <p>
221     * If a level value is specified, then the root level is set to the corresponding level. If no level value is specified,
222     * then the root level remains untouched.
223     * </p>
224     * <p>
225     * The root logger can be assigned multiple appenders.
226     * </p>
227     * <p>
228     * Each <i>appenderName</i> (separated by commas) will be added to the root logger. The named appender is defined using
229     * the appender syntax defined above.
230     * </p>
231     * <p>
232     * For non-root categories the syntax is almost the same:
233     * </p>
234     * <pre>
235     * log4j.logger.logger_name=[level|INHERITED|NULL], appenderName, appenderName, ...
236     * </pre>
237     * <p>
238     * The meaning of the optional level value is discussed above in relation to the root logger. In addition however, the
239     * value INHERITED can be specified meaning that the named logger should inherit its level from the logger hierarchy.
240     * </p>
241     * <p>
242     * If no level value is supplied, then the level of the named logger remains untouched.
243     * </p>
244     * <p>
245     * By default categories inherit their level from the hierarchy. However, if you set the level of a logger and later
246     * decide that that logger should inherit its level, then you should specify INHERITED as the value for the level value.
247     * NULL is a synonym for INHERITED.
248     * </p>
249     * <p>
250     * Similar to the root logger syntax, each <i>appenderName</i> (separated by commas) will be attached to the named
251     * logger.
252     * </p>
253     * <p>
254     * See the <a href="../../../../manual.html#additivity">appender additivity rule</a> in the user manual for the meaning
255     * of the <code>additivity</code> flag.
256     * </p>
257     * <pre>
258     * # Set options for appender named "A1". # Appender "A1" will be a SyslogAppender
259     * log4j.appender.A1=org.apache.log4j.net.SyslogAppender
260     *
261     * # The syslog daemon resides on www.abc.net log4j.appender.A1.SyslogHost=www.abc.net
262     *
263     * # A1's layout is a PatternLayout, using the conversion pattern # <b>%r %-5p %c{2} %M.%L %x - %m\n</b>. Thus, the log
264     * output will # include # the relative time since the start of the application in # milliseconds, followed by the level
265     * of the log request, # followed by the two rightmost components of the logger name, # followed by the callers method
266     * name, followed by the line number, # the nested diagnostic context and finally the message itself. # Refer to the
267     * documentation of {@link PatternLayout} for further information # on the syntax of the ConversionPattern key.
268     * log4j.appender.A1.layout=org.apache.log4j.PatternLayout log4j.appender.A1.layout.ConversionPattern=%-4r %-5p %c{2}
269     * %M.%L %x - %m\n
270     *
271     * # Set options for appender named "A2" # A2 should be a RollingFileAppender, with maximum file size of 10 MB # using
272     * at most one backup file. A2's layout is TTCC, using the # ISO8061 date format with context printing enabled.
273     * log4j.appender.A2=org.apache.log4j.RollingFileAppender log4j.appender.A2.MaxFileSize=10MB
274     * log4j.appender.A2.MaxBackupIndex=1 log4j.appender.A2.layout=org.apache.log4j.TTCCLayout
275     * log4j.appender.A2.layout.ContextPrinting=enabled log4j.appender.A2.layout.DateFormat=ISO8601
276     *
277     * # Root logger set to DEBUG using the A2 appender defined above. log4j.rootLogger=DEBUG, A2
278     *
279     * # Logger definitions: # The SECURITY logger inherits is level from root. However, it's output # will go to A1
280     * appender defined above. It's additivity is non-cumulative. log4j.logger.SECURITY=INHERIT, A1
281     * log4j.additivity.SECURITY=false
282     *
283     * # Only warnings or above will be logged for the logger "SECURITY.access". # Output will go to A1.
284     * log4j.logger.SECURITY.access=WARN
285     *
286     *
287     * # The logger "class.of.the.day" inherits its level from the # logger hierarchy. Output will go to the appender's of
288     * the root # logger, A2 in this case. log4j.logger.class.of.the.day=INHERIT
289     * </pre>
290     * <p>
291     * Refer to the <b>setOption</b> method in each Appender and Layout for class specific options.
292     * </p>
293     * <p>
294     * Use the <code>#</code> or <code>!</code> characters at the beginning of a line for comments.
295     * </p>
296     */
297    private void doConfigure(final Properties properties) {
298        String status = "error";
299        String value = properties.getProperty(DEBUG_KEY);
300        if (value == null) {
301            value = properties.getProperty("log4j.configDebug");
302            if (value != null) {
303                LOGGER.warn("[log4j.configDebug] is deprecated. Use [log4j.debug] instead.");
304            }
305        }
306
307        if (value != null) {
308            status = OptionConverter.toBoolean(value, false) ? "debug" : "error";
309        }
310
311        final StatusConfiguration statusConfig = new StatusConfiguration().withStatus(status);
312        statusConfig.initialize();
313
314        // if log4j.reset=true then reset hierarchy
315        final String reset = properties.getProperty(RESET_KEY);
316        if (reset != null && OptionConverter.toBoolean(reset, false)) {
317            LogManager.resetConfiguration();
318        }
319
320        final String threshold = OptionConverter.findAndSubst(THRESHOLD_KEY, properties);
321        if (threshold != null) {
322            final Level level = OptionConverter.convertLevel(threshold.trim(), Level.ALL);
323            addFilter(ThresholdFilter.createFilter(level, Result.NEUTRAL, Result.DENY));
324        }
325
326        configureRoot(properties);
327        parseLoggers(properties);
328
329        LOGGER.debug("Finished configuring.");
330    }

336    private void configureRoot(final Properties props) {
337        String effectiveFrefix = ROOT_LOGGER_PREFIX;
338        String value = OptionConverter.findAndSubst(ROOT_LOGGER_PREFIX, props);
339
340        if (value == null) {
341            value = OptionConverter.findAndSubst(ROOT_CATEGORY_PREFIX, props);
342            effectiveFrefix = ROOT_CATEGORY_PREFIX;
343        }
344
345        if (value == null) {
346            LOGGER.debug("Could not find root logger information. Is this OK?");
347        } else {
348            final LoggerConfig root = getRootLogger();
349            parseLogger(props, root, effectiveFrefix, INTERNAL_ROOT_NAME, value);
350        }
351    }

353    /**
354     * Parses non-root elements, such non-root categories and renderers.
355     */
356    private void parseLoggers(final Properties props) {
357        final Enumeration<?> enumeration = props.propertyNames();
358        while (enumeration.hasMoreElements()) {
359            final String key = Objects.toString(enumeration.nextElement(), null);
360            if (key.startsWith(CATEGORY_PREFIX) || key.startsWith(LOGGER_PREFIX)) {
361                String loggerName = null;
362                if (key.startsWith(CATEGORY_PREFIX)) {
363                    loggerName = key.substring(CATEGORY_PREFIX.length());
364                } else if (key.startsWith(LOGGER_PREFIX)) {
365                    loggerName = key.substring(LOGGER_PREFIX.length());
366                }
367                final String value = OptionConverter.findAndSubst(key, props);
368                LoggerConfig loggerConfig = getLogger(loggerName);
369                if (loggerConfig == null) {
370                    final boolean additivity = getAdditivityForLogger(props, loggerName);
371                    loggerConfig = new LoggerConfig(loggerName, org.apache.logging.log4j.Level.ERROR, additivity);
372                    addLogger(loggerName, loggerConfig);
373                }
374                parseLogger(props, loggerConfig, key, loggerName, value);
375            }
376        }
377    }

379    /**
380     * Parses the additivity option for a non-root category.
381     */
382    private boolean getAdditivityForLogger(final Properties props, final String loggerName) {
383        boolean additivity = true;
384        final String key = ADDITIVITY_PREFIX + loggerName;
385        final String value = OptionConverter.findAndSubst(key, props);
386        LOGGER.debug("Handling {}=[{}]", key, value);
387        // touch additivity only if necessary
388        if ((value != null) && (!value.equals(""))) {
389            additivity = OptionConverter.toBoolean(value, true);
390        }
391        return additivity;
392    }

394    /**
395     * This method must work for the root category as well.
396     */
397    private void parseLogger(final Properties props, final LoggerConfig loggerConfig, final String optionKey, final String loggerName, final String value) {
398
399        LOGGER.debug("Parsing for [{}] with value=[{}].", loggerName, value);
400        // We must skip over ',' but not white space
401        final StringTokenizer st = new StringTokenizer(value, ",");
402
403        // If value is not in the form ", appender.." or "", then we should set the level of the logger.
404
405        if (!(value.startsWith(",") || value.equals(""))) {
406
407            // just to be on the safe side...
408            if (!st.hasMoreTokens()) {
409                return;
410            }
411
412            final String levelStr = st.nextToken();
413            LOGGER.debug("Level token is [{}].", levelStr);
414
415            final org.apache.logging.log4j.Level level = levelStr == null ? org.apache.logging.log4j.Level.ERROR
416                : OptionConverter.convertLevel(levelStr, org.apache.logging.log4j.Level.DEBUG);
417            loggerConfig.setLevel(level);
418            LOGGER.debug("Logger {} level set to {}", loggerName, level);
419        }
420
421        Appender appender;
422        String appenderName;
423        while (st.hasMoreTokens()) {
424            appenderName = st.nextToken().trim();
425            if (appenderName == null || appenderName.equals(",")) {
426                continue;
427            }
428            LOGGER.debug("Parsing appender named \"{}\".", appenderName);
429            appender = parseAppender(props, appenderName);
430            if (appender != null) {
431                LOGGER.debug("Adding appender named [{}] to loggerConfig [{}].", appenderName, loggerConfig.getName());
432                loggerConfig.addAppender(getAppender(appenderName), null, null);
433            } else {
434                LOGGER.debug("Appender named [{}] not found.", appenderName);
435            }
436        }
437    }

439    public Appender parseAppender(final Properties props, final String appenderName) {
440        Appender appender = registry.get(appenderName);
441        if ((appender != null)) {
442            LOGGER.debug("Appender \"" + appenderName + "\" was already parsed.");
443            return appender;
444        }
445        // Appender was not previously initialized.
446        final String prefix = APPENDER_PREFIX + appenderName;
447        final String layoutPrefix = prefix + ".layout";
448        final String filterPrefix = APPENDER_PREFIX + appenderName + ".filter.";
449        final String className = OptionConverter.findAndSubst(prefix, props);
450        if (className == null) {
451            LOGGER.debug("Appender \"" + appenderName + "\" does not exist.");
452            return null;
453        }
454        appender = manager.parseAppender(appenderName, className, prefix, layoutPrefix, filterPrefix, props, this);
455        if (appender == null) {
456            appender = buildAppender(appenderName, className, prefix, layoutPrefix, filterPrefix, props);
457        } else {
458            registry.put(appenderName, appender);
459            if (appender instanceof AppenderWrapper) {
460                addAppender(((AppenderWrapper) appender).getAppender());
461            } else {
462                addAppender(new AppenderAdapter(appender).getAdapter());
463            }
464        }
465        return appender;
466    }

468    private Appender buildAppender(final String appenderName, final String className, final String prefix, final String layoutPrefix, final String filterPrefix,
469        final Properties props) {
470        final Appender appender = newInstanceOf(className, "Appender");
471        if (appender == null) {
472            return null;
473        }
474        appender.setName(appenderName);
475        appender.setLayout(parseLayout(layoutPrefix, appenderName, props));
476        final String errorHandlerPrefix = prefix + ".errorhandler";
477        final String errorHandlerClass = OptionConverter.findAndSubst(errorHandlerPrefix, props);
478        if (errorHandlerClass != null) {
479            final ErrorHandler eh = parseErrorHandler(props, errorHandlerPrefix, errorHandlerClass, appender);
480            if (eh != null) {
481                appender.setErrorHandler(eh);
482            }
483        }
484        appender.addFilter(parseAppenderFilters(props, filterPrefix, appenderName));
485        final String[] keys = new String[] {layoutPrefix};
486        addProperties(appender, keys, props, prefix);
487        if (appender instanceof AppenderWrapper) {
488            addAppender(((AppenderWrapper) appender).getAppender());
489        } else {
490            addAppender(new AppenderAdapter(appender).getAdapter());
491        }
492        registry.put(appenderName, appender);
493        return appender;
494    }

496    public Layout parseLayout(final String layoutPrefix, final String appenderName, final Properties props) {
497        final String layoutClass = OptionConverter.findAndSubst(layoutPrefix, props);
498        if (layoutClass == null) {
499            return null;
500        }
501        Layout layout = manager.parse(layoutClass, layoutPrefix, props, this);
502        if (layout == null) {
503            layout = buildLayout(layoutPrefix, layoutClass, appenderName, props);
504        }
505        return layout;
506    }
507
508    private Layout buildLayout(final String layoutPrefix, final String className, final String appenderName, final Properties props) {
509        final Layout layout = newInstanceOf(className, "Layout");
510        if (layout == null) {
511            return null;
512        }
513        LOGGER.debug("Parsing layout options for \"{}\".", appenderName);
514        PropertySetter.setProperties(layout, props, layoutPrefix + ".");
515        LOGGER.debug("End of parsing for \"{}\".", appenderName);
516        return layout;
517    }

519    public ErrorHandler parseErrorHandler(final Properties props, final String errorHandlerPrefix, final String errorHandlerClass, final Appender appender) {
520        final ErrorHandler eh = newInstanceOf(errorHandlerClass, "ErrorHandler");
521        final String[] keys = new String[] {
522            // @formatter:off
523            errorHandlerPrefix + "." + ROOT_REF,
524            errorHandlerPrefix + "." + LOGGER_REF,
525            errorHandlerPrefix + "." + APPENDER_REF_TAG};
526            // @formatter:on
527        addProperties(eh, keys, props, errorHandlerPrefix);
528        return eh;
529    }

531    public void addProperties(final Object obj, final String[] keys, final Properties props, final String prefix) {
532        final Properties edited = new Properties();
533        props.stringPropertyNames().stream().filter(name -> {
534            if (name.startsWith(prefix)) {
535                for (final String key : keys) {
536                    if (name.equals(key)) {
537                        return false;
538                    }
539                }
540                return true;
541            }
542            return false;
543        }).forEach(name -> edited.put(name, props.getProperty(name)));
544        PropertySetter.setProperties(obj, edited, prefix + ".");
545    }

547    public Filter parseAppenderFilters(final Properties props, final String filterPrefix, final String appenderName) {
548        // extract filters and filter options from props into a hashtable mapping
549        // the property name defining the filter class to a list of pre-parsed
550        // name-value pairs associated to that filter
551        final int fIdx = filterPrefix.length();
552        final SortedMap<String, List<NameValue>> filters = new TreeMap<>();
553        final Enumeration<?> e = props.keys();
554        String name = "";
555        while (e.hasMoreElements()) {
556            final String key = (String) e.nextElement();
557            if (key.startsWith(filterPrefix)) {
558                final int dotIdx = key.indexOf('.', fIdx);
559                String filterKey = key;
560                if (dotIdx != -1) {
561                    filterKey = key.substring(0, dotIdx);
562                    name = key.substring(dotIdx + 1);
563                }
564                final List<NameValue> filterOpts = filters.computeIfAbsent(filterKey, k -> new ArrayList<>());
565                if (dotIdx != -1) {
566                    final String value = OptionConverter.findAndSubst(key, props);
567                    filterOpts.add(new NameValue(name, value));
568                }
569            }
570        }
571
572        Filter head = null;
573        for (final Map.Entry<String, List<NameValue>> entry : filters.entrySet()) {
574            final String clazz = props.getProperty(entry.getKey());
575            Filter filter = null;
576            if (clazz != null) {
577                filter = manager.parse(clazz, entry.getKey(), props, this);
578                if (filter == null) {
579                    LOGGER.debug("Filter key: [{}] class: [{}] props: {}", entry.getKey(), clazz, entry.getValue());
580                    filter = buildFilter(clazz, appenderName, entry.getValue());
581                }
582            }
583            head = FilterAdapter.addFilter(head, filter);
584        }
585        return head;
586    }
587
588    private Filter buildFilter(final String className, final String appenderName, final List<NameValue> props) {
589        final Filter filter = newInstanceOf(className, "Filter");
590        if (filter != null) {
591            final PropertySetter propSetter = new PropertySetter(filter);
592            for (final NameValue property : props) {
593                propSetter.setProperty(property.key, property.value);
594            }
595            propSetter.activate();
596        }
597        return filter;
598    }
599
600    private static <T> T newInstanceOf(final String className, final String type) {
601        try {
602            return LoaderUtil.newInstanceOf(className);
603        } catch (ReflectiveOperationException ex) {
604            LOGGER.error("Unable to create {} {} due to {}:{}", type, className, ex.getClass().getSimpleName(), ex.getMessage(), ex);
605            return null;
606        }
607    }

609    private static class NameValue {
610        String key, value;
611
612        NameValue(final String key, final String value) {
613            this.key = key;
614            this.value = value;
615        }
616
617        @Override
618        public String toString() {
619            return key + "=" + value;
620        }
621    }

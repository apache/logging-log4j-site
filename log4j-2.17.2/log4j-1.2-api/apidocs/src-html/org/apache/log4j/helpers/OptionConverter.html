068    public static String[] concatanateArrays(final String[] l, final String[] r) {
069        final int len = l.length + r.length;
070        final String[] a = new String[len];
071
072        System.arraycopy(l, 0, a, 0, l.length);
073        System.arraycopy(r, 0, a, l.length, r.length);
074
075        return a;
076    }

078    public static  org.apache.logging.log4j.Level convertLevel(final Level level) {
079        if (level == null) {
080            return org.apache.logging.log4j.Level.ERROR;
081        }
082        if (level.isGreaterOrEqual(Level.FATAL)) {
083            return org.apache.logging.log4j.Level.FATAL;
084        } else if (level.isGreaterOrEqual(Level.ERROR)) {
085            return org.apache.logging.log4j.Level.ERROR;
086        } else if (level.isGreaterOrEqual(Level.WARN)) {
087            return org.apache.logging.log4j.Level.WARN;
088        } else if (level.isGreaterOrEqual(Level.INFO)) {
089            return org.apache.logging.log4j.Level.INFO;
090        } else if (level.isGreaterOrEqual(Level.DEBUG)) {
091            return org.apache.logging.log4j.Level.DEBUG;
092        } else if (level.isGreaterOrEqual(Level.TRACE)) {
093            return org.apache.logging.log4j.Level.TRACE;
094        }
095        return org.apache.logging.log4j.Level.ALL;
096    }

099    public static Level convertLevel(final org.apache.logging.log4j.Level level) {
100        if (level == null) {
101            return Level.ERROR;
102        }
103        switch (level.getStandardLevel()) {
104            case FATAL:
105                return Level.FATAL;
106            case WARN:
107                return Level.WARN;
108            case INFO:
109                return Level.INFO;
110            case DEBUG:
111                return Level.DEBUG;
112            case TRACE:
113                return Level.TRACE;
114            case ALL:
115                return Level.ALL;
116            case OFF:
117                return Level.OFF;
118            default:
119                return Level.ERROR;
120        }
121    }

123    public static org.apache.logging.log4j.Level convertLevel(final String level,
124            final org.apache.logging.log4j.Level defaultLevel) {
125        final Level l = toLevel(level, null);
126        return l != null ? convertLevel(l) : defaultLevel;
127    }

129    public static String convertSpecialChars(final String s) {
130        char c;
131        final int len = s.length();
132        final StringBuilder sbuf = new StringBuilder(len);
133
134        int i = 0;
135        while (i < len) {
136            c = s.charAt(i++);
137            if (c == '\\') {
138                c = s.charAt(i++);
139                for (final CharMap entry : charMap) {
140                    if (entry.key == c) {
141                        c = entry.replacement;
142                    }
143                }
144            }
145            sbuf.append(c);
146        }
147        return sbuf.toString();
148    }

150    /**
151     * Find the value corresponding to <code>key</code> in
152     * <code>props</code>. Then perform variable substitution on the
153     * found value.
154     * @param key The key used to locate the substitution string.
155     * @param props The properties to use in the substitution.
156     * @return The substituted string.
157     */
158    public static String findAndSubst(final String key, final Properties props) {
159        final String value = props.getProperty(key);
160        if (value == null) {
161            return null;
162        }
163
164        try {
165            return substVars(value, props);
166        } catch (final IllegalArgumentException e) {
167            LOGGER.error("Bad option value [{}].", value, e);
168            return value;
169        }
170    }

174     * Very similar to <code>System.getProperty</code> except
175     * that the {@link SecurityException} is hidden.
176     *
177     * @param key The key to search for.
178     * @param def The default value to return.
179     * @return the string value of the system property, or the default
180     * value if there is no property with that key.
181     * @since 1.1
182     */
183    public static String getSystemProperty(final String key, final String def) {
184        try {
185            return System.getProperty(key, def);
186        } catch (final Throwable e) { // MS-Java throws com.ms.security.SecurityExceptionEx
187            LOGGER.debug("Was not allowed to read system property \"{}\".", key);
188            return def;
189        }

192    /**
193     * Instantiate an object given a class name. Check that the
194     * <code>className</code> is a subclass of
195     * <code>superClass</code>. If that test fails or the object could
196     * not be instantiated, then <code>defaultValue</code> is returned.
197     *
198     * @param className    The fully qualified class name of the object to instantiate.
199     * @param superClass   The class to which the new object should belong.
200     * @param defaultValue The object to return in case of non-fulfillment
201     * @return The created object.
202     */
203    public static Object instantiateByClassName(final String className, final Class<?> superClass,
204            final Object defaultValue) {
205        if (className != null) {
206            try {
207                final Object obj = LoaderUtil.newInstanceOf(className);
208                if (!superClass.isAssignableFrom(obj.getClass())) {
209                    LOGGER.error("A \"{}\" object is not assignable to a \"{}\" variable", className,
210                            superClass.getName());
211                    return defaultValue;
212                }
213                return obj;
214            } catch (final ReflectiveOperationException e) {
215                LOGGER.error("Could not instantiate class [" + className + "].", e);
216            }
217        }
218        return defaultValue;
219    }
220
221    public static Object instantiateByKey(final Properties props, final String key, final Class superClass, final Object defaultValue) {
222
223        // Get the value of the property in string form
224        final String className = findAndSubst(key, props);
225        if (className == null) {
226            LogLog.error("Could not find value for key " + key);
227            return defaultValue;
228        }
229        // Trim className to avoid trailing spaces that cause problems.
230        return OptionConverter.instantiateByClassName(className.trim(), superClass, defaultValue);
231    }

233    /**
234     * Configure log4j given an {@link InputStream}.
235     * <p>
236     * The InputStream will be interpreted by a new instance of a log4j configurator.
237     * </p>
238     * <p>
239     * All configurations steps are taken on the <code>hierarchy</code> passed as a parameter.
240     * </p>
241     *
242     * @param inputStream The configuration input stream.
243     * @param clazz The class name, of the log4j configurator which will parse the <code>inputStream</code>. This must be a
244     *        subclass of {@link Configurator}, or null. If this value is null then a default configurator of
245     *        {@link PropertyConfigurator} is used.
246     * @param hierarchy The {@link LoggerRepository} to act on.
247     * @since 1.2.17
248     */
249    static public void selectAndConfigure(final InputStream inputStream, final String clazz, final LoggerRepository hierarchy) {
250        Configurator configurator = null;
251
252        if (clazz != null) {
253            LOGGER.debug("Preferred configurator class: " + clazz);
254            configurator = (Configurator) instantiateByClassName(clazz, Configurator.class, null);
255            if (configurator == null) {
256                LOGGER.error("Could not instantiate configurator [" + clazz + "].");
257                return;
258            }
259        } else {
260            configurator = new PropertyConfigurator();
261        }
262
263        configurator.doConfigure(inputStream, hierarchy);
264    }

266    /**
267     * Configure log4j given a URL.
268     * <p>
269     * The url must point to a file or resource which will be interpreted by a new instance of a log4j configurator.
270     * </p>
271     * <p>
272     * All configurations steps are taken on the <code>hierarchy</code> passed as a parameter.
273     * </p>
274     *
275     * @param url The location of the configuration file or resource.
276     * @param clazz The classname, of the log4j configurator which will parse the file or resource at <code>url</code>. This
277     *        must be a subclass of {@link Configurator}, or null. If this value is null then a default configurator of
278     *        {@link PropertyConfigurator} is used, unless the filename pointed to by <code>url</code> ends in '.xml', in
279     *        which case {@link org.apache.log4j.xml.DOMConfigurator} is used.
280     * @param hierarchy The {@link LoggerRepository} to act on.
281     *
282     * @since 1.1.4
283     */
284    static public void selectAndConfigure(final URL url, String clazz, final LoggerRepository hierarchy) {
285        Configurator configurator = null;
286        final String filename = url.getFile();
287
288        if (clazz == null && filename != null && filename.endsWith(".xml")) {
289            clazz = "org.apache.log4j.xml.DOMConfigurator";
290        }
291
292        if (clazz != null) {
293            LOGGER.debug("Preferred configurator class: " + clazz);
294            configurator = (Configurator) instantiateByClassName(clazz, Configurator.class, null);
295            if (configurator == null) {
296                LOGGER.error("Could not instantiate configurator [" + clazz + "].");
297                return;
298            }
299        } else {
300            configurator = new PropertyConfigurator();
301        }
302
303        configurator.doConfigure(url, hierarchy);
304    }

306    /**
307     * Perform variable substitution in string <code>val</code> from the
308     * values of keys found in the system propeties.
309     *
310     * <p>The variable substitution delimeters are <b>${</b> and <b>}</b>.
311     *
312     * <p>For example, if the System properties contains "key=value", then
313     * the call
314     * <pre>
315     * String s = OptionConverter.substituteVars("Value of key is ${key}.");
316     * </pre>
317     * <p>
318     * will set the variable <code>s</code> to "Value of key is value.".
319     *
320     * <p>If no value could be found for the specified key, then the
321     * <code>props</code> parameter is searched, if the value could not
322     * be found there, then substitution defaults to the empty string.
323     *
324     * <p>For example, if system propeties contains no value for the key
325     * "inexistentKey", then the call
326     *
327     * <pre>
328     * String s = OptionConverter.subsVars("Value of inexistentKey is [${inexistentKey}]");
329     * </pre>
330     * will set <code>s</code> to "Value of inexistentKey is []"
331     *
332     * <p>An {@link IllegalArgumentException} is thrown if
333     * <code>val</code> contains a start delimeter "${" which is not
334     * balanced by a stop delimeter "}". </p>
335     *
336     * @param val The string on which variable substitution is performed.
337     * @param props The properties to use for the substitution.
338     * @return The substituted string.
339     * @throws IllegalArgumentException if <code>val</code> is malformed.
340     */
341    public static String substVars(final String val, final Properties props) throws IllegalArgumentException {
342        return substVars(val, props, new ArrayList<>());
343    }
344
345    private static String substVars(final String val, final Properties props, final List<String> keys)
346            throws IllegalArgumentException {
347        if (val == null) {
348            return null;
349        }
350        final StringBuilder sbuf = new StringBuilder();
351
352        int i = 0;
353        int j;
354        int k;
355
356        while (true) {
357            j = val.indexOf(DELIM_START, i);
358            if (j == -1) {
359                // no more variables
360                if (i == 0) { // this is a simple string
361                    return val;
362                }
363                // add the tail string which contails no variables and return the result.
364                sbuf.append(val.substring(i));
365                return sbuf.toString();
366            }
367            sbuf.append(val.substring(i, j));
368            k = val.indexOf(DELIM_STOP, j);
369            if (k == -1) {
370                throw new IllegalArgumentException(Strings.dquote(val)
371                        + " has no closing brace. Opening brace at position " + j
372                        + '.');
373            }
374            j += DELIM_START_LEN;
375            final String key = val.substring(j, k);
376            // first try in System properties
377            String replacement = PropertiesUtil.getProperties().getStringProperty(key, null);
378            // then try props parameter
379            if (replacement == null && props != null) {
380                replacement = props.getProperty(key);
381            }
382
383            if (replacement != null) {
384
385                // Do variable substitution on the replacement string
386                // such that we can solve "Hello ${x2}" as "Hello p1"
387                // the where the properties are
388                // x1=p1
389                // x2=${x1}
390                if (!keys.contains(key)) {
391                    final List<String> usedKeys = new ArrayList<>(keys);
392                    usedKeys.add(key);
393                    final String recursiveReplacement = substVars(replacement, props, usedKeys);
394                    sbuf.append(recursiveReplacement);
395                } else {
396                    sbuf.append(replacement);
397                }
398
399            }
400            i = k + DELIM_STOP_LEN;
401        }
402    }

404    /**
405     * If <code>value</code> is "true", then <code>true</code> is
406     * returned. If <code>value</code> is "false", then
407     * <code>true</code> is returned. Otherwise, <code>default</code> is
408     * returned.
409     *
410     * <p>Case of value is unimportant.
411     * @param value The value to convert.
412     * @param dEfault The default value.
413     * @return the value of the result.
414     */
415    public static boolean toBoolean(final String value, final boolean dEfault) {
416        if (value == null) {
417            return dEfault;
418        }
419        final String trimmedVal = value.trim();
420        if ("true".equalsIgnoreCase(trimmedVal)) {
421            return true;
422        }
423        if ("false".equalsIgnoreCase(trimmedVal)) {
424            return false;
425        }
426        return dEfault;
427    }

429    public static long toFileSize(final String value, final long defaultValue) {
430        if (value == null) {
431            return defaultValue;
432        }
433
434        String s = value.trim().toUpperCase();
435        long multiplier = 1;
436        int index;
437
438        if ((index = s.indexOf("KB")) != -1) {
439            multiplier = 1024;
440            s = s.substring(0, index);
441        } else if ((index = s.indexOf("MB")) != -1) {
442            multiplier = 1024 * 1024;
443            s = s.substring(0, index);
444        } else if ((index = s.indexOf("GB")) != -1) {
445            multiplier = 1024 * 1024 * 1024;
446            s = s.substring(0, index);
447        }
448        if (s != null) {
449            try {
450                return Long.valueOf(s).longValue() * multiplier;
451            } catch (final NumberFormatException e) {
452                LogLog.error("[" + s + "] is not in proper int form.");
453                LogLog.error("[" + value + "] not in expected format.", e);
454            }
455        }
456        return defaultValue;
457    }

459    public static int toInt(final String value, final int dEfault) {
460        if (value != null) {
461            final String s = value.trim();
462            try {
463                return Integer.valueOf(s).intValue();
464            } catch (final NumberFormatException e) {
465                LogLog.error("[" + s + "] is not in proper int form.");
466                e.printStackTrace();
467            }
468        }
469        return dEfault;
470    }

472    /**
473     * Converts a standard or custom priority level to a Level
474     * object.  <p> If <code>value</code> is of form
475     * "level#classname", then the specified class' toLevel method
476     * is called to process the specified level string; if no '#'
477     * character is present, then the default {@link org.apache.log4j.Level}
478     * class is used to process the level value.
479     *
480     * <p>As a special case, if the <code>value</code> parameter is
481     * equal to the string "NULL", then the value <code>null</code> will
482     * be returned.
483     *
484     * <p> If any error occurs while converting the value to a level,
485     * the <code>defaultValue</code> parameter, which may be
486     * <code>null</code>, is returned.
487     *
488     * <p> Case of <code>value</code> is insignificant for the level level, but is
489     * significant for the class name part, if present.
490     * @param value The value to convert.
491     * @param defaultValue The default value.
492     * @return the value of the result.
493     *
494     * @since 1.1
495     */
496    public static Level toLevel(String value, final Level defaultValue) {
497        if (value == null) {
498            return defaultValue;
499        }
500
501        value = value.trim();
502
503        final int hashIndex = value.indexOf('#');
504        if (hashIndex == -1) {
505            if ("NULL".equalsIgnoreCase(value)) {
506                return null;
507            }
508            // no class name specified : use standard Level class
509            return Level.toLevel(value, defaultValue);
510        }
511
512        Level result = defaultValue;
513
514        final String clazz = value.substring(hashIndex + 1);
515        final String levelName = value.substring(0, hashIndex);
516
517        // This is degenerate case but you never know.
518        if ("NULL".equalsIgnoreCase(levelName)) {
519            return null;
520        }
521
522        LOGGER.debug("toLevel" + ":class=[" + clazz + "]"
523                + ":pri=[" + levelName + "]");
524
525        try {
526            final Class<?> customLevel = LoaderUtil.loadClass(clazz);
527
528            // get a ref to the specified class' static method
529            // toLevel(String, org.apache.log4j.Level)
530            final Class<?>[] paramTypes = new Class[] { String.class, org.apache.log4j.Level.class };
531            final java.lang.reflect.Method toLevelMethod =
532                    customLevel.getMethod("toLevel", paramTypes);
533
534            // now call the toLevel method, passing level string + default
535            final Object[] params = new Object[]{levelName, defaultValue};
536            final Object o = toLevelMethod.invoke(null, params);
537
538            result = (Level) o;
539        } catch (final ClassNotFoundException e) {
540            LOGGER.warn("custom level class [" + clazz + "] not found.");
541        } catch (final NoSuchMethodException e) {
542            LOGGER.warn("custom level class [" + clazz + "]"
543                    + " does not have a class function toLevel(String, Level)", e);
544        } catch (final java.lang.reflect.InvocationTargetException e) {
545            if (e.getTargetException() instanceof InterruptedException
546                    || e.getTargetException() instanceof InterruptedIOException) {
547                Thread.currentThread().interrupt();
548            }
549            LOGGER.warn("custom level class [" + clazz + "]"
550                    + " could not be instantiated", e);
551        } catch (final ClassCastException e) {
552            LOGGER.warn("class [" + clazz
553                    + "] is not a subclass of org.apache.log4j.Level", e);
554        } catch (final IllegalAccessException e) {
555            LOGGER.warn("class [" + clazz +
556                    "] cannot be instantiated due to access restrictions", e);
557        } catch (final RuntimeException e) {
558            LOGGER.warn("class [" + clazz + "], level [" + levelName +
559                    "] conversion failed.", e);
560        }
561        return result;
562    }

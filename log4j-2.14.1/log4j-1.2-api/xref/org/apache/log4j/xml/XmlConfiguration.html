121     /**
122      * Configures log4j by reading in a log4j.dtd compliant XML
123      * configuration file.
124      */
125     @Override
126     public void doConfigure() throws FactoryConfigurationError {
127         ConfigurationSource source = getConfigurationSource();
128         ParseAction action = new ParseAction() {
129             @Override
130             public Document parse(final DocumentBuilder parser) throws SAXException, IOException {
131                 @SuppressWarnings("resource") // The ConfigurationSource and its caller manages the InputStream.
132                 InputSource inputSource = new InputSource(source.getInputStream());
133                 inputSource.setSystemId("dummy://log4j.dtd");
134                 return parser.parse(inputSource);
135             }
136 
137             @Override
138             public String toString() {
139                 return getConfigurationSource().getLocation();
140             }
141         };
142         doConfigure(action);
143     }
144 
145     private void doConfigure(final ParseAction action) throws FactoryConfigurationError {
146         DocumentBuilderFactory dbf;
147         try {
148             LOGGER.debug("System property is : {}", OptionConverter.getSystemProperty(dbfKey, null));
149             dbf = DocumentBuilderFactory.newInstance();
150             LOGGER.debug("Standard DocumentBuilderFactory search succeded.");
151             LOGGER.debug("DocumentBuilderFactory is: " + dbf.getClass().getName());
152         } catch (FactoryConfigurationError fce) {
153             Exception e = fce.getException();
154             LOGGER.debug("Could not instantiate a DocumentBuilderFactory.", e);
155             throw fce;
156         }
157 
158         try {
159             dbf.setValidating(true);
160 
161             DocumentBuilder docBuilder = dbf.newDocumentBuilder();
162 
163             docBuilder.setErrorHandler(new SAXErrorHandler());
164             docBuilder.setEntityResolver(new Log4jEntityResolver());
165 
166             Document doc = action.parse(docBuilder);
167             parse(doc.getDocumentElement());
168         } catch (Exception e) {
169             if (e instanceof InterruptedException || e instanceof InterruptedIOException) {
170                 Thread.currentThread().interrupt();
171             }
172             // I know this is miserable...
173             LOGGER.error("Could not parse " + action.toString() + ".", e);
174         }
175     }

177     @Override
178     public Configuration reconfigure() {
179         try {
180             final ConfigurationSource source = getConfigurationSource().resetInputStream();
181             if (source == null) {
182                 return null;
183             }
184             final XmlConfigurationFactory factory = new XmlConfigurationFactory();
185             final XmlConfiguration config =
186                     (XmlConfiguration) factory.getConfiguration(getLoggerContext(), source);
187             return config == null || config.getState() != State.INITIALIZING ? null : config;
188         } catch (final IOException ex) {
189             LOGGER.error("Cannot locate file {}: {}", getConfigurationSource(), ex);
190         }
191         return null;
192     }

195      * Delegates unrecognized content to created instance if it supports UnrecognizedElementParser.
196      *
197      * @param instance instance, may be null.
198      * @param element  element, may not be null.
199      * @param props    properties
200      * @throws IOException thrown if configuration of owner object should be abandoned.
201      */
202     private void parseUnrecognizedElement(final Object instance, final Element element,
203             final Properties props) throws Exception {
204         boolean recognized = false;
205         if (instance instanceof UnrecognizedElementHandler) {
206             recognized = ((UnrecognizedElementHandler) instance).parseUnrecognizedElement(
207                     element, props);
208         }
209         if (!recognized) {
210             LOGGER.warn("Unrecognized element {}", element.getNodeName());
211         }
212     }

270     /**
271      * Creates an object and processes any nested param elements
272      * but does not call activateOptions.  If the class also supports
273      * UnrecognizedElementParser, the parseUnrecognizedElement method
274      * will be call for any child elements other than param.
275      *
276      * @param element       element, may not be null.
277      * @param props         properties
278      * @param expectedClass interface or class expected to be implemented
279      *                      by created class
280      * @return created class or null.
281      * @throws Exception thrown if the contain object should be abandoned.
282      * @since 1.2.15
283      */
284     public Object parseElement(final Element element, final Properties props,
285             @SuppressWarnings("rawtypes") final Class expectedClass) throws Exception {
286         String clazz = subst(element.getAttribute("class"), props);
287         Object instance = OptionConverter.instantiateByClassName(clazz,
288                 expectedClass, null);
289 
290         if (instance != null) {
291             PropertySetter propSetter = new PropertySetter(instance);
292             NodeList children = element.getChildNodes();
293             final int length = children.getLength();
294 
295             for (int loop = 0; loop < length; loop++) {
296                 Node currentNode = children.item(loop);
297                 if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
298                     Element currentElement = (Element) currentNode;
299                     String tagName = currentElement.getTagName();
300                     if (tagName.equals("param")) {
301                         setParameter(currentElement, propSetter, props);
302                     } else {
303                         parseUnrecognizedElement(instance, currentElement, props);
304                     }
305                 }
306             }
307             return instance;
308         }
309         return null;
310     }

312     /**
313      * Used internally to parse appenders by IDREF name.
314      */
315     private Appender findAppenderByName(Document doc, String appenderName) {
316         Appender appender = appenderMap.get(appenderName);
317 
318         if (appender != null) {
319             return appender;
320         }
321         // Endre's hack:
322         Element element = null;
323         NodeList list = doc.getElementsByTagName("appender");
324         for (int t = 0; t < list.getLength(); t++) {
325             Node node = list.item(t);
326             NamedNodeMap map = node.getAttributes();
327             Node attrNode = map.getNamedItem("name");
328             if (appenderName.equals(attrNode.getNodeValue())) {
329                 element = (Element) node;
330                 break;
331             }
332         }
333         // Hack finished.
334 
335         if (element == null) {
336 
337             LOGGER.error("No appender named [{}] could be found.", appenderName);
338             return null;
339         }
340         appender = parseAppender(element);
341         if (appender != null) {
342             appenderMap.put(appenderName, appender);
343         }
344         return appender;
345     }

359     public Appender parseAppender(Element appenderElement) {
360         String className = subst(appenderElement.getAttribute(CLASS_ATTR));
361         LOGGER.debug("Class name: [" + className + ']');
362         Appender appender = manager.parseAppender(className, appenderElement, this);

369     private Appender buildAppender(String className, Element appenderElement) {
370         try {
371             Appender appender = LoaderUtil.newInstanceOf(className);
372             PropertySetter propSetter = new PropertySetter(appender);
373 
374             appender.setName(subst(appenderElement.getAttribute(NAME_ATTR)));
375             forEachElement(appenderElement.getChildNodes(), currentElement -> {
376                 // Parse appender parameters
377                 switch (currentElement.getTagName()) {
378                     case PARAM_TAG:
379                         setParameter(currentElement, propSetter);
380                         break;
381                     case LAYOUT_TAG:
382                         appender.setLayout(parseLayout(currentElement));
383                         break;
384                     case FILTER_TAG:
385                         Filter filter = parseFilters(currentElement);
386                         if (filter != null) {
387                             LOGGER.debug("Adding filter of type [{}] to appender named [{}]",
388                                     filter.getClass(), appender.getName());
389                             appender.addFilter(filter);
390                         }
391                         break;
392                     case ERROR_HANDLER_TAG:
393                         parseErrorHandler(currentElement, appender);
394                         break;
395                     case APPENDER_REF_TAG:
396                         String refName = subst(currentElement.getAttribute(REF_ATTR));
397                         if (appender instanceof AppenderAttachable) {
398                             AppenderAttachable aa = (AppenderAttachable) appender;
399                             Appender child = findAppenderByReference(currentElement);
400                             LOGGER.debug("Attaching appender named [{}] to appender named [{}].", refName,
401                                     appender.getName());
402                             aa.addAppender(child);
403                         } else {
404                             LOGGER.error("Requesting attachment of appender named [{}] to appender named [{}]"
405                                             + "which does not implement org.apache.log4j.spi.AppenderAttachable.",
406                                     refName, appender.getName());
407                         }
408                         break;
409                     default:
410                         try {
411                             parseUnrecognizedElement(appender, currentElement, props);
412                         } catch (Exception ex) {
413                             throw new ConsumerException(ex);
414                         }
415                 }
416             });
417             propSetter.activate();
418             return appender;
419         } catch (ConsumerException ex) {
420             Throwable t = ex.getCause();
421             if (t instanceof InterruptedException || t instanceof InterruptedIOException) {
422                 Thread.currentThread().interrupt();
423             }
424             LOGGER.error("Could not create an Appender. Reported error follows.", t);
425         } catch (Exception oops) {
426             if (oops instanceof InterruptedException || oops instanceof InterruptedIOException) {
427                 Thread.currentThread().interrupt();
428             }
429             LOGGER.error("Could not create an Appender. Reported error follows.", oops);
430         }
431         return null;
432     }

517     /**
518      * Used internally to parse an category element.
519      */
520     private void parseCategory(Element loggerElement) {
521         // Create a new org.apache.log4j.Category object from the <category> element.
522         String catName = subst(loggerElement.getAttribute(NAME_ATTR));
523         boolean additivity = OptionConverter.toBoolean(subst(loggerElement.getAttribute(ADDITIVITY_ATTR)), true);
524         LoggerConfig loggerConfig = getLogger(catName);
525         if (loggerConfig == null) {
526             loggerConfig = new LoggerConfig(catName, org.apache.logging.log4j.Level.ERROR, additivity);
527             addLogger(catName, loggerConfig);
528         } else {
529             loggerConfig.setAdditive(additivity);
530         }
531         parseChildrenOfLoggerElement(loggerElement, loggerConfig, false);
532     }

542     /**
543      * Used internally to parse the children of a LoggerConfig element.
544      */
545     private void parseChildrenOfLoggerElement(Element catElement, LoggerConfig loggerConfig, boolean isRoot) {
546 
547         final PropertySetter propSetter = new PropertySetter(loggerConfig);
548         loggerConfig.getAppenderRefs().clear();
549         forEachElement(catElement.getChildNodes(), currentElement -> {
550             switch (currentElement.getTagName()) {
551                 case APPENDER_REF_TAG: {
552                     Appender appender = findAppenderByReference(currentElement);
553                     String refName = subst(currentElement.getAttribute(REF_ATTR));
554                     if (appender != null) {
555                         LOGGER.debug("Adding appender named [{}] to loggerConfig [{}].", refName,
556                                 loggerConfig.getName());
557                         loggerConfig.addAppender(getAppender(refName), null, null);
558                     } else {
559                         LOGGER.debug("Appender named [{}] not found.", refName);
560                     }
561                     break;
562                 }
563                 case LEVEL_TAG: case PRIORITY_TAG: {
564                     parseLevel(currentElement, loggerConfig, isRoot);
565                     break;
566                 }
567                 case PARAM_TAG: {
568                     setParameter(currentElement, propSetter);
569                     break;
570                 }
571                 default: {
572                     quietParseUnrecognizedElement(loggerConfig, currentElement, props);
573                 }
574             }
575         });
576         propSetter.activate();
577     }

626     /**
627      * Used internally to parse a level  element.
628      */
629     private void parseLevel(Element element, LoggerConfig logger, boolean isRoot) {
630         String catName = logger.getName();
631         if (isRoot) {
632             catName = "root";
633         }
634 
635         String priStr = subst(element.getAttribute(VALUE_ATTR));
636         LOGGER.debug("Level value for {} is [{}].", catName, priStr);
637 
638         if (INHERITED.equalsIgnoreCase(priStr) || NULL.equalsIgnoreCase(priStr)) {
639             if (isRoot) {
640                 LOGGER.error("Root level cannot be inherited. Ignoring directive.");
641             } else {
642                 logger.setLevel(null);
643             }
644         } else {
645             String className = subst(element.getAttribute(CLASS_ATTR));
646             if (EMPTY_STR.equals(className)) {
647                 logger.setLevel(OptionConverter.convertLevel(priStr, org.apache.logging.log4j.Level.DEBUG));
648             } else {
649                 LOGGER.debug("Desired Level sub-class: [{}]", className);
650                 try {
651                     Class<?> clazz = LoaderUtil.loadClass(className);
652                     Method toLevelMethod = clazz.getMethod("toLevel", ONE_STRING_PARAM);
653                     Level pri = (Level) toLevelMethod.invoke(null, priStr);
654                     logger.setLevel(OptionConverter.convertLevel(pri));
655                 } catch (Exception oops) {
656                     if (oops instanceof InterruptedException || oops instanceof InterruptedIOException) {
657                         Thread.currentThread().interrupt();
658                     }
659                     LOGGER.error("Could not create level [" + priStr +
660                             "]. Reported error follows.", oops);
661                     return;
662                 }
663             }
664         }
665         LOGGER.debug("{} level set to {}", catName,  logger.getLevel());
666     }

676      * Used internally to configure the log4j framework by parsing a DOM
677      * tree of XML elements based on <a
678      * href="doc-files/log4j.dtd">log4j.dtd</a>.
679      */
680     private void parse(Element element) {
681         String rootElementName = element.getTagName();
682 
683         if (!rootElementName.equals(CONFIGURATION_TAG)) {
684             if (rootElementName.equals(OLD_CONFIGURATION_TAG)) {
685                 LOGGER.warn("The <" + OLD_CONFIGURATION_TAG +
686                         "> element has been deprecated.");
687                 LOGGER.warn("Use the <" + CONFIGURATION_TAG + "> element instead.");
688             } else {
689                 LOGGER.error("DOM element is - not a <" + CONFIGURATION_TAG + "> element.");
690                 return;
691             }
692         }
693 
694 
695         String debugAttrib = subst(element.getAttribute(INTERNAL_DEBUG_ATTR));
696 
697         LOGGER.debug("debug attribute= \"" + debugAttrib + "\".");
698         // if the log4j.dtd is not specified in the XML file, then the
699         // "debug" attribute is returned as the empty string.
700         String status = "error";
701         if (!debugAttrib.equals("") && !debugAttrib.equals("null")) {
702             status = OptionConverter.toBoolean(debugAttrib, true) ? "debug" : "error";
703 
704         } else {
705             LOGGER.debug("Ignoring " + INTERNAL_DEBUG_ATTR + " attribute.");
706         }
707 
708         String confDebug = subst(element.getAttribute(CONFIG_DEBUG_ATTR));
709         if (!confDebug.equals("") && !confDebug.equals("null")) {
710             LOGGER.warn("The \"" + CONFIG_DEBUG_ATTR + "\" attribute is deprecated.");
711             LOGGER.warn("Use the \"" + INTERNAL_DEBUG_ATTR + "\" attribute instead.");
712             status = OptionConverter.toBoolean(confDebug, true) ? "debug" : "error";
713         }
714 
715         final StatusConfiguration statusConfig = new StatusConfiguration().withStatus(status);
716         statusConfig.initialize();
717 
718         forEachElement(element.getChildNodes(), currentElement -> {
719             switch (currentElement.getTagName()) {
720                 case CATEGORY: case LOGGER_ELEMENT:
721                     parseCategory(currentElement);
722                     break;
723                 case ROOT_TAG:
724                     parseRoot(currentElement);
725                     break;
726                 case RENDERER_TAG:
727                     LOGGER.warn("Renderers are not supported by Log4j 2 and will be ignored.");
728                     break;
729                 case THROWABLE_RENDERER_TAG:
730                     LOGGER.warn("Throwable Renderers are not supported by Log4j 2 and will be ignored.");
731                     break;
732                 case CATEGORY_FACTORY_TAG: case LOGGER_FACTORY_TAG:
733                     LOGGER.warn("Log4j 1 Logger factories are not supported by Log4j 2 and will be ignored.");
734                     break;
735                 case APPENDER_TAG:
736                     Appender appender = parseAppender(currentElement);
737                     appenderMap.put(appender.getName(), appender);
738                     if (appender instanceof AppenderWrapper) {
739                         addAppender(((AppenderWrapper) appender).getAppender());
740                     } else {
741                         addAppender(new AppenderAdapter(appender).getAdapter());
742                     }
743                     break;
744                 default:
745                     quietParseUnrecognizedElement(null, currentElement, props);
746             }
747         });
748     }
749 
750     private String subst(final String value) {
751         return getStrSubstitutor().replace(value);
752     }
753 
754     public static void forEachElement(NodeList list, Consumer<Element> consumer) {
755         final int length = list.getLength();
756         for (int loop = 0; loop < length; loop++) {
757             Node currentNode = list.item(loop);
758 
759             if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
760                 Element currentElement = (Element) currentNode;
761                 consumer.accept(currentElement);
762             }
763         }
764     }

770     private static class SAXErrorHandler implements org.xml.sax.ErrorHandler {
771         private static final org.apache.logging.log4j.Logger LOGGER = StatusLogger.getLogger();
772 
773         @Override
774         public void error(final SAXParseException ex) {
775             emitMessage("Continuable parsing error ", ex);
776         }
777 
778         @Override
779         public void fatalError(final SAXParseException ex) {
780             emitMessage("Fatal parsing error ", ex);
781         }
782 
783         @Override
784         public void warning(final SAXParseException ex) {
785             emitMessage("Parsing warning ", ex);
786         }
787 
788         private static void emitMessage(final String msg, final SAXParseException ex) {
789             LOGGER.warn("{} {} and column {}", msg, ex.getLineNumber(), ex.getColumnNumber());
790             LOGGER.warn(ex.getMessage(), ex.getException());
791         }

021    import org.apache.logging.log4j.LoggingException;
022    import org.apache.logging.log4j.Marker;
023    import org.apache.logging.log4j.core.LogEvent;
024    import org.apache.logging.log4j.core.helpers.UUIDUtil;
025    import org.apache.logging.log4j.message.MapMessage;
026    import org.apache.logging.log4j.message.Message;
027    import org.apache.logging.log4j.message.StructuredDataId;
028    import org.apache.logging.log4j.message.StructuredDataMessage;
029    
030    import java.io.ByteArrayOutputStream;
031    import java.io.IOException;
032    import java.util.Arrays;
033    import java.util.HashMap;
034    import java.util.List;
035    import java.util.Map;
036    import java.util.Stack;
037    import java.util.zip.GZIPOutputStream;

039    /**
040     * Class that is both a Flume and Log4j Event.
041     */
042    public class FlumeEvent extends EventBaseImpl implements LogEvent {
043    
044        private static final String DEFAULT_MDC_PREFIX = "mdc:";
045    
046        private static final String DEFAULT_EVENT_PREFIX = "";
047    
048        private static final String EVENT_TYPE = "eventType";
049    
050        private static final String EVENT_ID = "eventId";
051    
052        private static final String GUID = "guId";
053    
054        private final LogEvent event;
055    
056        private byte[] body;
057    
058        private final String hostname;
059    
060        private final Map<String, String> ctx = new HashMap<String, String>();
061    
062        private final boolean compress;
063    
064        /**
065         * Construct the FlumeEvent.
066         * @param event The Log4j LogEvent.
067         * @param hostname The host name.
068         * @param includes A comma separated list of MDC elements to include.
069         * @param excludes A comma separated list of MDC elements to exclude.
070         * @param required A comma separated list of MDC elements that are required to be defined.
071         * @param mdcPrefix The value to prefix to MDC keys.
072         * @param eventPrefix The value to prefix to event keys.
073         * @param compress If true the event body should be compressed.
074         */
075        public FlumeEvent(LogEvent event, String hostname, String includes, String excludes, String required,
076                          String mdcPrefix, String eventPrefix, boolean compress) {
077            this.event = event;
078            this.hostname = hostname;
079            this.compress = compress;
080            if (mdcPrefix == null) {
081                mdcPrefix = DEFAULT_MDC_PREFIX;
082            }
083            if (eventPrefix == null) {
084                eventPrefix = DEFAULT_EVENT_PREFIX;
085            }
086            this.fields = new HashMap<String, byte[]>();
087            Map<String, String> mdc = event.getContextMap();
088            if (includes != null) {
089                String[] array = includes.split(",");
090                if (array.length > 0) {
091                    for (String str : array) {
092                        if (mdc.containsKey(str)) {
093                            ctx.put(str, mdc.get(str));
094                        }
095                    }
096                }
097            } else if (excludes != null) {
098                String[] array = excludes.split(",");
099                if (array.length > 0) {
100                    List<String> list = Arrays.asList(array);
101                    for (Map.Entry<String, String> entry : mdc.entrySet()) {
102                        if (!list.contains(entry.getKey())) {
103                            ctx.put(entry.getKey(), entry.getValue());
104                        }
105                    }
106                }
107            } else {
108                ctx.putAll(mdc);
109            }
110    
111            if (required != null) {
112                String[] array = required.split(",");
113                if (array.length > 0) {
114                    for (String str : array) {
115                        if (!mdc.containsKey(str)) {
116                            throw new LoggingException("Required key " + str + " is missing from the MDC");
117                        }
118                    }
119                }
120            }
121            Message message = event.getMessage();
122            if (message instanceof MapMessage) {
123                if (message instanceof StructuredDataMessage) {
124                    addStructuredData(eventPrefix, fields, (StructuredDataMessage) message);
125                }
126                addMapData(eventPrefix, fields, (MapMessage) message);
127            }
128    
129            addContextData(mdcPrefix, fields, ctx);
130    
131            addGuid(fields);
132        }
133    
134        protected void addStructuredData(String prefix, Map<String, byte[]> fields, StructuredDataMessage msg) {
135            fields.put(prefix + EVENT_TYPE, msg.getType().getBytes());
136            StructuredDataId id = msg.getId();
137            fields.put(prefix + EVENT_ID, id.getName().getBytes());
138        }
139    
140        protected void addMapData(String prefix, Map<String, byte[]> fields, MapMessage msg) {
141            Map<String, String> data = msg.getData();
142            for (Map.Entry<String, String> entry : data.entrySet()) {
143                fields.put(prefix + entry.getKey(), entry.getValue().getBytes());
144            }
145        }
146    
147        protected void addContextData(String prefix, Map<String, byte[]> fields, Map<String, String> context) {
148            for (Map.Entry<String, String> entry : ctx.entrySet()) {
149                fields.put(prefix + entry.getKey(), entry.getValue().toString().getBytes());
150            }
151        }
152    
153        protected void addGuid(Map<String, byte[]> fields) {
154            fields.put(GUID, UUIDUtil.getTimeBasedUUID().toString().getBytes());
155        }
156    
157        /**
158         * Set the body in the event.
159         * @param body The body to add to the event.
160         */
161        public void setBody(byte[] body) {
162            if (body == null || body.length == 0) {
163                this.body = new byte[0];
164                return;
165            }
166            if (compress) {
167                ByteArrayOutputStream baos = new ByteArrayOutputStream();
168                try {
169                    GZIPOutputStream os = new GZIPOutputStream(baos);
170                    os.write(body);
171                    os.close();
172                } catch (IOException ioe) {
173                    throw new LoggingException("Unable to compress message", ioe);
174                }
175                this.body = baos.toByteArray();
176            } else {
177                this.body = body;
178            }
179        }
180    
181        @Override
182        public byte[] getBody() {
183            return this.body;
184        }
185    
186        @Override
187        public Priority getPriority() {
188            switch (event.getLevel()) {
189                case INFO:
190                    return Priority.INFO;
191                case ERROR:
192                    return Priority.ERROR;
193                case DEBUG:
194                    return Priority.DEBUG;
195                case WARN:
196                    return Priority.WARN;
197                case TRACE:
198                    return Priority.TRACE;
199                case FATAL:
200                    return Priority.FATAL;
201            }
202            return Priority.INFO;
203        }
204    
205        /**
206         * Get the Frequently Qualified Class Name.
207         * @return the FQCN String.
208         */
209        public String getFQCN() {
210            return event.getFQCN();
211        }
212    
213        @Override
214        public long getTimestamp() {
215            return event.getMillis();
216        }
217    
218        @Override
219        public long getNanos() {
220            return System.nanoTime();
221        }
222    
223        @Override
224        public String getHost() {
225            return hostname;
226        }
227    
228        /**
229         * Return the logging Level.
230         * @return the Level.
231         */
232        public Level getLevel() {
233            return event.getLevel();
234        }
235    
236        /**
237         * Return the logger name.
238         * @return the logger name.
239         */
240        public String getLoggerName() {
241            return event.getLoggerName();
242        }
243    
244        /**
245         * Return the StackTraceElement for the caller of the logging API.
246         * @return the StackTraceElement of the caller.
247         */
248        public StackTraceElement getSource() {
249            return event.getSource();
250        }
251    
252        /**
253         * Return the Message.
254         * @return the Message.
255         */
256        public Message getMessage() {
257            return event.getMessage();
258        }
259    
260        /**
261         * Return the Marker.
262         * @return the Marker.
263         */
264        public Marker getMarker() {
265            return event.getMarker();
266        }
267    
268        /**
269         * Return the name of the Thread.
270         * @return the name of the Thread.
271         */
272        public String getThreadName() {
273            return event.getThreadName();
274        }
275    
276        /**
277         * Return the event timestamp.
278         * @return the event timestamp.
279         */
280        public long getMillis() {
281            return event.getMillis();
282        }
283    
284        /**
285         * Return the Throwable associated with the event, if any.
286         * @return the Throwable.
287         */
288        public Throwable getThrown() {
289            return event.getThrown();
290        }
291    
292        /**
293         * Return a copy of the context Map.
294         * @return a copy of the context Map.
295         */
296        public Map<String, String> getContextMap() {
297            return ctx;
298        }
299    
300        /**
301         * Return a copy of the context stack.
302         * @return a copy of the context stack.
303         */
304        public Stack<String> getContextStack() {
305            return event.getContextStack();
306        }

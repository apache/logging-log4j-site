017    package org.apache.logging.log4j.message;
018    
019    import java.io.InvalidObjectException;
020    import java.io.ObjectInputStream;
021    import java.io.Serializable;
022    import java.lang.management.ManagementFactory;
023    import java.lang.management.ThreadInfo;
024    import java.lang.management.ThreadMXBean;
025    import java.lang.reflect.Method;
026    import java.util.HashMap;
027    import java.util.Map;
028    
029    /**
030     * Captures information about all running Threads.
031     */
032    public class ThreadDumpMessage implements Message {
033    
034        private static final long serialVersionUID = -1103400781608841088L;
035    
036        private static ThreadInfoFactory factory;
037    
038        private volatile Map<ThreadInformation, StackTraceElement[]> threads;
039    
040        private final String title;
041    
042        private String formattedMessage = null;
043    
044        static {
045            Method[] methods = ThreadInfo.class.getMethods();
046            boolean basic = true;
047            for (Method method : methods) {
048                if (method.getName().equals("getLockInfo")) {
049                    basic = false;
050                    break;
051                }
052            }
053            factory = basic ? new BasicThreadInfoFactory() : new ExtendedThreadInfoFactory();
054        }
055    
056        /**
057         * Generate a ThreadDumpMessage with no title.
058         */
059        public ThreadDumpMessage() {
060            this(null);
061    
062        }
063    
064        /**
065         * Generate a ThreadDumpMessage with a title.
066         * @param title The title.
067         */
068        public ThreadDumpMessage(String title) {
069            this.title = title == null ? "" : title;
070            threads = factory.createThreadInfo();
071        }
072    
073        private ThreadDumpMessage(String formattedMsg, String title) {
074            this.formattedMessage = formattedMsg;
075            this.title = title;
076        }
077    
078        /**
079         * Return the ThreadDump in printable format.
080         * @return the ThreadDump suitable for logging.
081         */
082        public String getFormattedMessage() {
083            if (formattedMessage != null) {
084                return formattedMessage;
085            }
086            StringBuilder sb = new StringBuilder(title);
087            if (title.length() > 0) {
088                sb.append("\n");
089            }
090            for (Map.Entry<ThreadInformation, StackTraceElement[]> entry : threads.entrySet()) {
091                ThreadInformation info = entry.getKey();
092                info.printThreadInfo(sb);
093                info.printStack(sb, entry.getValue());
094                sb.append("\n");
095            }
096            return sb.toString();
097        }
098    
099        /**
100         * Returns the title.
101         * @return the title.
102         */
103        public String getMessageFormat() {
104            return title == null ? "" : title;
105        }
106    
107        /**
108         * Returns an array with a single element, a Map containing the ThreadInformation as the key.
109         * and the StackTraceElement array as the value;
110         * @return the "parameters" to this Message.
111         */
112        public Object[] getParameters() {
113            return null;
114        }
115    
116            /**
117         * Creates a ThreadDumpMessageProxy that can be serialized.
118         * @return a ThreadDumpMessageProxy.
119         */
120        protected Object writeReplace() {
121            return new ThreadDumpMessageProxy(this);
122        }
123    
124        private void readObject(ObjectInputStream stream)
125            throws InvalidObjectException {
126            throw new InvalidObjectException("Proxy required");
127        }
128    
129        /**
130         * Proxy pattern used to serialize the ThreadDumpMessage.
131         */
132        private static class ThreadDumpMessageProxy implements Serializable {
133    
134            private static final long serialVersionUID = -3476620450287648269L;
135            private String formattedMsg;
136            private String title;
137    
138            public ThreadDumpMessageProxy(ThreadDumpMessage msg) {
139                this.formattedMsg = msg.getFormattedMessage();
140                this.title = msg.title;
141            }
142    
143            /**
144             * Return a ThreadDumpMessage using the data in the proxy.
145             * @return a ThreadDumpMessage.
146             */
147            protected Object readResolve() {
148                return new ThreadDumpMessage(formattedMsg, title);
149            }
150        }
151    
152        /**
153         * Factory to create Thread information.
154         */
155        private interface ThreadInfoFactory {
156            Map<ThreadInformation, StackTraceElement[]> createThreadInfo();
157        }
158    
159        /**
160         * Factory to create basic thread information.
161         */
162        private static class BasicThreadInfoFactory implements ThreadInfoFactory {
163            public Map<ThreadInformation, StackTraceElement[]> createThreadInfo() {
164                Map<Thread, StackTraceElement[]> map = Thread.getAllStackTraces();
165                Map<ThreadInformation, StackTraceElement[]> threads =
166                    new HashMap<ThreadInformation, StackTraceElement[]>(map.size());
167                for (Map.Entry<Thread, StackTraceElement[]> entry : map.entrySet()) {
168                    threads.put(new BasicThreadInformation(entry.getKey()), entry.getValue());
169                }
170                return threads;
171            }
172        }
173    
174        /**
175         * Factory to create extended thread information.
176         */
177        private static class ExtendedThreadInfoFactory implements ThreadInfoFactory {
178            public Map<ThreadInformation, StackTraceElement[]> createThreadInfo() {
179                ThreadMXBean bean = ManagementFactory.getThreadMXBean();
180                ThreadInfo[] array = bean.dumpAllThreads(true, true);
181    
182                Map<ThreadInformation, StackTraceElement[]>  threads =
183                    new HashMap<ThreadInformation, StackTraceElement[]>(array.length);
184                for (ThreadInfo info : array) {
185                    threads.put(new ExtendedThreadInformation(info), info.getStackTrace());
186                }
187                return threads;
188            }
189        }
190    }

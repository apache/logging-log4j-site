017package org.apache.logging.log4j.spring.cloud.config.client;
018
019import java.util.UUID;
020import java.util.concurrent.ConcurrentHashMap;
021import java.util.concurrent.ConcurrentMap;
022
023import org.apache.logging.log4j.core.util.WatchEventService;
024import org.apache.logging.log4j.core.util.WatchManager;
025
026/**
027 *
028 */
029public class WatchEventManager implements WatchEventService {
030        private static final ConcurrentMap<UUID, WatchManager> watchManagers = new ConcurrentHashMap<>();
031
032        public static void publishEvent() {
033                for (WatchManager manager : watchManagers.values()) {
034                        manager.checkFiles();
035                }
036        }
037
038        @Override
039        public void subscribe(WatchManager manager) {
040                watchManagers.put(manager.getId(), manager);
041
042        }
043
044        @Override
045        public void unsubscribe(WatchManager manager) {
046                watchManagers.remove(manager.getId());
047        }
048}

065/**
066 * Override Spring's implementation of the Log4j 2 Logging System to properly support Spring Cloud Config.
067 */
068public class Log4j2CloudConfigLoggingSystem extends Log4J2LoggingSystem {
069
070    /**
071     * Property that disables the usage of this {@link LoggingSystem}.
072     */
073    public static final String LOG4J2_DISABLE_CLOUD_CONFIG_LOGGING_SYSTEM = "log4j2.disableCloudConfigLoggingSystem";
074
075    public static final String ENVIRONMENT_KEY = "SpringEnvironment";
076    private static final String HTTPS = "https";
077    private static final String OVERRIDE_PARAM = "override";
078    private static Logger LOGGER = StatusLogger.getLogger();
079    private static final int PRECEDENCE = 0;
080
081    public Log4j2CloudConfigLoggingSystem(ClassLoader loader) {
082        super(loader);
083    }
084
085    /**
086     * Set the environment into the ExternalContext field so that it can be obtained by SpringLookup when it
087     * is constructed. Spring will replace the ExternalContext field with a String once initialization is
088     * complete.
089     * @param initializationContext The initialization context.
090     * @param configLocation The configuration location.
091     * @param logFile the log file.
092     */
093    @Override
094    public void initialize(LoggingInitializationContext initializationContext, String configLocation, LogFile logFile) {
095        getLoggerContext().putObjectIfAbsent(ENVIRONMENT_KEY, initializationContext.getEnvironment());
096        super.initialize(initializationContext, configLocation, logFile);
097    }
098
099    @Override
100    protected String[] getStandardConfigLocations() {
101        String[] locations = super.getStandardConfigLocations();
102        PropertiesUtil props = new PropertiesUtil(new Properties());
103        String location = props.getStringProperty(ConfigurationFactory.CONFIGURATION_FILE_PROPERTY);
104        if (location != null) {
105            List<String> list = new ArrayList<>(Arrays.asList(super.getStandardConfigLocations()));
106            list.add(location);
107            locations = list.toArray(Strings.EMPTY_ARRAY);
108        }
109        return locations;
110    }
111
112    @Override
113    protected void loadDefaults(LoggingInitializationContext initializationContext, LogFile logFile) {
114        if (logFile != null) {
115            this.loadConfiguration(this.getBootPackagedConfigFile("log4j2-file.xml"), logFile);
116        } else {
117            this.loadConfiguration(this.getBootPackagedConfigFile("log4j2.xml"), logFile);
118        }
119    }
120
121    private String getBootPackagedConfigFile(String fileName) {
122        String defaultPath = ClassUtils.getPackageName(Log4J2LoggingSystem.class);
123        defaultPath = defaultPath.replace('.', '/');
124        defaultPath = defaultPath + "/" + fileName;
125        defaultPath = "classpath:" + defaultPath;
126        return defaultPath;
127    }
128
129    /**
130     * This method is removed from Spring in 3.x and is scheduled to be removed in 2.8.x. It must be left in
131     * to support older Spring releases.
132     * @param location the location
133     * @param logFile log file configuration
134     */
135    @Override
136    protected void loadConfiguration(String location, LogFile logFile) {
137        loadConfiguration(location, logFile, Collections.emptyList());
138    }
139
140    /**
141     * Added in Spring 2.6.0, the overrides parameter allows override files to be specified in the
142     * "logging.log4j2.config.override" property. However, spring does not support passing credentials
143     * when accessing the location. We do.
144     * @param location The location of the primary configuration.
145     * @param logFile log file configuration.
146     * @param overrides Any override files.
147     */
148    @Override
149    protected void loadConfiguration(String location, LogFile logFile, List<String> overrides) {
150        Assert.notNull(location, "Location must not be null");
151        try {
152            LoggerContext ctx = getLoggerContext();
153            List<String> locations = parseConfigLocations(location);
154            if (overrides != null) {
155                locations.addAll(overrides);
156            }
157            if (locations.size() == 1) {
158                final URL url = ResourceUtils.getURL(location);
159                final ConfigurationSource source = getConfigurationSource(url);
160                if (source != null) {
161                    ctx.start(ConfigurationFactory.getInstance().getConfiguration(ctx, source));
162                }
163            } else {
164                final List<AbstractConfiguration> configs = new ArrayList<>();
165                boolean first = true;
166                for (final String sourceLocation : locations) {
167                    final ConfigurationSource source = getConfigurationSource(ResourceUtils.getURL(sourceLocation));
168                    if (source != null) {
169                        try {
170                            final Configuration config = ConfigurationFactory.getInstance().getConfiguration(ctx, source);
171                            if (config instanceof AbstractConfiguration) {
172                                configs.add((AbstractConfiguration) config);
173                            } else {
174                                LOGGER.warn("Configuration at {} cannot be combined in a CompositeConfiguration", sourceLocation);
175                                return;
176                            }
177                        } catch (Exception ex) {
178                            if (!first) {
179                                LOGGER.warn("Error accessing {}: {}. Ignoring override", sourceLocation, ex.getMessage());
180                            } else {
181                                throw ex;
182                            }
183                        }
184                    }
185                    first = false;
186                }
187                if (configs.size() > 1) {
188                    ctx.start(new CompositeConfiguration(configs));
189                } else {
190                    ctx.start(configs.get(0));
191                }
192            }
193        }
194        catch (Exception ex) {
195            throw new IllegalStateException(
196                    "Could not initialize Log4J2 logging from " + location, ex);
197        }
198    }
199
200    @Override
201    public void cleanUp() {
202        getLoggerContext().removeObject(ENVIRONMENT_KEY);
203        super.cleanUp();
204    }
205
206    private List<String> parseConfigLocations(String configLocations) {
207        final String[] uris = configLocations.split("\\?");
208        final List<String> locations = new ArrayList<>();
209        locations.add(uris[0]);
210        if (uris.length > 1) {
211            try {
212                final URL url = new URL(configLocations);
213                final String[] pairs = url.getQuery().split("&");
214                for (String pair : pairs) {
215                    final int idx = pair.indexOf("=");
216                    try {
217                        final String key = idx > 0 ? URLDecoder.decode(pair.substring(0, idx), "UTF-8") : pair;
218                        if (key.equalsIgnoreCase(OVERRIDE_PARAM)) {
219                            locations.add(URLDecoder.decode(pair.substring(idx + 1), "UTF-8"));
220                        }
221                    } catch (UnsupportedEncodingException ex) {
222                        LOGGER.warn("Bad data in configuration string: {}", pair);
223                    }
224                }
225                return locations;
226            } catch (MalformedURLException ex) {
227                LOGGER.warn("Unable to parse configuration URL {}", configLocations);
228            }
229        }
230        return locations;
231    }
232
233    private ConfigurationSource getConfigurationSource(URL url) throws IOException, URISyntaxException {
234        AuthorizationProvider provider = ConfigurationFactory.authorizationProvider(PropertiesUtil.getProperties());
235        SslConfiguration sslConfiguration = url.getProtocol().equals(HTTPS)
236                ? SslConfigurationFactory.getSslConfiguration() : null;
237        URLConnection urlConnection = UrlConnectionFactory.createConnection(url, 0, sslConfiguration,
238                provider);
239
240        File file = FileUtils.fileFromUri(url.toURI());
241        try {
242            if (file != null) {
243                return new ConfigurationSource(urlConnection.getInputStream(), FileUtils.fileFromUri(url.toURI()));
244            }
245            return new ConfigurationSource(urlConnection.getInputStream(), url, urlConnection.getLastModified());
246        } catch (FileNotFoundException ex) {
247            LOGGER.info("Unable to locate file {}, ignoring.", url.toString());
248            return null;
249        }
250    }
251
252    private LoggerContext getLoggerContext() {
253        return (LoggerContext) LogManager.getContext(false);
254    }
255
256    @Order(PRECEDENCE)
257    public static class Factory implements LoggingSystemFactory {
258
259        @Override
260        public LoggingSystem getLoggingSystem(ClassLoader classLoader) {
261            if (PropertiesUtil.getProperties().getBooleanProperty(LOG4J2_DISABLE_CLOUD_CONFIG_LOGGING_SYSTEM)) {
262                return null;
263            }
264            return new Log4j2CloudConfigLoggingSystem(classLoader);
265        }
266
267    }

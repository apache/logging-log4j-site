017package org.apache.logging.log4j.layout.template.json.resolver;
018
019import org.apache.logging.log4j.core.config.plugins.Plugin;
020import org.apache.logging.log4j.core.config.plugins.PluginFactory;
021
022/**
023 * {@link PatternResolver} factory.
024 */
025@Plugin(name = "PatternResolverFactory", category = TemplateResolverFactory.CATEGORY)
026public final class PatternResolverFactory implements EventResolverFactory {
027
028    private static final PatternResolverFactory INSTANCE = new PatternResolverFactory();
029
030    private PatternResolverFactory() {}
031
032    @PluginFactory
033    public static PatternResolverFactory getInstance() {
034        return INSTANCE;
035    }
036
037    @Override
038    public String getName() {
039        return PatternResolver.getName();
040    }
041
042    @Override
043    public PatternResolver create(
044            final EventResolverContext context,
045            final TemplateResolverConfig config) {
046        return new PatternResolver(context, config);
047    }
048
049}

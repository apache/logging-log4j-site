096    private JsonTemplateLayout(final Builder builder) {
097        this.charset = builder.charset;
098        this.contentType = "application/json; charset=" + charset;
099        final String eventDelimiterSuffix = builder.isNullEventDelimiterEnabled() ? "\0" : "";
100        this.eventDelimiter = builder.eventDelimiter + eventDelimiterSuffix;
101        final Configuration configuration = builder.configuration;
102        final JsonWriter jsonWriter = JsonWriter
103                .newBuilder()
104                .setMaxStringLength(builder.maxStringLength)
105                .setTruncatedStringSuffix(builder.truncatedStringSuffix)
106                .build();
107        this.eventResolver = createEventResolver(
108                builder,
109                configuration,
110                charset,
111                jsonWriter);
112        this.contextRecycler = createContextRecycler(builder, jsonWriter);
113    }

115    private TemplateResolver<LogEvent> createEventResolver(
116            final Builder builder,
117            final Configuration configuration,
118            final Charset charset,
119            final JsonWriter jsonWriter) {
120
121        // Inject resolver factory and interceptor plugins.
122        final List<String> pluginPackages = configuration.getPluginPackages();
123        final Map<String, EventResolverFactory> resolverFactoryByName =
124                EventResolverFactories.populateResolverFactoryByName(pluginPackages);
125        final List<EventResolverInterceptor> resolverInterceptors =
126                EventResolverInterceptors.populateInterceptors(pluginPackages);
127        final EventResolverStringSubstitutor substitutor =
128                new EventResolverStringSubstitutor(configuration.getStrSubstitutor());
129
130        // Read event and stack trace element templates.
131        final String eventTemplate = readEventTemplate(builder);
132        final String stackTraceElementTemplate = readStackTraceElementTemplate(builder);
133
134        // Determine the max. string byte count.
135        final float maxByteCountPerChar = builder.charset.newEncoder().maxBytesPerChar();
136        final int maxStringByteCount =
137                Math.toIntExact(Math.round(Math.ceil(
138                        maxByteCountPerChar * builder.maxStringLength)));
139
140        // Replace null event template additional fields with an empty array.
141        final EventTemplateAdditionalField[] eventTemplateAdditionalFields =
142                builder.eventTemplateAdditionalFields != null
143                        ? builder.eventTemplateAdditionalFields
144                        : EventTemplateAdditionalField.EMPTY_ARRAY;
145
146        // Create the resolver context.
147        final EventResolverContext resolverContext = EventResolverContext
148                .newBuilder()
149                .setConfiguration(configuration)
150                .setResolverFactoryByName(resolverFactoryByName)
151                .setResolverInterceptors(resolverInterceptors)
152                .setSubstitutor(substitutor)
153                .setCharset(charset)
154                .setJsonWriter(jsonWriter)
155                .setRecyclerFactory(builder.recyclerFactory)
156                .setMaxStringByteCount(maxStringByteCount)
157                .setTruncatedStringSuffix(builder.truncatedStringSuffix)
158                .setLocationInfoEnabled(builder.locationInfoEnabled)
159                .setStackTraceEnabled(builder.stackTraceEnabled)
160                .setStackTraceElementTemplate(stackTraceElementTemplate)
161                .setEventTemplateRootObjectKey(builder.eventTemplateRootObjectKey)
162                .setEventTemplateAdditionalFields(eventTemplateAdditionalFields)
163                .build();
164
165        // Compile the resolver template.
166        return TemplateResolvers.ofTemplate(resolverContext, eventTemplate);
167
168    }

170    private static String readEventTemplate(final Builder builder) {
171        return readTemplate(
172                builder.eventTemplate,
173                builder.eventTemplateUri,
174                builder.charset);
175    }
176
177    private static String readStackTraceElementTemplate(final Builder builder) {
178        return readTemplate(
179                builder.stackTraceElementTemplate,
180                builder.stackTraceElementTemplateUri,
181                builder.charset);
182    }
183
184    private static String readTemplate(
185            final String template,
186            final String templateUri,
187            final Charset charset) {
188        return Strings.isBlank(template)
189                ? Uris.readUri(templateUri, charset)
190                : template;
191    }

193    private static Recycler<Context> createContextRecycler(
194            final Builder builder,
195            final JsonWriter jsonWriter) {
196        final Supplier<Context> supplier =
197                createContextSupplier(builder.charset, jsonWriter);
198        return builder
199                .recyclerFactory
200                .create(supplier, Context::close);
201    }
202
203    private static Supplier<Context> createContextSupplier(
204            final Charset charset,
205            final JsonWriter jsonWriter) {
206        return () -> {
207            final JsonWriter clonedJsonWriter = jsonWriter.clone();
208            final Encoder<StringBuilder> encoder = new StringBuilderEncoder(charset);
209            return new Context(clonedJsonWriter, encoder);
210        };
211    }

213    /**
214     * {@link org.apache.logging.log4j.core.layout.StringBuilderEncoder} clone replacing thread-local allocations with instance fields.
215     */
216    private static final class StringBuilderEncoder implements Encoder<StringBuilder> {
217
218        private final Charset charset;
219
220        private final CharsetEncoder charsetEncoder;
221
222        private final CharBuffer charBuffer;
223
224        private final ByteBuffer byteBuffer;
225
226        private StringBuilderEncoder(final Charset charset) {
227            this.charset = charset;
228            this.charsetEncoder = charset
229                    .newEncoder()
230                    .onMalformedInput(CodingErrorAction.REPLACE)
231                    .onUnmappableCharacter(CodingErrorAction.REPLACE);
232            this.charBuffer = CharBuffer.allocate(Constants.ENCODER_CHAR_BUFFER_SIZE);
233            this.byteBuffer = ByteBuffer.allocate(Constants.ENCODER_BYTE_BUFFER_SIZE);
234        }
235
236        @Override
237        public void encode(
238                final StringBuilder source,
239                final ByteBufferDestination destination) {
240            try {
241                TextEncoderHelper.encodeText(charsetEncoder, charBuffer, byteBuffer, source, destination);
242            } catch (final Exception error) {
243                fallbackEncode(charset, source, destination, error);
244            }
245        }
246
247        private /* for JIT-ergonomics: */ static void fallbackEncode(
248                final Charset charset,
249                final StringBuilder source,
250                final ByteBufferDestination destination,
251                final Exception error) {
252            StatusLogger
253                    .getLogger()
254                    .error("TextEncoderHelper.encodeText() failure", error);
255            final byte[] bytes = source.toString().getBytes(charset);
256            destination.writeBytes(bytes, 0, bytes.length);
257        }
258
259    }
260
261    @Override
262    public byte[] toByteArray(final LogEvent event) {
263        final String eventJson = toSerializable(event);
264        return StringEncoder.toBytes(eventJson, charset);
265    }
266
267    @Override
268    public String toSerializable(final LogEvent event) {
269
270        // Acquire a context.
271        final Recycler<Context> contextRecycler = this.contextRecycler;
272        final Context context = contextRecycler.acquire();
273        final JsonWriter jsonWriter = context.jsonWriter;
274        final StringBuilder stringBuilder = jsonWriter.getStringBuilder();
275
276        // Render the JSON.
277        try {
278            eventResolver.resolve(event, jsonWriter);
279            stringBuilder.append(eventDelimiter);
280            return stringBuilder.toString();
281        }
282
283        // Release the context.
284        finally {
285            contextRecycler.release(context);
286        }
287
288    }
289
290    @Override
291    public void encode(final LogEvent event, final ByteBufferDestination destination) {
292
293        // Acquire a context.
294        final Recycler<Context> contextRecycler = this.contextRecycler;
295        final Context context = contextRecycler.acquire();
296        final JsonWriter jsonWriter = context.jsonWriter;
297        final StringBuilder stringBuilder = jsonWriter.getStringBuilder();
298        final Encoder<StringBuilder> encoder = context.encoder;
299
300        // Render & write the JSON.
301        try {
302            eventResolver.resolve(event, jsonWriter);
303            stringBuilder.append(eventDelimiter);
304            encoder.encode(stringBuilder, destination);
305        }
306
307        // Release the context.
308        finally {
309            contextRecycler.release(context);
310        }
311
312    }

575    @Plugin(name = "EventTemplateAdditionalField",
576            category = Node.CATEGORY,
577            printObject = true)
578    public static final class EventTemplateAdditionalField {
579
580        static final EventTemplateAdditionalField[] EMPTY_ARRAY = {};
581
582        public enum Format { STRING, JSON }
583
584        private final String key;
585
586        private final String value;
587
588        private final Format format;
589
590        private EventTemplateAdditionalField(final Builder builder) {
591            this.key = builder.key;
592            this.value = builder.value;
593            this.format = builder.format;
594        }
595
596        public String getKey() {
597            return key;
598        }
599
600        public String getValue() {
601            return value;
602        }
603
604        public Format getFormat() {
605            return format;
606        }
607
608        @Override
609        public boolean equals(Object object) {
610            if (this == object) {
611                return true;
612            }
613            if (object == null || getClass() != object.getClass()) {
614                return false;
615            }
616            EventTemplateAdditionalField that = (EventTemplateAdditionalField) object;
617            return key.equals(that.key) &&
618                    value.equals(that.value) &&
619                    format == that.format;
620        }
621
622        @Override
623        public int hashCode() {
624            return Objects.hash(key, value, format);
625        }
626
627        @Override
628        public String toString() {
629            final String formattedValue = Format.STRING.equals(format)
630                    ? String.format("\"%s\"", value)
631                    : value;
632            return String.format("%s=%s", key, formattedValue);
633        }
634
635        @PluginBuilderFactory
636        public static EventTemplateAdditionalField.Builder newBuilder() {
637            return new EventTemplateAdditionalField.Builder();
638        }

018package org.apache.log4j.jmx;
019
020import java.beans.BeanInfo;
021import java.beans.IntrospectionException;
022import java.beans.Introspector;
023import java.beans.PropertyDescriptor;
024import java.io.InterruptedIOException;
025import java.lang.reflect.Constructor;
026import java.lang.reflect.InvocationTargetException;
027import java.lang.reflect.Method;
028import java.util.Hashtable;
029import java.util.Vector;
030
031import javax.management.Attribute;
032import javax.management.AttributeNotFoundException;
033import javax.management.InvalidAttributeValueException;
034import javax.management.MBeanAttributeInfo;
035import javax.management.MBeanConstructorInfo;
036import javax.management.MBeanException;
037import javax.management.MBeanInfo;
038import javax.management.MBeanNotificationInfo;
039import javax.management.MBeanOperationInfo;
040import javax.management.MBeanParameterInfo;
041import javax.management.ReflectionException;
042import javax.management.RuntimeOperationsException;
043
044import org.apache.log4j.Layout;
045import org.apache.log4j.Level;
046import org.apache.log4j.Logger;
047import org.apache.log4j.helpers.OptionConverter;
048import org.apache.log4j.spi.OptionHandler;
049
050public class LayoutDynamicMBean extends AbstractDynamicMBean {
051
052    // This category instance is for logging.
053    private static Logger cat = Logger.getLogger(LayoutDynamicMBean.class);
054    private final MBeanConstructorInfo[] dConstructors = new MBeanConstructorInfo[1];
055    private final Vector dAttributes = new Vector();
056
057    private final String dClassName = this.getClass().getName();
058    private final Hashtable dynamicProps = new Hashtable(5);
059    private final MBeanOperationInfo[] dOperations = new MBeanOperationInfo[1];
060
061    private final String dDescription = "This MBean acts as a management facade for log4j layouts.";
062
063    // We wrap this layout instance.
064    private final Layout layout;
065
066    public LayoutDynamicMBean(final Layout layout) throws IntrospectionException {
067        this.layout = layout;
068        buildDynamicMBeanInfo();
069    }
070
071    private void buildDynamicMBeanInfo() throws IntrospectionException {
072        final Constructor[] constructors = this.getClass().getConstructors();
073        dConstructors[0] = new MBeanConstructorInfo("LayoutDynamicMBean(): Constructs a LayoutDynamicMBean instance", constructors[0]);
074
075        final BeanInfo bi = Introspector.getBeanInfo(layout.getClass());
076        final PropertyDescriptor[] pd = bi.getPropertyDescriptors();
077
078        final int size = pd.length;
079
080        for (int i = 0; i < size; i++) {
081            final String name = pd[i].getName();
082            final Method readMethod = pd[i].getReadMethod();
083            final Method writeMethod = pd[i].getWriteMethod();
084            if (readMethod != null) {
085                final Class returnClass = readMethod.getReturnType();
086                if (isSupportedType(returnClass)) {
087                    String returnClassName;
088                    if (returnClass.isAssignableFrom(Level.class)) {
089                        returnClassName = "java.lang.String";
090                    } else {
091                        returnClassName = returnClass.getName();
092                    }
093
094                    dAttributes.add(new MBeanAttributeInfo(name, returnClassName, "Dynamic", true, writeMethod != null, false));
095                    dynamicProps.put(name, new MethodUnion(readMethod, writeMethod));
096                }
097            }
098        }
099
100        final MBeanParameterInfo[] params = new MBeanParameterInfo[0];
101
102        dOperations[0] = new MBeanOperationInfo("activateOptions", "activateOptions(): add an layout", params, "void", MBeanOperationInfo.ACTION);
103    }
104
105    @Override
106    public Object getAttribute(final String attributeName) throws AttributeNotFoundException, MBeanException, ReflectionException {
107
108        // Check attributeName is not null to avoid NullPointerException later on
109        if (attributeName == null) {
110            throw new RuntimeOperationsException(new IllegalArgumentException("Attribute name cannot be null"),
111                "Cannot invoke a getter of " + dClassName + " with null attribute name");
112        }
113
114        final MethodUnion mu = (MethodUnion) dynamicProps.get(attributeName);
115
116        cat.debug("----name=" + attributeName + ", mu=" + mu);
117
118        if (mu != null && mu.readMethod != null) {
119            try {
120                return mu.readMethod.invoke(layout, null);
121            } catch (final InvocationTargetException e) {
122                if (e.getTargetException() instanceof InterruptedException || e.getTargetException() instanceof InterruptedIOException) {
123                    Thread.currentThread().interrupt();
124                }
125                return null;
126            } catch (final IllegalAccessException e) {
127                return null;
128            } catch (final RuntimeException e) {
129                return null;
130            }
131        }
132
133        // If attributeName has not been recognized throw an AttributeNotFoundException
134        throw (new AttributeNotFoundException("Cannot find " + attributeName + " attribute in " + dClassName));
135
136    }
137
138    @Override
139    protected Logger getLogger() {
140        return cat;
141    }
142
143    @Override
144    public MBeanInfo getMBeanInfo() {
145        cat.debug("getMBeanInfo called.");
146
147        final MBeanAttributeInfo[] attribs = new MBeanAttributeInfo[dAttributes.size()];
148        dAttributes.toArray(attribs);
149
150        return new MBeanInfo(dClassName, dDescription, attribs, dConstructors, dOperations, new MBeanNotificationInfo[0]);
151    }
152
153    @Override
154    public Object invoke(final String operationName, final Object params[], final String signature[]) throws MBeanException, ReflectionException {
155
156        if (operationName.equals("activateOptions") && layout instanceof OptionHandler) {
157            final OptionHandler oh = (OptionHandler) layout;
158            oh.activateOptions();
159            return "Options activated.";
160        }
161        return null;
162    }
163
164    private boolean isSupportedType(final Class clazz) {
165        if (clazz.isPrimitive() || (clazz == String.class) || clazz.isAssignableFrom(Level.class)) {
166            return true;
167        }
168
169        return false;
170    }
171
172    @Override
173    public void setAttribute(final Attribute attribute) throws AttributeNotFoundException, InvalidAttributeValueException, MBeanException, ReflectionException {
174
175        // Check attribute is not null to avoid NullPointerException later on
176        if (attribute == null) {
177            throw new RuntimeOperationsException(new IllegalArgumentException("Attribute cannot be null"),
178                "Cannot invoke a setter of " + dClassName + " with null attribute");
179        }
180        final String name = attribute.getName();
181        Object value = attribute.getValue();
182
183        if (name == null) {
184            throw new RuntimeOperationsException(new IllegalArgumentException("Attribute name cannot be null"),
185                "Cannot invoke the setter of " + dClassName + " with null attribute name");
186        }
187
188        final MethodUnion mu = (MethodUnion) dynamicProps.get(name);
189
190        if (mu != null && mu.writeMethod != null) {
191            final Object[] o = new Object[1];
192
193            final Class[] params = mu.writeMethod.getParameterTypes();
194            if (params[0] == org.apache.log4j.Priority.class) {
195                value = OptionConverter.toLevel((String) value, (Level) getAttribute(name));
196            }
197            o[0] = value;
198
199            try {
200                mu.writeMethod.invoke(layout, o);
201
202            } catch (final InvocationTargetException e) {
203                if (e.getTargetException() instanceof InterruptedException || e.getTargetException() instanceof InterruptedIOException) {
204                    Thread.currentThread().interrupt();
205                }
206                cat.error("FIXME", e);
207            } catch (final IllegalAccessException e) {
208                cat.error("FIXME", e);
209            } catch (final RuntimeException e) {
210                cat.error("FIXME", e);
211            }
212        } else {
213            throw (new AttributeNotFoundException("Attribute " + name + " not found in " + this.getClass().getName()));
214        }
215    }
216}

018package org.apache.log4j.helpers;
019
020import java.util.ArrayList;
021import java.util.List;
022
023import org.apache.log4j.Level;
024
025/**
026 * An extension of the Level class that provides support for java.util.logging Levels.
027 */
028public class UtilLoggingLevel extends Level {
029
030    /**
031     * Serialization version id.
032     */
033    private static final long serialVersionUID = 909301162611820211L;
034
035    /**
036     * Numerical value for SEVERE.
037     */
038    public static final int SEVERE_INT = 22000;
039    /**
040     * Numerical value for WARNING.
041     */
042    public static final int WARNING_INT = 21000;
043
044    // INFO level defined in parent as 20000..no need to redefine here
045
046    /**
047     * Numerical value for CONFIG.
048     */
049    public static final int CONFIG_INT = 14000;
050
051    /**
052     * Numerical value for FINE.
053     */
054    public static final int FINE_INT = 13000;
055    
056    /**
057     * Numerical value for FINER.
058     */
059    public static final int FINER_INT = 12000;
060    
061    /**
062     * Numerical value for FINEST.
063     */
064    public static final int FINEST_INT = 11000;
065    
066    /**
067     * Numerical value for UNKNOWN.
068     */
069    public static final int UNKNOWN_INT = 10000;
070
071    /**
072     * SEVERE.
073     */
074    public static final UtilLoggingLevel SEVERE = new UtilLoggingLevel(SEVERE_INT, "SEVERE", 0);
075    
076    /**
077     * WARNING.
078     */
079    public static final UtilLoggingLevel WARNING = new UtilLoggingLevel(WARNING_INT, "WARNING", 4);
080    
081    /**
082     * INFO.
083     */
084    // note: we've aligned the int values of the java.util.logging INFO level with log4j's level
085    public static final UtilLoggingLevel INFO = new UtilLoggingLevel(INFO_INT, "INFO", 5);
086    
087    /**
088     * CONFIG.
089     */
090    public static final UtilLoggingLevel CONFIG = new UtilLoggingLevel(CONFIG_INT, "CONFIG", 6);
091    
092    /**
093     * FINE.
094     */
095    public static final UtilLoggingLevel FINE = new UtilLoggingLevel(FINE_INT, "FINE", 7);
096    
097    /**
098     * FINER.
099     */
100    public static final UtilLoggingLevel FINER = new UtilLoggingLevel(FINER_INT, "FINER", 8);
101    
102    /**
103     * FINEST.
104     */
105    public static final UtilLoggingLevel FINEST = new UtilLoggingLevel(FINEST_INT, "FINEST", 9);
106
107    /**
108     * Create new instance.
109     *
110     * @param level numeric value for level.
111     * @param levelStr symbolic name for level.
112     * @param syslogEquivalent Equivalent syslog severity.
113     */
114    protected UtilLoggingLevel(final int level, final String levelStr, final int syslogEquivalent) {
115        super(level, levelStr, syslogEquivalent);
116    }
117
118    /**
119     * Convert an integer passed as argument to a level. If the conversion fails, then this method returns the specified
120     * default.
121     *
122     * @param val numeric value.
123     * @param defaultLevel level to be returned if no level matches numeric value.
124     * @return matching level or default level.
125     */
126    public static UtilLoggingLevel toLevel(final int val, final UtilLoggingLevel defaultLevel) {
127        switch (val) {
128        case SEVERE_INT:
129            return SEVERE;
130
131        case WARNING_INT:
132            return WARNING;
133
134        case INFO_INT:
135            return INFO;
136
137        case CONFIG_INT:
138            return CONFIG;
139
140        case FINE_INT:
141            return FINE;
142
143        case FINER_INT:
144            return FINER;
145
146        case FINEST_INT:
147            return FINEST;
148
149        default:
150            return defaultLevel;
151        }
152    }
153
154    /**
155     * Gets level matching numeric value.
156     *
157     * @param val numeric value.
158     * @return matching level or UtilLoggerLevel.FINEST if no match.
159     */
160    public static Level toLevel(final int val) {
161        return toLevel(val, FINEST);
162    }
163
164    /**
165     * Gets list of supported levels.
166     *
167     * @return list of supported levels.
168     */
169    public static List getAllPossibleLevels() {
170        final ArrayList<UtilLoggingLevel> list = new ArrayList<>();
171        list.add(FINE);
172        list.add(FINER);
173        list.add(FINEST);
174        list.add(INFO);
175        list.add(CONFIG);
176        list.add(WARNING);
177        list.add(SEVERE);
178        return list;
179    }
180
181    /**
182     * Get level with specified symbolic name.
183     *
184     * @param s symbolic name.
185     * @return matching level or Level.DEBUG if no match.
186     */
187    public static Level toLevel(final String s) {
188        return toLevel(s, Level.DEBUG);
189    }
190
191    /**
192     * Get level with specified symbolic name.
193     *
194     * @param sArg symbolic name.
195     * @param defaultLevel level to return if no match.
196     * @return matching level or defaultLevel if no match.
197     */
198    public static Level toLevel(final String sArg, final Level defaultLevel) {
199        if (sArg == null) {
200            return defaultLevel;
201        }
202
203        final String s = sArg.toUpperCase();
204
205        if (s.equals("SEVERE")) {
206            return SEVERE;
207        }
208
209        // if(s.equals("FINE")) return Level.FINE;
210        if (s.equals("WARNING")) {
211            return WARNING;
212        }
213
214        if (s.equals("INFO")) {
215            return INFO;
216        }
217
218        if (s.equals("CONFIG")) {
219            return CONFIG;
220        }
221
222        if (s.equals("FINE")) {
223            return FINE;
224        }
225
226        if (s.equals("FINER")) {
227            return FINER;
228        }
229
230        if (s.equals("FINEST")) {
231            return FINEST;
232        }
233        return defaultLevel;
234    }
235
236}

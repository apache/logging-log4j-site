46  /**
47   * Use this class to initialize the log4j environment using a DOM tree.
48   *
49   * <p>
50   * The DTD is specified in <a href="doc-files/log4j.dtd"><b>log4j.dtd</b></a>.
51   *
52   * <p>
53   * Sometimes it is useful to see how log4j is reading configuration files. You can enable log4j internal logging by
54   * defining the <b>log4j.debug</b> variable on the java command line. Alternatively, set the <code>debug</code>
55   * attribute in the <code>log4j:configuration</code> element. As in
56   *
57   * <pre>
58  &lt;log4j:configuration <b>debug="true"</b> xmlns:log4j="http://jakarta.apache.org/log4j/">
59  ...
60  &lt;/log4j:configuration>
61   * </pre>
62   *
63   * <p>
64   * There are sample XML files included in the package.
65   *
66   * @since 0.8.3
67   */
68  public class DOMConfigurator {
69  
70      public static void configure(final Element element) {
71      }
72  
73      public static void configure(final String fileName) throws FactoryConfigurationError {
74          final Path path = Paths.get(fileName);
75          try (final InputStream inputStream = Files.newInputStream(path)) {
76              final ConfigurationSource source = new ConfigurationSource(inputStream, path);
77              final LoggerContext context = (LoggerContext) org.apache.logging.log4j.LogManager.getContext(false);
78              Configuration configuration;
79              configuration = new XmlConfigurationFactory().getConfiguration(context, source);
80              LogManager.getRootLogger().removeAllAppenders();
81              Configurator.reconfigure(configuration);
82          } catch (final IOException e) {
83              throw new FactoryConfigurationError(e);
84          }
85      }
86  
87      public static void configure(final URL url) throws FactoryConfigurationError {
88          new DOMConfigurator().doConfigure(url, LogManager.getLoggerRepository());
89      }
90  
91      public static void configureAndWatch(final String fileName) {
92          // TODO Watch
93          configure(fileName);
94      }
95  
96      public static void configureAndWatch(final String fileName, final long delay) {
97          XMLWatchdog xdog = new XMLWatchdog(fileName);
98          xdog.setDelay(delay);
99          xdog.start();
100     }

114     private void doConfigure(final ConfigurationSource source) {
115         final LoggerContext context = (LoggerContext) org.apache.logging.log4j.LogManager.getContext(false);
116         Configuration configuration;
117         configuration = new XmlConfigurationFactory().getConfiguration(context, source);
118         Configurator.reconfigure(configuration);
119     }
120 
121     public void doConfigure(final Element element, final LoggerRepository repository) {
122     }
123 
124     public void doConfigure(final InputStream inputStream, final LoggerRepository repository) throws FactoryConfigurationError {
125         try {
126             doConfigure(new ConfigurationSource(inputStream));
127         } catch (final IOException e) {
128             throw new FactoryConfigurationError(e);
129         }
130     }
131 
132     public void doConfigure(final Reader reader, final LoggerRepository repository) throws FactoryConfigurationError {
133         try {
134             final StringWriter sw = new StringWriter();
135             IOUtils.copy(reader, sw);
136             doConfigure(new ConfigurationSource(new ByteArrayInputStream(sw.toString().getBytes(StandardCharsets.UTF_8))));
137         } catch (final IOException e) {
138             throw new FactoryConfigurationError(e);
139         }
140     }
141 
142     public void doConfigure(final String fileName, final LoggerRepository repository) {
143         configure(fileName);
144     }
145 
146     public void doConfigure(final URL url, final LoggerRepository repository) {
147         try {
148             final URLConnection connection = UrlConnectionFactory.createConnection(url);
149             try (InputStream inputStream = connection.getInputStream()) {
150                 doConfigure(new ConfigurationSource(inputStream, url));
151             }
152         } catch (final IOException e) {
153             throw new FactoryConfigurationError(e);
154         }
155     }

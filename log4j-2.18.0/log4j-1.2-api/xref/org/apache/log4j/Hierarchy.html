47  /**
48   * This class is specialized in retrieving loggers by name and also maintaining the logger hierarchy.
49   *
50   * <p>
51   * <em>The casual user does not have to deal with this class directly.</em>
52   * </p>
53   * <p>
54   * The structure of the logger hierarchy is maintained by the {@link #getLogger} method. The hierarchy is such that
55   * children link to their parent but parents do not have any pointers to their children. Moreover, loggers can be
56   * instantiated in any order, in particular descendant before ancestor.
57   * </p>
58   * <p>
59   * In case a descendant is created before a particular ancestor, then it creates a provision node for the ancestor and
60   * adds itself to the provision node. Other descendants of the same ancestor add themselves to the previously created
61   * provision node.
62   * </p>
63   */
64  public class Hierarchy implements LoggerRepository, RendererSupport, ThrowableRendererSupport {

144         ht = new Hashtable();

148         setThreshold(Level.ALL);
149         this.root.setHierarchy(this);
150         rendererMap = new RendererMap();
151         defaultFactory = new DefaultCategoryFactory();
152     }

219     @Override
220     public void fireAddAppenderEvent(final Category logger, final Appender appender) {
221         if (listeners != null) {
222             final int size = listeners.size();
223             HierarchyEventListener listener;
224             for (int i = 0; i < size; i++) {
225                 listener = (HierarchyEventListener) listeners.elementAt(i);
226                 listener.addAppenderEvent(logger, appender);
227             }
228         }
229     }

231     void fireRemoveAppenderEvent(final Category logger, final Appender appender) {
232         if (listeners != null) {
233             final int size = listeners.size();
234             HierarchyEventListener listener;
235             for (int i = 0; i < size; i++) {
236                 listener = (HierarchyEventListener) listeners.elementAt(i);
237                 listener.removeAppenderEvent(logger, appender);
238             }
239         }
240     }

379     /**
380      * Resets all values contained in this hierarchy instance to their default. This removes all appenders from all
381      * categories, sets the level of all non-root categories to <code>null</code>, sets their additivity flag to
382      * <code>true</code> and sets the level of the root logger to {@link Level#DEBUG DEBUG}. Moreover, message disabling is
383      * set its default "off" value.
384      *
385      * <p>
386      * Existing categories are not removed. They are just reset.
387      * </p>
388      *
389      * <p>
390      * This method should be used sparingly and with care as it will block all logging until it is completed.
391      * </p>
392      *
393      * @since 0.8.5
394      */
395     @Override
396     public void resetConfiguration() {
397         resetConfiguration(getContext());
398     }
399 
400     void resetConfiguration(final ClassLoader classLoader) {
401         resetConfiguration(getContext(classLoader));
402     }

404     void resetConfiguration(final LoggerContext loggerContext) {
405         getLoggersMap(loggerContext).clear();
406 
407         getRootLogger().setLevel(Level.DEBUG);
408         root.setResourceBundle(null);
409         setThreshold(Level.ALL);
410 
411         // the synchronization is needed to prevent JDK 1.2.x hashtable
412         // surprises
413         synchronized (ht) {
414             shutdown(); // nested locks are OK
415 
416             final Enumeration cats = getCurrentLoggers();
417             while (cats.hasMoreElements()) {
418                 final Logger c = (Logger) cats.nextElement();
419                 c.setLevel(null);
420                 c.setAdditivity(true);
421                 c.setResourceBundle(null);
422             }
423         }
424         rendererMap.clear();
425         throwableRenderer = null;
426     }

530     /**
531      * We update the links for all the children that placed themselves in the provision node 'pn'. The second argument 'cat'
532      * is a reference for the newly created Logger, parent of all the children in 'pn'
533      *
534      * We loop on all the children 'c' in 'pn':
535      *
536      * If the child 'c' has been already linked to a child of 'cat' then there is no need to update 'c'.
537      *
538      * Otherwise, we set cat's parent field to c's parent and set c's parent field to cat.
539      *
540      */
541     final private void updateChildren(final ProvisionNode pn, final Logger logger) {
542         // System.out.println("updateChildren called for " + logger.name);
543         final int last = pn.size();
544 
545         for (int i = 0; i < last; i++) {
546             final Logger l = (Logger) pn.elementAt(i);
547             // System.out.println("Updating child " +p.name);
548 
549             // Unless this child already points to a correct (lower) parent,
550             // make cat.parent point to l.parent and l.parent to cat.
551             if (!l.parent.name.startsWith(logger.name)) {
552                 logger.parent = l.parent;
553                 l.parent = logger;
554             }
555         }
556     }

558     /**
559      * This method loops through all the *potential* parents of 'cat'. There 3 possible cases:
560      *
561      * 1) No entry for the potential parent of 'cat' exists
562      *
563      * We create a ProvisionNode for this potential parent and insert 'cat' in that provision node.
564      *
565      * 2) There entry is of type Logger for the potential parent.
566      *
567      * The entry is 'cat's nearest existing parent. We update cat's parent field with this entry. We also break from the
568      * loop because updating our parent's parent is our parent's responsibility.
569      *
570      * 3) There entry is of type ProvisionNode for this potential parent.
571      *
572      * We add 'cat' to the list of children for this potential parent.
573      */
574     final private void updateParents(final Logger cat) {
575         final String name = cat.name;
576         final int length = name.length();
577         boolean parentFound = false;
578 
579         // System.out.println("UpdateParents called for " + name);
580 
581         // if name = "w.x.y.z", loop thourgh "w.x.y", "w.x" and "w", but not "w.x.y.z"
582         for (int i = name.lastIndexOf('.', length - 1); i >= 0; i = name.lastIndexOf('.', i - 1)) {
583             final String substr = name.substring(0, i);
584 
585             // System.out.println("Updating parent : " + substr);
586             final CategoryKey key = new CategoryKey(substr); // simple constructor
587             final Object o = ht.get(key);
588             // Create a provision node for a future parent.
589             if (o == null) {
590                 // System.out.println("No parent "+substr+" found. Creating ProvisionNode.");
591                 final ProvisionNode pn = new ProvisionNode(cat);
592                 ht.put(key, pn);
593             } else if (o instanceof Category) {
594                 parentFound = true;
595                 cat.parent = (Category) o;
596                 // System.out.println("Linking " + cat.name + " -> " + ((Category) o).name);
597                 break; // no need to update the ancestors of the closest ancestor
598             } else if (o instanceof ProvisionNode) {
599                 ((ProvisionNode) o).addElement(cat);
600             } else {
601                 final Exception e = new IllegalStateException("unexpected object type " + o.getClass() + " in ht.");
602                 e.printStackTrace();
603             }
604         }
605         // If we could not find any existing parents, then link with root.
606         if (!parentFound) {
607             cat.parent = root;
608         }
609     }

18  package org.apache.log4j.varia;
19  
20  import org.apache.log4j.Level;
21  import org.apache.log4j.helpers.OptionConverter;
22  import org.apache.log4j.spi.Filter;
23  import org.apache.log4j.spi.LoggingEvent;
24  
25  /**
26   * Simple filter based on level matching.
27   *
28   * <p>
29   * The filter admits two options <b>LevelToMatch</b> and <b>AcceptOnMatch</b>. If there is an exact match between the
30   * value of the <b>LevelToMatch</b> option and the level of the {@link LoggingEvent}, then the {@link #decide} method
31   * returns {@link Filter#ACCEPT} in case the <b>AcceptOnMatch</b> option value is set to <code>true</code>, if it is
32   * <code>false</code> then {@link Filter#DENY} is returned. If there is no match, {@link Filter#NEUTRAL} is returned.
33   * </p>
34   *
35   * @since 1.2
36   */
37  public class LevelMatchFilter extends Filter {
38  
39      /**
40       * Do we return ACCEPT when a match occurs. Default is <code>true</code>.
41       */
42      boolean acceptOnMatch = true;
43  
44      /**
45       */
46      Level levelToMatch;
47  
48      /**
49       * Return the decision of this filter.
50       *
51       * Returns {@link Filter#NEUTRAL} if the <b>LevelToMatch</b> option is not set or if there is not match. Otherwise, if
52       * there is a match, then the returned decision is {@link Filter#ACCEPT} if the <b>AcceptOnMatch</b> property is set to
53       * <code>true</code>. The returned decision is {@link Filter#DENY} if the <b>AcceptOnMatch</b> property is set to false.
54       *
55       */
56      @Override
57      public int decide(final LoggingEvent event) {
58          if (this.levelToMatch == null) {
59              return Filter.NEUTRAL;
60          }
61  
62          boolean matchOccured = false;
63          if (this.levelToMatch.equals(event.getLevel())) {
64              matchOccured = true;
65          }
66  
67          if (matchOccured) {
68              if (this.acceptOnMatch) {
69                  return Filter.ACCEPT;
70              }
71              return Filter.DENY;
72          }
73          return Filter.NEUTRAL;
74      }
75  
76      public boolean getAcceptOnMatch() {
77          return acceptOnMatch;
78      }
79  
80      public String getLevelToMatch() {
81          return levelToMatch == null ? null : levelToMatch.toString();
82      }
83  
84      public void setAcceptOnMatch(final boolean acceptOnMatch) {
85          this.acceptOnMatch = acceptOnMatch;
86      }
87  
88      public void setLevelToMatch(final String level) {
89          levelToMatch = OptionConverter.toLevel(level, null);
90      }
91  }

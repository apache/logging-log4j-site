47  public class LoggerDynamicMBean extends AbstractDynamicMBean implements NotificationListener {
48  
49      // This Logger instance is for logging.
50      private static Logger cat = Logger.getLogger(LoggerDynamicMBean.class);
51      private final MBeanConstructorInfo[] dConstructors = new MBeanConstructorInfo[1];
52  
53      private final MBeanOperationInfo[] dOperations = new MBeanOperationInfo[1];
54      private final Vector dAttributes = new Vector();
55  
56      private final String dClassName = this.getClass().getName();
57  
58      private final String dDescription = "This MBean acts as a management facade for a org.apache.log4j.Logger instance.";
59  
60      // We wrap this Logger instance.
61      private final Logger logger;
62  
63      public LoggerDynamicMBean(final Logger logger) {
64          this.logger = logger;
65          buildDynamicMBeanInfo();
66      }
67  
68      void addAppender(final String appenderClass, final String appenderName) {
69          cat.debug("addAppender called with " + appenderClass + ", " + appenderName);
70          final Appender appender = (Appender) OptionConverter.instantiateByClassName(appenderClass, org.apache.log4j.Appender.class, null);
71          appender.setName(appenderName);
72          logger.addAppender(appender);
73  
74          // appenderMBeanRegistration();
75  
76      }
77  
78      void appenderMBeanRegistration() {
79          final Enumeration enumeration = logger.getAllAppenders();
80          while (enumeration.hasMoreElements()) {
81              final Appender appender = (Appender) enumeration.nextElement();
82              registerAppenderMBean(appender);
83          }
84      }
85  
86      private void buildDynamicMBeanInfo() {
87          final Constructor[] constructors = this.getClass().getConstructors();
88          dConstructors[0] = new MBeanConstructorInfo("HierarchyDynamicMBean(): Constructs a HierarchyDynamicMBean instance", constructors[0]);
89  
90          dAttributes.add(new MBeanAttributeInfo("name", "java.lang.String", "The name of this Logger.", true, false, false));
91  
92          dAttributes.add(new MBeanAttributeInfo("priority", "java.lang.String", "The priority of this logger.", true, true, false));
93  
94          final MBeanParameterInfo[] params = new MBeanParameterInfo[2];
95          params[0] = new MBeanParameterInfo("class name", "java.lang.String", "add an appender to this logger");
96          params[1] = new MBeanParameterInfo("appender name", "java.lang.String", "name of the appender");
97  
98          dOperations[0] = new MBeanOperationInfo("addAppender", "addAppender(): add an appender", params, "void", MBeanOperationInfo.ACTION);
99      }
100 
101     @Override
102     public Object getAttribute(final String attributeName) throws AttributeNotFoundException, MBeanException, ReflectionException {
103 
104         // Check attributeName is not null to avoid NullPointerException later on
105         if (attributeName == null) {
106             throw new RuntimeOperationsException(new IllegalArgumentException("Attribute name cannot be null"),
107                 "Cannot invoke a getter of " + dClassName + " with null attribute name");
108         }
109 
110         // Check for a recognized attributeName and call the corresponding getter
111         if (attributeName.equals("name")) {
112             return logger.getName();
113         } else if (attributeName.equals("priority")) {
114             final Level l = logger.getLevel();
115             if (l == null) {
116                 return null;
117             } else {
118                 return l.toString();
119             }
120         } else if (attributeName.startsWith("appender=")) {
121             try {
122                 return new ObjectName("log4j:" + attributeName);
123             } catch (final MalformedObjectNameException e) {
124                 cat.error("Could not create ObjectName" + attributeName);
125             } catch (final RuntimeException e) {
126                 cat.error("Could not create ObjectName" + attributeName);
127             }
128         }
129 
130         // If attributeName has not been recognized throw an AttributeNotFoundException
131         throw (new AttributeNotFoundException("Cannot find " + attributeName + " attribute in " + dClassName));
132 
133     }
134 
135     @Override
136     protected Logger getLogger() {
137         return logger;
138     }

140     @Override
141     public MBeanInfo getMBeanInfo() {
142         // cat.debug("getMBeanInfo called.");
143 
144         final MBeanAttributeInfo[] attribs = new MBeanAttributeInfo[dAttributes.size()];
145         dAttributes.toArray(attribs);
146 
147         final MBeanInfo mb = new MBeanInfo(dClassName, dDescription, attribs, dConstructors, dOperations, new MBeanNotificationInfo[0]);
148         // cat.debug("getMBeanInfo exit.");
149         return mb;
150     }
151 
152     @Override
153     public void handleNotification(final Notification notification, final Object handback) {
154         cat.debug("Received notification: " + notification.getType());
155         registerAppenderMBean((Appender) notification.getUserData());
156 
157     }
158 
159     @Override
160     public Object invoke(final String operationName, final Object params[], final String signature[]) throws MBeanException, ReflectionException {
161 
162         if (operationName.equals("addAppender")) {
163             addAppender((String) params[0], (String) params[1]);
164             return "Hello world.";
165         }
166 
167         return null;
168     }
169 
170     @Override
171     public void postRegister(final java.lang.Boolean registrationDone) {
172         appenderMBeanRegistration();
173     }
174 
175     void registerAppenderMBean(final Appender appender) {
176         final String name = getAppenderName(appender);
177         cat.debug("Adding AppenderMBean for appender named " + name);
178         ObjectName objectName = null;
179         try {
180             final AppenderDynamicMBean appenderMBean = new AppenderDynamicMBean(appender);
181             objectName = new ObjectName("log4j", "appender", name);
182             if (!server.isRegistered(objectName)) {
183                 registerMBean(appenderMBean, objectName);
184                 dAttributes.add(new MBeanAttributeInfo("appender=" + name, "javax.management.ObjectName", "The " + name + " appender.", true, true, false));
185             }
186 
187         } catch (final JMException e) {
188             cat.error("Could not add appenderMBean for [" + name + "].", e);
189         } catch (final java.beans.IntrospectionException e) {
190             cat.error("Could not add appenderMBean for [" + name + "].", e);
191         } catch (final RuntimeException e) {
192             cat.error("Could not add appenderMBean for [" + name + "].", e);
193         }
194     }
195 
196     @Override
197     public void setAttribute(final Attribute attribute) throws AttributeNotFoundException, InvalidAttributeValueException, MBeanException, ReflectionException {
198 
199         // Check attribute is not null to avoid NullPointerException later on
200         if (attribute == null) {
201             throw new RuntimeOperationsException(new IllegalArgumentException("Attribute cannot be null"),
202                 "Cannot invoke a setter of " + dClassName + " with null attribute");
203         }
204         final String name = attribute.getName();
205         final Object value = attribute.getValue();
206 
207         if (name == null) {
208             throw new RuntimeOperationsException(new IllegalArgumentException("Attribute name cannot be null"),
209                 "Cannot invoke the setter of " + dClassName + " with null attribute name");
210         }
211 
212         if (name.equals("priority")) {
213             if (value instanceof String) {
214                 final String s = (String) value;
215                 Level p = logger.getLevel();
216                 if (s.equalsIgnoreCase("NULL")) {
217                     p = null;
218                 } else {
219                     p = OptionConverter.toLevel(s, p);
220                 }
221                 logger.setLevel(p);
222             }
223         } else {
224             throw (new AttributeNotFoundException("Attribute " + name + " not found in " + this.getClass().getName()));
225         }
226     }

018package org.apache.logging.log4j.core.appender.db;
019
020import java.io.Flushable;
021import java.io.Serializable;
022import java.util.ArrayList;
023import java.util.concurrent.TimeUnit;
024
025import org.apache.logging.log4j.core.Layout;
026import org.apache.logging.log4j.core.LogEvent;
027import org.apache.logging.log4j.core.appender.AbstractManager;
028import org.apache.logging.log4j.core.appender.ManagerFactory;
029import org.apache.logging.log4j.core.config.Configuration;

031/**
032 * Manager that allows database appenders to have their configuration reloaded without losing events.
033 */
034public abstract class AbstractDatabaseManager extends AbstractManager implements Flushable {
035    
036    /**
037     * Implementations should extend this class for passing data between the getManager method and the manager factory
038     * class.
039     */
040    protected abstract static class AbstractFactoryData extends AbstractManager.AbstractFactoryData {
041        private final int bufferSize;
042        private final Layout<? extends Serializable> layout;
043
044        /**
045         * Constructs the base factory data.
046         *
047         * @param bufferSize The size of the buffer.
048         * @param layout The appender-level layout
049         * @deprecated Use {@link AbstractFactoryData#AbstractFactoryData(Configuration, int, Layout)}.
050         */
051        protected AbstractFactoryData(final int bufferSize, final Layout<? extends Serializable> layout) {
052            this(null, bufferSize, layout);
053        }
054
055        /**
056         * Constructs the base factory data.
057         * @param configuration Configuration creating this instance.
058         * @param bufferSize The size of the buffer.
059         * @param layout The appender-level layout
060         */
061        protected AbstractFactoryData(final Configuration configuration, final int bufferSize, final Layout<? extends Serializable> layout) {
062            super(configuration);
063            this.bufferSize = bufferSize;
064            this.layout = layout;
065        }
066
067        /**
068         * Gets the buffer size.
069         *
070         * @return the buffer size.
071         */
072        public int getBufferSize() {
073            return bufferSize;
074        }
075
076        /**
077         * Gets the layout.
078         *
079         * @return the layout.
080         */
081        public Layout<? extends Serializable> getLayout() {
082            return layout;
083        }
084
085    }
086
087    /**
088     * Implementations should define their own getManager method and call this method from that to create or get
089     * existing managers.
090     *
091     * @param name The manager name, which should include any configuration details that one might want to be able to
092     *             reconfigure at runtime, such as database name, username, (hashed) password, etc.
093     * @param data The concrete instance of {@link AbstractFactoryData} appropriate for the given manager.
094     * @param factory A factory instance for creating the appropriate manager.
095     * @param <M> The concrete manager type.
096     * @param <T> The concrete {@link AbstractFactoryData} type.
097     * @return a new or existing manager of the specified type and name.
098     */
099    protected static <M extends AbstractDatabaseManager, T extends AbstractFactoryData> M getManager(final String name, final T data,
100        final ManagerFactory<M, T> factory) {
101        return AbstractManager.getManager(name, factory, data);
102    }
103
104    private final ArrayList<LogEvent> buffer;
105    private final int bufferSize;
106
107    private final Layout<? extends Serializable> layout;
108
109    private boolean running;
110
111    /**
112     * Constructs the base manager.
113     *
114     * @param name The manager name, which should include any configuration details that one might want to be able to
115     *             reconfigure at runtime, such as database name, username, (hashed) password, etc.
116     * @param bufferSize The size of the log event buffer.
117     * @deprecated Use {@link AbstractDatabaseManager#AbstractDatabaseManager(String, int, Layout, Configuration)}.
118     */
119    @Deprecated
120    protected AbstractDatabaseManager(final String name, final int bufferSize) {
121        this(name, bufferSize, null);
122    }
123
124    /**
125     * Constructs the base manager.
126     *
127     * @param name The manager name, which should include any configuration details that one might want to be able to
128     *             reconfigure at runtime, such as database name, username, (hashed) password, etc.
129     * @param layout the Appender-level layout.
130     * @param bufferSize The size of the log event buffer.
131     * @deprecated Use {@link AbstractDatabaseManager#AbstractDatabaseManager(String, int, Layout, Configuration)}.
132     */
133    @Deprecated
134    protected AbstractDatabaseManager(final String name, final int bufferSize, final Layout<? extends Serializable> layout) {
135        this(name, bufferSize, layout, null);
136    }
137
138    /**
139     * Constructs the base manager.
140     *
141     * @param name The manager name, which should include any configuration details that one might want to be able to
142     *             reconfigure at runtime, such as database name, username, (hashed) password, etc.
143     * @param layout the Appender-level layout.
144     * @param bufferSize The size of the log event buffer.
145     * @param configuration My configuration.
146     */
147    protected AbstractDatabaseManager(final String name, final int bufferSize, final Layout<? extends Serializable> layout, final Configuration configuration) {
148        // null configuration allowed for backward compatibility.
149        // TODO should super track Configuration instead of LoggerContext?
150        super(configuration != null ? configuration.getLoggerContext() : null, name);
151        this.bufferSize = bufferSize;
152        this.buffer = new ArrayList<>(bufferSize + 1);
153        this.layout = layout; // A null layout is allowed.
154    }
155
156    protected void buffer(final LogEvent event) {
157        this.buffer.add(event.toImmutable());
158        if (this.buffer.size() >= this.bufferSize || event.isEndOfBatch()) {
159            this.flush();
160        }
161    }
162
163    /**
164     * Commits any active transaction (if applicable) and disconnects from the database (returns the connection to the
165     * connection pool). With buffering enabled, this is called when flushing the buffer completes, after the last call
166     * to {@link #writeInternal}. With buffering disabled, this is called immediately after every invocation of
167     * {@link #writeInternal}.
168     * @return true if all resources were closed normally, false otherwise.
169     */
170    protected abstract boolean commitAndClose();
171
172    /**
173     * Connects to the database and starts a transaction (if applicable). With buffering enabled, this is called when
174     * flushing the buffer begins, before the first call to {@link #writeInternal}. With buffering disabled, this is
175     * called immediately before every invocation of {@link #writeInternal}.
176     */
177    protected abstract void connectAndStart();
178
179    /**
180     * This method is called automatically when the buffer size reaches its maximum or at the beginning of a call to
181     * {@link #shutdown()}. It can also be called manually to flush events to the database.
182     */
183    @Override
184    public final synchronized void flush() {
185        if (this.isRunning() && isBuffered()) {
186            this.connectAndStart();
187            try {
188                for (final LogEvent event : this.buffer) {
189                    this.writeInternal(event, layout != null ? layout.toSerializable(event) : null);
190                }
191            } finally {
192                this.commitAndClose();
193                // not sure if this should be done when writing the events failed
194                this.buffer.clear();
195            }
196        }
197    }
198
199    protected boolean isBuffered() {
200        return this.bufferSize > 0;
201    }
202
203    /**
204     * Indicates whether the manager is currently connected {@link #startup()} has been called and {@link #shutdown()}
205     * has not been called).
206     *
207     * @return {@code true} if the manager is connected.
208     */
209    public final boolean isRunning() {
210        return this.running;
211    }
212
213    @Override
214    public final boolean releaseSub(final long timeout, final TimeUnit timeUnit) {
215        return this.shutdown();
216    }
217
218    /**
219     * This method is called from the {@link #close()} method when the appender is stopped or the appender's manager
220     * is replaced. If it has not already been called, it calls {@link #shutdownInternal()} and catches any exceptions
221     * it might throw.
222     * @return true if all resources were closed normally, false otherwise.
223     */
224    public final synchronized boolean shutdown() {
225        boolean closed = true;
226        this.flush();
227        if (this.isRunning()) {
228            try {
229                closed &= this.shutdownInternal();
230            } catch (final Exception e) {
231                logWarn("Caught exception while performing database shutdown operations", e);
232                closed = false;
233            } finally {
234                this.running = false;
235            }
236        }
237        return closed;
238    }
239
240    /**
241     * Implementations should implement this method to perform any proprietary disconnection / shutdown operations. This
242     * method will never be called twice on the same instance, and it will only be called <em>after</em>
243     * {@link #startupInternal()}. It is safe to throw any exceptions from this method. This method does not
244     * necessarily disconnect from the database for the same reasons outlined in {@link #startupInternal()}.
245     * @return true if all resources were closed normally, false otherwise.
246     */
247    protected abstract boolean shutdownInternal() throws Exception;
248
249    /**
250     * This method is called within the appender when the appender is started. If it has not already been called, it
251     * calls {@link #startupInternal()} and catches any exceptions it might throw.
252     */
253    public final synchronized void startup() {
254        if (!this.isRunning()) {
255            try {
256                this.startupInternal();
257                this.running = true;
258            } catch (final Exception e) {
259                logError("Could not perform database startup operations", e);
260            }
261        }
262    }
263
264    /**
265     * Implementations should implement this method to perform any proprietary startup operations. This method will
266     * never be called twice on the same instance. It is safe to throw any exceptions from this method. This method
267     * does not necessarily connect to the database, as it is generally unreliable to connect once and use the same
268     * connection for hours.
269     */
270    protected abstract void startupInternal() throws Exception;
271
272    @Override
273    public final String toString() {
274        return this.getName();
275    }
276
277    /**
278     * This method manages buffering and writing of events.
279     *
280     * @param event The event to write to the database.
281     * @deprecated since 2.11.0 Use {@link #write(LogEvent, Serializable)}.
282     */
283    @Deprecated
284    public final synchronized void write(final LogEvent event) {
285        write(event, null);
286    }
287    
288    /**
289     * This method manages buffering and writing of events.
290     *
291     * @param event The event to write to the database.
292     * @param serializable Serializable event
293     */
294    public final synchronized void write(final LogEvent event, final Serializable serializable) {
295        if (isBuffered()) {
296            buffer(event);
297        } else {
298            writeThrough(event, serializable);
299        }
300    }
301
302    /**
303     * Performs the actual writing of the event in an implementation-specific way. This method is called immediately
304     * from {@link #write(LogEvent, Serializable)} if buffering is off, or from {@link #flush()} if the buffer has reached its limit.
305     *
306     * @param event The event to write to the database.
307     * @deprecated Use {@link #writeInternal(LogEvent, Serializable)}.
308     */
309    @Deprecated
310    protected void writeInternal(final LogEvent event) {
311        writeInternal(event, null);
312    }
313
314    /**
315     * Performs the actual writing of the event in an implementation-specific way. This method is called immediately
316     * from {@link #write(LogEvent, Serializable)} if buffering is off, or from {@link #flush()} if the buffer has reached its limit.
317     *
318     * @param event The event to write to the database.
319     */
320    protected abstract void writeInternal(LogEvent event, Serializable serializable);
321
322    protected void writeThrough(final LogEvent event, final Serializable serializable) {
323        this.connectAndStart();
324        try {
325            this.writeInternal(event, serializable);
326        } finally {
327            this.commitAndClose();
328        }
329    }

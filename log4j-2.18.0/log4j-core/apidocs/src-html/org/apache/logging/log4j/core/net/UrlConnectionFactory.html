017package org.apache.logging.log4j.core.net;
018
019import java.io.IOException;
020import java.net.HttpURLConnection;
021import java.net.JarURLConnection;
022import java.net.ProtocolException;
023import java.net.URL;
024import java.net.URLConnection;
025import java.util.Arrays;
026import java.util.List;
027import java.util.Locale;
028
029import javax.net.ssl.HttpsURLConnection;
030
031import org.apache.logging.log4j.core.config.ConfigurationFactory;
032import org.apache.logging.log4j.core.net.ssl.LaxHostnameVerifier;
033import org.apache.logging.log4j.core.net.ssl.SslConfiguration;
034import org.apache.logging.log4j.core.net.ssl.SslConfigurationFactory;
035import org.apache.logging.log4j.core.util.AuthorizationProvider;
036import org.apache.logging.log4j.util.PropertiesUtil;
037import org.apache.logging.log4j.util.Strings;
038
039/**
040 * Constructs an HTTPURLConnection. This class should be considered to be internal
041 */
042public class UrlConnectionFactory {
043
044    private static final int DEFAULT_TIMEOUT = 60000;
045    private static final int connectTimeoutMillis = DEFAULT_TIMEOUT;
046    private static final int readTimeoutMillis = DEFAULT_TIMEOUT;
047    private static final String JSON = "application/json";
048    private static final String XML = "application/xml";
049    private static final String PROPERTIES = "text/x-java-properties";
050    private static final String TEXT = "text/plain";
051    private static final String HTTP = "http";
052    private static final String HTTPS = "https";
053    private static final String JAR = "jar";
054    private static final String DEFAULT_ALLOWED_PROTOCOLS = "https, file, jar";
055    private static final String NO_PROTOCOLS = "_none";
056    public static final String ALLOWED_PROTOCOLS = "log4j2.Configuration.allowedProtocols";
057
058    @SuppressWarnings("unchecked")
059    public static <T extends URLConnection> T createConnection(final URL url, final long lastModifiedMillis,
060            final SslConfiguration sslConfiguration, final AuthorizationProvider authorizationProvider)
061        throws IOException {
062        final PropertiesUtil props = PropertiesUtil.getProperties();
063        final List<String> allowed = Arrays.asList(Strings.splitList(props
064                .getStringProperty(ALLOWED_PROTOCOLS, DEFAULT_ALLOWED_PROTOCOLS).toLowerCase(Locale.ROOT)));
065        if (allowed.size() == 1 && NO_PROTOCOLS.equals(allowed.get(0))) {
066            throw new ProtocolException("No external protocols have been enabled");
067        }
068        final String protocol = url.getProtocol();
069        if (protocol == null) {
070            throw new ProtocolException("No protocol was specified on " + url.toString());
071        }
072        if (!allowed.contains(protocol)) {
073            throw new ProtocolException("Protocol " + protocol + " has not been enabled as an allowed protocol");
074        }
075        URLConnection urlConnection;
076        if (url.getProtocol().equals(HTTP) || url.getProtocol().equals(HTTPS)) {
077            final HttpURLConnection httpURLConnection = (HttpURLConnection) url.openConnection();
078            if (authorizationProvider != null) {
079                authorizationProvider.addAuthorization(httpURLConnection);
080            }
081            httpURLConnection.setAllowUserInteraction(false);
082            httpURLConnection.setDoOutput(true);
083            httpURLConnection.setDoInput(true);
084            httpURLConnection.setRequestMethod("GET");
085            if (connectTimeoutMillis > 0) {
086                httpURLConnection.setConnectTimeout(connectTimeoutMillis);
087            }
088            if (readTimeoutMillis > 0) {
089                httpURLConnection.setReadTimeout(readTimeoutMillis);
090            }
091            final String[] fileParts = url.getFile().split("\\.");
092            final String type = fileParts[fileParts.length - 1].trim();
093            final String contentType = isXml(type) ? XML : isJson(type) ? JSON : isProperties(type) ? PROPERTIES : TEXT;
094            httpURLConnection.setRequestProperty("Content-Type", contentType);
095            if (lastModifiedMillis > 0) {
096                httpURLConnection.setIfModifiedSince(lastModifiedMillis);
097            }
098            if (url.getProtocol().equals(HTTPS) && sslConfiguration != null) {
099                ((HttpsURLConnection) httpURLConnection).setSSLSocketFactory(sslConfiguration.getSslSocketFactory());
100                if (!sslConfiguration.isVerifyHostName()) {
101                    ((HttpsURLConnection) httpURLConnection).setHostnameVerifier(LaxHostnameVerifier.INSTANCE);
102                }
103            }
104            urlConnection = httpURLConnection;
105        } else if (url.getProtocol().equals(JAR)) {
106            urlConnection = url.openConnection();
107            urlConnection.setUseCaches(false);
108        } else {
109            urlConnection = url.openConnection();
110        }
111        return (T) urlConnection;
112    }
113
114    public static URLConnection createConnection(final URL url) throws IOException {
115        URLConnection urlConnection = null;
116        if (url.getProtocol().equals(HTTPS) || url.getProtocol().equals(HTTP)) {
117            final AuthorizationProvider provider = ConfigurationFactory.authorizationProvider(PropertiesUtil.getProperties());
118            urlConnection = createConnection(url, 0, SslConfigurationFactory.getSslConfiguration(), provider);
119        } else {
120            urlConnection = url.openConnection();
121            if (urlConnection instanceof JarURLConnection) {
122                // A "jar:" URL file remains open after the stream is closed, so do not cache it.
123                urlConnection.setUseCaches(false);
124            }
125        }
126        return urlConnection;
127    }
128
129    private static boolean isXml(final String type) {
130        return type.equalsIgnoreCase("xml");
131    }
132
133    private static boolean isJson(final String type) {
134        return type.equalsIgnoreCase("json") || type.equalsIgnoreCase("jsn");
135    }
136
137    private static boolean isProperties(final String type) {
138        return type.equalsIgnoreCase("properties");
139    }
140}

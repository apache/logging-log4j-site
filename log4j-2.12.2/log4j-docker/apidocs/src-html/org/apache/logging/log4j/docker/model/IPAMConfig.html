017package org.apache.logging.log4j.docker.model;
018
019import com.fasterxml.jackson.annotation.JsonProperty;
020
021/**
022 *
023 */
024public class IPAMConfig {
025    
026    @JsonProperty("Subnet")
027    private String subnet;
028    
029    @JsonProperty("IPRange")
030    private String ipRange;
031    
032    @JsonProperty("Gateway")
033    private String gateway;
034
035    @JsonProperty("IPv4Address")
036    private String ipv4Address;
037
038    public String getSubnet() {
039        return subnet;
040    }
041
042    public void setSubnet(String subnet) {
043        this.subnet = subnet;
044    }
045
046    public String getIpRange() {
047        return ipRange;
048    }
049
050    public void setIpRange(String ipRange) {
051        this.ipRange = ipRange;
052    }
053
054    public String getGateway() {
055        return gateway;
056    }
057
058    public void setGateway(String gateway) {
059        this.gateway = gateway;
060    }
061
062    public String getIpv4Address() {
063        return ipv4Address;
064    }
065
066    public void setIpv4Address(String ipv4Address) {
067        this.ipv4Address = ipv4Address;
068    }
069}

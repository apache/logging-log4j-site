18  package org.apache.logging.log4j.core.appender.rolling.action;
19  
20  import java.nio.file.Files;
21  import java.nio.file.Path;
22  import java.nio.file.attribute.BasicFileAttributes;
23  import java.util.Arrays;
24  
25  /**
26   * Filter that accepts or rejects a candidate {@code Path} for deletion.
27   */
28  public interface PathCondition {
29      
30      /**
31       * The empty array.
32       */
33      static final PathCondition[] EMPTY_ARRAY = {};
34  
35      /**
36       * Copies the given input.
37       *
38       * @param source What to copy
39       * @return a copy, never null.
40       */
41      static PathCondition[] copy(PathCondition[] source) {
42          return source == null ? EMPTY_ARRAY : Arrays.copyOf(source, source.length);
43      }
44      
45      /**
46       * Invoked before a new {@linkplain Files#walkFileTree(Path, java.util.Set, int, java.nio.file.FileVisitor) file
47       * tree walk} is started. Stateful PathConditions can reset their state when this method is called.
48       */
49      void beforeFileTreeWalk();
50  
51      /**
52       * Returns {@code true} if the specified candidate path should be deleted, {@code false} otherwise.
53       *
54       * @param baseDir the directory from where to start scanning for deletion candidate files
55       * @param relativePath the candidate for deletion. This path is relative to the baseDir.
56       * @param attrs attributes of the candidate path
57       * @return whether the candidate path should be deleted
58       */
59      boolean accept(final Path baseDir, final Path relativePath, final BasicFileAttributes attrs);
60  }

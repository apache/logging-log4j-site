017package org.apache.logging.log4j.docker.model;
018
019import com.fasterxml.jackson.annotation.JsonProperty;
020
021/**
022 *
023 */
024public class PortDefinition {
025
026    @JsonProperty("IP")
027    private String ip;
028    
029    @JsonProperty("PrivatePort")
030    private Integer privatePort;
031
032    @JsonProperty("PublicPort")
033    private Integer publicPort;
034
035    @JsonProperty("Type")
036    private String type;
037
038    public String getIp() {
039        return ip;
040    }
041
042    public void setIp(String ip) {
043        this.ip = ip;
044    }
045
046    public Integer getPrivatePort() {
047        return privatePort;
048    }
049
050    public void setPrivatePort(Integer privatePort) {
051        this.privatePort = privatePort;
052    }
053
054    public Integer getPublicPort() {
055        return publicPort;
056    }
057
058    public void setPublicPort(Integer publicPort) {
059        this.publicPort = publicPort;
060    }
061
062    public String getType() {
063        return type;
064    }
065
066    public void setType(String type) {
067        this.type = type;
068    }
069}
